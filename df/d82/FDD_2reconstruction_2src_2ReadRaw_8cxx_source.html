32 void ReadRaw::readRawData(const LookUpTable& lut)
33 {
34  LOG(INFO) << "o2::fdd::ReadRaw::readRawData():Start.";
35  constexpr int CRUWordSize = 16;
36  o2::header::RAWDataHeader mRDH;
37  ChannelData chData; // output to store digits
38  EventHeader eventHeader; // raw data container
39  TCMdata mTCMdata; // raw data container
40  EventData eventData[NChPerMod]; // raw data container
41 
42  // Get input RAW file size
43  mRawFileIn.seekg(0, mRawFileIn.end);
44  long rawFileSize = mRawFileIn.tellg();
45  mRawFileIn.seekg(0);
46 
47  long posInFile = 0;
48  while (posInFile < rawFileSize - sizeof(mRDH)) {
49  // Read contents of the next RDH
50  mRawFileIn.seekg(posInFile);
51  mRawFileIn.read(reinterpret_cast<char*>(&mRDH), sizeof(mRDH));
52  int nwords = RDHUtils::getMemorySize(mRDH);
53  int offset = RDHUtils::getOffsetToNext(mRDH);
54  int link = RDHUtils::getLinkID(mRDH);
55  posInFile += offset; // posInFile is now prepared for the next iteration of the main loop
56 
57  int posPayload = 0;
58  if (nwords > sizeof(mRDH)) {
59  // RDHUtils::printRDH(mRDH);
60  // Read the payload following the RDH (only if there is anything to read)
61  posPayload = int(sizeof(mRDH));
62 
63  while (posPayload < nwords) {
64  mRawFileIn.read(reinterpret_cast<char*>(&eventHeader), sizeof(eventHeader));
65  posPayload += sizeof(eventHeader);
66  LOG(DEBUG) << " Read internal EventHeader for link: " << link
67  << " nWords: " << (int)eventHeader.nGBTWords
68  << " orbit: " << int(eventHeader.orbit)
69  << " BC: " << int(eventHeader.bc)
70  << " posInFile: " << posInFile
71  << " posPayload: " << posPayload;
72  o2::InteractionRecord intrec{uint16_t(eventHeader.bc), uint32_t(eventHeader.orbit)};
73 
74  if (link == lut.getTcmLink()) { // is TCM payload
75  mRawFileIn.read(reinterpret_cast<char*>(&mTCMdata), sizeof(mTCMdata));
76  posPayload += sizeof(mTCMdata);
77  LOG(DEBUG) << " Read TCM: posPayload: " << posPayload
78  << " posInFile: " << posInFile;
79  } else { // is PM payload
80  posPayload += CRUWordSize - o2::fdd::EventHeader::PayloadSize; // padding is enabled
81  for (int i = 0; i < eventHeader.nGBTWords; ++i) {
82  mRawFileIn.read(reinterpret_cast<char*>(&eventData[2 * i]), o2::fdd::EventData::PayloadSizeFirstWord);
83  posPayload += o2::fdd::EventData::PayloadSizeFirstWord;
84  chData = {Short_t(lut.getChannel(link, int(eventData[2 * i].channelID))),
85  Float_t(eventData[2 * i].time),
86  Short_t(eventData[2 * i].charge), 0};
87  mDigitAccum[intrec].emplace_back(chData);
88  LOG(DEBUG) << " Read 1st half-word: (PMchannel, globalChannel, Q, T, posPayload) = "
89  << std::setw(3) << int(eventData[2 * i].channelID)
90  << std::setw(4) << lut.getChannel(link, int(eventData[2 * i].channelID))
91  << std::setw(5) << int(eventData[2 * i].charge)
92  << std::setw(5) << float(eventData[2 * i].time)
93  << std::setw(5) << posPayload;
94 
95  Short_t channelIdFirstHalfWord = chData.mPMNumber;
96 
97  mRawFileIn.read(reinterpret_cast<char*>(&eventData[2 * i + 1]), EventData::PayloadSizeSecondWord);
98  posPayload += o2::fdd::EventData::PayloadSizeSecondWord;
99  chData = {Short_t(lut.getChannel(link, (eventData[2 * i + 1].channelID))),
100  Float_t(eventData[2 * i + 1].time),
101  Short_t(eventData[2 * i + 1].charge), 0};
102  if (chData.mPMNumber <= channelIdFirstHalfWord) {
103  // Don't save the second half-word if it is only filled with zeroes (empty-data)
104  // TODO: Verify if it works correctly with real data from readout
105  continue;
106  }
107  mDigitAccum[intrec].emplace_back(chData);
108  LOG(DEBUG) << " Read 2nd half-word: (PMchannel, globalChannel, Q, T, posPayload) = "
109  << std::setw(3) << int(eventData[2 * i + 1].channelID)
110  << std::setw(4) << lut.getChannel(link, int(eventData[2 * i + 1].channelID))
111  << std::setw(5) << int(eventData[2 * i + 1].charge)
112  << std::setw(5) << float(eventData[2 * i + 1].time)
113  << std::setw(5) << posPayload;
114  }
115  }
116  }
117  }
118  }
119  close();
120  LOG(INFO) << "o2::fdd::ReadRaw::readRawData():Finished.";
121 }

130 void ReadRaw::writeDigits(const std::string& outputDigitsFilePath)
131 {
132  TFile* outFile = new TFile(outputDigitsFilePath.data(), "RECREATE");
133  if (!outFile || outFile->IsZombie()) {
134  LOG(ERROR) << "Failed to open " << outputDigitsFilePath << " output file";
135  } else {
136  LOG(INFO) << "o2::fdd::ReadRaw::writeDigits(): Opened output file: " << outputDigitsFilePath;
137  }
138  TTree* outTree = new TTree("o2sim", "o2sim");
139  std::vector<ChannelData> chDataVecTree;
140  std::vector<Digit> chBcVecTree;
141 
142  for (auto& digit : mDigitAccum) {
143  LOG(DEBUG) << " IR (" << digit.first << ") (i, PMT, Q, T):";
144  for (uint16_t i = 0; i < digit.second.size(); i++) {
145  ChannelData* chd = &(digit.second.at(i));
146  LOG(DEBUG) << " " << std::setw(3) << i
147  << std::setw(4) << chd->mPMNumber
148  << std::setw(5) << chd->mChargeADC
149  << std::setw(5) << chd->mTime;
150  }
151 
152  size_t nStored = 0;
153  size_t first = chDataVecTree.size();
154  for (auto& sec : digit.second) {
155  chDataVecTree.emplace_back(int(sec.mPMNumber), float(sec.mTime), short(sec.mChargeADC), short(0));
156  nStored++;
157  }
158  chBcVecTree.emplace_back(first, nStored, digit.first, Triggers());
159  }
160 
161  outTree->Branch("FDDDigit", &chBcVecTree);
162  outTree->Branch("FDDDigitCh", &chDataVecTree);
163  outTree->Fill();
164 
165  outFile->cd();
166  outTree->Write();
167  outFile->Close();
168  LOG(INFO) << "o2::fdd::ReadRaw::writeDigits(): Finished converting " << chBcVecTree.size() << " events.";
169 }
