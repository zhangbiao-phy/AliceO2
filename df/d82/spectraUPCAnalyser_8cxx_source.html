29 struct UPCSpectraAnalyserTask {
30  float mPion = 0.13957;
31  HistogramRegistry histos{"Histos", {}, OutputObjHandlingPolicy::AnalysisObject};
32 
33  void init(o2::framework::InitContext&)
34  {
35  histos.add("fhMass", "Mass of mother", kTH1F, {{500, 0.55, 1.4}});
36  histos.add("fhPt", "Pt of mother", kTH1F, {{500, 0., 1.}});
37  }
38 
39  void process(aod::UDTrack const& track)
40  {
41  TLorentzVector p1, p2, p;
42  p1.SetXYZM(track.pt1() * TMath::Cos(track.phi1()), track.pt1() * TMath::Sin(track.phi1()), track.pt1() * TMath::SinH(track.eta1()), mPion);
43  p2.SetXYZM(track.pt2() * TMath::Cos(track.phi2()), track.pt2() * TMath::Sin(track.phi2()), track.pt2() * TMath::SinH(track.eta2()), mPion);
44  p = p1 + p2;
45  histos.fill(HIST("fhPt"), p.Pt());
46  float signalTPC1 = track.tpcSignal1();
47  float signalTPC2 = track.tpcSignal2();
48  if ((p.Pt() < 0.1) && (signalTPC1 + signalTPC2 < 140.)) {
49  histos.fill(HIST("fhMass"), p.M());
50  }
51  }
52 };
