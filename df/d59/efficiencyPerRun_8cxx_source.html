40  void init(InitContext&)
41  {
42  ccdb->setURL(url.value);
43  // Enabling object caching, otherwise each call goes to the CCDB server
44  ccdb->setCaching(true);
45  ccdb->setLocalObjectValidityChecking();
46  // Not later than now, will be replaced by the value of the train creation
47  // This avoids that users can replace objects **while** a train is running
48  ccdb->setCreatedNotAfter(nolaterthan.value);
49  }

51  void process(aod::Collision const& collision, aod::BCsWithTimestamps const&, aod::Tracks const& tracks)
52  {
53  auto bc = collision.bc_as<aod::BCsWithTimestamps>();
54  LOGF(info, "Getting object %s for run number %i from timestamp=%llu", path.value.data(), bc.runNumber(), bc.timestamp());
55  auto efficiency = ccdb->getForTimeStamp<TH1F>(path.value, bc.timestamp());
56  if (!efficiency) {
57  LOGF(fatal, "Efficiency object not found!");
58  }
59 
60  for (auto& track : tracks) {
61  pt->Fill(track.pt(), efficiency->GetBinContent(efficiency->FindBin(track.pt())));
62  //LOGF(info, "Efficiency %f for pt %f", efficiency->GetBinContent(efficiency->FindBin(track.pt())), track.pt());
63  }
64  }
65 };
