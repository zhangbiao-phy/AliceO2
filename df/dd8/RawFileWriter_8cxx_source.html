35 //_____________________________________________________________________
36 void RawFileWriter::close()
37 {
38  // finalize all links
39  if (mFName2File.empty()) {
40  return;
41  }
42 
43  if (mCachingStage) {
44  fillFromCache();
45  }
46 
47  if (!mFirstIRAdded.isDummy()) { // flushing and completing the last HBF makes sense only if data was added.
48  auto irmax = getIRMax();
49  for (auto& lnk : mSSpec2Link) {
50  lnk.second.close(irmax);
51  lnk.second.print();
52  }
53  }
54  //
55  // close all files
56  for (auto& flh : mFName2File) {
57  LOG(INFO) << "Closing output file " << flh.first;
58  fclose(flh.second.handler);
59  flh.second.handler = nullptr;
60  }
61  mFName2File.clear();
62  mTimer.Stop();
63  mTimer.Print();
64 }

66 //_____________________________________________________________________
67 void RawFileWriter::fillFromCache()
68 {
69  LOG(INFO) << "Filling links from cached trees";
70  mCachingStage = false;
71  for (const auto& cache : mCacheMap) {
72  for (const auto& entry : cache.second) {
73  auto& link = getLinkWithSubSpec(entry.first);
74  link.cacheTree->GetEntry(entry.second);
75  link.addData(cache.first, link.cacheBuffer.payload, link.cacheBuffer.preformatted, link.cacheBuffer.trigger);
76  }
77  }
78  mCacheFile->cd();
79  for (auto& linkEntry : mSSpec2Link) {
80  if (linkEntry.second.cacheTree) {
81  linkEntry.second.cacheTree->Write();
82  linkEntry.second.cacheTree.reset(nullptr);
83  }
84  }
85  std::string cacheName{mCacheFile->GetName()};
86  mCacheFile->Close();
87  mCacheFile.reset(nullptr);
88  unlink(cacheName.c_str());
89 }

118 RawFileWriter::LinkData& RawFileWriter::registerLink(uint16_t fee, uint16_t cru, uint8_t link, uint8_t endpoint, std::string_view outFileNameV)
119 {
120  // register the GBT link and its output file
121  std::string outFileName{outFileNameV};
122  auto sspec = RDHUtils::getSubSpec(cru, link, endpoint, fee);
123  auto& linkData = mSSpec2Link[sspec];
124  auto& file = mFName2File[std::string(outFileName)];
125  if (!file.handler) {
126  if (!(file.handler = fopen(outFileName.c_str(), "wb"))) { // if file does not exist, create it
127  LOG(ERROR) << "Failed to open output file " << outFileName;
128  throw std::runtime_error(std::string("cannot open link output file ") + outFileName);
129  }
130  }
131  if (!linkData.fileName.empty()) { // this link was already declared and associated with a file
132  if (linkData.fileName == outFileName) {
133  LOGF(INFO, "Link 0x%ux was already declared with same output, do nothing", sspec);
134  return linkData;
135  } else {
136  LOGF(ERROR, "Link 0x%ux was already connected to different output file %s", sspec, linkData.fileName);
137  throw std::runtime_error("redifinition of the link output file");
138  }
139  }
140  linkData.fileName = outFileName;
141  linkData.subspec = sspec;
142  RDHUtils::setVersion(linkData.rdhCopy, mUseRDHVersion);
143  RDHUtils::setFEEID(linkData.rdhCopy, fee);
144  RDHUtils::setCRUID(linkData.rdhCopy, cru);
145  RDHUtils::setLinkID(linkData.rdhCopy, link);
146  RDHUtils::setEndPointID(linkData.rdhCopy, endpoint);
147  if (mUseRDHVersion >= 6) {
148  RDHUtils::setSourceID(linkData.rdhCopy, o2::header::DAQID::O2toDAQ(mOrigin));
149  }
150  linkData.writer = this;
151  linkData.updateIR = mHBFUtils.getFirstIR();
152  linkData.buffer.reserve(mSuperPageSize);
153  RDHUtils::printRDH(linkData.rdhCopy);
154  LOGF(INFO, "Registered %s with output to %s", linkData.describe(), outFileName);
155  return linkData;
156 }

159 void RawFileWriter::addData(uint16_t feeid, uint16_t cru, uint8_t lnk, uint8_t endpoint, const IR& ir, const gsl::span<char> data, bool preformatted, uint32_t trigger)
160 {
161  // add payload to relevant links
162  if (isCRUDetector() && (data.size() % RDHUtils::GBTWord)) {
163  LOG(ERROR) << "provided payload size " << data.size() << " is not multiple of GBT word size";
164  throw std::runtime_error("payload size is not mutiple of GBT word size");
165  }
166  auto sspec = RDHUtils::getSubSpec(cru, lnk, endpoint, feeid);
167  auto& link = getLinkWithSubSpec(sspec);
168  if (ir < mHBFUtils.getFirstIR()) {
169  LOG(WARNING) << "provided " << ir << " precedes first TF " << mHBFUtils.getFirstIR() << " | discarding data for " << link.describe();
170  return;
171  }
172  if (ir < mFirstIRAdded) {
173  mFirstIRAdded = ir;
174  }
175  link.addData(ir, data, preformatted, trigger);
176 }

250 void RawFileWriter::LinkData::cacheData(const IR& ir, const gsl::span<char> data, bool preformatted, uint32_t trigger)
251 {
252  // cache data to temporary tree
253  std::lock_guard<std::mutex> lock(writer->mCacheFileMtx);
254  if (!cacheTree) {
255  writer->mCacheFile->cd();
256  cacheTree = std::make_unique<TTree>(o2::utils::concat_string("lnk", std::to_string(subspec)).c_str(), "cache");
257  cacheTree->Branch("cache", &cacheBuffer);
258  }
259  cacheBuffer.preformatted = preformatted;
260  cacheBuffer.trigger = trigger;
261  cacheBuffer.payload.resize(data.size());
262  if (!data.empty()) {
263  memcpy(cacheBuffer.payload.data(), data.data(), data.size());
264  }
265  writer->mCacheMap[ir].emplace_back(subspec, cacheTree->GetEntries());
266  cacheTree->Fill();
267  return;
268 }

270 //___________________________________________________________________________________
271 void RawFileWriter::LinkData::addData(const IR& ir, const gsl::span<char> data, bool preformatted, uint32_t trigger)
272 {
273  // add payload corresponding to IR
274  LOG(DEBUG) << "Adding " << data.size() << " bytes in IR " << ir << " to " << describe();
275  std::lock_guard<std::mutex> lock(mtx);
276 
277  if (writer->mCachingStage) {
278  cacheData(ir, data, preformatted, trigger);
279  return;
280  }
281 
282  if (startOfRun && writer->isRORCDetector()) { // in RORC mode we write separate RDH with SOX in the very beginning of the run
283  writer->mHBFUtils.updateRDH<RDHAny>(rdhCopy, writer->mHBFUtils.getFirstIR(), false);
284  RDHUtils::setTriggerType(rdhCopy, 0);
285  openHBFPage(rdhCopy); // open new HBF just to report the SOX
286  // closeHBFPage();
287  }
288 
289  int dataSize = data.size();
290  if (ir >= updateIR) { // new IR exceeds or equal IR of next HBF to open, insert missed HBFs if needed
291  fillEmptyHBHs(ir, true);
292  }
293  // we are guaranteed to be under the valid RDH + possibly some data
294 
295  if (trigger) {
296  auto& rdh = *getLastRDH();
297  RDHUtils::setTriggerType(rdh, RDHUtils::getTriggerType(rdh) | trigger);
298  }
299 
300  if (!dataSize) {
301  return;
302  }
303  if (preformatted) { // in case detectors wants to add new CRU page of predefined size
304  addPreformattedCRUPage(data);
305  return;
306  }
307 
308  // if we are at the beginning of the page, detector may want to add some header
309  if (isNewPage() && writer->newRDHFunc) {
310  std::vector<char> newPageHeader;
311  writer->newRDHFunc(getLastRDH(), false, newPageHeader);
312  pushBack(newPageHeader.data(), newPageHeader.size());
313  }
314 
315  const char* ptr = &data[0];
316  // in case particular detector CRU pages need to be self-consistent, when carrying-over
317  // large payload to new CRU page we may need to write optional trailer and header before
318  // and after the new RDH.
319  bool carryOver = false, wasSplit = false, lastSplitPart = false;
320  int splitID = 0;
321  std::vector<char> carryOverHeader;
322  while (dataSize > 0) {
323 
324  if (carryOver) { // check if there is carry-over header to write in the buffer
325  addHBFPage(); // start new CRU page, if needed, the completed superpage is flushed
326  if (writer->newRDHFunc) {
327  std::vector<char> newPageHeader;
328  writer->newRDHFunc(getLastRDH(), false, newPageHeader);
329  pushBack(newPageHeader.data(), newPageHeader.size());
330  }
331 
332  // for sure after the carryOver we have space on the CRU page, no need to check
333  LOG(DEBUG) << "Adding carryOverHeader " << carryOverHeader.size()
334  << " bytes in IR " << ir << " to " << describe();
335  pushBack(carryOverHeader.data(), carryOverHeader.size());
336  carryOverHeader.clear();
337  carryOver = false;
338  }
339  int sizeLeftSupPage = writer->mSuperPageSize - buffer.size();
340  int sizeLeftCRUPage = RDHUtils::MAXCRUPage - (int(buffer.size()) - lastRDHoffset);
341  int sizeLeft = sizeLeftCRUPage < sizeLeftSupPage ? sizeLeftCRUPage : sizeLeftSupPage;
342  if (!sizeLeft) { // this page is just over, open a new one
343  addHBFPage(); // start new CRU page, if needed, the completed superpage is flushed
344  if (writer->newRDHFunc) {
345  std::vector<char> newPageHeader;
346  writer->newRDHFunc(getLastRDH(), false, newPageHeader);
347  pushBack(newPageHeader.data(), newPageHeader.size());
348  }
349  continue;
350  }
351 
352  if (dataSize <= sizeLeft) {
353  if (wasSplit && writer->mApplyCarryOverToLastPage) {
354  lastSplitPart = true;
355  carryOver = true;
356  }
357  } else {
358  carryOver = true;
359  wasSplit = true;
360  }
361 
362  if (!carryOver) { // add all remaining data
363  LOG(DEBUG) << "Adding payload " << dataSize << " bytes in IR " << ir << " (carryover=" << carryOver << " ) to " << describe();
364  pushBack(ptr, dataSize);
365  dataSize = 0;
366  } else { // need to carryOver payload, determine 1st wsize bytes to write starting from ptr
367  if (sizeLeft > dataSize) {
368  sizeLeft = dataSize;
369  }
370  int sizeActual = sizeLeft;
371  std::vector<char> carryOverTrailer;
372  if (writer->carryOverFunc) {
373  sizeActual = writer->carryOverFunc(&rdhCopy, data, ptr, sizeLeft, splitID++, carryOverTrailer, carryOverHeader);
374  }
375  LOG(DEBUG) << "Adding carry-over " << splitID - 1 << " fitted payload " << sizeActual << " bytes in IR " << ir << " to " << describe();
376  if (sizeActual < 0 || (!lastSplitPart && (sizeActual + carryOverTrailer.size() > sizeLeft))) {
377  throw std::runtime_error(std::string("wrong carry-over data size provided by carryOverMethod") + std::to_string(sizeActual));
378  }
379  // if there is carry-over trailer at the very last chunk, it must overwrite existing trailer
380  int trailerOffset = 0;
381  if (lastSplitPart) {
382  trailerOffset = carryOverTrailer.size();
383  if (sizeActual - trailerOffset < 0) {
384  throw std::runtime_error("trailer size of last split chunk cannot exceed actual size as it overwrites the existing trailer");
385  }
386  }
387  pushBack(ptr, sizeActual - trailerOffset); // write payload fitting to this page
388  dataSize -= sizeActual;
389  ptr += sizeActual;
390  LOG(DEBUG) << "Adding carryOverTrailer " << carryOverTrailer.size() << " bytes in IR " << ir << " to " << describe();
391  pushBack(carryOverTrailer.data(), carryOverTrailer.size());
392  }
393  }
394 }

396 //___________________________________________________________________________________
397 void RawFileWriter::LinkData::addPreformattedCRUPage(const gsl::span<char> data)
398 {
399  // add preformatted CRU page w/o any attempt of splitting
400 
401  // we are guaranteed to have a page with RDH open
402  int sizeLeftSupPage = writer->mSuperPageSize - buffer.size();
403  if (sizeLeftSupPage < data.size()) { // we are not allowed to split this payload
404  flushSuperPage(true); // flush all but the last added RDH
405  }
406  if (data.size() > RDHUtils::MAXCRUPage - sizeof(RDHAny)) {
407  LOG(ERROR) << "Preformatted payload size of " << data.size() << " bytes for " << describe()
408  << " exceeds max. size " << RDHUtils::MAXCRUPage - sizeof(RDHAny);
409  throw std::runtime_error("preformatted payload exceeds max size");
410  }
411  if (int(buffer.size()) - lastRDHoffset > sizeof(RDHAny)) { // we must start from empty page
412  addHBFPage(); // start new CRU page
413  }
414  pushBack(&data[0], data.size());
415 }

417 //___________________________________________________________________________________
418 void RawFileWriter::LinkData::addHBFPage(bool stop)
419 {
421 
422  if (lastRDHoffset < 0) {
423  return; // no page was open
424  }
425  // finalize last RDH
426  auto& lastRDH = *getLastRDH();
427  int psize = getCurrentPageSize(); // set the size for the previous header RDH
428  bool emptyPage = psize == sizeof(RDHAny);
429  if (stop && emptyPage && writer->emptyHBFFunc) { // we are closing an empty page, does detector want to add something?
430  std::vector<char> emtyHBFFiller; // working space for optional empty HBF filler
431  writer->emptyHBFFunc(&lastRDH, emtyHBFFiller);
432  if (emtyHBFFiller.size()) {
433  LOG(DEBUG) << "Adding empty HBF filler of size " << emtyHBFFiller.size() << " for " << describe();
434  pushBack(emtyHBFFiller.data(), emtyHBFFiller.size());
435  psize += emtyHBFFiller.size();
436  }
437  }
438  RDHUtils::setOffsetToNext(lastRDH, psize);
439  RDHUtils::setMemorySize(lastRDH, psize);
440 
441  rdhCopy = lastRDH;
442  bool add = true;
443  if (stop && !writer->mAddSeparateHBFStopPage) {
444  if (writer->isRDHStopUsed()) {
445  RDHUtils::setStop(lastRDH, stop);
446  }
447  add = false;
448  }
449  if (writer->mVerbosity > 2) {
450  RDHUtils::printRDH(lastRDH);
451  }
452  if (add) { // if we are in stopping HBF and new page is needed, add it
453  // check if the superpage reached the size where it hase to be flushed
454  int left = writer->mSuperPageSize - buffer.size();
455  if (left <= MarginToFlush) {
456  flushSuperPage();
457  }
458  RDHUtils::setPacketCounter(rdhCopy, packetCounter++);
459  RDHUtils::setPageCounter(rdhCopy, pageCnt++);
460  RDHUtils::setStop(rdhCopy, stop);
461  std::vector<char> userData;
462  int sz = sizeof(RDHAny);
463  if (stop && writer->newRDHFunc) { // detector may want to write something in closing page
464  writer->newRDHFunc(&rdhCopy, emptyPage, userData);
465  sz += userData.size();
466  }
467  RDHUtils::setOffsetToNext(rdhCopy, sz);
468  RDHUtils::setMemorySize(rdhCopy, sz);
469  lastRDHoffset = pushBack(rdhCopy); // entry of the new RDH
470  if (!userData.empty()) {
471  pushBack(userData.data(), userData.size());
472  }
473  }
474  if (stop) {
475  if (RDHUtils::getTriggerType(rdhCopy) & o2::trigger::TF) {
476  nTFWritten++;
477  }
478  if (writer->mVerbosity > 2 && add) {
479  RDHUtils::printRDH(rdhCopy);
480  }
481  lastRDHoffset = -1; // after closing, the previous RDH is not valid anymore
482  startOfRun = false; // signal that we are definitely not in the beginning of the run
483  }
484  //
485 }

487 //___________________________________________________________________________________
488 void RawFileWriter::LinkData::openHBFPage(const RDHAny& rdhn, uint32_t trigger)
489 {
491  bool forceNewPage = false;
492 
493  // for RORC detectors the TF flag is absent, instead the 1st trigger after the start of TF will define the 1st be interpreted as 1st TF
494  auto newTF_RORC = [this, &rdhn]() -> bool {
495  auto tfhbPrev = writer->mHBFUtils.getTFandHBinTF(this->updateIR.bc ? this->updateIR - 1 : this->updateIR); // updateIR was advanced by 1 BC wrt IR of the previous update
496  return this->writer->mHBFUtils.getTFandHBinTF(RDHUtils::getTriggerIR(rdhn)).first > tfhbPrev.first; // new TF_ID exceeds old one
497  };
498 
499  if ((RDHUtils::getTriggerType(rdhn) & o2::trigger::TF) || (writer->isRORCDetector() && newTF_RORC())) {
500  if (writer->mVerbosity > -10) {
501  LOGF(INFO, "Starting new TF for link FEEId 0x%04x", RDHUtils::getFEEID(rdhn));
502  }
503  if (writer->mStartTFOnNewSPage && nTFWritten) { // don't flush if 1st TF
504  forceNewPage = true;
505  }
506  }
507  int left = writer->mSuperPageSize - buffer.size();
508  if (forceNewPage || left <= MarginToFlush) {
509  flushSuperPage();
510  }
511  pageCnt = 0;
512  lastRDHoffset = pushBack(rdhn);
513  auto& newrdh = *getLastRDH(); // dress new RDH with correct counters
514  RDHUtils::setPacketCounter(newrdh, packetCounter++);
515  RDHUtils::setPageCounter(newrdh, pageCnt++);
516  RDHUtils::setStop(newrdh, 0);
517  RDHUtils::setMemorySize(newrdh, sizeof(RDHAny));
518  RDHUtils::setOffsetToNext(newrdh, sizeof(RDHAny));
519 
520  if (startOfRun && writer->isReadOutModeSet()) {
521  auto trg = RDHUtils::getTriggerType(newrdh) | (writer->isContinuousReadout() ? o2::trigger::SOC : o2::trigger::SOT);
522  RDHUtils::setTriggerType(newrdh, trg);
523  }
524  rdhCopy = newrdh;
525 }

527 //___________________________________________________________________________________
528 void RawFileWriter::LinkData::flushSuperPage(bool keepLastPage)
529 {
530  // write link superpage data to file (if requested, only up to the last page)
531  size_t pgSize = (lastRDHoffset < 0 || !keepLastPage) ? buffer.size() : lastRDHoffset;
532  if (writer->mVerbosity) {
533  LOGF(INFO, "Flushing super page of %u bytes for %s", pgSize, describe());
534  }
535  writer->mFName2File.find(fileName)->second.write(buffer.data(), pgSize);
536  auto toMove = buffer.size() - pgSize;
537  if (toMove) { // is there something left in the buffer, move it to the beginning of the buffer
538  if (toMove > pgSize) {
539  memcpy(buffer.data(), &buffer[pgSize], toMove);
540  } else {
541  memmove(buffer.data(), &buffer[pgSize], toMove);
542  }
543  buffer.resize(toMove);
544  lastRDHoffset -= pgSize;
545  } else {
546  buffer.clear();
547  lastRDHoffset = -1;
548  }
549 }

551 //___________________________________________________________________________________
552 void RawFileWriter::LinkData::close(const IR& irf)
553 {
554  // finalize link data
555  // close open HBF, write empty HBFs until the end of the TF corresponding to irfin and detach from the stream
556  if (writer->mFName2File.find(fileName) == writer->mFName2File.end()) {
557  return; // already closed
558  }
559  if (writer->isCRUDetector()) { // finalize last TF
560  auto irfin = irf;
561  if (irfin < updateIR) {
562  irfin = updateIR;
563  }
564  int tf = writer->mHBFUtils.getTF(irfin);

566  fillEmptyHBHs(finalIR, false);
567  }

572 //___________________________________________________________________________________
573 void RawFileWriter::LinkData::fillEmptyHBHs(const IR& ir, bool dataAdded)
574 {
575  // fill HBFs from last processed one to requested ir
576 
577  if (writer->isCRUDetector()) {
578  std::vector<o2::InteractionRecord> irw;
579  if (!writer->mHBFUtils.fillHBIRvector(irw, updateIR, ir)) {
580  return;
581  }
582  for (const auto& irdummy : irw) {
583  if (writer->mDontFillEmptyHBF &&
584  writer->mHBFUtils.getTFandHBinTF(irdummy).second != 0 &&
585  (!dataAdded || irdummy < ir)) {
586  // even if requested, we skip empty HBF filling only if
587  // 1) we are not at the new TF start
588  // 2) method was called from addData and the current IR is the one for which it was called (then it is not empty HB/trigger!)
589  continue;
590  }
591  if (writer->mVerbosity > 2) {
592  LOG(INFO) << "Adding HBF " << irdummy << " for " << describe();
593  }
594  closeHBFPage(); // close current HBF: add RDH with stop and update counters
595  RDHUtils::setTriggerType(rdhCopy, 0); // reset to avoid any detector specific flags in the dummy HBFs
596  writer->mHBFUtils.updateRDH<RDHAny>(rdhCopy, irdummy, writer->isCRUDetector()); // update HBF orbit/bc and trigger flags
597  openHBFPage(rdhCopy); // open new HBF
598  }
599  updateIR = irw.back() + o2::constants::lhc::LHCMaxBunches; // new HBF will be generated at >= this IR
600  } else { // RORC detector
601  if (writer->mVerbosity > 2) {
602  LOG(INFO) << "Adding HBF " << ir << " for " << describe();
603  }
604  closeHBFPage(); // close current HBF: add RDH with stop and update counters
605  RDHUtils::setTriggerType(rdhCopy, 0); // reset to avoid any detector specific flags in the dummy HBFs
606  writer->mHBFUtils.updateRDH<RDHAny>(rdhCopy, ir, false); // update HBF orbit/bc and trigger flags
607  openHBFPage(rdhCopy); // open new HBF
608  updateIR = ir + 1; // new Trigger in RORC detector will be generated at >= this IR
609  }
610 }
