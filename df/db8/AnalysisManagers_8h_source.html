228 template <typename T>

236  static bool prepare(ProcessingContext& pc, Spawns<T>& what)
237  {
238  auto original_table = soa::ArrowHelpers::joinTables(extractOriginals(what.sources_pack(), pc));
239  if (original_table->schema()->fields().empty() == true) {
240  using base_table_t = typename Spawns<T>::base_table_t;
241  original_table = makeEmptyTable<base_table_t>();
242  }
243 
244  what.extension = std::make_shared<typename Spawns<T>::extension_t>(o2::framework::spawner(what.pack(), original_table.get()));
245  what.table = std::make_shared<typename T::table_t>(soa::ArrowHelpers::joinTables({what.extension->asArrowTable(), original_table}));
246  return true;
247  }
248 
249  static bool finalize(ProcessingContext& pc, Spawns<T>& what)
250  {
251  pc.outputs().adopt(what.output(), what.asArrowTable());
252  return true;
253  }

358 template <typename T, typename IP>
359 struct OptionManager<Configurable<T, IP>> {
360  static bool appendOption(std::vector<ConfigParamSpec>& options, Configurable<T, IP>& what)
361  {
362  if constexpr (variant_trait_v<typename std::decay<T>::type> != VariantType::Unknown) {
363  options.emplace_back(ConfigParamSpec{what.name, variant_trait_v<std::decay_t<T>>, what.value, {what.help}});
364  } else {
365  auto specs = RootConfigParamHelpers::asConfigParamSpecs<T>(what.name, what.value);
366  options.insert(options.end(), specs.begin(), specs.end());
367  }
368  return true;
369  }
370 
371  static bool prepare(InitContext& context, Configurable<T, IP>& what)
372  {
373  if constexpr (variant_trait_v<typename std::decay<T>::type> != VariantType::Unknown) {
374  what.value = context.options().get<T>(what.name.c_str());
375  } else {
376  auto pt = context.options().get<boost::property_tree::ptree>(what.name.c_str());
377  what.value = RootConfigParamHelpers::as<T>(pt);
378  }
379  return true;
380  }
381 };
