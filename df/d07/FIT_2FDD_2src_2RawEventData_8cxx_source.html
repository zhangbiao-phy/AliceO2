96 void RawEventData::printHexEventHeader() const
97 {
98  std::stringstream ssheader;
99  ssheader << std::setfill('0') << std::setw(16) << std::hex << mEventHeader.word[0] << " " << std::setw(16) << mEventHeader.word[1] << "\n ";
100  ssheader << std::setw(3) << (0x00000fff & mEventHeader.bc) << " "
101  << std::setw(8) << (0xffffffff & mEventHeader.orbit) << " "
102  << std::setw(5) << (0x000fffff & mEventHeader.reservedField1) << " "
103  << std::setw(2) << (0x000000ff & mEventHeader.reservedField2) << " "
104  << std::setw(1) << (0x0000000f & mEventHeader.nGBTWords) << " "
105  << std::setw(1) << (0x0000000f & mEventHeader.startDescriptor) << " "
106  << std::setw(12) << (0xffffffffffff & mEventHeader.reservedField3);
107  LOG(DEBUG) << ssheader.str();
108 }
