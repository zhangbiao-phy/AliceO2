37 struct BTask {
38  OutputObj<TH2F> etaphiH{TH2F("etaphi", "etaphi", 100, 0., 2. * M_PI, 102, -2.01, 2.01)};
39  // Create a configurable which can be used inside the process method.
40  Configurable<float> phiCut{"phiCut", 6.29f, "A cut on phi"};
41 
42  void process(aod::Tracks const& tracks)
43  {
44  for (auto& track : tracks) {
45  // FIXME: this is until we have configurables which
46  // can be used in expressions.

50  }
51  }

54 struct CTask {

59  OutputObj<TH1F> trZ{"trZ", OutputObjHandlingPolicy::QAObject};
60  Configurable<float> pTCut{"pTCut", 0.5f, "Lower pT limit"};
61

70  // trZ.setObject({"Z","Z",100,-10.,10.}); <- creates new
