50 void JetFinder::setSub()
51 {
52  //if rho < 1e-6 it is set to 1e-6 in AliPhysics
53  if (bkgSubMode == BkgSubMode::rhoAreaSub) {
54  sub = decltype(sub){new fastjet::Subtractor{bkgE.get()}};
55  } else if (bkgSubMode == BkgSubMode::constSub) { //event or jetwise
56  constituentSub = decltype(constituentSub){new fastjet::contrib::ConstituentSubtractor{bkgE.get()}};
57  constituentSub->set_distance_type(fastjet::contrib::ConstituentSubtractor::deltaR);
58  constituentSub->set_max_distance(constSubRMax);
59  constituentSub->set_alpha(constSubAlpha);
60  constituentSub->set_ghost_area(ghostArea);
61  constituentSub->set_max_eta(bkgEtaMax);
62  constituentSub->set_background_estimator(bkgE.get()); //what about rho_m
63  } else {
64  if (bkgSubMode != BkgSubMode::none) {
65  LOGF(ERROR, "requested subtraction mode not implemented!");
66  }
