112  void process(soa::Join<aod::Collisions, aod::EvSels, aod::Cents>::iterator const& collision, soa::Filtered<aod::V0Datas> const& fullV0s)
113  {
114  if (!collision.alias()[kINT7]) {
115  return;
116  }
117  if (!collision.sel7()) {
118  return;
119  }
120 
121  for (auto& v0 : fullV0s) {
122  //FIXME: could not find out how to filter cosPA and radius variables (dynamic columns)
123  if (v0.v0radius() > v0radius && v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()) > v0cospa) {
124  if (TMath::Abs(v0.yLambda()) < rapidity) {
125  registry.fill(HIST("h3dMassLambda"), collision.centV0M(), v0.pt(), v0.mLambda());
126  registry.fill(HIST("h3dMassAntiLambda"), collision.centV0M(), v0.pt(), v0.mAntiLambda());
127  if (saveDcaHist == 1) {
128  registry.fill(HIST("h3dMassLambdaDca"), v0.dcaV0daughters(), v0.pt(), v0.mLambda());
129  registry.fill(HIST("h3dMassAntiLambdaDca"), v0.dcaV0daughters(), v0.pt(), v0.mAntiLambda());
130  }
131  }
132  if (TMath::Abs(v0.yK0Short()) < rapidity) {
133  registry.fill(HIST("h3dMassK0Short"), collision.centV0M(), v0.pt(), v0.mK0Short());
134  if (saveDcaHist == 1) {
135  registry.fill(HIST("h3dMassK0ShortDca"), v0.dcaV0daughters(), v0.pt(), v0.mK0Short());
136  }
137  }
138  }
139  }
140  }
141 };
