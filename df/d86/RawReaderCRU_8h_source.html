818 inline void GBTFrame::updateSyncCheck(SyncArray& syncArray)
819 {
820  const auto offset = mPrevHWpos ^ 4;
821 
822  for (int s = 0; s < 5; s++) {
823  for (int h = 0; h < 4; h++) {
824  const auto hPos = h + offset; // set position of last filled HW
825  // shift in a 1 if the halfword is 0x15
826  if (mFrameHalfWords[s][hPos] == 0x15) {
827  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1) | 1;
828  // shift in a 0 if the halfword is 0xA
829  } else if (mFrameHalfWords[s][hPos] == 0xA) {
830  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1);
831  // otherwise reset the register to 0
832  } else {
833  mSyncCheckRegister[s] = 0;
834  }
835  // std::cout << " SReg : " << s << " : " << std::hex << mSyncCheckRegister[s] << std::endl;
836  // check the register content for the SYNC pattern
837  // Patterh is : 1100.1100.1100.1100.1111.0000.1111.0000 = 0xCCCCF0F0
838  // alternative Pattern for first corrupted SYNC word : 0100.1100.1100.1100.1111.0000.1111.0000 = 4CCC.F0F0
839  if (mSyncCheckRegister[s] == 0xCCCCF0F0 or mSyncCheckRegister[s] == 0x4CCCF0F0 or mSyncCheckRegister[s] == 0x0CCCF0F0) {
840  syncArray[s].setPos(mPacketNum, mFrameNum, mFilePos, h);
841  };
842  }
843  };
844 }

880 inline void GBTFrame::getFrameHalfWords()
881 {
882  constexpr adc_t P[5][4] = {{19, 18, 17, 16}, {39, 38, 37, 36}, {63, 62, 61, 60}, {83, 82, 81, 80}, {107, 106, 105, 104}};
883  // i = Stream, j = Halfword
884  for (int i = 0; i < 5; i++) {
885  for (int j = 0; j < 4; j++) {
886  mFrameHalfWords[i][j + mPrevHWpos] = bit(P[i][j], adc_t(4)) |
887  bit(adc_t(P[i][j] - 4), adc_t(3)) |
888  bit(adc_t(P[i][j] - 8), adc_t(2)) |
889  bit(adc_t(P[i][j] - 12), adc_t(1)) |
890  bit(adc_t(P[i][j] - 16), adc_t(0));
891  // std::cout << " S : " << i << " H : " << j << " : " << std::hex << res << std::endl;
892  };
893  }

898 inline void GBTFrame::getAdcValues(ADCRawData& rawData)
899 {
900  // loop over all the 5 data streams
901  for (int s = 0; s < 5; s++) {
902  if (!mSyncPos[s].synched()) {
903  continue;
904  }
905 
906  // assemble the 2 ADC words from 4 half-words. Which halfwords are
907  // used depends on the position of the SYNC pattern.
908  const uint32_t pos = mSyncPos[s].getHalfWordPosition();
909  const uint32_t v0 = (mFrameHalfWords[s][(pos + 2 + mPrevHWpos) & 7] << 5) | mFrameHalfWords[s][(pos + 1 + mPrevHWpos) & 7];
910  const uint32_t v1 = (mFrameHalfWords[s][(pos + 4 + mPrevHWpos) & 7] << 5) | mFrameHalfWords[s][(pos + 3 + mPrevHWpos) & 7];
911 
912  // add the two ADC values for the current processed stream
913  rawData.add(s, v0, v1);
914  };
