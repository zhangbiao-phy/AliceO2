110  Int_t WhichParticle(Int_t pdgCode)
111  { //see in AlidNdPtTools
112  if (pdgCode == 0) {
113  return -1;
114  }
115  if (pdgCode == std::abs(211)) {
116  return 0; //pi+, pi-
117  }
118  if (pdgCode == std::abs(321)) {
119  return 1; //K+, K-
120  }
121  if (pdgCode == std::abs(2212)) {
122  return 2; //p, pbar
123  }
124  if (pdgCode == 3222 || pdgCode == -3112) {
125  return 3; //sigmaPlus, SigmaBarMinus
126  }
127  if (pdgCode == 3112 || pdgCode == -3222) {
128  return 4; //sigmaMinus, SigmaBarPlus
129  }
130  if (pdgCode == std::abs(11)) {
131  return 5; //e-, e+
132  }
133  if (pdgCode == std::abs(3312)) {
134  return 6; //XiP, XiM
135  }
136  if (pdgCode == std::abs(13)) {
137  return 7; //mu,antimu
138  }
139  if (pdgCode == std::abs(3334)) {
140  return 8; //OmegaP, OmegaM
141  }
142 
143  return 9; //other
144  }
145  Double_t MCCharge(Int_t pdgCode)
146  {

148  Double_t charge = par->Charge() / 3.0;
149  return charge;
150  }

154  void process(soa::Join<aod::Collisions, aod::McCollisionLabels, aod::EvSels, aod::Cents>::iterator const& collision, soa::Join<aod::Tracks, aod::McTrackLabels, aod::TrackSelection> const& tracks, aod::McParticles& mcParticles)
155  {
156  if (!collision.alias()[kINT7]) {
157  return;
158  }
159  if (!collision.sel7()) {
160  return;
161  }
162 
163  Double_t eventValues[3] = {0.0, 0.0, collision.posZ()};
164  fHistEvent->Fill(eventValues);
165 
166  for (auto& track : tracks) {
167  if (selectedTracks == 1 && !track.isGlobalTrack()) {
168  continue;
169  } else if (selectedTracks == 2 && !track.isGlobalTrackSDD()) {
170  continue;
171  }
172 
173  Double_t trackValues[4] = {0.0, 0.0, track.pt(), (Double_t)track.charge()};
174  fHistTrack->Fill(trackValues);
175 
176  Double_t mcInfoVal;
177  if (!isMC) {
178  continue;
179  }
180  if (MC::isPhysicalPrimary(mcParticles, track.mcParticle())) {
181  mcInfoVal = 0.0;
182  } else {
183  mcInfoVal = 1.0;
184  }
185 
186  Double_t MCpt = track.mcParticle().pt();
187  Double_t parType = (Double_t)WhichParticle(track.mcParticle().pdgCode());
188  Double_t MCcharge = (Double_t)track.charge();
189  Double_t MCvalues[4] = {MCpt, parType, mcInfoVal, MCcharge};
190 
191  fHistMC->Fill(MCvalues);
192  }
193  if (isMC) {
194  for (auto& mcParticle : mcParticles) {
195 
196  if (abs(mcParticle.eta()) > 0.8) {
197  continue;
198  }
199  if (!MC::isPhysicalPrimary(mcParticles, mcParticle)) {
200  continue;
201  }
202 
203  Double_t MCpt = mcParticle.pt();
204  Double_t parType = (Double_t)WhichParticle(mcParticle.pdgCode());
205  Int_t pdg = (Int_t)mcParticle.pdgCode();
206  Double_t MCcharge = MCCharge(pdg);
207 
208  if (MCcharge == 0.0) {
209  continue;
210  }
211  Double_t MCvalues[4] = {MCpt, parType, 2.0, MCcharge};
212  fHistMC->Fill(MCvalues);
213  }
214  }
215  }
