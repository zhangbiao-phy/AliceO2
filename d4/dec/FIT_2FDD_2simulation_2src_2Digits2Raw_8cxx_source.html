23 void Digits2Raw::readDigits(const std::string& outDir, const std::string& fileDigitsName)
24 {
25  LOG(INFO) << "==============FDD: Digits2Raw::convertDigits" << std::endl;
26  mWriter.setCarryOverCallBack(this);
27  LookUpTable lut(true);
28 
29  std::string outd = outDir;
30  if (outd.back() != '/') {
31  outd += '/';
32  }
33 
34  // Register PM links linearly
35  for (int iPmLink = 0; iPmLink < Nmodules; ++iPmLink) {
36  uint16_t feeId = uint16_t(iPmLink);
37  uint8_t linkId = uint8_t(iPmLink);
38  std::string outFileLink = mOutputPerLink ? (outd + "fdd_link" + std::to_string(iPmLink) + ".raw") : (outd + "fdd.raw");
39  LOG(INFO) << " Register PM link: " << iPmLink << " to file: " << outFileLink;
40  mWriter.registerLink(feeId, sCruId, linkId, sEndPointId, outFileLink);
41  }
42 
43  // Register TCM link separately
44  std::string outFileLink = mOutputPerLink ? (outd + "fdd_link" + std::to_string(sTcmLink) + ".raw") : (outd + "fdd.raw");
45  LOG(INFO) << " Register TCM link: " << outFileLink;
46  mWriter.registerLink(uint16_t(sTcmLink), sCruId, sTcmLink, sEndPointId, outFileLink);
47 
48  TFile* fdig = TFile::Open(fileDigitsName.data());
49  assert(fdig != nullptr);
50  LOG(INFO) << "Open digits file: " << fileDigitsName.data();
51 
52  TTree* digTree = (TTree*)fdig->Get("o2sim");
53  std::vector<o2::fdd::Digit> digitsBC, *fddDigitPtr = &digitsBC;
54  std::vector<o2::fdd::ChannelData> digitsCh, *fddChDataPtr = &digitsCh;
55  digTree->SetBranchAddress("FDDDigit", &fddDigitPtr);
56  digTree->SetBranchAddress("FDDDigitCh", &fddChDataPtr);
57 
58  for (int ient = 0; ient < digTree->GetEntries(); ient++) {
59  digTree->GetEntry(ient);
60  int nbc = digitsBC.size();
61  for (int ibc = 0; ibc < nbc; ibc++) {
62  auto& bcd = digitsBC[ibc];
63  auto channels = bcd.getBunchChannelData(digitsCh);
64 
65  if (!channels.empty()) {
66  LOG(DEBUG) << "o2::fdd::Digits2Raw::readDigits(): Start to convertDigits() at ibc = " << ibc << " " << bcd.mIntRecord
67  << " iCh0:" << bcd.ref.getFirstEntry() << " nentries:" << bcd.ref.getEntries();
68  convertDigits(bcd, channels, lut);
69  }
70  }
71  }
72 }

74 void Digits2Raw::convertDigits(o2::fdd::Digit bcdigits, gsl::span<const ChannelData> pmchannels,
75  const o2::fdd::LookUpTable& lut)
76 {
77  const o2::InteractionRecord intRecord = bcdigits.getIntRecord();
78  int prevPmLink = -1;
79  int iChannelPerLink = 0;
80  int nch = pmchannels.size();
81 
82  std::stringstream ss;
83  ss << " Number of channels: " << nch << " (Ch, PMT, Q, T)\n";
84  for (int ich = 0; ich < nch; ich++) {
85  if (pmchannels[ich].mChargeADC != 0) {
86  ss << " " << std::setw(2) << ich
87  << std::setw(3) << pmchannels[ich].mPMNumber
88  << std::setw(5) << pmchannels[ich].mChargeADC
89  << std::setw(7) << std::setprecision(3) << pmchannels[ich].mTime << "\n";
90  }
91  }
92  LOG(DEBUG) << ss.str().substr(0, ss.str().size() - 1);
93 
94  for (int ich = 0; ich < nch; ich++) {
95  int nLinkPm = lut.getLink(pmchannels[ich].mPMNumber);
96  if (nLinkPm != prevPmLink) {
97  if (prevPmLink >= 0) {
98  fillSecondHalfWordAndAddData(iChannelPerLink, prevPmLink, intRecord);
99  }
100  makeGBTHeader(mRawEventData.mEventHeader, nLinkPm, intRecord);
101  iChannelPerLink = 0;
102  prevPmLink = nLinkPm;
103  }
104  if (pmchannels[ich].mChargeADC != 0) {
105  LOG(DEBUG) << " Store data for channel: " << ich << " PmLink = " << nLinkPm << " ";
106  auto& newData = mRawEventData.mEventData[iChannelPerLink];
107  newData.charge = pmchannels[ich].mChargeADC;
108  newData.time = pmchannels[ich].mTime;
109  newData.generateFlags();
110  newData.channelID = lut.getModChannel(pmchannels[ich].mPMNumber);
111  iChannelPerLink++;
112  }
113  if (ich == nch - 1) {
114  fillSecondHalfWordAndAddData(iChannelPerLink, prevPmLink, intRecord);
115  }
116  }
117 
118  // TCM
119  makeGBTHeader(mRawEventData.mEventHeader, sTcmLink, intRecord);
120  mRawEventData.mEventHeader.nGBTWords = 1;
121  auto& tcmdata = mRawEventData.mTCMdata;
122  tcmdata.vertex = 1;
123  tcmdata.orA = 1;
124  tcmdata.orC = 0;
125  tcmdata.sCen = 0;
126  tcmdata.cen = 0;
127  tcmdata.nChanA = 0;
128  tcmdata.nChanC = 0;
129  tcmdata.amplA = 0;
130  tcmdata.amplC = 0;
131  tcmdata.timeA = 0;
132  tcmdata.timeC = 0;
133 
134  auto data = mRawEventData.to_vector(kTRUE); //for tcm module
135  uint32_t linkId = uint32_t(sTcmLink);
136  uint64_t feeId = uint64_t(sTcmLink);
137  mWriter.addData(feeId, sCruId, linkId, sEndPointId, intRecord, data);
138 
139  // fill mEventData[iChannelPerLink] with 0s to flag that this is a dummy data
140  uint nGBTWords = uint((iChannelPerLink + 1) / 2);
141  if ((iChannelPerLink % 2) == 1) {
142  mRawEventData.mEventData[iChannelPerLink] = {};
143  }
144  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
145  // LOG(DEBUG) << " last link: " << prevPmLink;
146 }

148 void Digits2Raw::makeGBTHeader(EventHeader& eventHeader, int link, o2::InteractionRecord const& mIntRecord)
149 {
150  eventHeader.startDescriptor = 0xf;
151  eventHeader.reservedField1 = 0;
152  eventHeader.reservedField2 = 0;
153  eventHeader.reservedField3 = 0;
154  eventHeader.bc = mIntRecord.bc;
155  eventHeader.orbit = mIntRecord.orbit;
156  LOG(DEBUG) << " makeGBTHeader for link: " << link;
157 }

159 void Digits2Raw::fillSecondHalfWordAndAddData(int iChannelPerLink, int prevPmLink, const o2::InteractionRecord& ir)
160 {
161  uint nGBTWords = uint((iChannelPerLink + 1) / 2);
162  if ((iChannelPerLink % 2) == 1) {
163  mRawEventData.mEventData[iChannelPerLink] = {};
164  LOG(DEBUG) << " Fill up empty second half-word.";
165  }
166  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
167  auto data = mRawEventData.to_vector(false);
168  uint32_t linkId = uint32_t(prevPmLink);
169  uint64_t feeId = uint64_t(prevPmLink);
170  mWriter.addData(feeId, sCruId, linkId, sEndPointId, ir, data);
171  LOG(DEBUG) << " Switch prevPmLink: " << prevPmLink << ". Save data with nGBTWords="
172  << nGBTWords << " in header. Last channel: " << iChannelPerLink;
173 }
