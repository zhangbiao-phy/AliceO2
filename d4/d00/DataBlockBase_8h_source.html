74  int nSteps = std::get<kNSTEPS>(sReadingLookupTable[nWords]);

76  memcpy(serializedBytes.data() + std::get<kSRCBYTEPOS>(sByteLookupTable[iStep]), srcAddress + std::get<kDESTBYTEPOS>(sByteLookupTable[iStep]), std::get<kNBYTES>(sByteLookupTable[iStep]));
77  countBytes += std::get<kSRCBYTEPOS>(sByteLookupTable[iStep]);
78  }
79  return std::move(vecBytes);

82  void deserialize(const gsl::span<const uint8_t> inputBytes, size_t nWords, size_t& srcPos)
83  {
84  mNelements = 0;
85  mNwords = 0;
86  if (nWords < MinNwords || nWords > MaxNwords || inputBytes.size() - srcPos < nWords * sizeWord) {
87  //in case of bad fields responsible for deserialization logic, byte position will be pushed to the end of binary sequence
88  srcPos = inputBytes.size();
89  mIsIncorrect = true;
90  return;
91  }
92  uint8_t* destAddress = (uint8_t*)mData;
93  size_t countBytes = 0;
94  int nSteps = std::get<kNSTEPS>(sReadingLookupTable[nWords]);
95  mNwords = nWords;
96  mNelements = std::get<kNELEMENTS>(sReadingLookupTable[nWords]);
97  for (int iStep = 0; iStep < nSteps; iStep++) {
98  memcpy(destAddress + std::get<kDESTBYTEPOS>(sByteLookupTable[iStep]), inputBytes.data() + std::get<kSRCBYTEPOS>(sByteLookupTable[iStep]) + srcPos, std::get<kNBYTES>(sByteLookupTable[iStep]));
99  countBytes += std::get<kSRCBYTEPOS>(sByteLookupTable[iStep]);
100  }
101  srcPos += mNwords * sizeWord;
102  }

104  static constexpr int MaxNwords = T::PayloadSize * T::MaxNelements / T::PayloadPerGBTword + (T::PayloadSize * T::MaxNelements % T::PayloadPerGBTword > 0); //calculating max GBT words per block
105  static constexpr int MaxNbytes = sizeWord * MaxNwords;

107  static constexpr int MinNwords = T::PayloadSize * T::MinNelements / T::PayloadPerGBTword + (T::PayloadSize * T::MinNelements % T::PayloadPerGBTword > 0); //calculating min GBT words per block

110  //get number of byte reading steps
111  static constexpr size_t getNsteps()
112  {
113  int count = 0;
114  size_t payloadFull = T::MaxNelements * T::PayloadSize;
115  size_t payloadInWord = T::PayloadPerGBTword;
116  size_t payloadPerElem = T::PayloadSize;
117  while (payloadFull > 0) {
118  if (payloadPerElem < payloadInWord) {
119  count++;
120  payloadFull -= payloadPerElem;
121  payloadInWord -= payloadPerElem;
122  payloadPerElem = 0;
123  } else {
124  count++;
125  payloadFull -= payloadInWord;
126  payloadPerElem -= payloadInWord;
127  payloadInWord = 0;
128  }
129  if (payloadInWord == 0) {
130  payloadInWord = T::PayloadPerGBTword;
131  }
132  if (payloadPerElem == 0) {
133  payloadPerElem = T::PayloadSize;
134  }
135  }
136  return count;
137  }
138  //enumerator for tuple access:
139  //[Index] is index of step to read bytes
140  //kNBYTES - number of bytes to read from source and to write into destination
141  //kSRCBYTEPOS - Byte position in the source(binary raw data sequence)
142  //kDESTBYTEPOS - Byte position to write at destionation(memory allocation of T-element array)
143  //kELEMENTINDEX - element index at current step
144  //kWORDINDEX - word index at current step
145  //
146  enum AccessByteLUT { kNBYTES,
147  kSRCBYTEPOS,
148  kDESTBYTEPOS,
149  kELEMENTINDEX,
150  kWORDINDEX };
151  static constexpr std::array<std::tuple<size_t, size_t, size_t, int, int>, getNsteps()> GetByteLookupTable()
152  {
153  std::array<std::tuple<size_t, size_t, size_t, int, int>, getNsteps()> seqBytes{};
154  int count = 0;
155  int countElement = 0;
156  int countWord = 0;
157  size_t destBytePosPerElem = 0;
158  size_t srcBytePos = 0;
159  size_t payloadFull = T::MaxNelements * T::PayloadSize;
160 
161  size_t bytesInWord = sizeWord;
162  size_t payloadInWord = T::PayloadPerGBTword;
163 
164  size_t payloadPerElem = T::PayloadSize;
165 
166  uint64_t indexElem = 0;
167  uint64_t indexLastElem = T::MaxNelements - 1;
168 
169  while (payloadFull > 0) {
170  if (payloadPerElem < payloadInWord) { //new element
171  std::get<kNBYTES>(seqBytes[count]) = payloadPerElem;
172  std::get<kSRCBYTEPOS>(seqBytes[count]) = srcBytePos;
173  std::get<kDESTBYTEPOS>(seqBytes[count]) = destBytePosPerElem;
174  std::get<kELEMENTINDEX>(seqBytes[count]) = countElement;
175  std::get<kWORDINDEX>(seqBytes[count]) = countWord;
176  srcBytePos += payloadPerElem;
177  count++;
178  payloadFull -= payloadPerElem;
179  payloadInWord -= payloadPerElem;
180  bytesInWord -= payloadPerElem;
181  payloadPerElem = 0;
182 
183  } else {
184  std::get<kNBYTES>(seqBytes[count]) = payloadInWord;
185  std::get<kSRCBYTEPOS>(seqBytes[count]) = srcBytePos;
186  std::get<kDESTBYTEPOS>(seqBytes[count]) = destBytePosPerElem;
187  std::get<kELEMENTINDEX>(seqBytes[count]) = countElement;
188  std::get<kWORDINDEX>(seqBytes[count]) = countWord;
189  srcBytePos += bytesInWord;
190  count++;
191  destBytePosPerElem += payloadInWord;
192 
193  payloadFull -= payloadInWord;
194  payloadPerElem -= payloadInWord;
195  payloadInWord = 0;
196  bytesInWord = 0;
197  }
198 
199  if (payloadInWord == 0) {
200  payloadInWord = T::PayloadPerGBTword;
201  }
202  if (payloadPerElem == 0) {
203  payloadPerElem = T::PayloadSize;
204  countElement++;
205  destBytePosPerElem = countElement * sizeof(T);
206  }
207  if (bytesInWord == 0) {
208  bytesInWord = sizeWord;
209  countWord++;
210  }
211  }
212  return seqBytes;
213  }
214  static constexpr std::array<std::tuple<size_t, size_t, size_t, int, int>, getNsteps()> sByteLookupTable = GetByteLookupTable();

216  //enumerator for tuple access:
217  //[Index] is word index position, i.e. "Index" number of words will be deserialized
218  //kNELEMENTS - number of T elements will be fully deserialized in "Index+1" words
219  //kNSTEPS - number of steps for reading "Index" words
220  //kISPARTED - if one T-element is parted at current word,i.e. current word contains partially deserialized T element at the end of the word
221  enum AccessReadingLUT { kNELEMENTS,
222  kNSTEPS,
223  kISPARTED };
224  static constexpr std::array<std::tuple<unsigned int, unsigned int, bool>, MaxNwords + 1> GetReadingLookupTable()
225  {
226  std::array<std::tuple<unsigned int, unsigned int, bool>, MaxNwords + 1> readingScheme{};
227  size_t payloadPerElem = T::PayloadSize;
228  std::get<kNSTEPS>(readingScheme[0]) = 0;
229  std::get<kNELEMENTS>(readingScheme[0]) = 0;
230  std::get<kISPARTED>(readingScheme[0]) = false;
231  int countWord = 1;
232  for (int iStep = 0; iStep < getNsteps(); iStep++) {
233  if (countWord - 1 < std::get<kWORDINDEX>((GetByteLookupTable())[iStep])) { //new word
234  std::get<kNSTEPS>(readingScheme[countWord]) = iStep;
235  std::get<kNELEMENTS>(readingScheme[countWord]) = std::get<kELEMENTINDEX>((GetByteLookupTable())[iStep]);
236  if (payloadPerElem > 0) {
237  std::get<kISPARTED>(readingScheme[countWord]) = true;
238  } else {
239  std::get<kISPARTED>(readingScheme[countWord]) = false;
240  }
241  countWord++;
242  }
243  if (payloadPerElem == 0) {
244  payloadPerElem = T::PayloadSize;
245  }
246  payloadPerElem -= std::get<kNBYTES>((GetByteLookupTable())[iStep]);
247  }
248  //Last step checking
249  std::get<kNSTEPS>(readingScheme[countWord]) = getNsteps();
250  if (payloadPerElem > 0) {
251  std::get<kISPARTED>(readingScheme[countWord]) = true;
252  std::get<kNELEMENTS>(readingScheme[countWord]) = std::get<kELEMENTINDEX>((GetByteLookupTable())[getNsteps() - 1]);
253  } else {
254  std::get<kISPARTED>(readingScheme[countWord]) = false;
255  std::get<kNELEMENTS>(readingScheme[countWord]) = std::get<kELEMENTINDEX>((GetByteLookupTable())[getNsteps() - 1]) + 1;
256  }
257  return readingScheme;
258  }
