53 int GPUCATracking::runTracking(GPUO2InterfaceIOPtrs* data, GPUInterfaceOutputs* outputs)
54 {
55  if ((int)(data->tpcZS != nullptr) + (int)(data->o2Digits != nullptr && (data->tpcZS == nullptr || data->o2DigitsMC == nullptr)) + (int)(data->clusters != nullptr) + (int)(data->compressedClusters != nullptr) != 1) {
56  throw std::runtime_error("Invalid input for gpu tracking");
57  }
58 
59  std::vector<o2::tpc::Digit> gpuDigits[Sector::MAXSECTOR];
60  o2::dataformats::MCTruthContainer<o2::MCCompLabel> gpuDigitsMC[Sector::MAXSECTOR];
61  ClusterNativeAccess::ConstMCLabelContainerViewWithBuffer gpuDigitsMCConst[Sector::MAXSECTOR];
62 
63  GPUTrackingInOutDigits gpuDigitsMap;
64  GPUTPCDigitsMCInput gpuDigitsMapMC;
65  GPUTrackingInOutPointers ptrs;
66 
67  ptrs.tpcCompressedClusters = data->compressedClusters;
68  ptrs.tpcZS = data->tpcZS;
69  if (data->o2Digits) {
70  const float zsThreshold = mTrackingCAO2Interface->getConfig().configReconstruction.tpcZSthreshold;
71  const int maxContTimeBin = mTrackingCAO2Interface->getConfig().configEvent.continuousMaxTimeBin;
72  for (int i = 0; i < Sector::MAXSECTOR; i++) {
73  const auto& d = (*(data->o2Digits))[i];
74  if (zsThreshold > 0 && data->tpcZS == nullptr) {
75  gpuDigits[i].reserve(d.size());
76  }
77  for (int j = 0; j < d.size(); j++) {
78  if (maxContTimeBin && d[j].getTimeStamp() >= maxContTimeBin) {
79  throw std::runtime_error("Digit time bin exceeds time frame length");
80  }
81  if (zsThreshold > 0 && data->tpcZS == nullptr) {
82  if (d[j].getChargeFloat() >= zsThreshold) {
83  if (data->o2DigitsMC) {
84  for (const auto& element : (*data->o2DigitsMC)[i]->getLabels(j)) {
85  gpuDigitsMC[i].addElement(gpuDigits[i].size(), element);
86  }
87  }
88  gpuDigits[i].emplace_back(d[j]);
89  }
90  }
91  }
92  if (zsThreshold > 0 && data->tpcZS == nullptr) {
93  gpuDigitsMap.tpcDigits[i] = gpuDigits[i].data();
94  gpuDigitsMap.nTPCDigits[i] = gpuDigits[i].size();
95  if (data->o2DigitsMC) {
96  gpuDigitsMC[i].flatten_to(gpuDigitsMCConst[i].first);
97  gpuDigitsMCConst[i].second = gpuDigitsMCConst[i].first;
98  gpuDigitsMapMC.v[i] = &gpuDigitsMCConst[i].second;
99  }
100  } else {
101  gpuDigitsMap.tpcDigits[i] = (*(data->o2Digits))[i].data();
102  gpuDigitsMap.nTPCDigits[i] = (*(data->o2Digits))[i].size();
103  if (data->o2DigitsMC) {
104  gpuDigitsMapMC.v[i] = (*data->o2DigitsMC)[i];
105  }
106  }
107  }
108  if (data->o2DigitsMC) {
109  gpuDigitsMap.tpcDigitsMC = &gpuDigitsMapMC;
110  }
111  ptrs.tpcPackedDigits = &gpuDigitsMap;
112  }
113  ptrs.clustersNative = data->clusters;
114  int retVal = mTrackingCAO2Interface->RunTracking(&ptrs, outputs);
115  if (data->o2Digits || data->tpcZS || data->compressedClusters) {
116  data->clusters = ptrs.clustersNative;
117  }
118  data->compressedClusters = ptrs.tpcCompressedClusters;
119  data->outputTracks = {ptrs.outputTracksTPCO2, ptrs.nOutputTracksTPCO2};
120  data->outputClusRefs = {ptrs.outputClusRefsTPCO2, ptrs.nOutputClusRefsTPCO2};
121  data->outputTracksMCTruth = {ptrs.outputTracksTPCO2MC, ptrs.outputTracksTPCO2MC ? ptrs.nOutputTracksTPCO2 : 0};
122 
123  if (retVal || mTrackingCAO2Interface->getConfig().configInterface.dumpEvents >= 2) {
124  return retVal;
125  }
126 
127  mTrackingCAO2Interface->Clear(false);
128 
129  return (retVal);
130 }
