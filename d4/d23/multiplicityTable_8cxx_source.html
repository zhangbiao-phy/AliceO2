90 
91  // using FT0 row index from event selection task
92  int64_t foundFT0 = collision.foundFT0();
93 
94  if (foundFT0 != -1) {
95  auto ft0 = ft0s.iteratorAt(foundFT0);
96  aAmplitudesA = ft0.amplitudeA();
97  aAmplitudesC = ft0.amplitudeC();
98  for (int i = 0; i < 96; i++) {
99  multT0A += aAmplitudesA[i];
100  }
101  for (int i = 0; i < 112; i++) {
102  multT0C += aAmplitudesC[i];
103  }
104  }
105 
106  LOGF(debug, "multV0A=%5.0f multV0C=%5.0f multT0A=%5.0f multT0C=%5.0f multZNA=%6.0f multZNC=%6.0f multTracklets=%i", multV0A, multV0C, multT0A, multT0C, multZNA, multZNC, multTracklets);
107  mult(multV0A, multV0C, multT0A, multT0C, multZNA, multZNC, multTracklets);
108  }
109  }
