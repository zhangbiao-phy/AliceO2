68 // HACK: In order to be able to deduce which kind of aod object is transmitted to the templated VarManager::Fill functions
69 // a constexpr static bit map must be defined and sent as template argument
70 // The user has to include in this bit map all the tables needed in analysis, as defined in VarManager::ObjTypes
71 // Additionally, one should make sure that the requested tables are actually provided in the process() function,
72 // otherwise a compile time error will be thrown.

102  void DefineCuts()
103  {
104  // default cut is "eventStandard" (kINT7 and vtxZ selection)
105  TString cutNamesStr = fConfigCuts.value;
106  if (!cutNamesStr.IsNull()) {

114  VarManager::SetUseVars(AnalysisCut::fgUsedVars); // provide the list of required variables so that VarManager knows what to fill

118  {
119  // Reset the fValues array
120  VarManager::ResetValues(0, VarManager::kNEventWiseVariables, fValues);
121 
122  VarManager::FillEvent<gkEventFillMap>(event, fValues);
123  fHistMan->FillHistClass("Event_BeforeCuts", fValues); // automatically fill all the histograms in the class Event

125  fHistMan->FillHistClass("Event_AfterCuts", fValues);
126  eventSel(1);
127  } else {
128  eventSel(0);

178  void process(MyEventsSelected::iterator const& event, MyBarrelTracks const& tracks)
179  {
180  VarManager::ResetValues(0, VarManager::kNBarrelTrackVariables, fValues);
181  // fill event information which might be needed in histograms that combine track and event properties
182  VarManager::FillEvent<gkEventFillMap>(event, fValues);
183 
184  uint8_t filterMap = uint8_t(0);
185 
186  trackSel.reserve(tracks.size());
187 
188  for (auto& track : tracks) {
189  filterMap = uint8_t(0);
190  VarManager::FillTrack<gkTrackFillMap>(track, fValues);
191  if (event.isEventSelected()) {
192  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", fValues);
193  }
194 
195  int i = 0;
196  for (auto cut = fTrackCuts.begin(); cut != fTrackCuts.end(); ++cut, ++i) {
197  if ((*cut).IsSelected(fValues)) {
198  filterMap |= (uint8_t(1) << i);
199  fHistMan->FillHistClass(Form("TrackBarrel_%s", (*cut).GetName()), fValues);
200  }
201  }
202  trackSel(filterMap);
203  }
204  }

266  void process(MyEventsVtxCovSelected::iterator const& event, MyBarrelTracksSelected const& tracks)
267  {
268  if (!event.isEventSelected()) {
269  return;
270  }
271  // Reset the fValues array
272  VarManager::ResetValues(0, VarManager::kNVars, fValues);
273 
274  VarManager::FillEvent<gkEventFillMap>(event, fValues);
275 
276  // Run the same event pairing for barrel tracks
277  // TODO: Use combinations() when this will work for Partitions
278  /* e.g.
279  * for (auto& [tpos, tneg] : combinations(posTracks, negTracks)) {
280  VarManager::FillPair(tpos, tneg);
281  fHistMan->FillHistClass("PairsBarrelPM", VarManager::fgValues);
282  }
283  */
284  uint8_t filter = 0;
285  for (auto tpos : posTracks) {
286  for (auto tneg : negTracks) { // +- pairs
287  filter = tpos.isBarrelSelected() & tneg.isBarrelSelected();
288  if (!filter) { // the tracks must have at least one filter bit in common to continue
289  continue;
290  }
291  VarManager::FillPair(tpos, tneg, fValues);
292  dileptonList(event, fValues[VarManager::kMass], fValues[VarManager::kPt], fValues[VarManager::kEta], fValues[VarManager::kPhi], 0, filter);
293  for (int i = 0; i < fNTrackCuts; ++i) {
294  if (filter & (uint8_t(1) << i)) {
295  fHistMan->FillHistClass(Form("PairsBarrelPM_%s", fTrkCutsNameArray->At(i)->GetName()), fValues);
296  }
297  }
298  }
299  for (auto tpos2 = tpos + 1; tpos2 != posTracks.end(); ++tpos2) { // ++ pairs
300  filter = tpos.isBarrelSelected() & tpos2.isBarrelSelected();
301  if (!filter) { // the tracks must have at least one filter bit in common to continue
302  continue;
303  }
304  VarManager::FillPair(tpos, tpos2, fValues);
305  dileptonList(event, fValues[VarManager::kMass], fValues[VarManager::kPt], fValues[VarManager::kEta], fValues[VarManager::kPhi], 2, filter);
306  for (int i = 0; i < fNTrackCuts; ++i) {
307  if (filter & (uint8_t(1) << i)) {
308  fHistMan->FillHistClass(Form("PairsBarrelPP_%s", fTrkCutsNameArray->At(i)->GetName()), fValues);
309  }
310  }
311  }
312  }
313  for (auto tneg : negTracks) { // -- pairs
314  for (auto tneg2 = tneg + 1; tneg2 != negTracks.end(); ++tneg2) {
315  filter = tneg.isBarrelSelected() & tneg2.isBarrelSelected();
316  if (!filter) { // the tracks must have at least one filter bit in common to continue
317  continue;
318  }
319  VarManager::FillPair(tneg, tneg2, fValues);
320  dileptonList(event, fValues[VarManager::kMass], fValues[VarManager::kPt], fValues[VarManager::kEta], fValues[VarManager::kPhi], -2, filter);
321  for (int i = 0; i < fNTrackCuts; ++i) {
322  if (filter & (uint8_t(1) << i)) {
323  fHistMan->FillHistClass(Form("PairsBarrelMM_%s", fTrkCutsNameArray->At(i)->GetName()), fValues);
324  }
325  }
326  }
327  }
328  }
329 };
