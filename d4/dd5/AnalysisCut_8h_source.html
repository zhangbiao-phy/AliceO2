71 template <typename T1, typename T2>
72 void AnalysisCut::AddCut(int var, T1 cutLow, T2 cutHigh, bool exclude,
73  int dependentVar, float depCutLow, float depCutHigh, bool depCutExclude,
74  int dependentVar2, float depCut2Low, float depCut2High, bool depCut2Exclude)
75 {
76  //
77  // Add a cut
78  //
79  CutContainer cut = {};
80 
81  if constexpr (std::is_same_v<T1, TF1*>) {
82  if (dependentVar < 0) {
83  return;
84  }
85  cut.fFuncLow = cutLow;
86  cut.fLow = -9999.0;
87  } else {
88  cut.fFuncLow = nullptr;
89  cut.fLow = cutLow;
90  }
91  if constexpr (std::is_same_v<T2, TF1*>) {
92  if (dependentVar < 0) {
93  return;
94  }
95  cut.fFuncHigh = cutHigh;
96  cut.fHigh = -9999.0;
97  } else {
98  cut.fFuncHigh = nullptr;
99  cut.fHigh = cutHigh;
100  }
101  cut.fVar = var;
102  cut.fExclude = exclude;
103  fgUsedVars.push_back(var);
104 
105  cut.fDepVar = dependentVar;
106  cut.fDepLow = depCutLow;
107  cut.fDepHigh = depCutHigh;
108  cut.fDepExclude = depCutExclude;
109  if (dependentVar != -1) {
110  fgUsedVars.push_back(dependentVar);
111  }
112 
113  cut.fDepVar2 = dependentVar2;
114  cut.fDep2Low = depCut2Low;
115  cut.fDep2High = depCut2High;
116  cut.fDep2Exclude = depCut2Exclude;
117  if (dependentVar2 != -1) {
118  fgUsedVars.push_back(dependentVar2);
119  }
120 
121  fCuts.push_back(cut);
122 }

125 inline bool AnalysisCut::IsSelected(float* values)
126 {
127  //
128  // apply the configured cuts
129  //
130  // iterate over cuts
131  for (std::vector<CutContainer>::iterator it = fCuts.begin(); it != fCuts.end(); ++it) {
132  // check whether a dependent variables were enabled and if they are in the requested range
133  if ((*it).fDepVar != -1) {
134  bool inRange = (values[(*it).fDepVar] > (*it).fDepLow && values[(*it).fDepVar] <= (*it).fDepHigh);
135  if (!inRange && !((*it).fDepExclude)) {
136  continue;
137  }
138  if (inRange && (*it).fDepExclude) {
139  continue;
140  }
141  }
142  if ((*it).fDepVar2 != -1) {
143  bool inRange = (values[(*it).fDepVar2] > (*it).fDep2Low && values[(*it).fDepVar2] <= (*it).fDep2High);
144  if (!inRange && !((*it).fDep2Exclude)) {
145  continue;
146  }
147  if (inRange && (*it).fDep2Exclude) {
148  continue;
149  }
150  }
151  // obtain the low and high cut values (either directly as a value or from a function)
152  float cutLow, cutHigh;
153  if ((*it).fFuncLow) {
154  cutLow = ((*it).fFuncLow)->Eval(values[(*it).fDepVar]);
155  } else {
156  cutLow = ((*it).fLow);
157  }
158  if ((*it).fFuncHigh) {
159  cutHigh = ((*it).fFuncHigh)->Eval(values[(*it).fDepVar]);
160  } else {
161  cutHigh = ((*it).fHigh);
162  }
163  // apply the cut and return the decision
164  bool inRange = (values[(*it).fVar] >= cutLow && values[(*it).fVar] <= cutHigh);
165  if (!inRange && !((*it).fExclude)) {
166  return false;
167  }
168  if (inRange && ((*it).fExclude)) {
169  return false;
170  }
171  }
172 
173  return true;
174 }
