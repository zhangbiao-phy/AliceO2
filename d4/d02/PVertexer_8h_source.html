154 inline void PVertexer::applyConstraint(VertexSeed& vtxSeed) const
155 {
156  // impose meanVertex constraint, i.e. account terms
157  // (V_i-Constrain_i)^2/sig2constr_i for i=X,Y in the fit chi2 definition
158  vtxSeed.cxx += mXYConstraintInvErr[0];
159  vtxSeed.cyy += mXYConstraintInvErr[1];
160  vtxSeed.cx0 += mXYConstraintInvErr[0] * mMeanVertex.getX();
161  vtxSeed.cy0 += mXYConstraintInvErr[1] * mMeanVertex.getY();
162 }

165 inline bool PVertexer::upscaleSigma(VertexSeed& vtxSeed) const
166 {
167  // scale upward the scaleSigma2 if needes
168  if (vtxSeed.scaleSigma2 < mPVParams->maxScale2) {
169  auto s = vtxSeed.scaleSigma2 * mPVParams->upscaleFactor;
170  vtxSeed.setScale(s > mPVParams->maxScale2 ? mPVParams->maxScale2 : s, mTukey2I);
171  return true;
172  }
173  return false;
174 }

177 template <typename TR>
178 void PVertexer::createTracksPool(const TR& tracks, gsl::span<const o2d::GlobalTrackID> gids)
179 {
180  // create pull of all candidate tracks in a global array ordered in time
181  mTracksPool.clear();
182  mSortedTrackID.clear();
183 
184  auto ntGlo = tracks.size();
185  mTracksPool.reserve(ntGlo);
186  // check all containers
187  float vtxErr2 = 0.5 * (mMeanVertex.getSigmaX2() + mMeanVertex.getSigmaY2());
188  float pullIniCut = 9.; // RS FIXME pullIniCut should be a parameter
189  o2d::DCA dca;
190 
191  for (uint32_t i = 0; i < ntGlo; i++) {
192  o2::track::TrackParCov trc = tracks[i];
193  if (!trc.propagateToDCA(mMeanVertex, mBz, &dca, mPVParams->dcaTolerance) ||
194  dca.getY() * dca.getY() / (dca.getSigmaY2() + vtxErr2) > mPVParams->pullIniCut) {
195  continue;
196  }
197  auto& tvf = mTracksPool.emplace_back(trc, tracks[i].getTimeMUS(), i);
198  mStatZErr.add(std::sqrt(trc.getSigmaZ2()));
199  mStatTErr.add(tvf.timeEst.getTimeStampError());
200  }
201  // TODO: try to narrow timestamps using tof times
202  auto [zerrMean, zerrRMS] = mStatZErr.getMeanRMS2<float>();
203 
204  auto [terrMean, terrRMS] = mStatTErr.getMeanRMS2<float>();
205 
206  if (mTracksPool.empty()) {
207  return;
208  }
209  //
210  mSortedTrackID.resize(mTracksPool.size());
211  std::iota(mSortedTrackID.begin(), mSortedTrackID.end(), 0);
212 
213  std::sort(mSortedTrackID.begin(), mSortedTrackID.end(), [this](int i, int j) {
214  return this->mTracksPool[i].timeEst.getTimeStamp() < this->mTracksPool[j].timeEst.getTimeStamp();
215  });
216  auto tMin = mTracksPool[mSortedTrackID.front()].timeEst.getTimeStamp();
217  auto tMax = mTracksPool[mSortedTrackID.back()].timeEst.getTimeStamp();
218 }
