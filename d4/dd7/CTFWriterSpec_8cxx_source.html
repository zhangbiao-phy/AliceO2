42 template <typename T>
43 void appendToTree(TTree& tree, const std::string brname, T& ptr)
44 {
45  auto* br = tree.GetBranch(brname.c_str());
46  auto* pptr = &ptr;
47  if (br) {
48  br->SetAddress(&pptr);
49  } else {
50  br = tree.Branch(brname.c_str(), &pptr);
51  }
52  br->Fill();
53  br->ResetAddress();
54 }

56 CTFWriterSpec::CTFWriterSpec(DetID::mask_t dm, uint64_t r, bool doCTF, bool doDict, bool dictPerDet)
57  : mDets(dm), mRun(r), mWriteCTF(doCTF), mCreateDict(doDict), mDictPerDetector(dictPerDet)
58 {
59  mTimer.Stop();
60  mTimer.Reset();
61 
62  if (doDict) { // make sure that there is no local dictonary
63  for (int id = 0; id < DetID::nDetectors; id++) {
64  DetID det(id);
65  if (isPresent(det)) {
66  auto dictName = dictionaryFileName(det.getName());
67  if (gSystem->AccessPathName(dictName.c_str()) == 0) {
68  throw std::runtime_error(o2::utils::concat_string("CTF dictionary creation is requested but ", dictName, " already exists, remove it!"));
69  }
70  if (!mDictPerDetector) {
71  break; // no point in checking further
72  }
73  }
74  }
75  }
76 }

83 void CTFWriterSpec::run(ProcessingContext& pc)
84 {
85  auto cput = mTimer.CpuTime();
86  mTimer.Start(false);
87  auto tfOrb = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getByPos(0))->firstTForbit;
88 
89  std::unique_ptr<TFile> fileOut;
90  std::unique_ptr<TTree> treeOut;
91  if (mWriteCTF) {
92  // fileOut.reset(TFile::Open(o2::base::NameConf::getCTFFileName(tfOrb).c_str(), "recreate"));
93  // RS Until the DPL will propagate the firstTForbit, we will use simple counter in CTF file name to avoid overwriting in case of multiple TFs
94  fileOut.reset(TFile::Open(o2::base::NameConf::getCTFFileName(mNTF).c_str(), "recreate"));
95  treeOut = std::make_unique<TTree>(std::string(o2::base::NameConf::CTFTREENAME).c_str(), "O2 CTF tree");
96  }
97 
98  // create header
99  CTFHeader header{mRun, tfOrb};
100 
101  processDet<o2::itsmft::CTF>(pc, DetID::ITS, header, treeOut.get());
102  processDet<o2::itsmft::CTF>(pc, DetID::MFT, header, treeOut.get());
103  processDet<o2::tpc::CTF>(pc, DetID::TPC, header, treeOut.get());
104  processDet<o2::tof::CTF>(pc, DetID::TOF, header, treeOut.get());
105  processDet<o2::ft0::CTF>(pc, DetID::FT0, header, treeOut.get());
106  processDet<o2::fv0::CTF>(pc, DetID::FV0, header, treeOut.get());
107  processDet<o2::fdd::CTF>(pc, DetID::FDD, header, treeOut.get());
108  processDet<o2::mid::CTF>(pc, DetID::MID, header, treeOut.get());
109  processDet<o2::emcal::CTF>(pc, DetID::EMC, header, treeOut.get());
110  processDet<o2::phos::CTF>(pc, DetID::PHS, header, treeOut.get());
111  processDet<o2::cpv::CTF>(pc, DetID::CPV, header, treeOut.get());
112  processDet<o2::zdc::CTF>(pc, DetID::ZDC, header, treeOut.get());
113 
114  mTimer.Stop();
115 
116  if (mWriteCTF) {
117  appendToTree(*treeOut.get(), "CTFHeader", header);
118  treeOut->SetEntries(1);
119  treeOut->Write();
120  treeOut.reset();
121  fileOut->Close();
122  LOG(INFO) << "TF#" << mNTF << ": wrote " << fileOut->GetName() << " with CTF{" << header << "} in " << mTimer.CpuTime() - cput << " s";
123  } else {
124  LOG(INFO) << "TF#" << mNTF << " CTF writing is disabled";
125  }
126  mNTF++;
127  if (mCreateDict && mSaveDictAfter > 0 && (mNTF % mSaveDictAfter) == 0) {
128  storeDictionaries();
129  }
130 }

143 void CTFWriterSpec::prepareDictionaryTreeAndFile(DetID det)
144 {
145  if (mDictPerDetector) {
146  if (mDictTreeOut) {
147  mDictTreeOut->SetEntries(1);
148  mDictTreeOut->Write();
149  mDictTreeOut.reset();
150  mDictFileOut.reset();
151  }
152  }
153  if (!mDictTreeOut) {
154  mDictFileOut.reset(TFile::Open(dictionaryFileName(det.getName()).c_str(), "recreate"));
155  mDictTreeOut = std::make_unique<TTree>(std::string(o2::base::NameConf::CTFDICT).c_str(), "O2 CTF dictionary");
156  }
157 }

159 std::string CTFWriterSpec::dictionaryFileName(const std::string& detName)
160 {
161  if (mDictPerDetector) {
162  if (detName.empty()) {
163  throw std::runtime_error("Per-detector dictionary files are requested but detector name is not provided");
164  }
165  return o2::utils::concat_string(detName, "_", o2::base::NameConf::CTFDICT, ".root");
166  } else {
167  return o2::utils::concat_string(o2::base::NameConf::CTFDICT, ".root");
168  }

171 void CTFWriterSpec::storeDictionaries()
172 {
173  CTFHeader header{mRun, uint32_t(mNTF)};
174  storeDictionary<o2::itsmft::CTF>(DetID::ITS, header);
175  storeDictionary<o2::itsmft::CTF>(DetID::MFT, header);
176  storeDictionary<o2::tpc::CTF>(DetID::TPC, header);
177  storeDictionary<o2::tof::CTF>(DetID::TOF, header);
178  storeDictionary<o2::ft0::CTF>(DetID::FT0, header);
179  storeDictionary<o2::fv0::CTF>(DetID::FV0, header);
180  storeDictionary<o2::fdd::CTF>(DetID::FDD, header);
181  storeDictionary<o2::mid::CTF>(DetID::MID, header);
182  storeDictionary<o2::emcal::CTF>(DetID::EMC, header);
183  storeDictionary<o2::phos::CTF>(DetID::PHS, header);
184  storeDictionary<o2::cpv::CTF>(DetID::CPV, header);
185  storeDictionary<o2::zdc::CTF>(DetID::ZDC, header);
186  // close remnants
187  if (mDictTreeOut) {
188  closeDictionaryTreeAndFile(header);
189  }
190  LOG(INFO) << "Saved CTF dictionary after " << mNTF << " TFs processed";
191 }

193 void CTFWriterSpec::closeDictionaryTreeAndFile(CTFHeader& header)
194 {
195  if (mDictTreeOut) {
196  appendToTree(*mDictTreeOut.get(), "CTFHeader", header);
197  mDictTreeOut->SetEntries(1);
198  mDictTreeOut->Write(mDictTreeOut->GetName(), TObject::kSingleKey);
199  mDictTreeOut.reset();
200  mDictFileOut.reset();
201  }

204 DataProcessorSpec getCTFWriterSpec(DetID::mask_t dets, uint64_t run, bool doCTF, bool doDict, bool dictPerDet)
205 {
206  std::vector<InputSpec> inputs;
207  LOG(INFO) << "Detectors list:";
208  for (auto id = DetID::First; id <= DetID::Last; id++) {
209  if (dets[id]) {
210  inputs.emplace_back(DetID::getName(id), DetID::getDataOrigin(id), "CTFDATA", 0, Lifetime::Timeframe);
211  LOG(INFO) << "Det " << DetID::getName(id) << " added";
212  }
213  }
214  return DataProcessorSpec{
215  "ctf-writer",
216  inputs,
217  Outputs{},
218  AlgorithmSpec{adaptFromTask<CTFWriterSpec>(dets, run, doCTF, doDict, dictPerDet)},
219  Options{{"save-dict-after", VariantType::Int, -1, {"In dictionary generation mode save it dictionary after certain number of TFs processed"}}}};
220 }
