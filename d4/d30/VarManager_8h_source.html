261 template <uint32_t fillMap, typename T>
262 void VarManager::FillEvent(T const& event, float* values)
263 {
264  if (!values) {
265  values = fgValues;
266  }
267 
268  if constexpr ((fillMap & BC) > 0) {
269  values[kRunNo] = event.bc().runNumber(); // accessed via Collisions table
270  values[kBC] = event.bc().globalBC();
271  }
272 
273  if constexpr ((fillMap & Collision) > 0) {
274  if (fgUsedVars[kIsINT7]) {
275  values[kIsINT7] = (event.alias()[kINT7] > 0);
276  }
277  if (fgUsedVars[kIsEMC7]) {
278  values[kIsEMC7] = (event.alias()[kEMC7] > 0);
279  }
280  if (fgUsedVars[kIsINT7inMUON]) {
281  values[kIsINT7inMUON] = (event.alias()[kINT7inMUON] > 0);
282  }
283  if (fgUsedVars[kIsMuonSingleLowPt7]) {
284  values[kIsMuonSingleLowPt7] = (event.alias()[kMuonSingleLowPt7] > 0);
285  }
286  if (fgUsedVars[kIsMuonUnlikeLowPt7]) {
287  values[kIsMuonUnlikeLowPt7] = (event.alias()[kMuonUnlikeLowPt7] > 0);
288  }
289  if (fgUsedVars[kIsMuonLikeLowPt7]) {
290  values[kIsMuonLikeLowPt7] = (event.alias()[kMuonLikeLowPt7] > 0);
291  }
292  values[kVtxX] = event.posX();
293  values[kVtxY] = event.posY();
294  values[kVtxZ] = event.posZ();
295  values[kVtxNcontrib] = event.numContrib();
296  values[kCollisionTime] = event.collisionTime();
297  values[kVtxCovXX] = event.covXX();
298  values[kVtxCovXY] = event.covXY();
299  values[kVtxCovXZ] = event.covXZ();
300  values[kVtxCovYY] = event.covYY();
301  values[kVtxCovYZ] = event.covYZ();
302  values[kVtxCovZZ] = event.covZZ();
303  values[kVtxChi2] = event.chi2();
304  }
305 
306  if constexpr ((fillMap & CollisionCent) > 0) {
307  values[kCentVZERO] = event.centV0M();
308  }
309 
310  // TODO: need to add EvSels and Cents tables, etc. in case of the central data model
311 
312  if constexpr ((fillMap & ReducedEvent) > 0) {
313  values[kRunNo] = event.runNumber();
314  values[kVtxX] = event.posX();
315  values[kVtxY] = event.posY();
316  values[kVtxZ] = event.posZ();
317  values[kVtxNcontrib] = event.numContrib();
318  }
319 
320  if constexpr ((fillMap & ReducedEventExtended) > 0) {
321  values[kBC] = event.globalBC();
322  values[kCentVZERO] = event.centV0M();
323  if (fgUsedVars[kIsINT7]) {
324  values[kIsINT7] = (event.triggerAlias() & (uint32_t(1) << kINT7)) > 0;
325  }
326  if (fgUsedVars[kIsEMC7]) {
327  values[kIsEMC7] = (event.triggerAlias() & (uint32_t(1) << kEMC7)) > 0;
328  }
329  if (fgUsedVars[kIsINT7inMUON]) {
330  values[kIsINT7inMUON] = (event.triggerAlias() & (uint32_t(1) << kINT7inMUON)) > 0;
331  }
332  if (fgUsedVars[kIsMuonSingleLowPt7]) {
333  values[kIsMuonSingleLowPt7] = (event.triggerAlias() & (uint32_t(1) << kMuonSingleLowPt7)) > 0;
334  }
335  if (fgUsedVars[kIsMuonUnlikeLowPt7]) {
336  values[kIsMuonUnlikeLowPt7] = (event.triggerAlias() & (uint32_t(1) << kMuonUnlikeLowPt7)) > 0;
337  }
338  if (fgUsedVars[kIsMuonLikeLowPt7]) {
339  values[kIsMuonLikeLowPt7] = (event.triggerAlias() & (uint32_t(1) << kMuonLikeLowPt7)) > 0;
340  }
341  }
342 
343  if constexpr ((fillMap & ReducedEventVtxCov) > 0) {
344  values[kVtxCovXX] = event.covXX();
345  values[kVtxCovXY] = event.covXY();
346  values[kVtxCovXZ] = event.covXZ();
347  values[kVtxCovYY] = event.covYY();
348  values[kVtxCovYZ] = event.covYZ();
349  values[kVtxCovZZ] = event.covZZ();
350  values[kVtxChi2] = event.chi2();
351  }
352 
353  FillEventDerived(values);
354 }

356 template <uint32_t fillMap, typename T>
357 void VarManager::FillTrack(T const& track, float* values)
358 {
359  if (!values) {
360  values = fgValues;
361  }
362 
363  if constexpr ((fillMap & Track) > 0 || (fillMap & ReducedTrack) > 0) {
364  values[kPt] = track.pt();
365  if (fgUsedVars[kPx]) {
366  values[kPx] = track.px();
367  }
368  if (fgUsedVars[kPy]) {
369  values[kPy] = track.py();
370  }
371  if (fgUsedVars[kPz]) {
372  values[kPz] = track.pz();
373  }
374  values[kEta] = track.eta();
375  values[kPhi] = track.phi();
376  values[kCharge] = track.charge();
377  }
378 
379  if constexpr ((fillMap & TrackExtra) > 0 || (fillMap & ReducedTrackBarrel) > 0) {
380  values[kPin] = track.tpcInnerParam();
381  if (fgUsedVars[kIsITSrefit]) {
382  values[kIsITSrefit] = (track.flags() & (uint32_t(1) << 0)) > 0;
383  }
384  if (fgUsedVars[kIsTPCrefit]) {
385  values[kIsTPCrefit] = (track.flags() & (uint32_t(1) << 1)) > 0;
386  }
387  if (fgUsedVars[kIsSPDfirst]) {
388  values[kIsSPDfirst] = (track.itsClusterMap() & uint8_t(1)) > 0;
389  }
390  if (fgUsedVars[kIsSPDboth]) {
391  values[kIsSPDboth] = (track.itsClusterMap() & uint8_t(3)) > 0;
392  }
393  if (fgUsedVars[kIsSPDany]) {
394  values[kIsSPDany] = (track.itsClusterMap() & uint8_t(1)) || (track.itsClusterMap() & uint8_t(2));
395  }
396  values[kITSchi2] = track.itsChi2NCl();
397  values[kTPCncls] = track.tpcNClsFound();
398  values[kTPCchi2] = track.tpcChi2NCl();
399  values[kTrackLength] = track.length();
400  values[kTPCnclsCR] = track.tpcNClsCrossedRows();
401 
402  if constexpr ((fillMap & TrackExtra) > 0) {
403  if (fgUsedVars[kITSncls]) {
404  values[kITSncls] = track.itsNCls(); // dynamic column
405  }
406  }
407  if constexpr ((fillMap & ReducedTrackBarrel) > 0) {
408  if (fgUsedVars[kITSncls]) {
409  values[kITSncls] = 0.0;
410  for (int i = 0; i < 6; ++i) {
411  values[kITSncls] += ((track.itsClusterMap() & (1 << i)) ? 1 : 0);
412  }
413  }
414  values[kTrackDCAxy] = track.dcaXY();
415  values[kTrackDCAz] = track.dcaZ();
416  }
417  }
418 
419  if constexpr ((fillMap & TrackDCA) > 0) {
420  values[kTrackDCAxy] = track.dcaXY();
421  values[kTrackDCAz] = track.dcaZ();
422  }
423 
424  if constexpr ((fillMap & TrackSelection) > 0) {
425  values[kIsGlobalTrack] = track.isGlobalTrack();
426  values[kIsGlobalTrackSDD] = track.isGlobalTrackSDD();
427  }
428 
429  if constexpr ((fillMap & TrackCov) > 0 || (fillMap & ReducedTrackBarrelCov) > 0) {
430  values[kTrackCYY] = track.cYY();
431  values[kTrackCZZ] = track.cZZ();
432  values[kTrackCSnpSnp] = track.cSnpSnp();
433  values[kTrackCTglTgl] = track.cTglTgl();
434  values[kTrackC1Pt21Pt2] = track.c1Pt21Pt2();
435  }
436 
437  if constexpr ((fillMap & TrackPID) > 0 || (fillMap & ReducedTrackBarrelPID) > 0) {
438  values[kTPCnSigmaEl] = track.tpcNSigmaEl();
439  values[kTPCnSigmaMu] = track.tpcNSigmaMu();
440  values[kTPCnSigmaPi] = track.tpcNSigmaPi();
441  values[kTPCnSigmaKa] = track.tpcNSigmaKa();
442  values[kTPCnSigmaPr] = track.tpcNSigmaPr();
443  values[kTOFnSigmaEl] = track.tofNSigmaEl();
444  values[kTOFnSigmaMu] = track.tofNSigmaMu();
445  values[kTOFnSigmaPi] = track.tofNSigmaPi();
446  values[kTOFnSigmaKa] = track.tofNSigmaKa();
447  values[kTOFnSigmaPr] = track.tofNSigmaPr();
448  values[kTPCsignal] = track.tpcSignal();
449  values[kTRDsignal] = track.trdSignal();
450  values[kTOFsignal] = track.tofSignal();
451  values[kTOFbeta] = track.beta();
452  }
453 
454  if constexpr ((fillMap & ReducedTrackMuon) > 0) {
455  values[kMuonInvBendingMomentum] = track.inverseBendingMomentum();
456  values[kMuonThetaX] = track.thetaX();
457  values[kMuonThetaY] = track.thetaY();
458  values[kMuonZMu] = track.zMu();
459  values[kMuonBendingCoor] = track.bendingCoor();
460  values[kMuonNonBendingCoor] = track.nonBendingCoor();
461  values[kMuonRAtAbsorberEnd] = track.rAtAbsorberEnd();
462  values[kMuonPDca] = track.pDca();
463  values[kMuonChi2] = track.chi2();
464  values[kMuonChi2MatchTrigger] = track.chi2MatchTrigger();
465  }
466 
467  if constexpr ((fillMap & Pair) > 0) {
468  values[kMass] = track.mass();
469  }
470 
471  FillTrackDerived(values);
472 }

474 template <typename T>
475 void VarManager::FillPair(T const& t1, T const& t2, float* values, PairCandidateType pairType)
476 {
477  if (!values) {
478  values = fgValues;
479  }
480 
481  float m1 = fgkElectronMass;
482  float m2 = fgkElectronMass;
483  if (pairType == kJpsiToMuMu) {
484  m1 = fgkMuonMass;
485  m2 = fgkMuonMass;
486  }
487 
488  ROOT::Math::PtEtaPhiMVector v1(t1.pt(), t1.eta(), t1.phi(), m1);
489  ROOT::Math::PtEtaPhiMVector v2(t2.pt(), t2.eta(), t2.phi(), m2);
490  ROOT::Math::PtEtaPhiMVector v12 = v1 + v2;
491  values[kMass] = v12.M();
492  values[kPt] = v12.Pt();
493  values[kEta] = v12.Eta();
494  values[kPhi] = v12.Phi();
495  values[kRap] = -v12.Rapidity();
496 }

498 template <typename T1, typename T2>
499 void VarManager::FillDileptonHadron(T1 const& dilepton, T2 const& hadron, float* values, float hadronMass)
500 {
501  if (!values) {
502  values = fgValues;
503  }
504 
505  if (fgUsedVars[kPairMass] || fgUsedVars[kPairPt] || fgUsedVars[kPairEta] || fgUsedVars[kPairPhi]) {
506  ROOT::Math::PtEtaPhiMVector v1(dilepton.pt(), dilepton.eta(), dilepton.phi(), dilepton.mass());
507  ROOT::Math::PtEtaPhiMVector v2(hadron.pt(), hadron.eta(), hadron.phi(), hadronMass);
508  ROOT::Math::PtEtaPhiMVector v12 = v1 + v2;
509  values[kPairMass] = v12.M();
510  values[kPairPt] = v12.Pt();
511  values[kPairEta] = v12.Eta();
512  values[kPairPhi] = v12.Phi();
513  }
514  if (fgUsedVars[kDeltaPhi]) {
515  double delta = dilepton.phi() - hadron.phi();
516  if (delta > 3.0 / 2.0 * TMath::Pi()) {
517  delta -= 2.0 * TMath::Pi();
518  }
519  if (delta < -0.5 * TMath::Pi()) {
520  delta += 2.0 * TMath::Pi();
521  }
522  values[kDeltaPhi] = delta;
523  }
524  if (fgUsedVars[kDeltaPhiSym]) {
525  double delta = TMath::Abs(dilepton.phi() - hadron.phi());
526  if (delta > TMath::Pi()) {
527  delta = 2 * TMath::Pi() - delta;
528  }
529  values[kDeltaPhiSym] = delta;
530  }
531  if (fgUsedVars[kDeltaEta]) {
532  values[kDeltaEta] = dilepton.eta() - hadron.eta();
533  }
534 }
