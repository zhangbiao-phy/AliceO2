39  inline bool run()
40  {
41  rewind();
42  if (mDecoderCONET) {
43  mDecoderPointerMax = reinterpret_cast<const uint32_t*>(mDecoderBuffer + mDecoderBufferSize);
44  while (mDecoderPointer < mDecoderPointerMax) {
45  mEventCounter++;
46  processDRM();
47  if (mDecoderFatal) {
48  mFatalCounter++;
49  }
50  if (mDecoderError) {
51  mErrorCounter++;
52  }
53  }
54  return false;
55  }
56  while (!processHBF()) {
57  ;
58  }
59  return false;
60  };
