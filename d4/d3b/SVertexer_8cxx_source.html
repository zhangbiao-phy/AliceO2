57 void SVertexer::process(const gsl::span<const PVertex>& vertices, // primary vertices
58  const gsl::span<const GIndex>& trackIndex, // Global ID's for associated tracks
59  const gsl::span<const VRef>& vtxRefs, // references from vertex to these track IDs
60  const o2d::GlobalTrackAccessor& tracksPool, // accessor to various tracks
61  std::vector<V0>& v0s, // found V0s
62  std::vector<RRef>& vtx2V0refs // references from PVertex to V0
63 )
64 {
65  std::unordered_map<uint64_t, int> cache; // cache for tested combinations, the value >0 will give the entry of prevalidated V0 in the v0sTmp
66  std::vector<V0> v0sTmp(1); // 1st one is dummy!
67  std::vector<int> v0sIdx; // id's in v0sTmp used attached to p.vertices
68  std::vector<RRef> pv2v0sRefs; // p.vertex to v0 index references
69  std::vector<char> selQ(trackIndex.size(), 0);
70 
71  auto selTrack = [&tracksPool, &trackIndex](int i) -> char {
72  auto id = trackIndex[i];
73  return id.isPVContributor() || (id.getSource() != GIndex::ITSTPC && id.getSource() != GIndex::ITS) ? 0 : tracksPool.get(id).getCharge();
74  };
75 
76  for (size_t i = 0; i < trackIndex.size(); i++) {
77  selQ[i] = selTrack(i);
78  }
79 
80  int nv = vertices.size();
81  size_t countTot = 0, countTotUnique = 0;
82  for (int iv = 0; iv < nv; iv++) {
83  //
84  // select vertices
85  auto& pvrefs = pv2v0sRefs.emplace_back(v0sIdx.size(), 0);
86  const auto& pv = vertices[iv];
87  if (pv.getNContributors() < 5) {
88  continue;
89  }
90  //
91  const auto& vtref = vtxRefs[iv];
92  //
93  int first = vtref.getFirstEntry();
94  int last = first + vtref.getEntries();
95  size_t count = 0, countUnique = 0;
96  for (int ipos = first; ipos < last; ipos++) {
97  if (selQ[ipos] != 1) {
98  continue;
99  }
100  auto idpos = trackIndex[ipos];
101  bool ambiguousPos = idpos.isAmbiguous(); // is this track compatible also with other vertex?
102  const auto& trPos = tracksPool.get(idpos);
103 
104  for (int ineg = first; ineg < last; ineg++) {
105  if (selQ[ineg] != -1) {
106  continue;
107  }
108  auto idneg = trackIndex[ineg];
109  count++;
110  bool accept = false, newPair = true, ambiguousV0 = ambiguousPos && idneg.isAmbiguous(); // V0 is ambiguous if both tracks are compatible with other vertices
111  uint64_t idPosNeg = getPairIdx(idpos, idneg);
112  int* resPair = nullptr;
113  if (ambiguousV0) { // check if it was already processed
114  resPair = &cache[idPosNeg];
115  if ((*resPair) < 0) { // was already checked and rejected
116  continue;
117  } else if ((*resPair) == 0) { // was not checked yet
118  countUnique++;
119  (*resPair) = -1; // this is rejection flag
120  } else {
121  newPair = false;
122  }
123  }
124  if (newPair) {
125  auto trNeg = tracksPool.get(idneg);
126  // LOG(INFO) << "0: " << idpos << " " << tr0.o2::track::TrackPar::asString();
127  // LOG(INFO) << "1: " << idneg << " " << tr1.o2::track::TrackPar::asString();
128  int nCand = mFitter2Prong.process(trPos, trNeg);
129  if (nCand == 0) { // discard this pair
130  continue;
131  }
132  const auto& v0pos = mFitter2Prong.getPCACandidate();
133  // check closeness to the beam-line
134  auto r2 = (v0pos[0] - mMeanVertex.getX()) * (v0pos[0] - mMeanVertex.getX()) + (v0pos[1] - mMeanVertex.getY()) * (v0pos[1] - mMeanVertex.getY());
135  if (r2 < mMinR2ToMeanVertex) {
136  continue;
137  }
138  if (!mFitter2Prong.isPropagateTracksToVertexDone() && !mFitter2Prong.propagateTracksToVertex()) {
139  continue;
140  }
141  auto& trPosProp = mFitter2Prong.getTrack(0);
142  auto& trNegProp = mFitter2Prong.getTrack(1);
143  std::array<float, 3> pPos, pNeg;
144  trPosProp.getPxPyPzGlo(pPos);
145  trNegProp.getPxPyPzGlo(pNeg);
146  // estimate DCA of neutral V0 track to beamline: straight line with parametric equation
147  // x = X0 + pV0[0]*t, y = Y0 + pV0[1]*t reaches DCA to beamline (Xv, Yv) at
148  // t = -[ (x0-Xv)*pV0[0] + (y0-Yv)*pV0[1]) ] / ( pT(pV0)^2 )
149  // Similar equation for 3D distance involving pV0[2]
150  std::array<float, 3> pV0 = {pPos[0] + pNeg[0], pPos[1] + pNeg[1], pPos[2] + pNeg[2]};
151  float dx = v0pos[0] - mMeanVertex.getX(), dy = v0pos[1] - mMeanVertex.getY();
152  float pt2V0 = pV0[0] * pV0[0] + pV0[1] * pV0[1], prodXY = dx * pV0[0] + dy * pV0[1], tDCAXY = -prodXY / pt2V0;
153  float dcaX = dx + pV0[0] * tDCAXY, dcaY = dy + pV0[1] * tDCAXY, dca2 = dcaX * dcaX + dcaY * dcaY;
154  if (dca2 > mMaxDCAXY2ToMeanVertex) {
155  continue;
156  }
157  float p2V0 = pt2V0 + pV0[2] * pV0[2], ptV0 = std::sqrt(pt2V0);
158 
159  // apply mass selections
160  float p2Pos = pPos[0] * pPos[0] + pPos[1] * pPos[1] + pPos[2] * pPos[2], p2Neg = pNeg[0] * pNeg[0] + pNeg[1] * pNeg[1] + pNeg[2] * pNeg[2];
161  bool goodHyp = false;
162  for (int ipid = 0; ipid < SVertexerParams::NPIDV0; ipid++) {
163  if (mV0Hyps[ipid].check(p2Pos, p2Neg, p2V0, ptV0)) {
164  goodHyp = true;
165  break;
166  }
167  }
168  if (!goodHyp) {
169  continue;
170  }
171  // check cos of pointing angle
172  float dz = v0pos[2] - pv.getZ(), cosPointingAngle = (prodXY + dz * pV0[2]) / std::sqrt((dx * dx + dy * dy + dz * dz) * p2V0);
173  if (cosPointingAngle < mMinCosPointingAngle) {
174  if (!ambiguousV0) {
175  continue; // no need to check this pair wrt other vertices
176  }
177  cosPointingAngle = mMinCosPointingAngle - 1e-6;
178  } else { // cuts passed, register v0
179  accept = true;
180  }
181  // preliminary checks passed, cache V0 and proceed to specific vertex check
182  int szV0tmp = v0sTmp.size();
183  if (resPair) {
184  *resPair = szV0tmp; // index of V0 to be created
185  }
186  // LOG(INFO) << "Adding new V0 " << szV0tmp;
187 
188  std::array<float, 3> v0posF = {float(v0pos[0]), float(v0pos[1]), float(v0pos[2])};
189  auto& v0 = v0sTmp.emplace_back(v0posF, pV0, trPosProp, trNegProp, idpos, idneg);
190  if (accept) {
191  v0.setCosPA(cosPointingAngle);
192  v0.setDCA(mFitter2Prong.getChi2AtPCACandidate());
193  v0.setVertexID(iv);
194  v0sIdx.push_back(szV0tmp);
195  pvrefs.changeEntriesBy(1);
196  }
197  } else { // check if already created V0 is good for this vertex
198  // LOG(INFO) << "Rechecking new V0 " << *resPair;
199  auto& v0 = v0sTmp[*resPair];
200  std::array<float, 3> posV0, momV0;
201  v0.getXYZGlo(posV0);
202  v0.getPxPyPzGlo(momV0);
203  float cosPointingAngle = posV0[0] * momV0[0] + posV0[1] * momV0[1] + posV0[2] * momV0[2];
204  cosPointingAngle /= std::sqrt((posV0[0] * posV0[0] + posV0[1] * posV0[1] + posV0[2] * posV0[2]) * (momV0[0] * momV0[0] + momV0[1] * momV0[1] + momV0[2] * momV0[2]));
205  if (cosPointingAngle > v0.getCosPA()) { // reassign
206  v0.setCosPA(cosPointingAngle);
207  v0.setVertexID(iv);
208  v0sIdx.push_back(*resPair);
209  pvrefs.changeEntriesBy(1);
210  }
211  }
212  }
213  }
214  countTot += count;
215  countTotUnique += countUnique;
216  // LOG(INFO) << "Tried " << count << " combs with " << countUnique << " uniques";
217  }
218  // finalize V0s
219  for (int iv = 0; iv < nv; iv++) {
220  const auto& pvtx = vertices[iv];
221  const auto& pvrefsTmp = pv2v0sRefs[iv];
222  auto& pvrefsFin = vtx2V0refs.emplace_back(v0s.size(), 0);
223  int from = pvrefsTmp.getFirstEntry(), to = from + pvrefsTmp.getEntries(), nAdded = 0;
224  for (int iv0 = from; iv0 < to; iv0++) {
225  const auto& v0tmp = v0sTmp[v0sIdx[iv0]];
226  if (v0tmp.getVertexID() != iv) { // this v0 - p.vertex association was reassigned
227  continue;
228  }
229  v0s.push_back(v0tmp);
230  nAdded++;
231  }
232  pvrefsFin.setEntries(nAdded);
233  LOG(INFO) << nAdded << " V0s added for vertex " << iv << " with " << pvtx.getNContributors() << " tracks, out of initial " << pvrefsTmp.getEntries();
234  pvtx.print();
235  }
236 
237  LOG(INFO) << "Tried " << countTot << " combs in total " << countTotUnique << " uniques";
238 }
