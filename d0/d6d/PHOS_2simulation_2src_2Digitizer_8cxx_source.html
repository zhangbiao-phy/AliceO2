51 void Digitizer::processHits(const std::vector<Hit>* hits, const std::vector<Digit>& digitsBg,
52  std::vector<Digit>& digitsOut, o2::dataformats::MCTruthContainer<MCLabel>& labels,
53  int collId, int source, double dt)
54 {
55  // Convert list of hits + possible Bg digits to digits:
56  // Add hits with energy deposition in same cell and same time
57  // Add energy corrections
58  // Apply time smearing
59  // //Despite sorting in Detector::EndEvent(), hits still can be unsorted due to splitting of processing different bunches of primary
60  for (int i = NCHANNELS; i--;) {
61  mArrayD[i].reset();
62  }
63 
64  if (digitsBg.size() == 0) { // no digits provided: try simulate noise
65  for (int i = NCHANNELS; i--;) {
66  float energy = simulateNoiseEnergy(i + OFFSET);
67  energy = uncalibrate(energy, i + OFFSET);
68  if (energy > o2::phos::PHOSSimParams::Instance().mDigitThreshold) {
69  float time = simulateNoiseTime();
70  mArrayD[i].setAmplitude(energy);
71  mArrayD[i].setTime(time);
72  mArrayD[i].setAbsId(i + OFFSET);
73  }
74  }
75  } else { //if digits exist, no noise should be added
76  for (auto& dBg : digitsBg) { //digits are sorted and unique
77  mArrayD[dBg.getAbsId() - OFFSET] = dBg;
78  }
79  }
80 
81  //add Hits
82  for (auto& h : *hits) {
83  short absId = h.GetDetectorID();
84  short i = absId - OFFSET;
85  float energy = h.GetEnergyLoss();
86  if (o2::phos::PHOSSimParams::Instance().mApplyNonLinearity) {
87  energy = nonLinearity(energy);
88  }
89  energy = uncalibrate(energy, absId);
90  float time = h.GetTime() + dt * 1.e-9;
91  if (o2::phos::PHOSSimParams::Instance().mApplyTimeResolution) {
92  time = uncalibrateT(timeResolution(time, energy), absId);
93  }
94  if (mArrayD[i].getAmplitude() > 0) {
95  //update energy and time
96  mArrayD[i].addEnergyTime(energy, time);
97  //if overflow occured?
98  if (mArrayD[i].isHighGain()) {
99  if (mArrayD[i].getAmplitude() > o2::phos::PHOSSimParams::Instance().mMCOverflow) { //10bit ADC
100  float hglgratio = mCalibParams->getHGLGRatio(absId);
101  mArrayD[i].setAmplitude(mArrayD[i].getAmplitude() / hglgratio);
102  mArrayD[i].setHighGain(false);
103  }
104  }
105  } else {
106  mArrayD[i].setHighGain(energy < o2::phos::PHOSSimParams::Instance().mMCOverflow); //10bit ADC
107  if (mArrayD[i].isHighGain()) {
108  mArrayD[i].setAmplitude(energy);
109  } else {
110  float hglgratio = mCalibParams->getHGLGRatio(absId);
111  mArrayD[i].setAmplitude(energy / hglgratio);
112  }
113  mArrayD[i].setTime(time);
114  mArrayD[i].setAbsId(absId);
115  }
116  //Add MC info
117  if (mProcessMC) {
118  int labelIndex = mArrayD[i].getLabel();
119  if (labelIndex == -1) { //no digit or noisy
120  labelIndex = labels.getIndexedSize();
121  MCLabel label(h.GetTrackID(), collId, source, false, h.GetEnergyLoss());
122  labels.addElement(labelIndex, label);
123  mArrayD[i].setLabel(labelIndex);
124  } else { //check if lable already exist
125  MCLabel label(h.GetTrackID(), collId, source, false, h.GetEnergyLoss());
126  gsl::span<MCLabel> sp = labels.getLabels(labelIndex);
127  bool found = false;
128  for (MCLabel& te : sp) {
129  if (te == label) {
130  found = true;
131  te.add(label, 1.);
132  break;
133  }
134  }
135  if (!found) {
136  //Highly inefficient management of Labels: commenting line below reeduces WHOLE digitization time by factor ~30
137  labels.addElementRandomAccess(labelIndex, label);
138  //sort MCLabels according to eDeposited
139  sp = labels.getLabels(labelIndex);
140  std::sort(sp.begin(), sp.end(),
141  [](o2::phos::MCLabel a, o2::phos::MCLabel b) { return a.getEdep() > b.getEdep(); });
142  }
143  }
144  } else {
145  mArrayD[i].setLabel(-1);
146  }
147  }
148  for (int i = 0; i < NCHANNELS; i++) {
149  if (mArrayD[i].getAmplitude() > PHOSSimParams::Instance().mZSthreshold) {
150  digitsOut.push_back(mArrayD[i]);
151  }
152  }
153 }

156 float Digitizer::nonLinearity(const float e)
157 {
158  float a = o2::phos::PHOSSimParams::Instance().mCellNonLineaityA;
159  float b = o2::phos::PHOSSimParams::Instance().mCellNonLineaityB;
160  float c = o2::phos::PHOSSimParams::Instance().mCellNonLineaityC;
161  return e * c * (1. + a * exp(-e * e / (2. * b * b)));
162 }

164 float Digitizer::uncalibrate(const float e, const int absId)
165 {
166  // Decalibrate EMC digit, i.e. transform from energy to ADC counts a factor read from CDB
167  float calib = mCalibParams->getGain(absId);
168  if (calib > 0) {
169  return floor(e / calib);
170  } else {
171  return 0;
172  }
173 }
174 //_______________________________________________________________________
175 float Digitizer::uncalibrateT(const float time, const int absId)
176 {
177  // Decalibrate EMC digit, i.e. transform from energy to ADC counts a factor read from CDB
178  // note time in seconds
179  return time + mCalibParams->getHGTimeCalib(absId);
180 }

182 float Digitizer::timeResolution(const float time, const float e)
183 {
184  // apply time resolution
185  // time measured in seconds
186 
187  float timeResolution = o2::phos::PHOSSimParams::Instance().mTimeResolutionA +
188  o2::phos::PHOSSimParams::Instance().mTimeResolutionB /
189  std::max(float(e), o2::phos::PHOSSimParams::Instance().mTimeResThreshold);
190  return gRandom->Gaus(time, timeResolution);
191 }
