118 void DCSProcessor::init(const std::vector<DPID>& pids)
119 {
120 
121  int nchars = 0, nints = 0, ndoubles = 0, nUints = 0,
122  nbools = 0, nstrings = 0, ntimes = 0, nbinaries = 0;
123  for (auto it = std::begin(pids); it != std::end(pids); ++it) {
124  if ((*it).get_type() == DeliveryType::RAW_CHAR) {
125  mPidschars.emplace_back((*it).get_alias(), DeliveryType::RAW_CHAR);
126  mPids[*it] = nchars;
127  nchars++;
128  }
129  if ((*it).get_type() == DeliveryType::RAW_INT) {
130  mPidsints.emplace_back((*it).get_alias(), DeliveryType::RAW_INT);
131  mPids[*it] = nints;
132  nints++;
133  }
134  if ((*it).get_type() == DeliveryType::RAW_DOUBLE) {
135  mPidsdoubles.emplace_back((*it).get_alias(), DeliveryType::RAW_DOUBLE);
136  mPids[*it] = ndoubles;
137  ndoubles++;
138  }
139  if ((*it).get_type() == DeliveryType::RAW_UINT) {
140  mPidsUints.emplace_back((*it).get_alias(), DeliveryType::RAW_UINT);
141  mPids[*it] = nUints;
142  nUints++;
143  }
144  if ((*it).get_type() == DeliveryType::RAW_BOOL) {
145  mPidsbools.emplace_back((*it).get_alias(), DeliveryType::RAW_BOOL);
146  mPids[*it] = nbools;
147  nbools++;
148  }
149  if ((*it).get_type() == DeliveryType::RAW_STRING) {
150  mPidsstrings.emplace_back((*it).get_alias(), DeliveryType::RAW_STRING);
151  mPids[*it] = nstrings;
152  nstrings++;
153  }
154  if ((*it).get_type() == DeliveryType::RAW_TIME) {
155  mPidstimes.emplace_back((*it).get_alias(), DeliveryType::RAW_TIME);
156  mPids[*it] = ntimes;
157  ntimes++;
158  }
159  if ((*it).get_type() == DeliveryType::RAW_BINARY) {
160  mPidsbinaries.emplace_back((*it).get_alias(), DeliveryType::RAW_BINARY);
161  mPids[*it] = nbinaries;
162  nbinaries++;
163  }
164  }
165 
166  mLatestTimestampchars.resize(nchars, 0);
167  mLatestTimestampints.resize(nints, 0);
168  mLatestTimestampdoubles.resize(ndoubles, 0);
169  mLatestTimestampUints.resize(nUints, 0);

171  mLatestTimestampstrings.resize(nstrings, 0);

178 int DCSProcessor::processMap(const std::unordered_map<DPID, DPVAL>& map, bool isDelta)
179 {
180 
181  // process function to do "something" with the DCS map that is passed
182 
183  // resetting the content of the CCDB object to be sent
184 
185  if (!isDelta) {
186  // full map sent
187  mFullMapSent = true;
188  } else {
189  if (!mFullMapSent) {
190  LOG(ERROR) << "We need first a full map!";
191  }
192  mNCyclesNoFullMap++;
193  if (mNCyclesNoFullMap > mMaxCyclesNoFullMap) {
194  LOG(ERROR) << "We expected a full map!";
195  }
196  }
197 
198  mIsDelta = isDelta;
199 
200  // we need to check if there are the Data Points that we need
201 
202  int foundChars = 0, foundInts = 0, foundDoubles = 0, foundUInts = 0,
203  foundBools = 0, foundStrings = 0, foundTimes = 0, foundBinaries = 0;
204 
205  // char type
206  foundChars = processArrayType(mPidschars, DeliveryType::RAW_CHAR, map, mLatestTimestampchars, mDpscharsmap);
207 
208  // int type
209  foundInts = processArrayType(mPidsints, DeliveryType::RAW_INT, map, mLatestTimestampints, mDpsintsmap);
210 
211  // double type
212  foundDoubles = processArrayType(mPidsdoubles, DeliveryType::RAW_DOUBLE, map, mLatestTimestampdoubles,
213  mDpsdoublesmap);
214 
215  // UInt type
216  foundUInts = processArrayType(mPidsUints, DeliveryType::RAW_UINT, map, mLatestTimestampUints, mDpsUintsmap);
217 
218  // Bool type
219  foundBools = processArrayType(mPidsbools, DeliveryType::RAW_BOOL, map, mLatestTimestampbools, mDpsboolsmap);
220 
221  // String type
222  foundStrings = processArrayType(mPidsstrings, DeliveryType::RAW_STRING, map, mLatestTimestampstrings,
223  mDpsstringsmap);
224 
225  // Time type
226  foundTimes = processArrayType(mPidstimes, DeliveryType::RAW_TIME, map, mLatestTimestamptimes, mDpstimesmap);
227 
228  // Binary type
229  foundBinaries = processArrayType(mPidsbinaries, DeliveryType::RAW_BINARY, map, mLatestTimestampbinaries,
230  mDpsbinariesmap);
231 
232  if (!isDelta) {
233  if (foundChars != mPidschars.size()) {
234  LOG(INFO) << "Not all expected char-typed DPs found!";
235  }
236  if (foundInts != mPidsints.size()) {
237  LOG(INFO) << "Not all expected int-typed DPs found!";
238  }
239  if (foundDoubles != mPidsdoubles.size()) {
240  LOG(INFO) << "Not all expected double-typed DPs found!";
241  }
242  if (foundUInts != mPidsUints.size()) {
243  LOG(INFO) << "Not all expected uint-typed DPs found!";
244  }
245  if (foundBools != mPidsbools.size()) {
246  LOG(INFO) << "Not all expected bool-typed DPs found!";
247  }
248  if (foundStrings != mPidsstrings.size()) {
249  LOG(INFO) << "Not all expected string-typed DPs found!";
250  }
251  if (foundTimes != mPidstimes.size()) {
252  LOG(INFO) << "Not all expected time-typed DPs found!";
253  }
254  if (foundBinaries != mPidsbinaries.size()) {
255  LOG(INFO) << "Not all expected binary-typed DPs found!";
256  }
257  }
258 
259  // filling CCDB info to be sent in output
260  std::map<std::string, std::string> md;
261  prepareCCDBobject(mccdbSimpleMovingAverage, mccdbSimpleMovingAverageInfo,
262  mName + "/TestDCS/SimpleMovingAverageDPs", mTF, md);
263 
264  LOG(DEBUG) << "Size of unordered_map for CCDB = " << mccdbSimpleMovingAverage.size();
265  LOG(DEBUG) << "CCDB entry for TF " << mTF << " will be:";
266  for (const auto& i : mccdbSimpleMovingAverage) {
267  LOG(DEBUG) << i.first << " --> " << i.second;
268  }
269 
270  return 0;
271 }

275 int DCSProcessor::processDP(const std::pair<DPID, DPVAL>& dpcom)
276 {
277 
278  // processing single DP
279 
280  DPID dpid = dpcom.first;
281  DeliveryType type = dpid.get_type();
282 
283  // first we check if the DP is in the list for the detector
284  auto el = mPids.find(dpid);
285  if (el == mPids.end()) {
286  LOG(ERROR) << "DP not found for this detector, please check";
287  return 1;
288  }
289  int posInVector = (*el).second;
290 
291  if (type == DeliveryType::RAW_CHAR) {
292  checkFlagsAndFill(dpcom, mLatestTimestampchars[posInVector], mDpscharsmap);
293  processCharDP(dpid);
294  }
295 
296  else if (type == DeliveryType::RAW_INT) {
297  checkFlagsAndFill(dpcom, mLatestTimestampints[posInVector], mDpsintsmap);
298  processIntDP(dpid);
299  }
300 
301  else if (type == DeliveryType::RAW_DOUBLE) {
302  checkFlagsAndFill(dpcom, mLatestTimestampdoubles[posInVector], mDpsdoublesmap);
303  processDoubleDP(dpid);
304  }
305 
306  else if (type == DeliveryType::RAW_UINT) {
307  checkFlagsAndFill(dpcom, mLatestTimestampUints[posInVector], mDpsUintsmap);
308  processUIntDP(dpid);
309  }
310 
311  else if (type == DeliveryType::RAW_BOOL) {
312  checkFlagsAndFill(dpcom, mLatestTimestampbools[posInVector], mDpsboolsmap);
313  processBoolDP(dpid);
314  }
315 
316  else if (type == DeliveryType::RAW_STRING) {
317  checkFlagsAndFill(dpcom, mLatestTimestampstrings[posInVector], mDpsstringsmap);
318  processStringDP(dpid);
319  }
320 
321  else if (type == DeliveryType::RAW_TIME) {
322  checkFlagsAndFill(dpcom, mLatestTimestamptimes[posInVector], mDpstimesmap);
323  processTimeDP(dpid);
324  }
325 
326  else if (type == DeliveryType::RAW_BINARY) {
327  checkFlagsAndFill(dpcom, mLatestTimestampbinaries[posInVector], mDpsbinariesmap);
328  processBinaryDP(dpid);
329  }
330 
331  return 0;
332 }

336 template <>
337 void DCSProcessor::checkFlagsAndFill(const std::pair<DPID, DPVAL>& dpcom, uint64_t& latestTimeStamp,
338  std::unordered_map<DPID, DQStrings>& destmap)
339 {
340 
341  // check the flags for the upcoming data, and if ok, fill the accumulator
342 
343  auto& dpid = dpcom.first;
344  auto& val = dpcom.second;
345  auto flags = val.get_flags();
346  if (processFlag(flags, dpid.get_alias()) == 0) {
347  auto etime = val.get_epoch_time();
348  // fill only if new value has a timestamp different from the timestamp of the previous one
349  LOG(DEBUG) << "destmap[pid].size() = " << destmap[dpid].size();
350  if (destmap[dpid].size() == 0 || etime != latestTimeStamp) {
351  auto& tmp = destmap[dpid].emplace_back();
352  std::strncpy(tmp.data(), (char*)&(val.payload_pt1), 56);
353  latestTimeStamp = etime;
354  }
355  }
356 }
