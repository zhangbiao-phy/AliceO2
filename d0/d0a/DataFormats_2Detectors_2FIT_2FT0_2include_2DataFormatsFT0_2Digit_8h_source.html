33 struct Triggers {
34  enum { bitA,
35  bitC,
36  bitVertex,
37  bitCen,
38  bitSCen };
39  uint8_t triggersignals = 0; // T0 trigger signals
40  int8_t nChanA = 0; // number of fired channels A side
41  int8_t nChanC = 0; // number of fired channels A side
42  int32_t amplA = -1000; // sum amplitude A side
43  int32_t amplC = -1000; // sum amplitude C side
44  int16_t timeA = -1000; // average time A side
45  int16_t timeC = -1000; // average time C side
46  uint8_t eventFlags = 0; // event conditions
47  Triggers() = default;
48  Triggers(uint8_t signals, int8_t chanA, int8_t chanC, int32_t aamplA, int32_t aamplC, int16_t atimeA, int16_t atimeC)
49  {
50  triggersignals = signals;
51  nChanA = chanA;
52  nChanC = chanC;
53  amplA = aamplA;
54  amplC = aamplC;
55  timeA = atimeA;
56  timeC = atimeC;
57  }
58  bool getOrA() const { return (triggersignals & (1 << bitA)) != 0; }
59  bool getOrC() const { return (triggersignals & (1 << bitC)) != 0; }
60  bool getVertex() const { return (triggersignals & (1 << bitVertex)) != 0; }
61  bool getCen() const { return (triggersignals & (1 << bitCen)) != 0; }
62  bool getSCen() const { return (triggersignals & (1 << bitSCen)) != 0; }

64  void setTriggers(Bool_t isA, Bool_t isC, Bool_t isVrtx, Bool_t isCnt, Bool_t isSCnt, int8_t chanA, int8_t chanC, int32_t aamplA,
65  int32_t aamplC, int16_t atimeA, int16_t atimeC)
66  {
67  triggersignals = (isA << bitA) | (isC << bitC) | (isVrtx << bitVertex) | (isCnt << bitCen) | (isSCnt << bitSCen);
68  nChanA = chanA;
69  nChanC = chanC;
70  amplA = aamplA;
71  amplC = aamplC;
72  timeA = atimeA;
73  timeC = atimeC;
74  }
75  void cleanTriggers()
76  {
77  triggersignals = 0;
78  nChanA = nChanC = 0;
79  amplA = amplC = -1000;
80  timeA = timeC = -1000;
81  }
