56 void Trap2CRU::readTrapData(const std::string& outputDir, const std::string& inputFilename, int superPageSizeInB)
57 {
58  //set things up, read the file and then deligate to convertTrapdata to do the conversion.
59  //
60  mRawData.reserve(1024 * 1024); //TODO take out the hardcoded 1MB its supposed to come in from the options
61  LOG(debug) << "Trap2CRU::readTrapData";
62  // data comes in index by event (triggerrecord) and link (linke record) both sequentially.
63  // first 15 links go to cru0a, second 15 links go to cru0b, 3rd 15 links go to cru1a ... first 90 links to flp0 and then repate for 12 flp
64  // then do next event
65 
66  // lets register our links
67  std::string prefix = outputDir;
68  if (!prefix.empty() && prefix.back() != '/') {
69  prefix += '/';
70  }
71 
72  for (int link = 0; link < NumberOfHalfCRU; link++) {
73  // FeeID *was* 0xFEED, now is indicates the cru Supermodule, side (A/C) and endpoint. See RawData.cxx for details.
74  int supermodule = link / 4;
75  int endpoint = link / 2;
76  int side = link % 2 ? 1 : 0;
77  mFeeID = buildTRDFeeID(supermodule, side, endpoint);
78  mCruID = link / 2;
79  mEndPointID = link % 2 ? 1 : 0; //TODO figure out a value ... endpoint needs a rebase to PR4106
80  mLinkID = TRDLinkID;
81  std::string trdside = link % 2 ? "c" : "a"; // the side of supermodule readout A or C, odd numbered CRU are A, even numbered CRU are C.
82  // filenmae structure of trd_cru_[CRU#]_[upper/lower].raw
83  std::string outputFilelink = o2::utils::concat_string(prefix, "trd_cru_", std::to_string(mCruID), "_", trdside, ".raw");
84  mWriter.registerLink(mFeeID, mCruID, mLinkID, mEndPointID, outputFilelink);
85  }
86  mTrapRawFile = TFile::Open(inputFilename.data());
87  assert(mTrapRawFile != nullptr);
88  LOG(debug) << "Trap Raw file open " << inputFilename;
89  mTrapRawTree = (TTree*)mTrapRawFile->Get("o2sim");
90 
91  mTrapRawTree->SetBranchAddress("TrapLinkRecord", &mLinkRecordsPtr); // branch with the link records
92  mTrapRawTree->SetBranchAddress("RawTriggerRecord", &mTriggerRecordsPtr); // branch with the trigger records
93  mTrapRawTree->SetBranchAddress("TrapRaw", &mTrapRawDataPtr); // branch with the actual incoming data.
94 
95  for (int entry = 0; entry < mTrapRawTree->GetEntries(); entry++) {
96  mTrapRawTree->GetEntry(entry);
97  LOG(debug) << "Before Trigger Reord loop Event starts at:" << mTriggerRecords[0].getFirstEntry() << " and has " << mTriggerRecords[0].getNumberOfObjects() << " entries";
98  uint32_t linkcount = 0;
99  for (auto trigger : mTriggerRecords) {
100  //get the event limits from TriggerRecord;
101  uint32_t eventstart = trigger.getFirstEntry();
102  uint32_t eventend = trigger.getFirstEntry() + trigger.getNumberOfObjects();
103  LOG(debug) << "Event starts at:" << eventstart << " and ends at :" << eventend;
104  convertTrapData(trigger);
105  }
106  }
107 }

128 void Trap2CRU::linkSizePadding(uint32_t linksize, uint32_t& crudatasize, uint32_t& padding)
129 {
130  // all data must be 256 bit aligned (8x64bit).
131  // if zero the whole 256 bit must be padded (empty link)
132  // crudatasize is the size to be stored in the cruheader, i.e. units of 256bits.
133  // linksize is the incoming link size from the linkrecord,
134  // padding is of course the amount of padding in 32bit words.
135  uint32_t rem = 0;
136  if (linksize != 0) {
137  //data, so figure out padding cru word, the other case is simple, full padding if size=0
138  rem = linksize % 8;
139  if (rem != 0) {
140  crudatasize = linksize / 8 + 1;
141  padding = 8 - rem;
142  } else {
143 
144  crudatasize = linksize / 8; // 32 bit word to 256 bit word.
145  padding = 0;
146  }
147  LOG(debug) << "We have data with linkdatasize=" << linksize << " with size number in header of:" << crudatasize << " padded with " << padding << " 32bit words";
148  } else {
149  //linksize is zero so no data, pad fully.
150  crudatasize = 1;
151  padding = 8;
152  LOG(debug) << "We have data with linkdatasize=" << linksize << " with size number in header of:" << crudatasize << " padded with " << padding << " 32bit words";
153  }
154  LOG(debug) << "linkSizePadding : CRUDATASIZE : " << crudatasize;
155 }

157 uint32_t Trap2CRU::buildCRUHeader(HalfCRUHeader& header, uint32_t bc, uint32_t halfcru, int startlinkrecord)
158 {
159  int bunchcrossing = bc;
160  int stopbits = 0x01; // do we care about this and eventtype in simulations?
161  int eventtype = 0x01;
162  int crurdhversion = 6;
163  int feeid = 0; //TODO must be in cruheader down the road, inside the reserved somewhere ...
164  int cruid = 0; //TODO """"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
165  uint32_t crudatasize = 0; //link size in units of 256 bits.
166  int endpoint = halfcru % 2 ? 1 : 0; //TODO figure out a value ... endpoint needs a rebase to PR4106
167  uint32_t padding = 0;
168  //setHalfCRUHeader(halfcruheader, rdhversion, bunchcrossing, stopbits, endpoint, eventtype); //TODO come back and pull this from somewhere.
169  setHalfCRUHeader(header, crurdhversion, bunchcrossing, stopbits, endpoint, eventtype, feeid, cruid); //TODO come back and pull this from somewhere.
170  // memset(&tmpLinkInfo[0],0,sizeof(tmpLinkInfo[0])*tmpLinkInfo.size());
171 
172  // halfcruheader from the relevant mLinkRecords.
173  int linkrecord = startlinkrecord;
174  int totallinkdatasize = 0; //in units of 256bits
175  for (int link = 0; link < NLinksPerHalfCRU; link++) {
176  int linkid = link + halfcru * NLinksPerHalfCRU; // TODO this might have to change to a lut I dont think the mapping is linear.
177  int errors = 0;
178  int linksize = 0; // linkSizePadding will convert it to 1 for the no data case.
179  if (mLinkRecords[linkrecord].getLinkId() == linkid) {
180  linksize = mLinkRecords[linkrecord].getNumberOfObjects();
181  // this can be done differently by keeping a pointer to halfcruheader and setting it after reading it all in and going back per link to set the size.
182  LOG(debug3) << "setting CRU HEADER for halfcru : " << halfcru << "and link : " << link << " contents" << header << ":" << link << ":" << linksize << ":" << errors;
183  linkrecord++; // increment locally for walking through linkrecords.
184  }
185  linkSizePadding(linksize, crudatasize, padding);
186  setHalfCRUHeaderLinkData(header, link, crudatasize, errors); // write one padding block for empty links.
187  totallinkdatasize += crudatasize;
188  }
189  return totallinkdatasize;

192 void Trap2CRU::convertTrapData(o2::trd::TriggerRecord const& TrigRecord)
193 {
194 
195  //build a HalfCRUHeader for this event/cru/endpoint
196  //loop over cru's
197  // loop over all half chambers, thankfully they data is sorted.
198  // check if current chamber has a link
199  // if not blank, else fill in data from link records
200  // dump data to rawwriter
201  //finished for event. this method is only called per event.
202  int currentlinkrecord = 0;
203  char* traprawdataptr = (char*)&mTrapRawData[0];
204  for (int halfcru = 0; halfcru < NumberOfHalfCRU; halfcru++) {
205  int supermodule = halfcru / 4;
206  int endpoint = halfcru / 2;
207  int side = halfcru % 2 ? 1 : 0;
208  mFeeID = buildTRDFeeID(supermodule, side, endpoint);
209  mCruID = halfcru / 2;
210  mLinkID = TRDLinkID;
211  mEndPointID = halfcru % 2 ? 1 : 0;
212 
213  memset(&mRawData[0], 0, sizeof(mRawData[0]) * mRawData.size()); // zero the rawdata storage
214  int numberofdetectors = o2::trd::constants::MAXCHAMBER;
215  HalfCRUHeader halfcruheader;
216  //now write the cruheader at the head of all the data for this halfcru.
217  LOG(debug) << "cru before building cruheader for halfcru index : " << halfcru << " with contents \n"
218  << halfcruheader;
219  uint32_t totalhalfcrudatasize = buildCRUHeader(halfcruheader, TrigRecord.getBCData().bc, halfcru, currentlinkrecord);
220 
221  std::vector<char> rawdatavector(totalhalfcrudatasize * 32 + sizeof(halfcruheader)); // sum of link sizes + padding in units of bytes and some space for the header (512 bytes).
222  char* rawdataptr = rawdatavector.data();
223 
224  //dumpHalfCRUHeader(halfcruheader);
225  memcpy(rawdataptr, (char*)&halfcruheader, sizeof(halfcruheader));
226  std::array<uint64_t, 8> raw{};
227  memcpy((char*)&raw[0], rawdataptr, sizeof(halfcruheader));
228  for (int i = 0; i < 2; i++) {
229  int index = 4 * i;
230  LOG(debug) << "[1/2rawdaptr " << i << " " << std::hex << raw[index + 3] << " " << raw[index + 2] << " " << raw[index + 1] << " " << raw[index + 0];
231  }
232  rawdataptr += sizeof(halfcruheader);
233 
234  int linkdatasize = 0; // in 32 bit words
235  int link = halfcru / 2;
236  for (int halfcrulink = 0; halfcrulink < NLinksPerHalfCRU; halfcrulink++) {
237  //links run from 0 to 14, so linkid offset is halfcru*15;
238  int linkid = halfcrulink + halfcru * NLinksPerHalfCRU;
239  LOG(debug) << "Currently checking for data on linkid : " << linkid << " from halfcru=" << halfcru << " and halfcrulink:" << halfcrulink << " ?? " << linkid << "==" << mLinkRecords[currentlinkrecord].getLinkId();
240  int errors = 0; // put no errors in for now.
241  int size = 0; // in 32 bit words
242  int datastart = 0; // in 32 bit words
243  int dataend = 0; // in 32 bit words
244  uint32_t paddingsize = 0; // in 32 bit words
245  uint32_t crudatasize = 0; // in 256 bit words.
246  if (mLinkRecords[currentlinkrecord].getLinkId() == linkid) {
247  //this link has data in the stream.
248  LOG(debug) << "+++ We have data on linkid = " << linkid << " halfcrulink : " << halfcrulink;
249  linkdatasize = mLinkRecords[currentlinkrecord].getNumberOfObjects();
250  datastart = mLinkRecords[currentlinkrecord].getFirstEntry();
251  dataend = datastart + size;
252  LOG(debug) << "We have data on linkid = " << linkid << " and linksize : " << linkdatasize << " so :" << linkdatasize / 8 << " 256 bit words";
253  currentlinkrecord++;
254  } else {
255  assert(mLinkRecords[currentlinkrecord].getLinkId() < linkid);
256  LOG(debug) << "---We do not have data on linkid = " << linkid << " halfcrulink : " << halfcrulink;
257  //blank data for this link
258  // put in a 1 256 bit word of data for the link and padd with 0xeeee x 8
259  linkdatasize = 0;
260  paddingsize = 8;
261  }
262  // now copy data to rawdata, padding as and where needed.
263  //
264  linkSizePadding(linkdatasize, crudatasize, paddingsize); //TODO this can come out as we have already called it, but previously we have lost the #padding words, solve to remove.
265 
266  LOG(debug) << "WRITING " << crudatasize << " 256 bit data words to output stream";
267  LOG(debug) << "setting CRY HEADER for " << halfcruheader << ":" << halfcrulink << ":" << crudatasize << ":" << errors;
268  // now pad ....
269  LOG(debug) << " now to pump data into the stream with : " << linkdatasize << " crudatasize:" << crudatasize << " paddingsize: " << paddingsize << " and rem:" << linkdatasize % 8;
270  char* olddataptr = rawdataptr; // store the old pointer so we can do some sanity checks for how far we advance.
271  //linkdatasize is the #of 32 bit words coming from the incoming tree.
272  //paddingsize is the number of padding words to add 0xeeee
273  uint32_t bytestocopy = linkdatasize * (sizeof(uint32_t));
274  LOG(debug) << "copying " << bytestocopy << " bytes of link tracklet data at pos:" << std::hex << static_cast<void*>(rawdataptr);
275  memcpy(rawdataptr, traprawdataptr, bytestocopy);
276  //increment pointer
277  rawdataptr += bytestocopy;
278  traprawdataptr += bytestocopy;
279  //now for padding
280  uint16_t padbytes = paddingsize * sizeof(uint32_t);
281  LOG(debug) << "writing " << padbytes << " bytes of padding data at pos:" << std::hex << static_cast<void*>(rawdataptr);
282  memset(rawdataptr, 0xee, padbytes);
283  //increment pointer.
284  rawdataptr += padbytes;
285  if (padbytes + bytestocopy != crudatasize * 32) {
286  LOG(debug) << "something wrong with data size writing padbytes:" << padbytes << " bytestocopy : " << bytestocopy << " crudatasize:" << crudatasize;
287  }
288  LOG(debug3) << std::hex << " rawdataptr:" << static_cast<void*>(rawdataptr) << " traprawdataptr " << static_cast<void*>(traprawdataptr);
289  //sanity check for now:
290  if (((char*)rawdataptr - (char*)olddataptr) != crudatasize * 32) { // cru words are 8 uint32 and comparison is in bytes.
291  LOG(debug) << "according to pointer arithmatic we have added " << rawdataptr - olddataptr << "bytes from " << static_cast<void*>(rawdataptr) << "-" << static_cast<void*>(olddataptr) << " when we should have added " << crudatasize * 8 * 4 << " because crudatasize=" << crudatasize;
292  }
293  if (crudatasize != o2::trd::getlinkdatasize(halfcruheader, halfcrulink)) {
294  // we have written the wrong amount of data ....
295  LOG(debug) << "crudata is ! = get link data size " << crudatasize << "!=" << o2::trd::getlinkdatasize(halfcruheader, halfcrulink);
296  }
297  LOG(debug) << "copied " << crudatasize * 32 << "bytes to halfcrurawdata which now has size of " << rawdatavector.size() << " for " << link << ":" << endpoint;
298  }
299  LOG(debug) << "writing to " << std::hex << mFeeID << std::dec << " : " << mCruID << " : " << mLinkID << " : " << mEndPointID;
300  mWriter.addData(mFeeID, mCruID, mLinkID, mEndPointID, TrigRecord.getBCData(), rawdatavector);
301  if (DebugDataWriting) {
302  std::ofstream out2("crutestdumprawdatavector");
303  out2.write(rawdatavector.data(), rawdatavector.size());
304  out2.flush();
305  halfcru = NumberOfHalfCRU; // exit loop after 1 half cru for now.
306  }
307  }
308 }
