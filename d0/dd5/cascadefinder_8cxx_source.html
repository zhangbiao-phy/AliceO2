11 // Cascade Finder task
12 // ===================
13 //
14 // This code loops over existing V0s and bachelor tracks and finds
15 // valid cascade candidates from scratch using a certain set of
16 // minimum (configurable) selection criteria.
17 //
18 // It is different than the producer: the producer merely
19 // loops over an *existing* list of cascades (V0+bachelor track
20 // indices) and calculates the corresponding full cascade information
21 //
22 // In both cases, any analysis should loop over the "CascData"
23 // table as that table contains all information.
24 //
25 // Comments, questions, complaints, suggestions?
26 // Please write to:
27 // david.dobrigkeit.chinellato@cern.ch
28 //

95 struct cascadeprefilter {
96  Configurable<float> dcabachtopv{"dcabachtopv", .1, "DCA Bach To PV"};
97  Configurable<int> mincrossedrows{"mincrossedrows", 70, "min crossed rows"};
98  Configurable<float> dcav0topv{"dcav0topv", .1, "DCA V0 To PV"};
99  Configurable<double> cospaV0{"cospaV0", .98, "CosPA V0"};
100  Configurable<double> v0radius{"v0radius", 0.9, "v0radius"};
101  Configurable<float> lambdamasswindow{"lambdamasswindow", .006, "Distance from Lambda mass"};
102  Configurable<float> dcav0dau{"dcav0dau", .6, "DCA V0 Daughters"};
103  Configurable<float> dcanegtopv{"dcanegtopv", .1, "DCA Neg To PV"};
104  Configurable<float> dcapostopv{"dcapostopv", .1, "DCA Pos To PV"};
105 
106  Produces<aod::CascGoodLambdas> cascGoodLambdas;
107  Produces<aod::CascGoodAntiLambdas> cascGoodAntiLambdas;
108  Produces<aod::CascGoodPosTracks> cascGoodPosTracks;
109  Produces<aod::CascGoodNegTracks> cascGoodNegTracks;
110 
111  Partition<soa::Join<aod::FullTracks, aod::TracksExtended>> goodPosTracks = aod::track::signed1Pt > 0.0f && aod::track::dcaXY > dcabachtopv;
112  Partition<soa::Join<aod::FullTracks, aod::TracksExtended>> goodNegTracks = aod::track::signed1Pt < 0.0f && aod::track::dcaXY < -dcabachtopv;
113 
114  Partition<aod::V0Datas> goodV0s = aod::v0data::dcapostopv > dcapostopv&& aod::v0data::dcanegtopv > dcanegtopv&& aod::v0data::dcaV0daughters < dcav0dau;
115 
116  using FullTracksExt = soa::Join<aod::FullTracks, aod::TracksExtended>;
117 
118  void process(aod::Collision const& collision,
119  FullTracksExt const& tracks,
120  aod::V0Datas const& V0s)
121  {
122  for (auto& t0 : goodPosTracks) {
123  if (!(t0.trackType() & o2::aod::track::TPCrefit)) {
124  continue; //TPC refit
125  }
126  if (t0.tpcNClsCrossedRows() < mincrossedrows) {
127  continue;
128  }
129  cascGoodPosTracks(t0.globalIndex(), t0.collisionId(), t0.dcaXY());
130  }
131  for (auto& t0 : goodNegTracks) {
132  if (!(t0.trackType() & o2::aod::track::TPCrefit)) {
133  continue; //TPC refit
134  }
135  if (t0.tpcNClsCrossedRows() < mincrossedrows) {
136  continue;
137  }
138  cascGoodNegTracks(t0.globalIndex(), t0.collisionId(), -t0.dcaXY());
139  }
140  for (auto& v0 : goodV0s) {
141  if (v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()) < cospaV0) {
142  continue;
143  }
144  if (v0.dcav0topv(collision.posX(), collision.posY(), collision.posZ()) < dcav0topv) {
145  continue;
146  }
147  if (v0.dcaV0daughters() > dcav0dau) {
148  continue;
149  }
150  if (v0.v0radius() < v0radius) {
151  continue;
152  }
153 
154  if (fabs(v0.mLambda() - 1.116) < lambdamasswindow) {
155  cascGoodLambdas(v0.globalIndex(), v0.collisionId());
156  }
157  if (fabs(v0.mAntiLambda() - 1.116) < lambdamasswindow) {
158  cascGoodAntiLambdas(v0.globalIndex(), v0.collisionId());
159  }
160  }
161  }
162 };

167  OutputObj<TH1F> hCandPerEvent{TH1F("hCandPerEvent", "", 100, 0, 100)};

169  //Configurables
170  Configurable<double> d_bz{"d_bz", +5.0, "bz field"};
171  Configurable<double> d_UseAbsDCA{"d_UseAbsDCA", kTRUE, "Use Abs DCAs"};
172 
173  //Selection criteria
174  Configurable<double> v0cospa{"casccospa", 0.998, "Casc CosPA"}; //double -> N.B. dcos(x)/dx = 0 at x=0)
175  Configurable<float> dcav0dau{"dcacascdau", 1.0, "DCA Casc Daughters"};
176  Configurable<float> v0radius{"cascradius", 1.0, "cascradius"};

178  //Process: subscribes to a lot of things!
179  void process(aod::Collision const& collision,
180  aod::FullTracks const& tracks,
181  aod::V0Datas const& V0s,
182  aod::CascGoodLambdas const& lambdas,
183  aod::CascGoodAntiLambdas const& antiLambdas,
184  aod::CascGoodPosTracks const& pBachtracks,
185  aod::CascGoodNegTracks const& nBachtracks)
186  {
187  //Define o2 fitter, 2-prong
188  o2::vertexing::DCAFitterN<2> fitterV0, fitterCasc;
189  fitterV0.setBz(d_bz);
190  fitterV0.setPropagateToPCA(true);
191  fitterV0.setMaxR(200.);
192  fitterV0.setMinParamChange(1e-3);
193  fitterV0.setMinRelChi2Change(0.9);
194  fitterV0.setMaxDZIni(1e9);
195  fitterV0.setMaxChi2(1e9);
196  fitterV0.setUseAbsDCA(d_UseAbsDCA);
197 
198  fitterCasc.setBz(d_bz);
199  fitterCasc.setPropagateToPCA(true);
200  fitterCasc.setMaxR(200.);
201  fitterCasc.setMinParamChange(1e-3);
202  fitterCasc.setMinRelChi2Change(0.9);
203  fitterCasc.setMaxDZIni(1e9);
204  fitterCasc.setMaxChi2(1e9);
205  fitterCasc.setUseAbsDCA(d_UseAbsDCA);
206 
207  Long_t lNCand = 0;
208 
209  std::array<float, 3> pVtx = {collision.posX(), collision.posY(), collision.posZ()};
210  std::array<float, 3> pos = {0.};
211  std::array<float, 3> posXi = {0.};
212  std::array<float, 3> pvecpos = {0.};
213  std::array<float, 3> pvecneg = {0.};
214  std::array<float, 3> pvecbach = {0.};
215 
216  //Cascades first
217  for (auto& v0id : lambdas) {
218  //required: de-reference the tracks for cascade building
219  auto v0 = v0id.goodLambda();
220  auto pTrack = getTrackParCov(v0.posTrack_as<aod::FullTracks>());
221  auto nTrack = getTrackParCov(v0.negTrack_as<aod::FullTracks>());
222  //Let's do the slow part first: the V0 recalculation from scratch
223  int nCand = fitterV0.process(pTrack, nTrack);
224  if (nCand != 0) {
225  fitterV0.propagateTracksToVertex();
226  const auto& v0vtx = fitterV0.getPCACandidate();
227  for (int i = 0; i < 3; i++) {
228  pos[i] = v0vtx[i];
229  }
230 
231  std::array<float, 21> cov0 = {0};
232  std::array<float, 21> cov1 = {0};
233  std::array<float, 21> covV0 = {0};
234 
235  //Covariance matrix calculation
236  const int momInd[6] = {9, 13, 14, 18, 19, 20}; // cov matrix elements for momentum component
237  fitterV0.getTrack(0).getPxPyPzGlo(pvecpos);
238  fitterV0.getTrack(1).getPxPyPzGlo(pvecneg);
239  fitterV0.getTrack(0).getCovXYZPxPyPzGlo(cov0);
240  fitterV0.getTrack(1).getCovXYZPxPyPzGlo(cov1);
241  for (int i = 0; i < 6; i++) {
242  int j = momInd[i];
243  covV0[j] = cov0[j] + cov1[j];
244  }
245  auto covVtxV0 = fitterV0.calcPCACovMatrix();
246  covV0[0] = covVtxV0(0, 0);
247  covV0[1] = covVtxV0(1, 0);
248  covV0[2] = covVtxV0(1, 1);
249  covV0[3] = covVtxV0(2, 0);
250  covV0[4] = covVtxV0(2, 1);
251  covV0[5] = covVtxV0(2, 2);
252 
253  const std::array<float, 3> vertex = {(float)v0vtx[0], (float)v0vtx[1], (float)v0vtx[2]};
254  const std::array<float, 3> momentum = {pvecpos[0] + pvecneg[0], pvecpos[1] + pvecneg[1], pvecpos[2] + pvecneg[2]};
255 
256  auto tV0 = o2::track::TrackParCov(vertex, momentum, covV0, 0);
257  tV0.setQ2Pt(0); //No bending, please
258 
259  for (auto& t0id : nBachtracks) {
260  auto t0 = t0id.goodNegTrack_as<aod::FullTracks>();
261  auto bTrack = getTrackParCov(t0);
262 
263  int nCand2 = fitterCasc.process(tV0, bTrack);
264  if (nCand2 != 0) {
265  fitterCasc.propagateTracksToVertex();
266  const auto& cascvtx = fitterCasc.getPCACandidate();
267  for (int i = 0; i < 3; i++) {
268  posXi[i] = cascvtx[i];
269  }
270  fitterCasc.getTrack(1).getPxPyPzGlo(pvecbach);
271 
272  lNCand++;
273  //If we got here, it means this is a good candidate!
274  cascdata(v0.globalIndex(), v0.posTrack().globalIndex(), v0.negTrack().collisionId(),
275  -1, posXi[0], posXi[1], posXi[2], pos[0], pos[1], pos[2],
276  pvecpos[0], pvecpos[1], pvecpos[2],
277  pvecneg[0], pvecneg[1], pvecneg[2],
278  pvecbach[0], pvecbach[1], pvecbach[2],
279  fitterV0.getChi2AtPCACandidate(), fitterCasc.getChi2AtPCACandidate(),
280  v0.dcapostopv(),
281  v0.dcanegtopv(),
282  t0id.dcaXY());
283  } //end if cascade recoed
284  } //end loop over bachelor
285  } //end if v0 recoed
286  } //end loop over cascades
287 
288  //Anticascades
289  for (auto& v0id : antiLambdas) {
290  //required: de-reference the tracks for cascade building
291  auto v0 = v0id.goodAntiLambda();
292  auto pTrack = getTrackParCov(v0.posTrack_as<aod::FullTracks>());
293  auto nTrack = getTrackParCov(v0.negTrack_as<aod::FullTracks>());
294  //Let's do the slow part first: the V0 recalculation from scratch
295  int nCand = fitterV0.process(pTrack, nTrack);
296  if (nCand != 0) {
297  fitterV0.propagateTracksToVertex();
298  const auto& v0vtx = fitterV0.getPCACandidate();
299  for (int i = 0; i < 3; i++) {
300  pos[i] = v0vtx[i];
301  }
302 
303  std::array<float, 21> cov0 = {0};
304  std::array<float, 21> cov1 = {0};
305  std::array<float, 21> covV0 = {0};
306 
307  //Covariance matrix calculation
308  const int momInd[6] = {9, 13, 14, 18, 19, 20}; // cov matrix elements for momentum component
309  fitterV0.getTrack(0).getPxPyPzGlo(pvecpos);
310  fitterV0.getTrack(1).getPxPyPzGlo(pvecneg);
311  fitterV0.getTrack(0).getCovXYZPxPyPzGlo(cov0);
312  fitterV0.getTrack(1).getCovXYZPxPyPzGlo(cov1);
313  for (int i = 0; i < 6; i++) {
314  int j = momInd[i];
315  covV0[j] = cov0[j] + cov1[j];
316  }
317  auto covVtxV0 = fitterV0.calcPCACovMatrix();
318  covV0[0] = covVtxV0(0, 0);
319  covV0[1] = covVtxV0(1, 0);
320  covV0[2] = covVtxV0(1, 1);
321  covV0[3] = covVtxV0(2, 0);
322  covV0[4] = covVtxV0(2, 1);
323  covV0[5] = covVtxV0(2, 2);
324 
325  const std::array<float, 3> vertex = {(float)v0vtx[0], (float)v0vtx[1], (float)v0vtx[2]};
326  const std::array<float, 3> momentum = {pvecpos[0] + pvecneg[0], pvecpos[1] + pvecneg[1], pvecpos[2] + pvecneg[2]};
327 
328  auto tV0 = o2::track::TrackParCov(vertex, momentum, covV0, 0);
329  tV0.setQ2Pt(0); //No bending, please
330 
331  for (auto& t0id : pBachtracks) {
332  auto t0 = t0id.goodPosTrack_as<aod::FullTracks>();
333  auto bTrack = getTrackParCov(t0);
334 
335  int nCand2 = fitterCasc.process(tV0, bTrack);
336  if (nCand2 != 0) {
337  fitterCasc.propagateTracksToVertex();
338  const auto& cascvtx = fitterCasc.getPCACandidate();
339  for (int i = 0; i < 3; i++) {
340  posXi[i] = cascvtx[i];
341  }
342  fitterCasc.getTrack(1).getPxPyPzGlo(pvecbach);
343 
344  lNCand++;
345  //If we got here, it means this is a good candidate!
346  cascdata(v0.globalIndex(), v0.posTrack().globalIndex(), v0.negTrack().collisionId(),
347  +1, posXi[0], posXi[1], posXi[2], pos[0], pos[1], pos[2],

353  v0.dcanegtopv(),

355  } //end if cascade recoed

360  hCandPerEvent->Fill(lNCand);

390  void process(soa::Join<aod::Collisions, aod::EvSels, aod::Cents>::iterator const& collision, soa::Filtered<aod::CascDataExt> const& Cascades)
391  {
392  if (!collision.alias()[kINT7]) {
393  return;
394  }
395  if (!collision.sel7()) {
396  return;
397  }
398  for (auto& casc : Cascades) {
399  //FIXME: dynamic columns cannot be filtered on?
400  if (casc.v0radius() > v0radius &&
401  casc.cascradius() > cascradius &&
402  casc.v0cosPA(collision.posX(), collision.posY(), collision.posZ()) > v0cospa &&
403  casc.casccosPA(collision.posX(), collision.posY(), collision.posZ()) > casccospa &&
404  casc.dcav0topv(collision.posX(), collision.posY(), collision.posZ()) > dcav0topv) {
405  if (casc.charge() < 0) { //FIXME: could be done better...
406  if (TMath::Abs(casc.yXi()) < 0.5) {
407  h3dMassXiMinus->Fill(collision.centV0M(), casc.pt(), casc.mXi());
408  }
409  if (TMath::Abs(casc.yOmega()) < 0.5) {
410  h3dMassOmegaMinus->Fill(collision.centV0M(), casc.pt(), casc.mOmega());
411  }
412  } else {
413  if (TMath::Abs(casc.yXi()) < 0.5) {
414  h3dMassXiPlus->Fill(collision.centV0M(), casc.pt(), casc.mXi());
415  }
416  if (TMath::Abs(casc.yOmega()) < 0.5) {
417  h3dMassOmegaPlus->Fill(collision.centV0M(), casc.pt(), casc.mOmega());
418  }
419  }
420  }
421  }
422  }
