34 template <typename T>
35 void makelogaxis(T h)
36 {
37  const int nbins = h->GetNbinsX();
38  double binp[nbins + 1];
39  double max = h->GetXaxis()->GetBinUpEdge(nbins);
40  double min = h->GetXaxis()->GetBinLowEdge(1);
41  if (min <= 0) {
42  min = 0.00001;
43  }
44  double lmin = TMath::Log10(min);
45  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
46  for (int i = 0; i < nbins; i++) {
47  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
48  }
49  binp[nbins] = max + 1;
50  h->GetXaxis()->Set(nbins, binp);
51 }
