51 // Grouping between MC particles and collisions
52 struct GeneratedTask {
53  OutputObj<TH2F> phiH{TH2F("phi", "phi;#phi;PDG code", 100, 0., 2. * M_PI, PDGBINNING)};
54  OutputObj<TH2F> etaH{TH2F("eta", "eta;#eta;PDG code", 102, -2.01, 2.01, PDGBINNING)};
55  OutputObj<TH2F> ptH{TH2F("pt", "pt;#it{p}_{T} (GeV/#it{c});PDG code", 500, 0, 20, PDGBINNING)};
56  OutputObj<TH2F> pH{TH2F("p", "p;#it{p} (GeV/#it{c});PDG code", 500, 0, 20, PDGBINNING)};
57  OutputObj<TH1F> pdgH{TH1F("pdg", "pdg;PDG code", PDGBINNING)};
58 
59  int events = 0;
60  int particles = 0;
61  int primaryparticles = 0;
62 
63  void process(aod::McCollision const& mcCollision, aod::McParticles& mcParticles)
64  {
65  LOGF(info, "MC. vtx-z = %f", mcCollision.posZ());
66  for (auto& mcParticle : mcParticles) {
67  if (abs(mcParticle.eta()) > 0.8) {
68  continue;
69  }
70  if (isPhysicalPrimary(mcParticles, mcParticle)) {
71  const auto pdg = Form("%i", mcParticle.pdgCode());
72  pdgH->Fill(pdg, 1);
73  const float pdgbin = pdgH->GetXaxis()->GetBinCenter(pdgH->GetXaxis()->FindBin(pdg));
74  phiH->Fill(mcParticle.phi(), pdgbin);
75  etaH->Fill(mcParticle.eta(), pdgbin);
76  pH->Fill(sqrt(mcParticle.px() * mcParticle.px() + mcParticle.py() * mcParticle.py() + mcParticle.pz() * mcParticle.pz()), pdgbin);
77  ptH->Fill(mcParticle.pt(), pdgbin);
78  primaryparticles++;
79  }
80  particles++;
81  }
82  LOGF(info, "Events %i", events++ + 1);
83  LOGF(info, "Particles %i", particles);
84  LOGF(info, "Primaries %i", primaryparticles);
85  }
86 };

98  OutputObj<TH2F> dcazsecH{TH2F("dcazsec", "dcazsec;DCA_{z} (cm)", 500, -10, 10, PDGBINNING)};
99  OutputObj<TH1F> pdgsecH{TH1F("pdgsec", "pdgsec;PDG code", PDGBINNING)};
100 
101  OutputObj<TH2F> phiDiff{TH2F("phiDiff", ";phi_{MC} - phi_{Rec}", 100, -M_PI, M_PI, PDGBINNING)};
102  OutputObj<TH2F> etaDiff{TH2F("etaDiff", ";eta_{MC} - eta_{Rec}", 100, -2, 2, PDGBINNING)};
103  OutputObj<TH2F> ptDiff{TH2F("ptDiff", "ptDiff;#it{p}_{T}_{MC} #it{p}_{T}_{Rec} (GeV/#it{c})", 500, -2, 2, PDGBINNING)};
104  OutputObj<TH2F> pDiff{TH2F("pDiff", "pDiff;#it{p}_{MC} #it{p}_{Rec} (GeV/#it{c})", 500, -2, 2, PDGBINNING)};
105 
106  Filter trackAcceptance = (nabs(aod::track::eta) < 0.8f);

109  void process(soa::Join<aod::Collisions, aod::McCollisionLabels>::iterator const& collision,
110  soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra, aod::TracksExtended, aod::McTrackLabels, aod::TrackSelection>> const& tracks,
111  aod::McParticles& mcParticles, aod::McCollisions const& mcCollisions)
112  {
113  LOGF(info, "vtx-z (data) = %f | vtx-z (MC) = %f", collision.posZ(), collision.mcCollision().posZ());
114  for (auto& track : tracks) {
115  const auto pdg = Form("%i", track.mcParticle().pdgCode());
116  if (!isPhysicalPrimary(mcParticles, track.mcParticle())) {
117  pdgsecH->Fill(pdg, 1);
118  const float pdgbinsec = pdgH->GetXaxis()->GetBinCenter(pdgsecH->GetXaxis()->FindBin(pdg));
119  dcaxysecH->Fill(track.dcaXY(), pdgbinsec);
120  dcazsecH->Fill(track.dcaZ(), pdgbinsec);
121  continue;
122  }
123  pdgH->Fill(pdg, 1); // Filling the first bin and check its bin
124  const float pdgbin = pdgH->GetXaxis()->GetBinCenter(pdgH->GetXaxis()->FindBin(pdg));
125  phiH->Fill(track.phi(), pdgbin);
126  etaH->Fill(track.eta(), pdgbin);
127  pH->Fill(track.p(), pdgbin);
128  ptH->Fill(track.pt(), pdgbin);
129  dcaxyH->Fill(track.dcaXY(), pdgbin);
130  dcazH->Fill(track.dcaZ(), pdgbin);
131 
132  etaDiff->Fill(track.mcParticle().eta() - track.eta(), pdgbin);
133  auto delta = track.mcParticle().phi() - track.phi();
134  if (delta > M_PI) {
135  delta -= 2 * M_PI;
136  }
137  if (delta < -M_PI) {
138  delta += 2 * M_PI;
139  }
140  phiDiff->Fill(delta, pdgbin);
141  pDiff->Fill(sqrt(track.mcParticle().px() * track.mcParticle().px() + track.mcParticle().py() * track.mcParticle().py() + track.mcParticle().pz() * track.mcParticle().pz()) - track.p(), pdgbin);
142  ptDiff->Fill(track.mcParticle().pt() - track.pt(), pdgbin);
143  }
144  }
145 };
