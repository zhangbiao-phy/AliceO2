23 CaloRawFitter::FitStatus CaloRawFitter::evaluate(const std::vector<Bunch>& bunchlist)
24 {
25 
26  mAmp.clear(); // used as mean in this mode
27  mTime.clear(); // used as RMS in pedestal mode
28  mChi2.clear(); // used as mean in this mode
29 
30  //Pedestal analysis mode
31  if (mPedestalRun) {
32  int nPed = 0;
33  float mean = 0.;
34  float rms = 0.;
35  for (auto b : bunchlist) {
36  const std::vector<uint16_t>& signal = b.getADC();
37  for (std::vector<uint16_t>::const_reverse_iterator it = signal.rbegin(); it != signal.rend(); ++it) {
38  uint16_t a = *it;
39  mean += a;
40  rms += a * a;
41  ++nPed;
42  }
43  }
44  if (nPed > 0) {
45  mean /= nPed;
46  rms = rms / nPed - mean * mean;
47  if (rms > 0.) {
48  rms = sqrt(rms);
49  }
50  }
51  mAmp.push_back(mean);
52  mTime.push_back(rms);
53  mOverflow.push_back(false);
54  mStatus = kOK;
55  return kOK;
56  }
57 
58  for (auto b : bunchlist) {
59  // Extract amplitude and time using maximum and k-level methods
60  FitStatus s = evalKLevel(b);
61  mStatus = s;
62 
63  // If this is Low Gain channel or explicitely requested, fit sample with Gamma2 function
64  if (makeFit || (mLowGain && s == kOverflow)) {
65  FitStatus s2 = fitGamma2(b);
66  }
67  //TODO: should we have separate status: overflow & fit OK, and overflow & fit failed, etc?
68  }
69  return mStatus;
70 }

72 CaloRawFitter::FitStatus CaloRawFitter::evalKLevel(const Bunch& b) //const ushort *signal, int sigStart, int sigLength)
73 {
74  // Calculate signal parameters (energy, time, quality) from array of samples
75  // Energy is a maximum sample minus pedestal 9
76  // Time is the first time bin
77  // Signal overflows is there are at least 3 samples of the same amplitude above 900
78 
79  float amp = 0.;
80  float time = 0.;
81  bool overflow = false;
82 
83  int sigLength = b.getBunchLength();
84  if (sigLength == 0) {
85  return kEmptyBunch;
86  }
87 
88  const short kSpikeThreshold = o2::phos::PHOSSimParams::Instance().mSpikeThreshold;
89  const short kBaseLine = o2::phos::PHOSSimParams::Instance().mBaseLine;
90  const short kPreSamples = o2::phos::PHOSSimParams::Instance().mPreSamples;
91 
92  float pedMean = 0;
93  float pedRMS = 0;
94  int nPed = 0;
95  mMaxSample = 0;
96  int nMax = 0; //number of consequitive maximal samples
97  bool spike = false;
98 
99  const std::vector<uint16_t>& signal = b.getADC();
100 
101  int ap = -1, app = -1; //remember previous values to evaluate spikes
102 
103  for (std::vector<uint16_t>::const_reverse_iterator it = signal.rbegin(); it != signal.rend(); ++it) {
104  uint16_t a = *it;
105  if (mPedSubtract) {
106  if (nPed < kPreSamples) { //inverse signal time order
107  nPed++;
108  pedMean += a;
109  pedRMS += a * a;
110  }
111  }
112  if (a > mMaxSample) {
113  mMaxSample = a;
114  nMax = 0;
115  }
116  if (a == mMaxSample) {
117  nMax++;
118  }
119  //check if there is a spike
120  if (app >= 0 && ap >= 0) {
121  spike |= (2 * ap - (a + app) > 2 * kSpikeThreshold);
122  }
123  app = ap;
124  ap = a;
125  }
126  amp = (float)mMaxSample;
127 
128  if (spike) {
129  mAmp.push_back(amp);
130  mTime.push_back(b.getStartTime() - 2);
131  mOverflow.push_back(false);
132  return kSpike;
133  }
134 
135  if (mMaxSample > 900 && nMax > 2) {
136  overflow = true;
137  }
138 
139  float pedestal = 0;
140  if (mPedSubtract) {
141  if (nPed > 0) {
142  pedRMS = (pedRMS - pedMean * pedMean / nPed) / nPed;
143  if (pedRMS > 0.) {
144  pedRMS = sqrt(pedRMS);
145  }
146  pedestal = pedMean / nPed;
147  } else {
148  mAmp.push_back(0);
149  mTime.push_back(b.getStartTime() - 2);
150  mOverflow.push_back(false);
151  return kBadPedestal;
152  }
153  }
154 
155  amp -= pedestal;
156  if (amp < kBaseLine) {
157  amp = 0;
158  }
159 
160  //Evaluate time
161  time = b.getStartTime() - 2;
162  const int nLine = 6; //Parameters of fitting
163  const float eMinTOF = 10.; //Choosed from beam-test and cosmic analyis
164  const float kAmp = 0.35; //Result slightly depends on them, so no getters
165  // Avoid too low peak:
166  if (amp < eMinTOF) {
167  mAmp.push_back(amp);
168  mTime.push_back(time);
169  mOverflow.push_back(false);
170  return kOK; //use estimated time
171  }
172 
173  // Find index posK (kLevel is a level of "timestamp" point Tk):
174  int posK = sigLength - 1; //last point before crossing k-level
175  float levelK = pedestal + kAmp * amp;
176  while (signal[posK] <= levelK && posK >= 0) {
177  posK--;
178  }
179  posK++;
180 
181  if (posK == 0 || posK == sigLength - 1) {
182  mAmp.push_back(amp);
183  mTime.push_back(time);
184  mOverflow.push_back(false);
185  return kNoTime; //
186  }
187 
188  // Find crossing point by solving linear equation (least squares method)
189  int np = 0;
190  int iup = posK - 1;
191  int idn = posK;
192  Double_t sx = 0., sy = 0., sxx = 0., sxy = 0.;
193  Double_t x, y;
194 
195  while (np < nLine) {
196  //point above crossing point
197  if (iup >= 0) {
198  x = sigLength - iup - 1;
199  y = signal[iup];
200  sx += x;
201  sy += y;
202  sxx += (x * x);
203  sxy += (x * y);
204  np++;
205  iup--;
206  }
207  //Point below crossing point
208  if (idn < sigLength) {
209  if (signal[idn] < pedestal) {
210  idn = sigLength - 1; //do not scan further
211  idn++;
212  continue;
213  }
214  x = sigLength - idn - 1;
215  y = signal[idn];
216  sx += x;
217  sy += y;
218  sxx += (x * x);
219  sxy += (x * y);
220  np++;
221  idn++;
222  }
223  if (idn >= sigLength && iup < 0) {
224  break; //can not fit futher
225  }
226  }
227 
228  Double_t det = np * sxx - sx * sx;
229  if (det == 0) {
230  mAmp.push_back(amp);
231  mTime.push_back(time);
232  mOverflow.push_back(false);
233  return kNoTime;
234  }
235  if (np == 0) {
236  mAmp.push_back(amp);
237  mTime.push_back(time);
238  mOverflow.push_back(false);
239  return kEmptyBunch;
240  }
241  Double_t c1 = (np * sxy - sx * sy) / det; //slope
242  Double_t c0 = (sy - c1 * sx) / np; //offset
243  if (c1 == 0) {
244  mAmp.push_back(amp);
245  mTime.push_back(time);
246  mOverflow.push_back(false);
247  return kNoTime;
248  }
249 
250  // Find where the line cross kLevel:
251  time += (levelK - c0) / c1 - 5.; //5: mean offset between k-Level and start times
252 
253  mAmp.push_back(amp);
254  mTime.push_back(time);
255  if (overflow) {
256  mOverflow.push_back(true);
257  return kOverflow;
258  } else {
259  mOverflow.push_back(false);
260  return kOK;
261  }
262 }

264 CaloRawFitter::FitStatus CaloRawFitter::fitGamma2(const Bunch& b)
265 {
266  // Fit bunch with gamma2 function
267  // TODO!!! validate method
268  //initial values
269  float A = mAmp.back();
270  float t0 = mTime.back();
271  const std::vector<uint16_t>& signal = b.getADC();
272  uint16_t tsart = b.getStartTime();
273 
274  const float alpha = 17.; //Decay time in units of 100ns //TODO!!! to be adjusted
275  const float kEpsilon = 1.e-6; //Accuracy of fit //TODO!!! to be adjusted
276  const int nIter = 10; //Maximal number of iterations //TODO!!! to be adjusted
277 
278  float chi2 = 0;
279  float derT = 0, derTprev = 0.;
280  float stepT = 0.1;
281  int iter = 0, i = 0;
282  do {
283  chi2 = 0.;
284  derTprev = derT;
285  derT = 0.;
286  i = 0;
287  float sA = 0., sB = 0.;
288  std::vector<uint16_t>::const_reverse_iterator it = signal.rbegin();
289  while (it != signal.rend()) {
290  uint16_t si = *it;
291  float ti = tsart + i - t0;
292  it++;
293  i++;
294  if (mOverflow.back() && si == mMaxSample) { //do not fit saturated samples
295  continue;
296  }
297  float fi = ti * ti * exp(-ti * alpha);
298  chi2 += (si - A * fi) * (si - A * fi);
299  sA += si * fi;
300  sB += fi * fi;
301  if (ti != 0.) {
302  derT += (2. / ti - alpha) * fi * (A * fi - si);
303  }
304  }
305  derT *= A;
306  //calculate time step and next time
307  if (derTprev != 0. && derT - derTprev != 0.) {
308  stepT = derT / (derT - derTprev) * stepT;
309  }
310  derTprev = derT;
311  t0 -= stepT;
312  if (sB > 0.) {
313  A = sA / sB;
314  }
315  } while (fabs(stepT) > kEpsilon && iter < nIter); // if time step is too small, stop
316 
317  if (iter >= nIter) { //Fit did not converge, keep old A and t0.
318  return kFitFailed;
319  }
320 
321  if (i > 0) { //chi2/NDF
322  chi2 /= i;
323  }
324 
325  mTime.back() = t0;
326  mAmp.back() = A;
327  return kOK;
328 }
