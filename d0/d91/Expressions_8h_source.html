61 template <typename T>
62 constexpr auto selectArrowType()
63 {
64  if constexpr (std::is_same_v<T, int>) {
65  return atype::INT32;
66  } else if constexpr (std::is_same_v<T, bool>) {
67  return atype::BOOL;
68  } else if constexpr (std::is_same_v<T, float>) {
69  return atype::FLOAT;
70  } else if constexpr (std::is_same_v<T, double>) {
71  return atype::DOUBLE;
72  } else if constexpr (std::is_same_v<T, uint8_t>) {
73  return atype::UINT8;
74  } else if constexpr (std::is_same_v<T, int8_t>) {
75  return atype::INT8;
76  } else if constexpr (std::is_same_v<T, uint16_t>) {
77  return atype::UINT16;
78  } else if constexpr (std::is_same_v<T, int16_t>) {
79  return atype::INT16;
80  } else if constexpr (std::is_same_v<T, int64_t>) {
81  return atype::INT64;
82  } else {
83  return atype::NA;
84  }
85  O2_BUILTIN_UNREACHABLE();
86 }

119 struct PlaceholderNode : LiteralNode {
120  template <typename T>
121  PlaceholderNode(Configurable<T> v) : LiteralNode{v.value}
122  {
123  if constexpr (variant_trait_v<typename std::decay<T>::type> != VariantType::Unknown) {
124  retrieve = [name = v.name](InitContext& context) { return LiteralNode::var_t{context.options().get<T>(name.c_str())}; };
125  } else {
126  retrieve = [name = v.name](InitContext& context) {
127  auto pt = context.options().get<boost::property_tree::ptree>(name.c_str());
128  return LiteralNode::var_t{RootConfigParamHelpers::as<T>(pt)};
129  };
130  }
131  }
132 
133  void reset(InitContext& context)
134  {
135  value = retrieve(context);
136  }
137 
138  std::function<LiteralNode::var_t(InitContext&)> retrieve;
139 };
