208 int FeeParam::getPadColFromADC(int irob, int imcm, int iadc)
209 {
210  //
211  // Return which pad is connected to this adc channel.
212  //
213  // Return virtual pad number even if ADC is outside chamber
214  // to keep compatibility of data processing at the edge MCM.
215  // User has to check that this is in the chamber if it is essential.
216  // Return -100 if iadc is invalid.
217  //
218  // Caution: ADC ordering in the online data is opposite to the pad column ordering.
219  // And it is not one-by-one correspondence. Precise drawing can be found in:
220  // http://wiki.kip.uni-heidelberg.de/ti/TRD/index.php/Image:ROB_MCM_numbering.pdf
221  //
222 
223  if (iadc < 0 || iadc > NADCMCM) {
224  return -100;
225  }
226  int mcmcol = imcm % NMCMROBINCOL + getRobSide(irob) * NMCMROBINCOL; // MCM column number on ROC [0..7]
227  int padcol = mcmcol * NCOLMCM + NCOLMCM + 1 - iadc;
228  if (padcol < 0 || padcol >= NCOLUMN) {
229  return -1; // this is commented because of reason above OK
230  }
231  return padcol;
232 }

357 int FeeParam::extAliToAli(unsigned int dest, unsigned short linkpair, unsigned short rocType, int* mcmList, int listSize)
358 {
359  // Converts an extended ALICE ID which identifies a single MCM or a group of MCMs to
360  // the corresponding list of MCMs. Only broadcasts (127) are encoded as 127
361  // The return value is the number of MCMs in the list
362 
363  mcmList[0] = -1;
364 
365  short nmcm = 0;
366  unsigned int mcm, rob, robAB;
367  unsigned int cmA = 0, cmB = 0; // Chipmask for each A and B side
368 
369  // Default chipmask for 4 linkpairs (each bit correponds each alice-mcm)
370  static const unsigned int gkChipmaskDefLp[4] = {0x1FFFF, 0x1FFFF, 0x3FFFF, 0x1FFFF};
371 
372  rob = dest >> 7; // Extract ROB pattern from dest.
373  mcm = dest & 0x07F; // Extract MCM pattern from dest.
374  robAB = getRobAB(rob, linkpair); // Get which ROB sides are selected.
375 
376  // Abort if no ROB is selected
377  if (robAB == 0) {
378  return 0;
379  }
380 
381  // Special case
382  if (mcm == 127) {
383  if (robAB == 3) { // This is very special 127 can stay only if two ROBs are selected
384  mcmList[0] = 127; // broadcase to ALL
385  mcmList[1] = -1;
386  return 1;
387  }
388  cmA = cmB = 0x3FFFF;
389  } else if ((mcm & 0x40) != 0) { // If top bit is 1 but not 127, this is chip group.
390  if ((mcm & 0x01) != 0) {
391  cmA |= 0x04444;
392  cmB |= 0x04444;
393  } // chip_cmrg
394  if ((mcm & 0x02) != 0) {
395  cmA |= 0x10000;
396  cmB |= 0x10000;
397  } // chip_bmrg
398  if ((mcm & 0x04) != 0 && rocType == 0) {
399  cmA |= 0x20000;
400  cmB |= 0x20000;
401  } // chip_hm3
402  if ((mcm & 0x08) != 0 && rocType == 1) {
403  cmA |= 0x20000;
404  cmB |= 0x20000;
405  } // chip_hm4
406  if ((mcm & 0x10) != 0) {
407  cmA |= 0x01111;
408  cmB |= 0x08888;
409  } // chip_edge
410  if ((mcm & 0x20) != 0) {
411  cmA |= 0x0aaaa;
412  cmB |= 0x03333;
413  } // chip_norm
414  } else { // Otherwise, this is normal chip ID, turn on only one chip.
415  cmA = 1 << mcm;
416  cmB = 1 << mcm;
417  }
418 
419  // Mask non-existing MCMs
420  cmA &= gkChipmaskDefLp[linkpair];
421  cmB &= gkChipmaskDefLp[linkpair];
422  // Remove if only one side is selected
423  if (robAB == 1) {
424  cmB = 0;
425  }
426  if (robAB == 2) {
427  cmA = 0;
428  }
429  if (robAB == 4 && linkpair != 2) {
430  cmA = cmB = 0; // Restrict to only T3A and T3B
431  }
432 
433  // Finally convert chipmask to list of slaves
434  nmcm = chipmaskToMCMlist(cmA, cmB, linkpair, mcmList, listSize);
435 
436  return nmcm;
437 }

439 short FeeParam::getRobAB(unsigned short robsel, unsigned short linkpair)
440 {
441  // Converts the ROB part of the extended ALICE ID to robs
442 
443  if ((robsel & 0x8) != 0) { // 1000 .. direct ROB selection. Only one of the 8 ROBs are used.
444  robsel = robsel & 7;
445  if ((robsel % 2) == 0 && (robsel / 2) == linkpair) {
446  return 1; // Even means A side (position 0,2,4,6)
447  }
448  if ((robsel % 2) == 1 && (robsel / 2) == linkpair) {
449  return 2; // Odd means B side (position 1,3,5,7)
450  }
451  return 0;
452  }
453 
454  // ROB group
455  if (robsel == 0) {
456  return 3;
457  } // Both ROB
458  if (robsel == 1) {
459  return 1;
460  } // A-side ROB
461  if (robsel == 2) {
462  return 2;
463  } // B-side ROB
464  if (robsel == 3) {
465  return 3;
466  } // Both ROB
467  if (robsel == 4) {
468  return 4;
469  } // Only T3A and T3B
470  // Other number 5 to 7 are ignored (not defined)
471 
472  return 0;
473 }

568 short FeeParam::chipmaskToMCMlist(unsigned int cmA, unsigned int cmB, unsigned short linkpair, int* mcmList, int listSize)
569 {
570  // Converts the chipmask to a list of MCMs
571 
572  short nmcm = 0;
573  short i;
574  for (i = 0; i < 18; i++) { // 18: number of MCMs on a ROB
575  if ((cmA & (1 << i)) != 0 && nmcm < listSize) {
576  mcmList[nmcm] = ((linkpair * 2) << 7) | i;
577  ++nmcm;
578  }
579  if ((cmB & (1 << i)) != 0 && nmcm < listSize) {
580  mcmList[nmcm] = ((linkpair * 2 + 1) << 7) | i;
581  ++nmcm;
582  }
583  }
584 
585  mcmList[nmcm] = -1;
586  return nmcm;
587 }

630 int FeeParam::getDyCorrection(int det, int rob, int mcm) const
631 {
632  // calculate the correction of the deflection
633  // i.e. Lorentz angle and tilt correction (if active)
634 
635  int layer = det % NLAYER;
636 
637  float dyTilt = (mgDriftLength * std::tan(mgTiltingAngle[layer] * M_PI / 180.) *
638  getLocalZ(det, rob, mcm) * mgInvX[layer]);
639 
640  // calculate Lorentz correction
641  float dyCorr = -mOmegaTau * mgDriftLength;
642 
643  if (mTiltCorr) {
644  dyCorr += dyTilt; // add tilt correction
645  }
646 
647  return (int)TMath::Nint(dyCorr * mgScalePad * mgInvWidthPad[layer]);
648 }

650 void FeeParam::getDyRange(int det, int rob, int mcm, int ch,
651  int& dyMinInt, int& dyMaxInt) const
652 {
653  // calculate the deflection range in which tracklets are accepted
654 
655  dyMinInt = mgDyMin;
656  dyMaxInt = mgDyMax;
657 
658  // deflection cut is considered for |B| > 0.1 T only
659  if (std::abs(mMagField) < 0.1) {
660  return;
661  }
662 
663  float e = 0.30;
664 
665  float maxDeflTemp = getPerp(det, rob, mcm, ch) / 2. * // Sekante/2 (cm)
666  (e * 1e-2 * std::abs(mMagField) * mInvPtMin); // 1/R (1/cm)
667 
668  float phi = getPhi(det, rob, mcm, ch);
669  if (maxDeflTemp < std::cos(phi)) {
670  float maxDeflAngle = std::asin(maxDeflTemp);
671 
672  float dyMin = (mgDriftLength *
673  std::tan(phi - maxDeflAngle));
674 
675  dyMinInt = int(dyMin / mgBinDy);
676  // clipping to allowed range
677  if (dyMinInt < mgDyMin) {
678  dyMinInt = mgDyMin;
679  } else if (dyMinInt > mgDyMax) {
680  dyMinInt = mgDyMax;
681  }
682 
683  float dyMax = (mgDriftLength *
684  std::tan(phi + maxDeflAngle));
685 
686  dyMaxInt = int(dyMax / mgBinDy);
687  // clipping to allowed range
688  if (dyMaxInt > mgDyMax) {
689  dyMaxInt = mgDyMax;
690  } else if (dyMaxInt < mgDyMin) {
691  dyMaxInt = mgDyMin;
692  }
693  } else if (maxDeflTemp < 0.) {
694  // this must not happen
695  printf("Inconsistent calculation of sin(alpha): %f\n", maxDeflTemp);
696  } else {
697  // TRD is not reached at the given pt threshold
698  // max range
699  }
700 
701  if ((dyMaxInt - dyMinInt) <= 0) {
702  LOG(debug) << "strange dy range: [" << dyMinInt << "," << dyMaxInt << "], using max range now";
703  dyMaxInt = mgDyMax;
704  dyMinInt = mgDyMin;
705  }
706 }

768 float FeeParam::getLocalZ(int det, int rob, int mcm) const
769 {
770  // get local z-position w.r.t. to the chamber boundary
771 
772  int stack = (det % NCHAMBERPERSEC) / NLAYER;
773  int layer = det % NLAYER;
774  int row = (rob / 2) * 4 + mcm / 4;
775 
776  if (stack == 2) {
777  if (row == 0) {
778  return (mgZrow[layer * NLAYER + stack] - 0.5 * mgLengthOuterPadC0);
779  } else if (row == 11) {
780  return (mgZrow[layer * NLAYER + stack] - 1.5 * mgLengthOuterPadC0 - (row - 1) * mgLengthInnerPadC0);
781  } else {
782  return (mgZrow[layer * NLAYER + stack] - mgLengthOuterPadC0 - (row - 0.5) * mgLengthInnerPadC0);
783  }
784  } else {
785  if (row == 0) {
786  return (mgZrow[layer * NLAYER + stack] - 0.5 * mgLengthOuterPadC1[layer]);
787  } else if (row == 15) {
788  return (mgZrow[layer * NLAYER + stack] - 1.5 * mgLengthOuterPadC1[layer] - (row - 1) * mgLengthInnerPadC1[layer]);
789  } else {
790  return (mgZrow[layer * NLAYER + stack] - mgLengthOuterPadC1[layer] - (row - 0.5) * mgLengthInnerPadC1[layer]);
791  }
792  }
793 }
