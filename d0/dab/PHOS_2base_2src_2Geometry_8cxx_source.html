45 short Geometry::relToAbsId(char moduleNumber, int strip, int cell)
46 {
47  // calculates absolute cell Id from moduleNumber, strip (number) and cell (number)
48  // PHOS layout parameters:
49  const short nStrpZ = 28; // Number of strips along z-axis
50  const short nCrystalsInModule = 56 * 64; // Total number of crystals in module
51  const short nCellsXInStrip = 8; // Number of crystals in strip unit along x-axis
52  const short nZ = 56; // nStripZ * nCellsZInStrip
53 
54  short row = nStrpZ - (strip - 1) % nStrpZ;
55  short col = (int)std::ceil((float)strip / (nStrpZ)) - 1;
56 
57  return moduleNumber * nCrystalsInModule + row * 2 + (col * nCellsXInStrip + (cell - 1) / 2) * nZ -
58  (cell & 1 ? 1 : 0);
59 }

61 bool Geometry::absToRelNumbering(short absId, char* relid)
62 {
63  // Converts the absolute numbering into the following array
64  // relid[0] = PHOS Module number 1:fNModules
65  // relid[1] = Row number inside a PHOS module (Z coordinate)
66  // relid[2] = Column number inside a PHOS module (Phi coordinate)
67  const short nZ = 56; // nStripZ * nCellsZInStrip
68  const short nPhi = 64; // nStripZ * nCellsZInStrip
69 
70  short phosmodulenumber = (absId - 1) / (nZ * nPhi);
71 
72  relid[0] = phosmodulenumber;
73  absId -= phosmodulenumber * nPhi * nZ;
74  relid[1] = 1 + (absId - 1) / nZ;
75  relid[2] = absId - (relid[1] - 1) * nZ;
76 
77  return true;
78 }

87 int Geometry::areNeighbours(short absId1, short absId2)
88 {
89 
90  // Gives the neighbourness of two digits = 0 are not neighbour but continue searching
91  // = 1 are neighbour
92  // = 2 are not neighbour but do not continue searching
93  // =-1 are not neighbour, continue searching, but do not look before d2 next
94  // time
95  // neighbours are defined as digits having at least a common vertex
96  // The order of d1 and d2 is important: first (d1) should be a digit already in a cluster
97  // which is compared to a digit (d2) not yet in a cluster
98 
99  char relid1[3];
100  absToRelNumbering(absId1, relid1);
101 
102  char relid2[3];
103  absToRelNumbering(absId2, relid2);
104 
105  if (relid1[0] == relid2[0]) { // inside the same PHOS module
106  char rowdiff = TMath::Abs(relid1[1] - relid2[1]);
107  char coldiff = TMath::Abs(relid1[2] - relid2[2]);
108 
109  if ((coldiff <= 1) && (rowdiff <= 1)) { // At least common vertex
110  return 1;
111  } else {
112  if ((relid2[1] > relid1[1]) && (relid2[2] > relid1[2] + 1)) {
113  return 2; // Difference in row numbers is too large to look further
114  }
115  }
116  return 0;
117 
118  } else {
119  if (relid1[0] > relid2[0]) { // we switched to the next module
120  return -1;
121  }
122  return 2;
123  }
124  return 0;
125 }
