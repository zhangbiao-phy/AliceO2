95 uint32_t Digit::Equipment2Pad(int Equi, int Colu, int Dilo, int Chan)
96 {
97  // Check the input data
98  if (Equi < 0 || Equi >= Geo::MAXEQUIPMENTS || Colu < 0 || Colu >= Geo::N_COLUMNS ||
99  Dilo < 0 || Dilo >= Geo::N_DILOGICS || Chan < 0 || Chan >= Geo::N_CHANNELS) {
100  return -1;
101  }
102 
103  int chan2y[6] = {3, 2, 4, 1, 5, 0}; // y coordinate translation for a channel address (index position) for even chamber
104 
105  bool isEven = (Equi % Geo::EQUIPMENTSPERMODULE) == 0 ? true : false; // Calculate the odd/even of geometry
106  int ch = Equi / Geo::EQUIPMENTSPERMODULE; // The Module
107 
108  // Calculate the x,y photo cathode relative coords For Odd equipment
109  int pc = (Colu / Geo::N_COLXSEGMENT) * 2 + 1; // col [0..23] -> [1,3,5]
110  int px = Geo::MAXXPHOTO - ((Dilo * Geo::DILOPADSROWS) + (Chan / Geo::DILOPADSCOLS));
111  int py = (Colu % Geo::DILOPADSROWS) * Geo::DILOPADSCOLS + chan2y[Chan % Geo::DILOPADSCOLS];
112  if (isEven) {
113  pc = 5 - pc;
114  py = Geo::MAXYPHOTO - py;
115  }
116  return Abs(ch, pc, px, py); // Pack the coords into the PadID word
117 }

125 void Digit::Pad2Equipment(uint32_t pad, int* Equi, int* Colu, int* Dilo, int* Chan)
126 {
127  int ch, ph, px, py;
128  int y2chan[6] = {5, 3, 1, 0, 2, 4};
129 
130  Pad2Photo(pad, &ch, &ph, &px, &py); // Unpak the pad ID in the photo cathode coords
131 
132  bool isEven = (ph % 2) == 0 ? true : false;
133  int eq = ch * Geo::EQUIPMENTSPERMODULE + 1;
134  px = Geo::MAXXPHOTO - px; // revert the X coord
135  if (isEven) {
136  eq--; // Correct the equipment number
137  py = Geo::MAXYPHOTO - py; // revert the Y coord
138  ph = 5 - ph; // revert the photo cathode index [0,2,4] -> [5,3,1]
139  }
140  *Dilo = px / Geo::DILOPADSROWS; // Calculate the Dilogic x [0..79] -> dil [0..9]
141  *Colu = ((ph / 2) * Geo::N_COLXSEGMENT) + (py / Geo::DILOPADSCOLS); // calculate the column (ph [1,3,5], y [0..47]) -> col [0..23]
142  *Chan = ((px % Geo::DILOPADSROWS) * Geo::DILOPADSCOLS) + y2chan[py % Geo::DILOPADSCOLS];
143  *Equi = eq;
144  return;
145 }

182 uint32_t Digit::Absolute2Pad(int Module, int x, int y)
183 {
184  int ph = (y / Geo::N_PHOTOCATODSY) * 2 + ((x >= Geo::HALFXROWS) ? 1 : 0);

186  int py = y % Geo::N_PHOTOCATODSY;

195 void Digit::Pad2Absolute(uint32_t pad, int* Module, int* x, int* y)
196 {
197  *Module = A2C(pad);
198  int ph = A2P(pad);
199  int px = A2X(pad);
200  int py = A2Y(pad);
201  *x = px + ((ph % 2 == 1) ? Geo::HALFXROWS : 0);
202  *y = ((ph >> 1) * Geo::N_PHOTOCATODSY) + py;
203  return;
204 }

270 Double_t Digit::QdcTot(Double_t e, Double_t time, Int_t pc, Int_t px, Int_t py, Double_t& localX, Double_t& localY)
271 {
272  //
273  // Arguments: e-
274  // Returns:
275  double Q = 0;
276  if (time > 1.2e-6) {
277  Q = 0;
278  }
279  if (py < 0) {
280  return 0;
281  } else {
282  double y = Param::LorsY(pc, py);
283  localY = ((y - localY) > 0) ? y - 0.2 : y + 0.2; //shift to the nearest anod wire
284 
285  double x = (localX > 66.6) ? localX - 66.6 : localX; //sagita is for PC (0-64) and not for chamber
286  double qdcEle = 34.06311 + 0.2337070 * x + 5.807476e-3 * x * x - 2.956471e-04 * x * x * x + 2.310001e-06 * x * x * x * x; //reparametrised from DiMauro
287 
288  int iNele = int((e / 26e-9) * 0.8);
289  if (iNele < 1) {
290  iNele = 1; //number of electrons created by hit, if photon e=0 implies iNele=1
291  }
292  for (Int_t i = 1; i <= iNele; i++) {
293  double rnd = gRandom->Rndm();
294  if (rnd == 0) {
295  rnd = 1e-12; //1e-12 is a protection against 0 from rndm
296  }
297  Q -= qdcEle * TMath::Log(rnd);
298  }
299  }
300  return Q;
301 }

310 Double_t Digit::IntPartMathiX(Double_t x, int pad)
311 {
312  Double_t shift1 = -LorsX(pad) + 0.5 * o2::hmpid::Param::SizePadX();
313  Double_t shift2 = -LorsX(pad) - 0.5 * o2::hmpid::Param::SizePadX();
314 
315  Double_t ux1 = o2::hmpid::Param::SqrtK3x() * TMath::TanH(o2::hmpid::Param::K2x() * (x + shift1) / o2::hmpid::Param::PitchAnodeCathode());
316  Double_t ux2 = o2::hmpid::Param::SqrtK3x() * TMath::TanH(o2::hmpid::Param::K2x() * (x + shift2) / o2::hmpid::Param::PitchAnodeCathode());
317 
318  return o2::hmpid::Param::K4x() * (TMath::ATan(ux2) - TMath::ATan(ux1));
319 }

328 Double_t Digit::IntPartMathiY(Double_t y, int pad)
329 {
330  Double_t shift1 = -LorsY(pad) + 0.5 * o2::hmpid::Param::SizePadY();
331  Double_t shift2 = -LorsY(pad) - 0.5 * o2::hmpid::Param::SizePadY();
332 
333  Double_t uy1 = o2::hmpid::Param::SqrtK3y() * TMath::TanH(o2::hmpid::Param::K2y() * (y + shift1) / o2::hmpid::Param::PitchAnodeCathode());
334  Double_t uy2 = o2::hmpid::Param::SqrtK3y() * TMath::TanH(o2::hmpid::Param::K2y() * (y + shift2) / o2::hmpid::Param::PitchAnodeCathode());
335 
336  return o2::hmpid::Param::K4y() * (TMath::ATan(uy2) - TMath::ATan(uy1));
337 }
