39  float calcMass2(float p2Pos, float p2Neg, float p2V0) const

41  // calculate v0 mass from squared momentum of its prongs and total momentum
42  float ePos = std::sqrt(p2Pos + getMass2PosProng()), eNeg = std::sqrt(p2Neg + getMass2NegProng()), eV0 = ePos + eNeg;
