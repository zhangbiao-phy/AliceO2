116 template <typename Source_IT>
117 SymbolStatistics::SymbolStatistics(const Source_IT begin, const Source_IT end, int64_t min, int64_t max, size_t scaleBits, size_t nUsedAlphabetSymbols) : mMin(min), mMax(max), mScaleBits(scaleBits), mFrequencyTable(), mNUsedAlphabetSymbols(nUsedAlphabetSymbols), mCumulativeFrequencyTable()
118 {
119 
120  using namespace internal;
121  LOG(trace) << "start building symbol statistics";
122  RANSTimer t;
123  t.start();
124 
125  if (begin == end) {
126  LOG(debug) << "Passed empty message to " << __func__; // RS this is ok for empty columns
127  return;
128  }
129 
130  // if we did not calculate renormalization size, calculate it.
131  if (scaleBits == 0) {
132  mScaleBits = [&, this]() {
133  const size_t minScale = MIN_SCALE;
134  const size_t maxScale = MAX_SCALE;
135  const size_t calculated = static_cast<uint32_t>(3 * std::ceil(std::log2(getNUsedAlphabetSymbols())) / 2 + 2);
136  return std::max(minScale, std::min(maxScale, calculated));
137  }();
138  }
139 
140  //additional bit for escape symbol
141  mFrequencyTable.reserve(std::distance(begin, end) + 1);
142 
143  mFrequencyTable.insert(mFrequencyTable.begin(), begin, end);
144  //incompressible symbol
145  mFrequencyTable.push_back(1);
146 
147  // range check
148  if (mMax - mMin > 1 << (MAX_RANGE - 1)) {
149  const std::string errmsg = [&]() {
150  std::stringstream ss;
151  ss << "Range of source message " << std::ceil(std::log2(mMax - mMin)) << "Bits surpasses maximal allowed range of " << MAX_RANGE << "Bits.";
152  return ss.str();
153  }();
154  LOG(error) << errmsg;
155  throw std::runtime_error(errmsg);
156  }
157 
158  buildCumulativeFrequencyTable();
159  rescale();
160 
161  assert(mFrequencyTable.size() > 1);
162  assert(mCumulativeFrequencyTable.size() > 2);
163  assert(mCumulativeFrequencyTable.size() - mFrequencyTable.size() == 1);
164 
165  t.stop();
166  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
167 
168 // advanced diagnostics in debug builds
169 #if !defined(NDEBUG)
170  LOG(debug2) << "SymbolStatistics: {"
171  << "entries: " << mFrequencyTable.size() << ", "
172  << "frequencyTableSizeB: " << mFrequencyTable.size() * sizeof(typename std::decay_t<decltype(mFrequencyTable)>::value_type) << ", "
173  << "CumulativeFrequencyTableSizeB: " << mCumulativeFrequencyTable.size() * sizeof(typename std::decay_t<decltype(mCumulativeFrequencyTable)>::value_type) << "}";
174 #endif
175 
176  LOG(trace) << "done building symbol statistics";
177 }
