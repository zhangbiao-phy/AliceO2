34 void VertexTrackMatcher::updateTPCTimeDependentParams()
35 {
36  // tpc time bin in microseconds
37  if (mMaxTPCDriftTimeMUS == 0) {
38  auto& gasParam = o2::tpc::ParameterGas::Instance();
39  auto& elParam = o2::tpc::ParameterElectronics::Instance();
40  auto& detParam = o2::tpc::ParameterDetector::Instance();
41  mTPCBin2MUS = elParam.ZbinWidth;
42  mMaxTPCDriftTimeMUS = detParam.TPClength / gasParam.DriftV;
43  }
44  if (mITSROFrameLengthMUS == 0) {
45  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom(o2::base::NameConf::getGRPFileName())};
46  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
47  mITSROFrameLengthMUS = grp->isDetContinuousReadOut(o2::detectors::DetID::ITS) ? alpParams.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingMS : alpParams.roFrameLengthTrig * 1.e-3;
48  }
49 }

51 void VertexTrackMatcher::process(const gsl::span<const PVertex>& vertices,
52  const gsl::span<const VTIndex>& v2tfitIDs,
53  const gsl::span<const VRef>& v2tfitRefs,
54  const o2::globaltracking::RecoContainer& recoData,
55  std::vector<VTIndex>& trackIndex,
56  std::vector<VRef>& vtxRefs)
57 {
58  updateTPCTimeDependentParams();
59 
60  int nv = vertices.size();
61  TmpMap tmpMap(nv);
62  // register vertex contributors
63  std::unordered_map<GIndex, bool> vcont;
64  std::vector<VtxTBracket> vtxOrdBrack; // vertex indices and brackets sorted in tmin
65  float maxVtxSpan = 0;
66  for (int iv = 0; iv < nv; iv++) {
67  int idMin = v2tfitRefs[iv].getFirstEntry(), idMax = idMin + v2tfitRefs[iv].getEntries();
68  auto& vtxIds = tmpMap[iv]; // global IDs of contibuting tracks
69  vtxIds.reserve(v2tfitRefs[iv].getEntries());
70  for (int id = idMin; id < idMax; id++) {
71  auto gid = v2tfitIDs[id];
72  vtxIds.emplace_back(gid).setPVContributor();
73  vcont[gid] = true;
74  }
75  const auto& vtx = vertices[iv];
76  const auto& vto = vtxOrdBrack.emplace_back(VtxTBracket{
77  {float((vtx.getIRMin().differenceInBC(mStartIR) - 0.5f) * o2::constants::lhc::LHCBunchSpacingMS),
78  float((vtx.getIRMax().differenceInBC(mStartIR) + 0.5f) * o2::constants::lhc::LHCBunchSpacingMS)},
79  iv});
80  if (vto.tBracket.delta() > maxVtxSpan) {
81  maxVtxSpan = vto.tBracket.delta();
82  }
83  }
84  // sort vertices in tmin
85  std::sort(vtxOrdBrack.begin(), vtxOrdBrack.end(), [](const VtxTBracket& a, const VtxTBracket& b) { return a.tBracket.getMin() < b.tBracket.getMin(); });
86 
87  extractTracks(recoData, vcont); // extract all track t-brackets, excluding those tracks which contribute to vertex (already attached)
88 
89  int ivStart = 0;
90  std::vector<int> vtxList; // list of vertices which match to checked track
91  for (const auto& tro : mTBrackets) {
92  vtxList.clear();
93  for (int iv = ivStart; iv < nv; iv++) {
94  const auto& vto = vtxOrdBrack[iv];
95  auto res = tro.tBracket.isOutside(vto.tBracket);
96 
97  if (res == TBracket::Below) { // vertex preceeds the track
98  if (tro.tBracket.getMin() > vto.tBracket.getMin() + maxVtxSpan) { // all following vertices will be preceeding all following tracks times
99  ivStart = ++iv;
100  break;
101  }
102  continue; // following vertex with longer span might still match this track
103  }
104  if (res == TBracket::Above) { // track preceeds the vertex, so will preceed also all following vertices
105  break;
106  }
107  // track matches to vertex, register
108  vtxList.push_back(vto.origID); // flag matching vertex
109  }
110  bool ambigous = vtxList.size() > 1; // did track match to multiple vertices?
111  for (auto v : vtxList) {
112  auto& ref = tmpMap[v].emplace_back(tro.origID);
113  if (ambigous) {
114  ref.setAmbiguous();
115  }
116  }
117  }
118 
119  // build final vector of global indices
120  trackIndex.clear();
121  vtxRefs.clear();
122 
123  for (int iv = 0; iv < nv; iv++) {
124  auto& trvec = tmpMap[iv];
125  // sort entries in each vertex track indices list according to the source
126  std::sort(trvec.begin(), trvec.end(), [](VTIndex a, VTIndex b) { return a.getSource() < b.getSource(); });
127 
128  auto entry0 = trackIndex.size(); // start of entries for this vertex
129  auto& vr = vtxRefs.emplace_back();
130  int oldSrc = -1;
131  for (const auto gid0 : trvec) {
132  int src = gid0.getSource();
133  while (oldSrc < src) {
134  oldSrc++;
135  vr.setFirstEntryOfSource(oldSrc, trackIndex.size()); // register start of new source
136  }
137  trackIndex.push_back(gid0);
138  }
139  while (++oldSrc < GIndex::NSources) {
140  vr.setFirstEntryOfSource(oldSrc, trackIndex.size());
141  }
142  vr.setEnd(trackIndex.size());
143  LOG(INFO) << "Vertxex " << iv << " Tracks " << vr;
144  }
145 }

148 void VertexTrackMatcher::extractTracks(const o2::globaltracking::RecoContainer& data, const std::unordered_map<GIndex, bool>& vcont)
149 {
150  // Scan all inputs and create tracks
151 
152  mTBrackets.clear();
153 
154  std::function<void(const o2::track::TrackParCov& _tr, float t0, float terr, GIndex _origID)> creator =
155  [this, &vcont](const o2::track::TrackParCov& _tr, float t0, float terr, GIndex _origID) {
156  if (vcont.find(_origID) != vcont.end()) { // track is contributor to vertex, already accounted
157  return;
158  }
159  if (_origID.getSource() == GIndex::TPC) { // convert TPC bins to \mus
160  t0 *= this->mTPCBin2MUS;
161  terr *= this->mTPCBin2MUS;
162  } else if (_origID.getSource() == GIndex::ITS) { // error is supplied a half-ROF duration, convert to \mus
163  t0 += this->mITSROFrameLengthMUS;
164  terr *= 0.5 * this->mITSROFrameLengthMUS;
165  } else {
166  //terr *= this->mMatchParams->nSigmaTError;
167  }
168  mTBrackets.emplace_back(TrackTBracket{{t0 - terr, t0 + terr}, _origID});
169  };
170 
171  data.createTracks(creator);
172 
173  // sort in increasing min.time
174  std::sort(mTBrackets.begin(), mTBrackets.end(), [](const TrackTBracket& a, const TrackTBracket& b) { return a.tBracket.getMin() < b.tBracket.getMin(); });
175 
176  LOG(INFO) << "collected " << mTBrackets.size() << " seeds";
177 }
