41  Configurable<float> vertexZCut{"vertexZCut", 10.0f, "Accepted z-vertex range"};
42  Configurable<float> trackPtCut{"trackPtCut", 0.1, "minimum constituent pT"};
43  Configurable<float> trackEtaCut{"trackEtaCut", 0.9, "constituent eta cut"};
44  Configurable<bool> DoRhoAreaSub{"DoRhoAreaSub", false, "do rho area subtraction"};
45  Configurable<bool> DoConstSub{"DoConstSub", false, "do constituent subtraction"};
46  Configurable<float> jetPtMin{"jetPtMin", 10.0, "minimum jet pT"};
47  Configurable<float> jetR{"jetR", 0.4, "jet resolution"};
48 
49  Filter collisionFilter = nabs(aod::collision::posZ) < vertexZCut;
50  Filter trackFilter = (nabs(aod::track::eta) < trackEtaCut) && (aod::track::isGlobalTrack == (uint8_t) true) && (aod::track::pt > trackPtCut);
51 
52  std::vector<fastjet::PseudoJet> jets;
53  std::vector<fastjet::PseudoJet> inputParticles;
54  JetFinder jetFinder; //should be a configurable but for now this cant be changed on hyperloop

56  void init(InitContext const&)
57  {
58  hJetPt.setObject(new TH1F("h_jet_pt", "jet p_{T};p_{T} (GeV/#it{c})",
59  100, 0., 100.));
60  hJetPhi.setObject(new TH1F("h_jet_phi", "jet #phi;#phi",
61  80, -1., 7.));
62  hJetEta.setObject(new TH1F("h_jet_eta", "jet #eta;#eta",
63  70, -0.7, 0.7));
64  hJetN.setObject(new TH1F("h_jet_n", "jet n;n constituents",
65  30, 0., 30.));
66  if (DoRhoAreaSub) {
67  jetFinder.setBkgSubMode(JetFinder::BkgSubMode::rhoAreaSub);
68  }
69  if (DoConstSub) {
70  jetFinder.setBkgSubMode(JetFinder::BkgSubMode::constSub);
71  }
72  jetFinder.jetPtMin = jetPtMin;
73  jetFinder.jetR = jetR;
74  }

76  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>::iterator const& collision,
77  soa::Filtered<soa::Join<aod::Tracks, aod::TrackSelection>> const& tracks)
78  {
79 
80  if (!collision.alias()[kINT7]) {
81  return; //remove hard code
82  }
83  if (!collision.sel7()) {
84  return; //remove hard code
85  }
86  jets.clear();
87  inputParticles.clear();
88 
89  for (auto& track : tracks) {
90  /* auto energy = std::sqrt(track.p() * track.p() + mPion*mPion);
91  inputParticles.emplace_back(track.px(), track.py(), track.pz(), energy);
92  inputParticles.back().set_user_index(track.globalIndex());*/
93  fillConstituents(track, inputParticles);
94  inputParticles.back().set_user_index(track.globalIndex());
95  }
96 
97  fastjet::ClusterSequenceArea clusterSeq(jetFinder.findJets(inputParticles, jets));
98 
99  for (const auto& jet : jets) {
100  jetsTable(collision, jet.pt(), jet.eta(), jet.phi(),
101  jet.E(), jet.m(), jet.area());
102  hJetPt->Fill(jet.pt());
103  hJetPhi->Fill(jet.phi());
104  hJetEta->Fill(jet.eta());
105  hJetN->Fill(jet.constituents().size());
106  for (const auto& constituent : jet.constituents()) { //event or jetwise
107  if (DoConstSub) {
108  constituentsSubTable(jetsTable.lastIndex(), constituent.pt(), constituent.eta(), constituent.phi(),
109  constituent.E(), constituent.m());
110  }
111  constituentsTable(jetsTable.lastIndex(), constituent.user_index());
112  }
113  }
114  }
115 };
