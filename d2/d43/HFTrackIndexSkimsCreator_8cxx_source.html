68  void process(aod::Collision const& collision,
69  soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra> const& tracks)
70  {
71  math_utils::Point3D<float> vtxXYZ(collision.posX(), collision.posY(), collision.posZ());
72  for (auto& track : tracks) {
73 
74  int status_prong = 3; // selection flag , 2 bits on
75 
76  auto trackPt = track.pt();
77  if (b_dovalplots.value) {
78  registry.get<TH1>(HIST("hpt_nocuts"))->Fill(trackPt);
79  }
80 
81  // pT cut
82  if (trackPt < ptmintrack_2prong) {
83  status_prong = status_prong & ~(1 << 0); // the bitwise operation & ~(1 << n) will set the nth bit to 0
84  }
85  if (trackPt < ptmintrack_3prong) {
86  status_prong = status_prong & ~(1 << 1);
87  }
88 
89  auto trackEta = track.eta();
90  // eta cut
91  if ((status_prong & (1 << 0)) && abs(trackEta) > etamax_2prong) {
92  status_prong = status_prong & ~(1 << 0);
93  }
94  if ((status_prong & (1 << 1)) && abs(trackEta) > etamax_3prong) {
95  status_prong = status_prong & ~(1 << 1);
96  }
97 
98  // quality cut
99  if (doCutQuality.value && status_prong > 0) { // FIXME to make a more complete selection e.g track.flags() & o2::aod::track::TPCrefit && track.flags() & o2::aod::track::GoldenChi2 &&
100  UChar_t clustermap = track.itsClusterMap();
101  if (!(track.tpcNClsFound() >= d_tpcnclsfound.value &&
102  track.flags() & o2::aod::track::ITSrefit &&
103  (TESTBIT(clustermap, 0) || TESTBIT(clustermap, 1)))) {
104  status_prong = 0;
105  }
106  }
107 
108  // DCA cut
109  array<float, 2> dca;
110  if (status_prong > 0) {
111  double dcatoprimxymin_2prong_ptdep = dcatoprimxymin_2prong * TMath::Max(0., (1 - TMath::Floor(trackPt / dcatoprimxy_2prong_maxpt)));
112  double dcatoprimxymin_3prong_ptdep = dcatoprimxymin_3prong * TMath::Max(0., (1 - TMath::Floor(trackPt / dcatoprimxy_3prong_maxpt)));
113  auto trackparvar0 = getTrackParCov(track);
114  if (!trackparvar0.propagateParamToDCA(vtxXYZ, d_bz, &dca, 100.)) { // get impact parameters
115  status_prong = 0;
116  }
117  if ((status_prong & (1 << 0)) && (abs(dca[0]) < dcatoprimxymin_2prong_ptdep || abs(dca[0]) > dcatoprimxymax_2prong)) {
118  status_prong = status_prong & ~(1 << 0);
119  }
120  if ((status_prong & (1 << 1)) && (abs(dca[0]) < dcatoprimxymin_3prong_ptdep || abs(dca[0]) > dcatoprimxymax_3prong)) {
121  status_prong = status_prong & ~(1 << 1);
122  }
123  }
124 
125  // fill histograms
126  if (b_dovalplots) {
127  if (status_prong & (1 << 0)) {
128  registry.get<TH1>(HIST("hpt_cuts_2prong"))->Fill(trackPt);
129  registry.get<TH1>(HIST("hdcatoprimxy_cuts_2prong"))->Fill(dca[0]);
130  registry.get<TH1>(HIST("heta_cuts_2prong"))->Fill(trackEta);
131  }
132  if (status_prong & (1 << 1)) {
133  registry.get<TH1>(HIST("hpt_cuts_3prong"))->Fill(trackPt);
134  registry.get<TH1>(HIST("hdcatoprimxy_cuts_3prong"))->Fill(dca[0]);
135  registry.get<TH1>(HIST("heta_cuts_3prong"))->Fill(trackEta);
136  }
137  }
138 
139  // fill table row
140  rowSelectedTrack(status_prong, dca[0], dca[1]);
141  }
142  }
143 };
