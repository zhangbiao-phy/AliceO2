77 Mapping::ErrorStatus Mapping::setMapping()
78 {
79  //Read mapping from data files a-la Run2
80 
81  o2::phos::Geometry* geom = o2::phos::Geometry::GetInstance();
82 
83  std::string p;
84  if (mPath.empty()) { //use default path
85  p = gSystem->Getenv("O2_ROOT");
86  p += "/share/Detectors/PHOS/files";
87  } else {
88  p = mPath.data();
89  }
90 
91  for (short m = 0; m < 4; m++) { //modules
92  for (short i = 0; i < 4; i++) { //RCU
93  if (m == 0 && (i < 2)) {
94  continue; //half of module: only RCU 2,3
95  }
96 
97  short numberOfChannels = 0;
98  short maxHWAddress = 0;
99  char fname[255];
100  snprintf(fname, 255, "%s/Mod%dRCU%d.data", p.data(), m, i);
101  std::ifstream* fIn = new std::ifstream(fname);
102  if (!*fIn) {
103  LOG(FATAL) << "Missing mapping file " << p << "/Mod" << m << "RCU" << i << ".data";
104  return kNotInitialized;
105  }
106  if (!(*fIn >> numberOfChannels)) {
107  LOG(FATAL) << "Syntax of mapping file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no numberOfChannels";
108  return kNotInitialized;
109  }
110  if (numberOfChannels != NHWPERDDL) {
111  LOG(FATAL) << "Unexpected number of channels: " << numberOfChannels << " expecting " << NHWPERDDL << " file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no numberOfChannels";
112  return kNotInitialized;
113  }
114  if (!(*fIn >> maxHWAddress)) {
115  LOG(FATAL) << "Syntax of mapping file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no maxHWAddress";
116  return kNotInitialized;
117  }
118  if (maxHWAddress > NMaxHWAddress) {
119  LOG(FATAL) << "Maximal HW address in file " << maxHWAddress << "larger than array size " << NMaxHWAddress << "for /Mod" << m << "RCU" << i << ".data is wrong: no maxHWAddress";
120  return kNotInitialized;
121  }
122  for (short ich = 0; ich < numberOfChannels; ich++) { // 1792 = 2*896 channels connected to each RCU
123  int hwAddress;
124  if (!(*fIn >> hwAddress)) {
125  LOG(FATAL) << "Syntax of mapping file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no HWadd for ch " << ich;
126  return kNotInitialized;
127  }
128  if (hwAddress > maxHWAddress) {
129  LOG(FATAL) << "Hardware (ALTRO) adress (" << hwAddress << ") outside the range (0 -> " << maxHWAddress << ") !";
130  return kNotInitialized;
131  }
132  int row, col, caloFlag;
133  if (!(*fIn >> row >> col >> caloFlag)) {
134  LOG(FATAL) << "Syntax of mapping file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no (raw col caloFlag)";
135  return kNotInitialized;
136  }
137 
138  if (caloFlag < 0 || caloFlag > 2) {
139  LOG(FATAL) << "Wrong CaloFlag value found (" << caloFlag << "). Should be 0, 1, 2 !";
140  return kNotInitialized;
141  }
142 
143  if (caloFlag == 2) { //TODO!!!! TRU mapping not known yet
144  continue;
145  }
146 
147  //convert ddl, col,raw caloFlag to AbsId
148  // Converts the absolute numbering into the following array
149  // relid[0] = PHOS Module number
150  // relid[1] = Row number inside a PHOS module (Phi coordinate)
151  // relid[2] = Column number inside a PHOS module (Z coordinate)
152  short ddl = 4 * m + i - 2;
153 
154  char relid[3] = {static_cast<char>(m), static_cast<char>(row + 1), static_cast<char>(col + 1)};
155  short absId;
156  geom->relToAbsNumbering(relid, absId);
157 
158  if (ddl < 0 || ddl >= NDDL) {
159  LOG(FATAL) << "Wrong ddl address found (" << ddl << "). Module= " << m << " RCU =" << i;
160  return kNotInitialized;
161  }
162 
163  mAbsId[ddl][hwAddress] = absId;
164  mCaloFlag[ddl][hwAddress] = (CaloFlag)caloFlag;
165  mAbsToHW[absId][caloFlag][0] = ddl;
166  mAbsToHW[absId][caloFlag][1] = hwAddress;
167  }
168  fIn->close();
169  } //RCU
170  } // module
171  mInitialized = true;
172  return kOK;
173 }
