104 double RCUTrailer::getTimeSample() const
105 {
106  unsigned char fq = (mAltroCFG2 >> 5) & 0xF;
107  double tSample;
108  switch (fq) {
109  case 0:
110  // 20 MHz
111  tSample = 2.0;
112  break;
113  case 1:
114  // 10 Mhz
115  tSample = 4.0;
116  break;
117  case 2:
118  // 5 MHz
119  tSample = 8.;
120  break;
121  default:
122  throw Error(Error::ErrorType_t::SAMPLINGFREQ_INVALID, fmt::format("Invalid sampling frequency value %d !", int(fq)).data());
123  }
124 
125  return tSample * o2::constants::lhc::LHCBunchSpacingNS * 1.e-9;
126 }

128 void RCUTrailer::setTimeSample(double timesample)
129 {
130  int fq = 0;
131  if (std::abs(timesample - 50) < DBL_EPSILON) {
132  fq = 0;
133  } else if (std::abs(timesample - 100) < DBL_EPSILON) {
134  fq = 1;
135  } else if (std::abs(timesample - 200) < DBL_EPSILON) {
136  fq = 2;
137  } else {
138  throw Error(Error::ErrorType_t::SAMPLINGFREQ_INVALID, fmt::format("invalid time sample: %f", timesample).data());
139  }
140  mAltroCFG2 = (mAltroCFG2 & 0x1F) | fq << 5;
141 }

143 double RCUTrailer::getL1Phase() const
144 {
145  double tSample = getTimeSample(),
146  phase = ((double)(mAltroCFG2 & 0x1F)) * o2::constants::lhc::LHCBunchSpacingNS * 1.e-9;
147  if (phase >= tSample) {
148  throw Error(Error::ErrorType_t::L1PHASE_INVALID, fmt::format("Invalid L1 trigger phase (%e s (phase) >= %e s (sampling time)) !", phase, tSample).data());
149  }
150  return phase;
151 }

159 std::vector<uint32_t> RCUTrailer::encode() const
160 {
161  std::vector<uint32_t> encoded;
162  encoded.emplace_back(mPayloadSize | 2 << 30);
163  encoded.emplace_back(mAltroCFG2 | 7 << 26 | 2 << 30);
164  encoded.emplace_back(mAltroCFG1 | 6 << 26 | 2 << 30);
165  encoded.emplace_back(mActiveFECsB | 5 << 26 | 2 << 30);
166  encoded.emplace_back(mActiveFECsA | 4 << 26 | 2 << 30);
167  encoded.emplace_back(mERRREG3 | 3 << 26 | 2 << 30);
168  encoded.emplace_back(mERRREG2 | 2 << 26 | 2 << 30);
169  encoded.emplace_back(mFECERRB >> 7 | (mFECERRA >> 7) << 13 | 1 << 26 | 2 << 30);
170 
171  uint32_t lasttrailerword = 3 << 30 | mFirmwareVersion << 16 | mRCUId << 7 | (encoded.size() + 1);
172  encoded.emplace_back(lasttrailerword);
173 
174  return encoded;
175 }

177 void RCUTrailer::printStream(std::ostream& stream) const
178 {
179  std::vector<std::string> errors;
180  double timesample = -1., l1phase = -1.;
181  try {
182  timesample = getTimeSample();
183  } catch (Error& e) {
184  errors.push_back(e.what());
185  }
186  try {
187  l1phase = getL1Phase();
188  } catch (Error& e) {
189  errors.push_back(e.what());
190  }
191 
192  stream << "RCU trailer (Format version 2):\n"
193  << "==================================================\n"
194  << "RCU ID: " << mRCUId << "\n"
195  << "Firmware version: " << int(mFirmwareVersion) << "\n"
196  << "Trailer size: " << mTrailerSize << "\n"
197  << "Payload size: " << mPayloadSize << "\n"
198  << "FECERRA: 0x" << std::hex << mFECERRA << "\n"
199  << "FECERRB: 0x" << std::hex << mFECERRB << "\n"
200  << "ERRREG2: 0x" << std::hex << mERRREG2 << "\n"
201  << "#channels skipped due to address mismatch: " << std::dec << getNumberOfChannelAddressMismatch() << "\n"
202  << "#channels skipped due to bad block length: " << std::dec << getNumberOfChannelLengthMismatch() << "\n"
203  << "Active FECs (branch A): 0x" << std::hex << mActiveFECsA << "\n"
204  << "Active FECs (branch B): 0x" << std::hex << mActiveFECsB << "\n"
205  << "Baseline corr: 0x" << std::hex << int(getBaselineCorrection()) << "\n"
206  << "Number of presamples: " << std::dec << int(getNumberOfPresamples()) << "\n"
207  << "Number of postsamples: " << std::dec << int(getNumberOfPostsamples()) << "\n"
208  << "Second baseline corr: " << (hasSecondBaselineCorr() ? "yes" : "no") << "\n"
209  << "GlitchFilter: " << std::dec << int(getGlitchFilter()) << "\n"
210  << "Number of non-ZS postsamples: " << std::dec << int(getNumberOfNonZeroSuppressedPostsamples()) << "\n"
211  << "Number of non-ZS presamples: " << std::dec << int(getNumberOfNonZeroSuppressedPresamples()) << "\n"
212  << "Number of ALTRO buffers: " << std::dec << getNumberOfAltroBuffers() << "\n"
213  << "Number of pretrigger samples: " << std::dec << int(getNumberOfPretriggerSamples()) << "\n"
214  << "Number of samples per channel: " << std::dec << getNumberOfSamplesPerChannel() << "\n"
215  << "Sparse readout: " << (isSparseReadout() ? "yes" : "no") << "\n"
216  << "AltroCFG1: 0x" << std::hex << mAltroCFG1 << "\n"
217  << "AltroCFG2: 0x" << std::hex << mAltroCFG2 << "\n"
218  << "Sampling time: " << std::scientific << timesample << " s\n"
219  << "L1 Phase: " << std::scientific << l1phase << " s\n"
220  << std::dec << std::fixed;
221  if (errors.size()) {
222  stream << "Errors: \n"
223  << "-------------------------------------------------\n";
224  for (const auto& e : errors) {
225  stream << e << "\n";
226  }
227  }
228  stream << "==================================================\n";
229 }
