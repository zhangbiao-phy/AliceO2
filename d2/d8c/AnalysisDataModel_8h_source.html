326 DECLARE_SOA_DYNAMIC_COLUMN(Eta, eta, [](float inverseBendingMomentum, float thetaX, float thetaY) -> float {
327  float pz = -std::sqrt(1.0 + std::tan(thetaY) * std::tan(thetaY)) / std::abs(inverseBendingMomentum);
328  float pt = std::abs(pz) * std::sqrt(std::tan(thetaX) * std::tan(thetaX) + std::tan(thetaY) * std::tan(thetaY));
329  float eta = std::acos(pz / std::sqrt(pt * pt + pz * pz));
330  eta = std::tan(0.5 * eta);
331  if (eta > 0.0)
332  return -std::log(eta);
333  else
334  return 0.0;
335 });
336 DECLARE_SOA_DYNAMIC_COLUMN(Phi, phi, [](float thetaX, float thetaY) -> float {
337  float phi = std::atan2(std::tan(thetaY), std::tan(thetaX));
338  constexpr float twopi = 2.0f * static_cast<float>(M_PI);
339  return (phi >= 0.0 ? phi : phi + twopi);
340 });
341 DECLARE_SOA_DYNAMIC_COLUMN(RAtAbsorberEnd, rAtAbsorberEnd, [](float bendingCoor, float nonBendingCoor, float zMu, float thetaX, float thetaY) -> float {
342  // linear extrapolation of the coordinates of the track to the position of the end of the absorber (-505 cm)
343  float dZ = -505. - zMu;
344  float NonBendingSlope = std::tan(thetaX);
345  float BendingSlope = std::tan(thetaY);
346  float xAbs = nonBendingCoor + NonBendingSlope * dZ;
347  float yAbs = bendingCoor + BendingSlope * dZ;
348  float rAtAbsorberEnd = std::sqrt(xAbs * xAbs + yAbs * yAbs);
349  return rAtAbsorberEnd;
350 });
351 DECLARE_SOA_DYNAMIC_COLUMN(PDca, pDca, [](float inverseBendingMomentum, float thetaX, float thetaY, float bendingCoor, float nonBendingCoor, float zMu) -> float {
352  // linear extrapolation of the coordinates of the track to the position of the end of the absorber (-505 cm)
353  float dca = std::sqrt(bendingCoor * bendingCoor + nonBendingCoor * nonBendingCoor + zMu * zMu);
354  float pz = -std::sqrt(1.0 + std::tan(thetaY) * std::tan(thetaY)) / std::abs(inverseBendingMomentum);
355  float pt = std::abs(pz) * std::sqrt(std::tan(thetaX) * std::tan(thetaX) + std::tan(thetaY) * std::tan(thetaY));
356  float pTot = std::sqrt(pt * pt + pz * pz);
357  float pDca = pTot * dca;
358  return pDca;
359 });
