134 template <typename DataT, size_t Nz, size_t Nr, size_t Nphi>
135 void testAlmostEqualArray(o2::tpc::DataContainer3D<DataT, Nz, Nr, Nphi>& analytical, o2::tpc::DataContainer3D<DataT, Nz, Nr, Nphi>& numerical)
136 {
137  for (size_t iPhi = 0; iPhi < Nphi; ++iPhi) {
138  for (size_t iR = 0; iR < Nr; ++iR) {
139  for (size_t iZ = 0; iZ < Nz; ++iZ) {
140  if (std::fabs(analytical(iZ, iR, iPhi)) < ABSTOLERANCE) {
141  BOOST_CHECK_SMALL(numerical(iZ, iR, iPhi) - analytical(iZ, iR, iPhi), ABSTOLERANCE);
142  } else {
143  BOOST_CHECK_CLOSE(numerical(iZ, iR, iPhi), analytical(iZ, iR, iPhi), TOLERANCE);
144  }
145  }
146  }
147  }
148 }

166 template <typename DataT, size_t Nz, size_t Nr, size_t Nphi>
167 void poissonSolver3D()
168 {
169  using GridProp = GridProperties<DataT, Nr, Nz, Nphi>;
170  const o2::tpc::RegularGrid3D<DataT, Nz, Nr, Nphi> grid3D{GridProp::ZMIN, GridProp::RMIN, GridProp::PHIMIN, GridProp::GRIDSPACINGZ, GridProp::GRIDSPACINGR, GridProp::GRIDSPACINGPHI};
171 
172  using DataContainer = o2::tpc::DataContainer3D<DataT, Nz, Nr, Nphi>;
173  DataContainer potentialNumerical{};
174  DataContainer potentialAnalytical{};
175  DataContainer charge{};
176 
177  const o2::tpc::AnalyticalFields<DataT> analyticalFields;
178  // set the boudnary and charge for numerical poisson solver
179  setChargeDensityFromFormula<DataT, Nz, Nr, Nphi>(analyticalFields, grid3D, charge);
180  setPotentialBoundaryFromFormula<DataT, Nz, Nr, Nphi>(analyticalFields, grid3D, potentialNumerical);
181 
182  // set analytical potential
183  setPotentialFromFormula<DataT, Nz, Nr, Nphi>(analyticalFields, grid3D, potentialAnalytical);
184 
185  //calculate numerical potential
186  PoissonSolver<DataT, Nz, Nr, Nphi> poissonSolver(grid3D);
187  const int symmetry = 0;
188  poissonSolver.poissonSolver3D(potentialNumerical, charge, symmetry);
189 
190  // compare numerical with analytical solution of the potential
191  testAlmostEqualArray<DataT, Nz, Nr, Nphi>(potentialAnalytical, potentialNumerical);
192 }
