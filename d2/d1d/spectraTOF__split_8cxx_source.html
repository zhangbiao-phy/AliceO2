24 struct TOFSpectraTaskSplit {
25  static constexpr int Np = 9;
26  static constexpr const char* pT[Np] = {"e", "#mu", "#pi", "K", "p", "d", "t", "^{3}He", "#alpha"};
27  static constexpr std::string_view hp[Np] = {"p/El", "p/Mu", "p/Pi", "p/Ka", "p/Pr", "p/De", "p/Tr", "p/He", "p/Al"};
28  static constexpr std::string_view hpt[Np] = {"pt/El", "pt/Mu", "pt/Pi", "pt/Ka", "pt/Pr", "pt/De", "pt/Tr", "pt/He", "pt/Al"};
29  HistogramRegistry histos{"Histos", {}, OutputObjHandlingPolicy::AnalysisObject};
30 
31  void init(o2::framework::InitContext&)
32  {
33  histos.add("p/Unselected", "Unselected;#it{p} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
34  histos.add("pt/Unselected", "Unselected;#it{p}_{T} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
35  for (int i = 0; i < Np; i++) {
36  histos.add(hp[i].data(), Form("%s;#it{p} (GeV/#it{c})", pT[i]), kTH1F, {{100, 0, 20}});
37  histos.add(hpt[i].data(), Form("%s;#it{p}_{T} (GeV/#it{c})", pT[i]), kTH1F, {{100, 0, 20}});
38  }
39  }

41  template <std::size_t i, typename T>
42  void fillParticleHistos(const T& track, const float& nsigma)
43  {
44  if (abs(nsigma) > nsigmacut.value) {
45  return;
46  }
47  histos.fill(HIST(hp[i]), track.p());
48  histos.fill(HIST(hpt[i]), track.pt());
49  }

51  Configurable<float> cfgCutVertex{"cfgCutVertex", 10.0f, "Accepted z-vertex range"};
52  Configurable<float> cfgCutEta{"cfgCutEta", 0.8f, "Eta range for tracks"};
53  Configurable<float> nsigmacut{"nsigmacut", 3, "Value of the Nsigma cut"};
54 
55  Filter collisionFilter = nabs(aod::collision::posZ) < cfgCutVertex;
56  Filter trackFilter = (nabs(aod::track::eta) < cfgCutEta) && (aod::track::isGlobalTrack == (uint8_t) true) && (aod::track::tofSignal > 0.f);
57  using TrackCandidates = soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra,
58  aod::pidRespTOFEl, aod::pidRespTOFMu, aod::pidRespTOFPi,
59  aod::pidRespTOFKa, aod::pidRespTOFPr, aod::pidRespTOFDe,
60  aod::pidRespTOFTr, aod::pidRespTOFHe, aod::pidRespTOFAl,
61  aod::TrackSelection>>;
62  void process(TrackCandidates::iterator const& track)
63  {
64  histos.fill(HIST("p/Unselected"), track.p());
65  histos.fill(HIST("pt/Unselected"), track.pt());
66 
67  fillParticleHistos<0>(track, track.tofNSigmaEl());
68  fillParticleHistos<1>(track, track.tofNSigmaMu());
69  fillParticleHistos<2>(track, track.tofNSigmaPi());
70  fillParticleHistos<3>(track, track.tofNSigmaKa());
71  fillParticleHistos<4>(track, track.tofNSigmaPr());
72  fillParticleHistos<5>(track, track.tofNSigmaDe());
73  fillParticleHistos<6>(track, track.tofNSigmaTr());
74  fillParticleHistos<7>(track, track.tofNSigmaHe());
75  fillParticleHistos<8>(track, track.tofNSigmaAl());
76  }
77 };
