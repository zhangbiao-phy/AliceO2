35 void RawToCellConverterSpec::init(framework::InitContext& ctx)
36 {
37  LOG(DEBUG) << "[EMCALRawToCellConverter - init] Initialize converter ";
38  if (!mGeometry) {
39  mGeometry = o2::emcal::Geometry::GetInstanceFromRunNumber(223409);
40  }
41  if (!mGeometry) {
42  LOG(ERROR) << "Failure accessing geometry";
43  }
44 
45  if (!mMapper) {
46  mMapper = std::unique_ptr<o2::emcal::MappingHandler>(new o2::emcal::MappingHandler);
47  }
48  if (!mMapper) {
49  LOG(ERROR) << "Failed to initialize mapper";
50  }
51 
52  auto fitmethod = ctx.options().get<std::string>("fitmethod");
53  if (fitmethod == "standard") {
54  LOG(INFO) << "Using standard raw fitter";
55  mRawFitter = std::unique_ptr<o2::emcal::CaloRawFitter>(new o2::emcal::CaloRawFitterStandard);
56  } else if (fitmethod == "gamma2") {
57  mRawFitter = std::unique_ptr<o2::emcal::CaloRawFitter>(new o2::emcal::CaloRawFitterGamma2);
58  }
59 
60  mRawFitter->setAmpCut(mNoiseThreshold);
61  mRawFitter->setL1Phase(0.);
62 }

64 void RawToCellConverterSpec::run(framework::ProcessingContext& ctx)
65 {
66  LOG(DEBUG) << "[EMCALRawToCellConverter - run] called";
67  const double CONVADCGEV = 0.016; // Conversion from ADC counts to energy: E = 16 MeV / ADC
68 
69  // Cache cells from for bunch crossings as the component reads timeframes from many links consecutively
70  std::map<o2::InteractionRecord, std::shared_ptr<std::vector<o2::emcal::Cell>>> cellBuffer; // Internal cell buffer
71  std::map<o2::InteractionRecord, uint32_t> triggerBuffer;
72 
73  int firstEntry = 0;
74  for (const auto& rawData : framework::InputRecordWalker(ctx.inputs())) {
75  //o2::emcal::RawReaderMemory<o2::header::RAWDataHeaderV4> rawreader(gsl::span(rawData.payload, o2::framework::DataRefUtils::getPayloadSize(rawData)));
76 
77  o2::emcal::RawReaderMemory rawreader(o2::framework::DataRefUtils::as<const char>(rawData));
78 
79  // loop over all the DMA pages
80  while (rawreader.hasNext()) {
81 
82  rawreader.next();
83 
84  auto& header = rawreader.getRawHeader();
85  auto triggerBC = o2::raw::RDHUtils::getTriggerBC(header);
86  auto triggerOrbit = o2::raw::RDHUtils::getTriggerOrbit(header);
87  auto feeID = o2::raw::RDHUtils::getFEEID(header);
88  auto triggerbits = o2::raw::RDHUtils::getTriggerType(header);
89 
90  o2::InteractionRecord currentIR(triggerBC, triggerOrbit);
91  std::shared_ptr<std::vector<o2::emcal::Cell>> currentCellContainer;
92  auto found = cellBuffer.find(currentIR);
93  if (found == cellBuffer.end()) {
94  currentCellContainer = std::make_shared<std::vector<o2::emcal::Cell>>();
95  cellBuffer[currentIR] = currentCellContainer;
96  // also add trigger bits
97  triggerBuffer[currentIR] = triggerbits;
98  } else {
99  currentCellContainer = found->second;
100  }
101 
102  if (feeID > 40) {
103  continue; //skip STU ddl
104  }
105 
106  //std::cout<<rawreader.getRawHeader()<<std::endl;
107 
108  // use the altro decoder to decode the raw data, and extract the RCU trailer
109  o2::emcal::AltroDecoder decoder(rawreader);
110  decoder.decode();
111 
112  LOG(DEBUG) << decoder.getRCUTrailer();
113 
114  o2::emcal::Mapper map = mMapper->getMappingForDDL(feeID);
115  int iSM = feeID / 2;
116 
117  // Loop over all the channels
118  for (auto& chan : decoder.getChannels()) {
119 
120  int iRow, iCol;
121  ChannelType_t chantype;
122  try {
123  iRow = map.getRow(chan.getHardwareAddress());
124  iCol = map.getColumn(chan.getHardwareAddress());
125  chantype = map.getChannelType(chan.getHardwareAddress());
126  } catch (Mapper::AddressNotFoundException& ex) {
127  std::cerr << ex.what() << std::endl;
128  continue;
129  };
130 
131  int CellID = mGeometry->GetAbsCellIdFromCellIndexes(iSM, iRow, iCol);
132 
133  // define the conatiner for the fit results, and perform the raw fitting using the stadnard raw fitter
134  o2::emcal::CaloFitResults fitResults = mRawFitter->evaluate(chan.getBunches(), 0, 0);
135  if (fitResults.getAmp() < 0) {
136  fitResults.setAmp(0.);
137  }
138  if (fitResults.getTime() < 0) {
139  fitResults.setTime(0.);
140  }
141  currentCellContainer->emplace_back(CellID, fitResults.getAmp() * CONVADCGEV, fitResults.getTime(), chantype);
142  }
143  }
144  }
145 
146  // Loop over BCs, sort cells with increasing tower ID and write to output containers
147  mOutputCells.clear();
148  mOutputTriggerRecords.clear();
149  for (auto [bc, cells] : cellBuffer) {
150  mOutputTriggerRecords.emplace_back(bc, triggerBuffer[bc], mOutputCells.size(), cells->size());
151  if (cells->size()) {
152  // Sort cells according to cell ID
153  std::sort(cells->begin(), cells->end(), [](o2::emcal::Cell& lhs, o2::emcal::Cell& rhs) { return lhs.getTower() < rhs.getTower(); });
154  for (auto cell : *cells) {
155  mOutputCells.push_back(cell);
156  }
157  }
158  }
159 
160  LOG(DEBUG) << "[EMCALRawToCellConverter - run] Writing " << mOutputCells.size() << " cells ...";
161  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLS", 0, o2::framework::Lifetime::Timeframe}, mOutputCells);
162  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLSTRGR", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggerRecords);
163 }
