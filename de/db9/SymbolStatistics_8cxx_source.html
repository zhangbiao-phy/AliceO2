30 void SymbolStatistics::rescale()
31 {
32 
33  auto getFrequency = [this](size_t i) { return mCumulativeFrequencyTable[i + 1] - mCumulativeFrequencyTable[i]; };
34 
35  using namespace internal;
36  LOG(trace) << "start rescaling frequency table";
37  RANSTimer t;
38  t.start();
39 
40  if (mFrequencyTable.empty()) {
41  LOG(warning) << "rescaling Frequency Table for empty message";
42  return;
43  }
44 
45  const size_t newCumulatedFrequency = bitsToRange(mScaleBits);
46  assert(newCumulatedFrequency >= this->getNUsedAlphabetSymbols() + 1);
47 
48  size_t cumulatedFrequencies = mCumulativeFrequencyTable.back();
49 
50  std::vector<uint32_t> sortIdx;
51  sortIdx.reserve(getNUsedAlphabetSymbols());
52 
53  // resample distribution based on cumulative frequencies_
54  for (size_t i = 0; i < mFrequencyTable.size(); i++) {
55  if (mFrequencyTable[i]) {
56  sortIdx.push_back(i); // we will sort only those memorize only those entries which can be used
57  }
58  }
59 
60  std::sort(sortIdx.begin(), sortIdx.end(), [&](uint32_t i, uint32_t j) { return getFrequency(i) < getFrequency(j); });
61  size_t need_shift = 0;
62  for (size_t i = 0; i < sortIdx.size(); i++) {
63  if (static_cast<uint64_t>(getFrequency(sortIdx[i])) * (newCumulatedFrequency - need_shift) / cumulatedFrequencies >= 1) {
64  break;
65  }
66  need_shift++;
67  }
68 
69  size_t shift = 0;
70  auto beforeUpdate = mCumulativeFrequencyTable[0];
71  for (size_t i = 0; i < mFrequencyTable.size(); i++) {
72  if (mFrequencyTable[i] && static_cast<uint64_t>(mCumulativeFrequencyTable[i + 1] - beforeUpdate) * (newCumulatedFrequency - need_shift) / cumulatedFrequencies < 1) {
73  shift++;
74  }
75  beforeUpdate = mCumulativeFrequencyTable[i + 1];
76  mCumulativeFrequencyTable[i + 1] = (static_cast<uint64_t>(newCumulatedFrequency - need_shift) * mCumulativeFrequencyTable[i + 1]) / cumulatedFrequencies + shift;
77  }
78  assert(shift == need_shift);
79 
80  // calculate updated freqs and make sure we didn't screw anything up
81  assert(mCumulativeFrequencyTable.front() == 0 &&
82  mCumulativeFrequencyTable.back() == newCumulatedFrequency);
83 
84  for (size_t i = 0; i < mFrequencyTable.size(); i++) {
85  if (mFrequencyTable[i] == 0) {
86  assert(mCumulativeFrequencyTable[i + 1] == mCumulativeFrequencyTable[i]);
87  } else {
88  assert(mCumulativeFrequencyTable[i + 1] > mCumulativeFrequencyTable[i]);
89  }
90 
91  // calc updated freq
92  mFrequencyTable[i] = getFrequency(i);
93  }
94  // for(int i = 0; i<static_cast<int>(freqs.getNumSymbols()); i++){
95  // std::cout << i << ": " << i + min_ << " " << freqs[i] << " " <<
96  // cummulatedFrequencies_[i] << std::endl;
97  // }
98  // std::cout << cummulatedFrequencies_.back() << std::endl;
99 
100  t.stop();
101  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
102 
103  LOG(trace) << "done rescaling frequency table";
104 }
