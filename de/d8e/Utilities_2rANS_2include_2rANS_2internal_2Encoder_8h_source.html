36 template <typename State_T, typename Stream_T>
37 class Encoder
38 {
39  __extension__ typedef unsigned __int128 uint128;
40 
41  // the Coder works either with a 64Bit state and 32 bit streaming or
42  //a 32 Bit state and 8 Bit streaming We need to make sure it gets initialized with
43  //the right template arguments at compile time.
44  static_assert((sizeof(State_T) == sizeof(uint32_t) && sizeof(Stream_T) == sizeof(uint8_t)) ||
45  (sizeof(State_T) == sizeof(uint64_t) && sizeof(Stream_T) == sizeof(uint32_t)),
46  "Coder can either be 32Bit with 8 Bit stream type or 64 Bit Type with 32 Bit stream type");
47 
48  public:
49  Encoder();
50 
51  // Encodes a single symbol with range start "start" and frequency "freq".
52  // All frequencies are assumed to sum to "1 << scale_bits", and the
53  // resulting bytes get written to ptr (which is updated).
54  //
55  // NOTE: With rANS, you need to encode symbols in *reverse order*, i.e. from
56  // beginning to end! Likewise, the output bytestream is written *backwards*:
57  // ptr starts pointing at the end of the output buffer and keeps decrementing.
58  template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool> = true>
59  Stream_IT put(Stream_IT iter, uint32_t start, uint32_t freq, uint32_t scale_bits);
60 
61  // Flushes the rANS encoder.
62  template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool> = true>
63  Stream_IT flush(Stream_IT iter);
64 
65  // Encodes a given symbol. This is faster than straight RansEnc since we can do
66  // multiplications instead of a divide.
67  //
68  // See Rans32EncSymbolInit for a description of how this works.
69  template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool> = true>
70  Stream_IT putSymbol(Stream_IT iter, const EncoderSymbol<State_T>& sym, uint32_t scale_bits);
71 
72  private:
73  State_T mState;
74 
75  // Renormalize the encoder.
76  template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool> = true>
77  std::tuple<State_T, Stream_IT> renorm(State_T x, Stream_IT iter, uint32_t freq, uint32_t scale_bits);
78 
79  // L ('l' in the paper) is the lower bound of our normalization interval.
80  // Between this and our byte-aligned emission, we use 31 (not 32!) bits.
81  // This is done intentionally because exact reciprocals for 31-bit uints
82  // fit in 32-bit uints: this permits some optimizations during encoding.
83  inline static constexpr State_T LOWER_BOUND = needs64Bit<State_T>() ? (1u << 31) : (1u << 23); // lower bound of our normalization interval
84 
85  inline static constexpr State_T STREAM_BITS = sizeof(Stream_T) * 8; // lower bound of our normalization interval
86 };

89 Encoder<State_T, Stream_T>::Encoder() : mState(LOWER_BOUND){};

91 template <typename State_T, typename Stream_T>
92 template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool>>
93 Stream_IT Encoder<State_T, Stream_T>::put(Stream_IT iter, uint32_t start, uint32_t freq, uint32_t scale_bits)
94 {
95  // renormalize
96  Stream_IT streamPos;
97  State_T x;
98  std::tie(x, streamPos) = renorm(mState, iter, freq, scale_bits);
99 
100  // x = C(s,x)
101  mState = ((x / freq) << scale_bits) + (x % freq) + start;
102  return streamPos;
103 };

105 template <typename State_T, typename Stream_T>
106 template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool>>
107 Stream_IT Encoder<State_T, Stream_T>::flush(Stream_IT iter)
108 {
109 
110  Stream_IT streamPos = iter;
111 
112  if constexpr (needs64Bit<State_T>()) {
113  ++streamPos;
114  *streamPos = static_cast<Stream_T>(mState >> 32);
115  ++streamPos;
116  *streamPos = static_cast<Stream_T>(mState >> 0);
117  assert(std::distance(iter, streamPos) == 2);
118  } else {
119  ++streamPos;
120  *streamPos = static_cast<Stream_T>(mState >> 24);
121  ++streamPos;
122  *streamPos = static_cast<Stream_T>(mState >> 16);
123  ++streamPos;
124  *streamPos = static_cast<Stream_T>(mState >> 8);
125  ++streamPos;
126  *streamPos = static_cast<Stream_T>(mState >> 0);
127  assert(std::distance(iter, streamPos) == 4);
128  }
129 
130  mState = 0;
131  return streamPos;
132 };

134 template <typename State_T, typename Stream_T>
135 template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool>>
136 Stream_IT Encoder<State_T, Stream_T>::putSymbol(Stream_IT iter, const EncoderSymbol<State_T>& sym, uint32_t scale_bits)
137 {
138 
139  assert(sym.freq != 0); // can't encode symbol with freq=0
140 
141  // renormalize
142  Stream_IT streamPos;
143  State_T x;
144  std::tie(x, streamPos) = renorm(mState, iter, sym.freq, scale_bits);
145 
146  // x = C(s,x)
147  State_T q = 0;
148 
149  if constexpr (needs64Bit<State_T>()) {
150  // This code needs support for 64-bit long multiplies with 128-bit result
151  // (or more precisely, the top 64 bits of a 128-bit result).
152  q = static_cast<State_T>((static_cast<uint128>(x) * sym.rcp_freq) >> 64);
153  } else {
154  q = static_cast<State_T>((static_cast<uint64_t>(x) * sym.rcp_freq) >> 32);
155  }
156  q = q >> sym.rcp_shift;
157 
158  mState = x + sym.bias + q * sym.cmpl_freq;
159  return streamPos;
160 };

162 template <typename State_T, typename Stream_T>
163 template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool>>
164 inline std::tuple<State_T, Stream_IT> Encoder<State_T, Stream_T>::renorm(State_T x, Stream_IT iter, uint32_t freq, uint32_t scale_bits)
165 {
166  Stream_IT streamPos = iter;
167 
168  State_T x_max = ((LOWER_BOUND >> scale_bits) << STREAM_BITS) * freq; // this turns into a shift.
169  if (x >= x_max) {
170  if constexpr (needs64Bit<State_T>()) {
171  ++streamPos;
172  *streamPos = static_cast<Stream_T>(x);
173  x >>= STREAM_BITS;
174  assert(x < x_max);
175  } else {
176  do {
177  ++streamPos;
178  *streamPos = static_cast<Stream_T>(x & 0xff);
179  x >>= STREAM_BITS;
180  } while (x >= x_max);
181  }
182  }
183  return std::make_tuple(x, streamPos);
184 };
