71 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
72 {
73  WorkflowSpec specs;
74 
75  // the lane configuration defines the subspecification ids to be distributed among the lanes.
76  auto nLanes = cfgc.options().get<int>("tof-lanes");
77  auto inputType = cfgc.options().get<std::string>("input-type");
78  auto outputType = cfgc.options().get<std::string>("output-type");
79 
80  bool writematching = 0;
81  bool writecalib = 0;
82 
83  if (outputType.rfind("matching-info") < outputType.size()) {
84  writematching = 1;
85  }
86  if (outputType.rfind("calib-info") < outputType.size()) {
87  writecalib = 1;
88  }
89 
90  bool clusterinput = 0;
91  bool trackinput = 0;
92  bool fitinput = 0;
93 
94  if (inputType.rfind("clusters") < inputType.size()) {
95  clusterinput = 1;
96  }
97  if (inputType.rfind("tracks") < inputType.size()) {
98  trackinput = 1;
99  }
100  auto useMC = !cfgc.options().get<bool>("disable-mc");
101  auto useCCDB = cfgc.options().get<bool>("use-ccdb");
102  auto useFIT = cfgc.options().get<bool>("use-fit");
103  bool disableRootInput = cfgc.options().get<bool>("disable-root-input");
104  bool disableRootOutput = cfgc.options().get<bool>("disable-root-output");
105 
106  if (inputType.rfind("fit") < inputType.size()) {
107  fitinput = 1;
108  useFIT = 1;
109  }
110 
111  LOG(INFO) << "TOF RECO WORKFLOW configuration";
112  LOG(INFO) << "TOF input = " << cfgc.options().get<std::string>("input-type");
113  LOG(INFO) << "TOF output = " << cfgc.options().get<std::string>("output-type");
114  LOG(INFO) << "TOF sectors = " << cfgc.options().get<std::string>("tof-sectors");
115  LOG(INFO) << "TOF disable-mc = " << cfgc.options().get<std::string>("disable-mc");
116  LOG(INFO) << "TOF lanes = " << cfgc.options().get<std::string>("tof-lanes");
117  LOG(INFO) << "TOF use-ccdb = " << cfgc.options().get<std::string>("use-ccdb");
118  LOG(INFO) << "TOF use-fit = " << cfgc.options().get<std::string>("use-fit");
119  LOG(INFO) << "TOF disable-root-input = " << disableRootInput;
120  LOG(INFO) << "TOF disable-root-output = " << disableRootOutput;
121 
122  if (!disableRootInput) { // input data loaded from root files
123  if (clusterinput) {
124  LOG(INFO) << "Insert TOF Cluster Reader";
125  specs.emplace_back(o2::tof::getClusterReaderSpec(useMC));
126  }
127  if (trackinput) {
128  LOG(INFO) << "Insert ITS-TPC Track Reader";
129  specs.emplace_back(o2::globaltracking::getTrackTPCITSReaderSpec(useMC));
130  }
131 
132  if (fitinput) {
133  LOG(INFO) << "Insert FIT RecPoint Reader";
134  specs.emplace_back(o2::ft0::getRecPointReaderSpec(useMC));
135  }
136  }
137  LOG(INFO) << "Insert TOF Matching";
138  specs.emplace_back(o2::tof::getTOFRecoWorkflowSpec(useMC, useFIT));
139 
140  if (!disableRootOutput) {
141  if (writematching) {
142  LOG(INFO) << "Insert TOF Matched Info Writer";
143  specs.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC));
144  }
145  if (writecalib) {
146  LOG(INFO) << "Insert TOF Calib Info Writer";
147  specs.emplace_back(o2::tof::getTOFCalibWriterSpec());
148  }
149  }
150  LOG(INFO) << "Number of active devices = " << specs.size();
151 
152  return std::move(specs);
153 }
