42 int PadPlane::getPadRowNumber(double z) const
43 {
44  //
45  // Finds the pad row number for a given z-position in local supermodule system
46  //
47 
48  int row = 0;
49  int nabove = 0;
50  int nbelow = 0;
51  int middle = 0;
52 
53  if ((z > getRow0()) || (z < getRowEnd())) {
54  row = -1;
55 
56  } else {
57  nabove = mNrows + 1;
58  nbelow = 0;
59  while (nabove - nbelow > 1) {
60  middle = (nabove + nbelow) / 2;
61  if (z == (mPadRow[middle - 1] + mPadRowSMOffset)) {
62  row = middle;
63  }
64  if (z > (mPadRow[middle - 1] + mPadRowSMOffset)) {
65  nabove = middle;
66  } else {
67  nbelow = middle;
68  }
69  }
70  row = nbelow - 1;
71  }
72 
73  return row;
74 }
