63  void process(soa::Filtered<soa::Join<aod::HfCandProng2, aod::HFSelD0Candidate>> const& candidates)
64  {
65  //Printf("Candidates: %d", candidates.size());
66  for (auto& candidate : candidates) {
67  if (!(candidate.hfflag() & 1 << D0ToPiK)) {
68  continue;
69  }
70  if (cutYCandMax >= 0. && std::abs(YD0(candidate)) > cutYCandMax) {
71  //Printf("Candidate: Y rejection: %g", YD0(candidate));
72  continue;
73  }
74  if (candidate.isSelD0() >= d_selectionFlagD0) {
75  registry.fill(HIST("hmass"), InvMassD0(candidate));
76  }
77  if (candidate.isSelD0bar() >= d_selectionFlagD0bar) {
78  registry.fill(HIST("hmass"), InvMassD0bar(candidate));
79  }
80  registry.fill(HIST("hptcand"), candidate.pt());
81  registry.fill(HIST("hptprong0"), candidate.ptProng0());
82  registry.fill(HIST("hptprong1"), candidate.ptProng1());
83  registry.fill(HIST("hdeclength"), candidate.decayLength());
84  registry.fill(HIST("hdeclengthxy"), candidate.decayLengthXY());
85  registry.fill(HIST("hd0Prong0"), candidate.impactParameter0());
86  registry.fill(HIST("hd0Prong1"), candidate.impactParameter1());
87  registry.fill(HIST("hd0d0"), candidate.impactParameterProduct());
88  registry.fill(HIST("hCTS"), CosThetaStarD0(candidate));
89  registry.fill(HIST("hCt"), CtD0(candidate));
90  registry.fill(HIST("hCPA"), candidate.cpa());
91  registry.fill(HIST("hEta"), candidate.eta());
92  registry.fill(HIST("hselectionstatus"), candidate.isSelD0() + (candidate.isSelD0bar() * 2));
93  registry.fill(HIST("hImpParErr"), candidate.errorImpactParameter0());
94  registry.fill(HIST("hImpParErr"), candidate.errorImpactParameter1());
95  registry.fill(HIST("hDecLenErr"), candidate.errorDecayLength());
96  registry.fill(HIST("hDecLenXYErr"), candidate.errorDecayLengthXY());
97  }
98  }
99 };

121  void process(soa::Filtered<soa::Join<aod::HfCandProng2, aod::HFSelD0Candidate, aod::HfCandProng2MCRec>> const& candidates,
122  soa::Join<aod::McParticles, aod::HfCandProng2MCGen> const& particlesMC, aod::BigTracksMC const& tracks)
123  {
124  // MC rec.
125  //Printf("MC Candidates: %d", candidates.size());
126  for (auto& candidate : candidates) {
127  if (!(candidate.hfflag() & 1 << D0ToPiK)) {
128  continue;
129  }
130  if (cutYCandMax >= 0. && std::abs(YD0(candidate)) > cutYCandMax) {
131  //Printf("MC Rec.: Y rejection: %g", YD0(candidate));
132  continue;
133  }
134  if (std::abs(candidate.flagMCMatchRec()) == 1 << D0ToPiK) {
135  // Get the corresponding MC particle.
136  auto indexMother = RecoDecay::getMother(particlesMC, candidate.index0_as<aod::BigTracksMC>().mcParticle_as<soa::Join<aod::McParticles, aod::HfCandProng2MCGen>>(), 421, true);
137  auto particleMother = particlesMC.iteratorAt(indexMother);
138  registry.fill(HIST("hPtGenSig"), particleMother.pt()); // gen. level pT
139  registry.fill(HIST("hPtRecSig"), candidate.pt()); // rec. level pT
140  registry.fill(HIST("hCPARecSig"), candidate.cpa());
141  registry.fill(HIST("hEtaRecSig"), candidate.eta());
142  } else {
143  registry.fill(HIST("hPtRecBg"), candidate.pt());
144  registry.fill(HIST("hCPARecBg"), candidate.cpa());
145  registry.fill(HIST("hEtaRecBg"), candidate.eta());
146  }
147  }
148  // MC gen.
149  //Printf("MC Particles: %d", particlesMC.size());
150  for (auto& particle : particlesMC) {
151  if (std::abs(particle.flagMCMatchGen()) == 1 << D0ToPiK) {
152  if (cutYCandMax >= 0. && std::abs(RecoDecay::Y(array{particle.px(), particle.py(), particle.pz()}, RecoDecay::getMassPDG(particle.pdgCode()))) > cutYCandMax) {
153  //Printf("MC Gen.: Y rejection: %g", RecoDecay::Y(array{particle.px(), particle.py(), particle.pz()}, RecoDecay::getMassPDG(particle.pdgCode())));
154  continue;
155  }
156  registry.fill(HIST("hPtGen"), particle.pt());
157  registry.fill(HIST("hEtaGen"), particle.eta());
158  }
159  }
160  }
