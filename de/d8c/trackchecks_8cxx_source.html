49 //function to convert eta to y
50 double eta2y(double pt, double m, double eta)
51 {
52  double mt = sqrt(m * m + pt * pt);
53  return asinh(pt / mt * sinh(eta));
54 }

106  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>::iterator const& col,
107  soa::Join<aod::Tracks, aod::TracksExtra, aod::McTrackLabels>& tracks, aod::McParticles& mcParticles)
108  {
109 
110  //event selection
111  if (!isMC && !col.alias()[kINT7]) { // trigger (should be skipped in MC)
112  return;
113  }
114  if (!col.sel7()) { //additional cuts
115  return;
116  }
117 
118  //Loop on tracks
119  for (auto& track : tracks) {
120  double y = -999.;
121  bool isPrimary = false;
122 
123  if (isMC) { //determine particle species base on MC truth and if it is primary or not
124  int pdgcode = track.mcParticle().pdgCode();
125 
126  if (MC::isPhysicalPrimary(mcParticles, track.mcParticle())) { //is primary?
127  isPrimary = true;
128  }
129 
130  //calculate rapidity
131  int pint = getparticleint(pdgcode);
132  if (pint == kNull) {
133  y = (double)kNull;
134  } else {
135  y = eta2y(track.pt(), mass[pint], track.eta());
136  }
137 
138  if (isPrimary && abs(y) < cfgCutY) {
139  //histograms with generated distribution (after event selection)
140  if (pdgcode == 11) {
141  hTrkPrimAftEvSel_truepid->Fill<kEl>(track.pt());
142  } else if (pdgcode == 211) {
143  hTrkPrimAftEvSel_truepid->Fill<kPi>(track.pt());
144  } else if (pdgcode == 321) {
145  hTrkPrimAftEvSel_truepid->Fill<kKa>(track.pt());
146  } else if (pdgcode == 2212) {
147  hTrkPrimAftEvSel_truepid->Fill<kPr>(track.pt());
148  } else if (pdgcode == 1000010020) {
149  hTrkPrimAftEvSel_truepid->Fill<kDe>(track.pt());
150  }
151 
152  hTrkPrimAftEvSel->Fill(track.pt()); //charged particles
153  }
154  }
155  } // end loop on tracks
156  } //end of process
157 }; //end struct TrackCheckTaskEvSel
