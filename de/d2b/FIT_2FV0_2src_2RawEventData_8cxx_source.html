58 void RawEventData::printHexEventHeader() const

62  ssheader << std::setw(3) << (0x00000fff & mEventHeader.bc) << " "
63  << std::setw(8) << (0xffffffff & mEventHeader.orbit) << " "
64  << std::setw(5) << (0x000fffff & mEventHeader.reservedField1) << " "
65  << std::setw(2) << (0x000000ff & mEventHeader.reservedField2) << " "
66  << std::setw(1) << (0x0000000f & mEventHeader.nGBTWords) << " "
67  << std::setw(1) << (0x0000000f & mEventHeader.startDescriptor) << " "
68  << std::setw(12) << (0xffffffffffff & mEventHeader.reservedField3);
69  LOG(DEBUG) << ssheader.str();
