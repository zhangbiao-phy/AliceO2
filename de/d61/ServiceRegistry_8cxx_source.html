31 void ServiceRegistry::registerService(hash_type typeHash, void* service, ServiceKind kind, uint64_t threadId, const char* name) const
32 {
33  hash_type id = typeHash & MAX_SERVICES_MASK;
34  hash_type threadHashId = (typeHash ^ threadId) & MAX_SERVICES_MASK;
35  // If kind is not stream, there is only one copy of our service.
36  // So we look if it is already registered and reused it if it is.
37  // If not, we register it as thread id 0 and as the passed one.
38  if (kind != ServiceKind::Stream && threadId != 0) {
39  void* oldService = this->get(typeHash, 0, kind);
40  if (oldService == nullptr) {
41  registerService(typeHash, service, kind, 0);
42  } else {
43  service = oldService;
44  }
45  }
46  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
47  // If the service slot was not taken, take it atomically
48  bool expected = false;
49  if (mServicesBooked[i + threadHashId].compare_exchange_strong(expected, true,
50  std::memory_order_seq_cst)) {
51  mServicesValue[i + threadHashId] = service;
52  mServicesMeta[i + threadHashId] = ServiceMeta{kind, threadId};
53  mServicesKey[i + threadHashId] = typeHash;
54  std::atomic_thread_fence(std::memory_order_release);
55  return;
56  }
57  }
58  throw std::runtime_error(std::string("Unable to find a spot in the registry for service ") +
59  std::to_string(typeHash) +
60  ". Make sure you use const / non-const correctly.");
61 }

63 void ServiceRegistry::declareService(ServiceSpec const& spec, DeviceState& state, fair::mq::ProgOptions& options)
64 {
65  mSpecs.push_back(spec);
66  // Services which are not stream must have a single instance created upfront.
67  if (spec.kind != ServiceKind::Stream) {
68  ServiceHandle handle = spec.init(*this, state, options);
69  this->registerService(handle.hash, handle.instance, handle.kind, 0, handle.name.c_str());
70  this->bindService(spec, handle.instance);
71  }
72 }

74 void ServiceRegistry::bindService(ServiceSpec const& spec, void* service)
75 {
76  static TracyLockableN(std::mutex, bindMutex, "bind mutex");
77  std::scoped_lock<LockableBase(std::mutex)> lock(bindMutex);
78  if (spec.preProcessing) {
79  mPreProcessingHandles.push_back(ServiceProcessingHandle{spec.preProcessing, service});
80  }
81  if (spec.postProcessing) {
82  mPostProcessingHandles.push_back(ServiceProcessingHandle{spec.postProcessing, service});
83  }
84  if (spec.preDangling) {
85  mPreDanglingHandles.push_back(ServiceDanglingHandle{spec.preDangling, service});
86  }
87  if (spec.postDangling) {
88  mPostDanglingHandles.push_back(ServiceDanglingHandle{spec.postDangling, service});
89  }
90  if (spec.preEOS) {
91  mPreEOSHandles.push_back(ServiceEOSHandle{spec.preEOS, service});
92  }
93  if (spec.postEOS) {
94  mPostEOSHandles.push_back(ServiceEOSHandle{spec.postEOS, service});
95  }
96  if (spec.postDispatching) {
97  mPostDispatchingHandles.push_back(ServiceDispatchingHandle{spec.postDispatching, service});
98  }
99  if (spec.exit) {
100  mPreExitHandles.push_back(ServiceExitHandle{spec.exit, service});
101  }
102 }
