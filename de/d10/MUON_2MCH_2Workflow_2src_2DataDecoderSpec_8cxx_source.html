78  //_________________________________________________________________________________________________
79  // the decodeTF() function processes the messages generated by the (sub)TimeFrame builder
80  void decodeTF(framework::ProcessingContext& pc)
81  {
82  // get the input buffer
83  auto& inputs = pc.inputs();
84  DPLRawParser parser(inputs, o2::framework::select(mInputSpec.c_str()));
85 
86  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
87  auto const* raw = it.raw();
88  if (!raw) {
89  continue;
90  }
91  size_t payloadSize = it.size();
92 
93  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(raw), sizeof(RDH) + payloadSize);
94  mDecoder->decodeBuffer(buffer);
95  }
96  }
97 
98  //_________________________________________________________________________________________________
99  // the decodeReadout() function processes the messages generated by o2-mch-cru-page-reader-workflow
100  void decodeReadout(const o2::framework::DataRef& input)
101  {
102  static int nFrame = 1;
103  // get the input buffer
104  if (input.spec->binding != "readout") {
105  return;
106  }
107 
108  const auto* header = o2::header::get<header::DataHeader*>(input.header);
109  if (!header) {
110  return;
111  }
112 
113  auto const* raw = input.payload;
114  // size of payload
115  size_t payloadSize = header->payloadSize;
116 
117  if (mDebug) {
118  std::cout << nFrame << " payloadSize=" << payloadSize << std::endl;
119  }
120  nFrame += 1;
121  if (payloadSize == 0) {
122  return;
123  }
124 
125  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(raw), payloadSize);
126  mDecoder->decodeBuffer(buffer);
127  }

129  //_________________________________________________________________________________________________
130  void run(framework::ProcessingContext& pc)
131  {
132  auto createBuffer = [&](auto& vec, size_t& size) {
133  size = vec.empty() ? 0 : sizeof(*(vec.begin())) * vec.size();
134  char* buf = nullptr;
135  if (size > 0) {
136  buf = (char*)malloc(size);
137  if (buf) {
138  char* p = buf;
139  size_t sizeofElement = sizeof(*(vec.begin()));
140  for (auto& element : vec) {
141  memcpy(p, &element, sizeofElement);
142  p += sizeofElement;
143  }
144  }
145  }
146  return buf;
147  };
148 
149  mDecoder->reset();
150  decodeTF(pc);
151  for (auto&& input : pc.inputs()) {
152  if (input.spec->binding == "readout") {
153  decodeReadout(input);
154  }
155  }
156 
157  auto& digits = mDecoder->getOutputDigits();
158  auto& orbits = mDecoder->getOrbits();
159 
160  if (mDebug) {
161  for (auto d : digits) {
162  std::cout << " DE# " << d.getDetID() << " PadId " << d.getPadID() << " ADC " << d.getADC() << " time " << d.getTime().sampaTime << std::endl;
163  }
164  }
165  // send the output buffer via DPL
166  size_t digitsSize, orbitsSize;
167  char* digitsBuffer = createBuffer(digits, digitsSize);
168  char* orbitsBuffer = createBuffer(orbits, orbitsSize);
169 
170  // create the output message
171  auto freefct = [](void* data, void*) { free(data); };
172  pc.outputs().adoptChunk(Output{"MCH", "DIGITS", 0}, digitsBuffer, digitsSize, freefct, nullptr);
173  pc.outputs().adoptChunk(Output{"MCH", "ORBITS", 0}, orbitsBuffer, orbitsSize, freefct, nullptr);
174  }
175 
176  private:
177  std::string mInputSpec;
178  bool mDebug = {false};
179  DataDecoder* mDecoder = {nullptr};
180 };
