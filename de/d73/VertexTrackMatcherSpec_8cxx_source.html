39 void VertexTrackMatcherSpec::run(ProcessingContext& pc)
40 {
41  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
42  mTimer.Start(false);
43 
44  o2::globaltracking::RecoContainer recoData;
45 
46  // RS FIXME this will not have effect until the 1st orbit is propagated, until that will work only for TF starting at orbit 0
47  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().getByPos(0).header);
48  mMatcher.setStartIR({0, dh->firstTForbit});
49  recoData.collectData(pc, dataRequestV2T);
50 
51  const auto vertices = pc.inputs().get<gsl::span<o2::dataformats::PrimaryVertex>>("vertices");
52  const auto vtxTracks = pc.inputs().get<gsl::span<o2::dataformats::VtxTrackIndex>>("vtxTracks");
53  const auto vtxTrackRefs = pc.inputs().get<gsl::span<o2::dataformats::VtxTrackRef>>("vtxTrackRefs");
54 
55  std::vector<o2::dataformats::VtxTrackIndex> trackIndex;
56  std::vector<o2::dataformats::VtxTrackRef> vtxRefs;
57 
58  mMatcher.process(vertices, vtxTracks, vtxTrackRefs, recoData, trackIndex, vtxRefs);
59 
60  pc.outputs().snapshot(Output{"GLO", "PVTX_TRMTC", 0, Lifetime::Timeframe}, trackIndex);
61  pc.outputs().snapshot(Output{"GLO", "PVTX_TRMTCREFS", 0, Lifetime::Timeframe}, vtxRefs);
62 
63  mTimer.Stop();
64  LOG(INFO) << "Made " << trackIndex.size() << " track associationgs for " << vertices.size() << " vertices, timing: CPU: "
65  << mTimer.CpuTime() - timeCPU0 << " Real: " << mTimer.RealTime() - timeReal0 << " s";
66 }

74 DataProcessorSpec getVertexTrackMatcherSpec(GTrackID::mask_t src)
75 {
76  std::vector<OutputSpec> outputs;
77 
78  if (src[GTrackID::ITS]) {
79  dataRequestV2T.requestITSTracks(false);
80  }
81  if (src[GTrackID::TPC]) {
82  dataRequestV2T.requestTPCTracks(false);
83  }
84  if (src[GTrackID::ITSTPC] || src[GTrackID::ITSTPCTOF]) { // ITSTPCTOF does not provide tracks, only matchInfo
85  dataRequestV2T.requestITSTPCTracks(false);
86  }
87  if (src[GTrackID::ITSTPCTOF]) {
88  dataRequestV2T.requestTOFMatches(false);
89  dataRequestV2T.requestTOFClusters(false);
90  }
91  if (src[GTrackID::TPCTOF]) {
92  dataRequestV2T.requestTPCTOFTracks(false);
93  }
94 
95  auto& inputs = dataRequestV2T.inputs;
96  inputs.emplace_back("vertices", "GLO", "PVTX", 0, Lifetime::Timeframe);
97  inputs.emplace_back("vtxTracks", "GLO", "PVTX_CONTID", 0, Lifetime::Timeframe);
98  inputs.emplace_back("vtxTrackRefs", "GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe);
99 
100  outputs.emplace_back("GLO", "PVTX_TRMTC", 0, Lifetime::Timeframe);
101  outputs.emplace_back("GLO", "PVTX_TRMTCREFS", 0, Lifetime::Timeframe);
102 
103  return DataProcessorSpec{
104  "pvertex-track-matching",
105  inputs,
106  outputs,
107  AlgorithmSpec{adaptFromTask<VertexTrackMatcherSpec>()},
108  Options{}};
109 }
