199 template <typename S, typename T>
200 struct variant_helper {
201  static void set(S* store, T value)
202  {
203  new (reinterpret_cast<T*>(store)) T{};
204  *(reinterpret_cast<T*>(store)) = value;
205  }
206  static void set(S* store, T values, size_t size)
207  {
208  *reinterpret_cast<T*>(store) = reinterpret_cast<T>(std::memcpy(std::malloc(size * sizeof(std::remove_pointer_t<T>)), reinterpret_cast<void*>(values), size * sizeof(std::remove_pointer_t<T>)));
209  }
210 
211  static T get(const S* store) { return *(reinterpret_cast<const T*>(store)); }
212 };
213 
214 template <typename S>
215 struct variant_helper<S, const char*> {
216  static const char* get(const S* store) { return *reinterpret_cast<const char* const*>(store); }
217 
218  static void set(S* store, const char* value) { *reinterpret_cast<char**>(store) = strdup(value); }
219 };

243  public:
244  Variant(VariantType type = VariantType::Unknown) : mType{type}, mSize{1} {}
245 
246  template <typename T>
247  Variant(T value) : mType{variant_trait_v<T>}, mSize{1}
248  {
249  variant_helper<storage_t, decltype(value)>::set(&mStore, value);
250  }

252  template <typename T>
253  Variant(T values, size_t size) : mType{variant_trait_v<T>}, mSize{size}

256  }
257 
258  template <typename T>
259  Variant(std::vector<T>& values) : mType{variant_trait_v<T*>}, mSize{values.size()}
260  {

262  }
263 
264  template <typename T>
265  Variant(std::initializer_list<T>) : mType{VariantType::Unknown}, mSize{1}

267  static_assert(sizeof(T) == 0,
268  "brace-enclosed initializer list forbidden for Variant"
269  "\n did you accidentally put braces around the default value?");

272  Variant(const Variant& other) : mType(other.mType)
273  {
274  // In case this is an array we need to duplicate it to avoid
275  // double deletion.
276  switch (mType) {
277  case variant_trait_v<const char*>:
278  mSize = other.mSize;
279  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
280  return;
281  case variant_trait_v<int*>:
282  mSize = other.mSize;
283  variant_helper<storage_t, int*>::set(&mStore, other.get<int*>(), mSize);
284  return;
285  case variant_trait_v<float*>:
286  mSize = other.mSize;
287  variant_helper<storage_t, float*>::set(&mStore, other.get<float*>(), mSize);
288  return;
289  case variant_trait_v<double*>:
290  mSize = other.mSize;
291  variant_helper<storage_t, double*>::set(&mStore, other.get<double*>(), mSize);
292  return;
293  case variant_trait_v<bool*>:
294  mSize = other.mSize;
295  variant_helper<storage_t, bool*>::set(&mStore, other.get<bool*>(), mSize);
296  return;
297  case variant_trait_v<std::string*>:
298  mSize = other.mSize;
299  variant_helper<storage_t, std::string*>::set(&mStore, other.get<std::string*>(), mSize);
300  return;
301  default:
302  mStore = other.mStore;
303  mSize = other.mSize;
304  }
305  }
306 
307  Variant(Variant&& other) : mType(other.mType)

317  return;

320  return;
321  case variant_trait_v<double*>:
322  *reinterpret_cast<double**>(&(other.mStore)) = nullptr;
323  return;
324  case variant_trait_v<bool*>:
325  *reinterpret_cast<bool**>(&(other.mStore)) = nullptr;
326  return;
327  case variant_trait_v<std::string*>:
328  *reinterpret_cast<std::string**>(&(other.mStore)) = nullptr;
329  default:
330  return;
331  }

334  ~Variant()
335  {
336  // In case we allocated an array, we
337  // should delete it.
338  switch (mType) {
339  case variant_trait_v<const char*>:
340  case variant_trait_v<int*>:
341  case variant_trait_v<float*>:
342  case variant_trait_v<double*>:
343  case variant_trait_v<bool*>:
344  case variant_trait_v<std::string*>:
345  if (reinterpret_cast<void**>(&mStore) != nullptr) {
346  free(*reinterpret_cast<void**>(&mStore));
347  }
348  return;
349  default:
350  return;
351  }
352  }
