98 template <typename H, typename... T>
99 constexpr auto make_originals_from_type()
100 {
101  using decayed = std::decay_t<H>;
102  if constexpr (sizeof...(T) == 0) {
103  if constexpr (is_type_with_originals_v<decayed>) {
104  return typename decayed::originals{};
105  } else if constexpr (is_type_with_originals_v<typename decayed::table_t>) {
106  return typename decayed::table_t::originals{};
107  } else if constexpr (is_type_with_parent_v<decayed>) {
108  return make_originals_from_type<typename decayed::parent_t>();
109  } else {
110  return framework::pack<decayed>{};
111  }
112  } else if constexpr (is_type_with_originals_v<decayed>) {
113  return framework::concatenate_pack(typename decayed::originals{}, make_originals_from_type<T...>());
114  } else if constexpr (is_type_with_originals_v<typename decayed::table_t>) {
115  return framework::concatenate_pack(typename decayed::table_t::originals{}, make_originals_from_type<T...>());
116  } else {
117  return framework::concatenate_pack(framework::pack<decayed>{}, make_originals_from_type<T...>());
118  }

138 template <typename T, typename ChunkingPolicy = Chunked>
139 class ColumnIterator : ChunkingPolicy
140 {
141  static constexpr char SCALE_FACTOR = std::is_same_v<std::decay_t<T>, bool> ? 3 : 0;
142 
143  public:
148  ColumnIterator(arrow::ChunkedArray const* column)
149  : mColumn{column},
150  mCurrentPos{nullptr},
151  mFirstIndex{0},
152  mCurrentChunk{0}
153  {
154  auto array = getCurrentArray();
155  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset();
156  mLast = mCurrent + array->length();
157  }
158 
159  ColumnIterator() = default;
160  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
161  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
162 
163  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
164  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
165 
167  void nextChunk() const
168  {
169  auto previousArray = getCurrentArray();
170  mFirstIndex += previousArray->length();
171 
172  mCurrentChunk++;
173  auto array = getCurrentArray();
174  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
175  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
176  }
177 
178  void prevChunk() const
179  {
180  auto previousArray = getCurrentArray();
181  mFirstIndex -= previousArray->length();
182 
183  mCurrentChunk--;
184  auto array = getCurrentArray();
185  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
186  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
187  }
188 
189  void moveToChunk(int chunk)
190  {
191  if (mCurrentChunk < chunk) {
192  while (mCurrentChunk != chunk) {
193  nextChunk();
194  }
195  } else {
196  while (mCurrentChunk != chunk) {
197  prevChunk();
198  }
199  }
200  }
201 
203  void moveToEnd()
204  {
205  mCurrentChunk = mColumn->num_chunks() - 1;
206  auto array = getCurrentArray();
207  mFirstIndex = mColumn->length() - array->length();
208  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
209  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
210  }
211 
212  decltype(auto) operator*() const
213  {
214  if constexpr (ChunkingPolicy::chunked) {
215  if (O2_BUILTIN_UNLIKELY(((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast))) {
216  nextChunk();
217  }
218  }
219  if constexpr (std::is_same_v<bool, std::decay_t<T>>) {
220  // FIXME: notice that due to the type punning we cannot simply convert the
221  // masked char to a bool, because it's undefined behavior.
222  // FIXME: check if shifting the masked bit to the first position is better than != 0
223  return (*((char*)mCurrent + (*mCurrentPos >> SCALE_FACTOR)) & (1 << (*mCurrentPos & 0x7))) != 0;
224  } else {
225  return *(mCurrent + (*mCurrentPos >> SCALE_FACTOR));
226  }
227  }
228 
229  // Move to the chunk which containts element pos
230  ColumnIterator<T>& moveToPos()
231  {
232  // If we get outside range of the current chunk, go to the next.
233  if constexpr (ChunkingPolicy::chunked) {
234  while (O2_BUILTIN_UNLIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast)) {
235  nextChunk();
236  }
237  }
238  return *this;
239  }
240 
241  // Move to the chunk which containts element pos
242  ColumnIterator<T>& checkNextChunk()
243  {
244  if constexpr (ChunkingPolicy::chunked) {
245  if (O2_BUILTIN_LIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) <= mLast)) {
246  return *this;
247  }
248  nextChunk();
249  }
250  return *this;
251  }
252 
253  mutable T const* mCurrent;
254  int64_t const* mCurrentPos;
255  mutable T const* mLast;
256  arrow::ChunkedArray const* mColumn;
257  mutable int mFirstIndex;
258  mutable int mCurrentChunk;
259 
260  private:
262  auto getCurrentArray() const
263  {
264  std::shared_ptr<arrow::Array> chunkToUse = mColumn->chunk(mCurrentChunk);
265  if constexpr (std::is_same_v<arrow_array_for_t<T>, arrow::FixedSizeListArray>) {
266  chunkToUse = std::dynamic_pointer_cast<arrow::FixedSizeListArray>(chunkToUse)->values();
267  return std::static_pointer_cast<arrow_array_for_t<value_for_t<T>>>(chunkToUse);
268  } else {
269  return std::static_pointer_cast<arrow_array_for_t<T>>(chunkToUse);
270  }
271  }
272 };

870 auto getSliceFor(int value, char const* key, std::shared_ptr<arrow::Table> const& input, std::shared_ptr<arrow::Table>& output, uint64_t& offset)
871 {
872  arrow::Datum value_counts;
873  auto options = arrow::compute::CountOptions::Defaults();
874  ARROW_ASSIGN_OR_RAISE(value_counts,
875  arrow::compute::CallFunction("value_counts", {input->GetColumnByName(key)},
876  &options));
877  auto pair = static_cast<arrow::StructArray>(value_counts.array());
878  auto values = static_cast<arrow::NumericArray<arrow::Int32Type>>(pair.field(0)->data());
879  auto counts = static_cast<arrow::NumericArray<arrow::Int64Type>>(pair.field(1)->data());
880 
881  int slice;
882  for (slice = 0; slice < values.length(); ++slice) {
883  if (values.Value(slice) == value) {
884  offset = slice;
885  output = input->Slice(slice, counts.Value(slice));
886  return arrow::Status::OK();
887  }
888  }
889  output = input->Slice(0, 0);
890  return arrow::Status::OK();
891 }
