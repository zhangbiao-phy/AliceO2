22 bool readFromTree(TTree& tree, const std::string brname, T& dest, int ev = 0)
23 {
24  auto* br = tree.GetBranch(brname.c_str());
25  if (br && br->GetEntries() > ev) {
26  auto* ptr = &dest;
27  br->SetAddress(&ptr);
28  br->GetEntry(ev);
29  br->ResetAddress();
30  return true;
31  }
32  return false;
33 }

35 std::unique_ptr<TFile> CTFCoderBase::loadDictionaryTreeFile(const std::string& dictPath, bool mayFail)
36 {
37  TDirectory* curd = gDirectory;
38  std::unique_ptr<TFile> fileDict(gSystem->AccessPathName(dictPath.c_str()) ? nullptr : TFile::Open(dictPath.c_str()));
39  if (!fileDict || fileDict->IsZombie()) {
40  if (mayFail) {
41  LOG(INFO) << "CTF dictionary file " << dictPath << " for detector " << mDet.getName() << " is absent, will use dictionaries stored in CTF";
42  fileDict.reset();
43  return std::move(fileDict);
44  }
45  LOG(ERROR) << "Failed to open CTF dictionary file " << dictPath << " for detector " << mDet.getName();
46  throw std::runtime_error("Failed to open dictionary file");
47  }
48  auto tnm = std::string(o2::base::NameConf::CTFDICT);
49  std::unique_ptr<TTree> tree((TTree*)fileDict->Get(tnm.c_str()));
50  if (!tree) {
51  fileDict.reset();
52  LOG(ERROR) << "Did not find CTF dictionary tree " << tnm << " in " << dictPath;
53  throw std::runtime_error("Did not fine CTF dictionary tree in the file");
54  }
55  CTFHeader ctfHeader;
56  if (!readFromTree(*tree.get(), "CTFHeader", ctfHeader) || !ctfHeader.detectors[mDet]) {
57  tree.reset();
58  fileDict.reset();
59  LOG(ERROR) << "Did not find CTF dictionary header or Detector " << mDet.getName() << " in it";
60  if (!mayFail) {
61  throw std::runtime_error("did not find CTFHeader with needed detector");
62  }
63  } else {
64  LOG(INFO) << "Found CTF dictionary for " << mDet.getName() << " in " << dictPath;
65  }
66  return fileDict;
67 }
