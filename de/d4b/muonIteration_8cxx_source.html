18 // This task shows how to access the Muons belong to a collision
19 // The association is made through the BC column (and in Run 3 may not be unique!)
20 //
21 // Note that one has to subscribe to aod::Collisions const& to load
22 // the relevant data even if you access the data itself through m.collision()
23 // This uses the exclusive matcher, so you only get BCs which have a collision
24 // If you want also BCs without collision, see the example IterateMuonsSparse below
25 struct IterateMuons {
26  void process(aod::MatchedBCCollisionsExclusive::iterator const& m, aod::Collisions const&, aod::Muons const& muons)
27  {
28  LOGF(INFO, "Vertex = %f has %d muons", m.collision().posZ(), muons.size());
29  for (auto& muon : muons) {
30  LOGF(info, " pT = %.2f", muon.pt());
31  }
32  }
33 };
34 
35 // This uses the sparase matcher, so you also get BCs without a collision.
36 // You need to check with m.has_collision()
37 struct IterateMuonsSparse {
38  void process(aod::MatchedBCCollisionsSparse::iterator const& m, aod::Collisions const&, aod::Muons const& muons)
39  {
40  if (m.has_collision()) {
41  LOGF(INFO, "Vertex = %f has %d muons", m.collision().posZ(), muons.size());
42  } else {
43  LOGF(INFO, "BC without collision has %d muons", muons.size());
44  }
45  for (auto& muon : muons) {
46  LOGF(info, " pT = %.2f", muon.pt());
47  }
48  }
49 };
