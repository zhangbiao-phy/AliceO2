52 class TrackAtVertexTask
53 {
54  public:
55  //_________________________________________________________________________________________________
56  void init(framework::InitContext& ic)
57  {
59 
60  LOG(INFO) << "initializing track extrapolation to vertex";
61 
62  if (!gGeoManager) {
63  o2::base::GeometryManager::loadGeometry("O2geometry.root");
64  if (!gGeoManager) {
65  throw std::runtime_error("cannot load the geometry");
66  }
67  }
68 
69  if (!TGeoGlobalMagField::Instance()->GetField()) {
70  auto l3Current = ic.options().get<float>("l3Current");
71  auto dipoleCurrent = ic.options().get<float>("dipoleCurrent");
72  auto field = o2::field::MagneticField::createFieldMap(l3Current, dipoleCurrent, o2::field::MagneticField::kConvLHC, false,
73  3500., "A-A", "$(O2_ROOT)/share/Common/maps/mfchebKGI_sym.root");
74  TGeoGlobalMagField::Instance()->SetField(field);
75  TGeoGlobalMagField::Instance()->Lock();
76  TrackExtrap::setField();
77  }
78 
79  auto stop = [this]() {
80  LOG(INFO) << "track propagation to vertex duration = " << mElapsedTime.count() << " s";
81  };
82  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, stop);
83  }
84 
85  //_________________________________________________________________________________________________
86  void run(framework::ProcessingContext& pc)
87  {
89 
90  // get the ROFs, tracks and vertices
91  auto rofs = pc.inputs().get<gsl::span<ROFRecord>>("rofs");
92  auto tracks = pc.inputs().get<gsl::span<TrackMCH>>("tracks");
93  auto vertices = pc.inputs().get<gsl::span<math_utils::Point3D<double>>>("vertices");
94 
95  if (vertices.size() != rofs.size()) {
96  throw length_error("number of vertices different from number of events");
97  }
98 
99  // for each event, propagate the tracks to the vertex
100  mTracksAtVtx.clear();
101  int iVertex(-1);
102  int nTracksTot(0);
103  for (const auto& rof : rofs) {
104  auto tStart = std::chrono::high_resolution_clock::now();
105  nTracksTot += extrapTracksToVertex(tracks.subspan(rof.getFirstIdx(), rof.getNEntries()), vertices[++iVertex]);
106  auto tEnd = std::chrono::high_resolution_clock::now();
107  mElapsedTime += tEnd - tStart;
108  }
109 
110  // create the output message
111  auto msgOut = pc.outputs().make<char>(Output{"MCH", "TRACKSATVERTEX", 0, Lifetime::Timeframe},
112  mTracksAtVtx.size() * sizeof(int) + nTracksTot * sizeof(TrackAtVtxStruct));
113 
114  // write the tracks
115  writeTracks(msgOut.data());
116  }
117 
118  private:
119  struct TrackAtVtxStruct {
120  TrackParamStruct paramAtVertex{};
121  double dca = 0.;
122  double rAbs = 0.;
123  int mchTrackIdx = 0;
124  };
125 
126  //_________________________________________________________________________________________________
127  int extrapTracksToVertex(gsl::span<const TrackMCH> tracks, const math_utils::Point3D<double>& vertex)
128  {
131 
132  auto& tracksAtVtx = mTracksAtVtx.emplace_back();
133  int trackIdx(-1);
134 
135  for (const auto& track : tracks) {
136 
137  // create a new track at vertex pointing to the current track (index within the current event)
138  auto& trackAtVtx = tracksAtVtx.emplace_back();
139  trackAtVtx.mchTrackIdx = ++trackIdx;
140 
141  // extrapolate to vertex
142  TrackParam trackParamAtVertex(track.getZ(), track.getParameters());
143  if (!TrackExtrap::extrapToVertex(&trackParamAtVertex, vertex.x(), vertex.y(), vertex.z(), 0., 0.)) {
144  tracksAtVtx.pop_back();
145  continue;
146  }
147  trackAtVtx.paramAtVertex.x = trackParamAtVertex.getNonBendingCoor();
148  trackAtVtx.paramAtVertex.y = trackParamAtVertex.getBendingCoor();
149  trackAtVtx.paramAtVertex.z = trackParamAtVertex.getZ();
150  trackAtVtx.paramAtVertex.px = trackParamAtVertex.px();
151  trackAtVtx.paramAtVertex.py = trackParamAtVertex.py();
152  trackAtVtx.paramAtVertex.pz = trackParamAtVertex.pz();
153  trackAtVtx.paramAtVertex.sign = trackParamAtVertex.getCharge();
154 
155  // extrapolate to DCA
156  TrackParam trackParamAtDCA(track.getZ(), track.getParameters());
157  if (!TrackExtrap::extrapToVertexWithoutBranson(&trackParamAtDCA, vertex.z())) {
158  tracksAtVtx.pop_back();
159  continue;
160  }
161  double dcaX = trackParamAtDCA.getNonBendingCoor() - vertex.x();
162  double dcaY = trackParamAtDCA.getBendingCoor() - vertex.y();
163  trackAtVtx.dca = TMath::Sqrt(dcaX * dcaX + dcaY * dcaY);
164 
165  // extrapolate to the end of the absorber
166  TrackParam trackParamAtRAbs(track.getZ(), track.getParameters());
167  if (!TrackExtrap::extrapToZ(&trackParamAtRAbs, -505.)) {
168  tracksAtVtx.pop_back();
169  continue;
170  }
171  double xAbs = trackParamAtRAbs.getNonBendingCoor();
172  double yAbs = trackParamAtRAbs.getBendingCoor();
173  trackAtVtx.rAbs = TMath::Sqrt(xAbs * xAbs + yAbs * yAbs);
174  }
175 
176  return tracksAtVtx.size();
177  }
178 
179  //_________________________________________________________________________________________________
180  void writeTracks(char* bufferPtr) const
181  {
183 
184  for (const auto& tracksAtVtx : mTracksAtVtx) {
185 
186  // write the number of tracks
187  int nTracks = tracksAtVtx.size();
188  memcpy(bufferPtr, &nTracks, sizeof(int));
189  bufferPtr += sizeof(int);
190 
191  // write the tracks
192  if (nTracks > 0) {
193  memcpy(bufferPtr, tracksAtVtx.data(), nTracks * sizeof(TrackAtVtxStruct));
194  bufferPtr += nTracks * sizeof(TrackAtVtxStruct);
195  }
196  }
197  }
198 
199  std::vector<std::vector<TrackAtVtxStruct>> mTracksAtVtx{};
200  std::chrono::duration<double> mElapsedTime{};
201 };
