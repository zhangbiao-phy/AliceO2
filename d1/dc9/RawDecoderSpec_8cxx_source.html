49  void run(of::ProcessingContext& pc)
50  {
51  auto tStart = std::chrono::high_resolution_clock::now();
52 
53  auto tAlgoStart = std::chrono::high_resolution_clock::now();
54  of::DPLRawParser parser(pc.inputs());
55 
56  if (!mDecoder) {
57  auto const* rdhPtr = reinterpret_cast<const o2::header::RDHAny*>(parser.begin().raw());
58  mDecoder = createDecoder(*rdhPtr, mIsDebugMode, mElectronicsDelay, mCrateMasks, mFeeIdConfig);
59  }
60 
61  mDecoder->clear();
62  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
63  auto const* rdhPtr = reinterpret_cast<const o2::header::RDHAny*>(it.raw());
64  gsl::span<const uint8_t> payload(it.data(), it.size());
65  mDecoder->process(payload, *rdhPtr);
66  }
67 
68  mTimerAlgo += std::chrono::high_resolution_clock::now() - tAlgoStart;
69 
70  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DECODED", mSubSpec}, mDecoder->getData());
71  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DECODEDROF", mSubSpec}, mDecoder->getROFRecords());
72 
73  mTimer += std::chrono::high_resolution_clock::now() - tStart;
74  mNROFs += mDecoder->getROFRecords().size();
75  }
76 
77  private:
78  std::unique_ptr<Decoder> mDecoder{nullptr};
79  bool mIsDebugMode{false};
80  FEEIdConfig mFeeIdConfig{};
81  CrateMasks mCrateMasks{};
82  ElectronicsDelay mElectronicsDelay{};
