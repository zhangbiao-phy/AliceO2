77 template <typename T, typename U, typename V>
78 auto getRotatedCovMatrixXX(const T& matrix, U phi, V theta)
79 {
80  auto cp = std::cos(phi);
81  auto sp = std::sin(phi);
82  auto ct = std::cos(theta);
83  auto st = std::sin(theta);
84  return matrix[0] * cp * cp * ct * ct // covXX
85  + matrix[1] * 2. * cp * sp * ct * ct // covXY
86  + matrix[2] * sp * sp * ct * ct // covYY
87  + matrix[3] * 2. * cp * ct * st // covXZ
88  + matrix[4] * 2. * sp * ct * st // covYZ
89  + matrix[5] * st * st; // covZZ
90 }
