35  void process(soa::Join<aod::Collisions, aod::EvSels>::iterator const& collision, soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra, aod::TrackSelection>> const& tracks)
36  {
37  bool checkV0 = collision.bbV0A() || collision.bbV0C() || collision.bgV0A() || collision.bgV0C();
38  if (checkV0) {
39  return;
40  }
41  bool checkFDD = collision.bbFDA() || collision.bbFDC() || collision.bgFDA() || collision.bgFDC();
42  if (checkFDD) {
43  return;
44  }
45  if (!collision.alias()[kCUP9]) {
46  return;
47  }
48  if (tracks.size() != 2) {
49  return;
50  }
51  auto track1 = tracks.begin();
52  auto track2 = track1 + 1;
53  if (track1.charge() * track2.charge() >= 0) {
54  return;
55  }
56  UChar_t clustermap1 = track1.itsClusterMap();
57  UChar_t clustermap2 = track2.itsClusterMap();
58  bool checkClusMap = TESTBIT(clustermap1, 0) && TESTBIT(clustermap1, 1) && TESTBIT(clustermap2, 0) && TESTBIT(clustermap2, 1);
59  if (!checkClusMap) {
60  return;
61  }
62  /*
63  if ((p.Pt() >= 0.1) || (signalTPC1 + signalTPC2 > 140.)) {
64  return;
65  }
66  */
67  outputTracks(track1.pt(), track1.eta(), track1.phi(), track1.tpcSignal(),
68  track2.pt(), track2.eta(), track2.phi(), track2.tpcSignal());
69  }
70 };
