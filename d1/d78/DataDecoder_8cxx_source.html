68 static void patchPage(gsl::span<const std::byte> rdhBuffer, bool verbose)
69 {
70  static int mNrdhs = 0;
71  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(rdhBuffer[0])));
72  mNrdhs++;
73 
74  auto cruId = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF;
75  auto flags = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF00;
76  auto endpoint = o2::raw::RDHUtils::getEndPointID(rdhAny);
77  uint32_t feeId = cruId * 2 + endpoint + flags;
78  o2::raw::RDHUtils::setFEEID(rdhAny, feeId);
79 
80  if (verbose) {
81  std::cout << mNrdhs << "--\n";
82  o2::raw::RDHUtils::printRDH(rdhAny);
83  }
84 };

106 void DataDecoder::decodeBuffer(gsl::span<const std::byte> page)
107 {
108  size_t ndigits{0};
109 
110  uint32_t orbit;
111 
112  auto channelHandler = [&](DsElecId dsElecId, uint8_t channel, o2::mch::raw::SampaCluster sc) {
113  if (mChannelHandler) {
114  mChannelHandler(dsElecId, channel, sc);
115  }
116 
117  if (mDs2manu) {
118  channel = ds2manu(int(channel));
119  }
120 
121  uint32_t digitadc = sc.sum();
122 
123  int deId{-1};
124  int dsIddet{-1};
125  if (auto opt = mElec2Det(dsElecId); opt.has_value()) {
126  DsDetId dsDetId = opt.value();
127  dsIddet = dsDetId.dsId();
128  deId = dsDetId.deId();
129  }
130  if (mDebug) {
131  auto s = asString(dsElecId);
132  auto ch = fmt::format("{}-CH{:02d}", s, channel);
133  std::cout << ch << " "
134  << "deId " << deId << " dsIddet " << dsIddet << std::endl;
135  }
136 
137  if (deId < 0 || dsIddet < 0 || !isValidDeID(deId)) {
138  return;
139  }
140 
141  int padId = -1;
142  const Segmentation& segment = segmentation(deId);
143 
144  padId = segment.findPadByFEE(dsIddet, int(channel));
145  if (mDebug) {
146  auto s = asString(dsElecId);
147  auto ch = fmt::format("{}-CH{:02d}", s, channel);
148  std::cout << ch << " "
149  << fmt::format("PAD ({:04d} {:04d} {:04d})\tADC {:06d} TIME ({} {} {:02d}) SIZE {} END {}",
150  deId, dsIddet, padId, digitadc, orbit, sc.bunchCrossing, sc.sampaTime, sc.nofSamples(), (sc.sampaTime + sc.nofSamples() - 1))
151  << (((sc.sampaTime + sc.nofSamples() - 1) >= 98) ? " *" : "") << std::endl;
152  }
153 
154  // skip channels not associated to any pad
155  if (padId < 0) {
156  return;
157  }
158 
159  Digit::Time time;
160  time.sampaTime = sc.sampaTime;
161  time.bunchCrossing = sc.bunchCrossing;
162  time.orbit = orbit;
163 
164  mOutputDigits.emplace_back(o2::mch::Digit(deId, padId, digitadc, time, sc.nofSamples()));
165 
166  if (mDebug) {
167  std::cout << "DIGIT STORED:\nADC " << mOutputDigits.back().getADC() << " DE# " << mOutputDigits.back().getDetID() << " PadId " << mOutputDigits.back().getPadID() << " time " << mOutputDigits.back().getTime().sampaTime << std::endl;
168  }
169  ++ndigits;
170  };
171 
172  const auto dumpDigits = [&](bool bending) {
173  for (auto d : mOutputDigits) {
174  if (d.getPadID() < 0) {
175  continue;
176  }
177  const Segmentation& segment = segmentation(d.getDetID());
178  bool bend = segment.isBendingPad(d.getPadID());
179  if (bending != segment.isBendingPad(d.getPadID())) {
180  continue;
181  }
182  float X = segment.padPositionX(d.getPadID());
183  float Y = segment.padPositionY(d.getPadID());
184  std::cout << fmt::format(" DE {:4d} PAD {:5d} ADC {:6d} TIME ({} {} {:4d})",
185  d.getDetID(), d.getPadID(), d.getADC(), d.getTime().orbit, d.getTime().bunchCrossing, d.getTime().sampaTime);
186  std::cout << fmt::format("\tC {} PAD_XY {:+2.2f} , {:+2.2f}", (bending ? (int)0 : (int)1), X, Y);
187  std::cout << std::endl;
188  }
189  };
190 
191  patchPage(page, mDebug);
192 
193  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(page[0])));
194  orbit = o2::raw::RDHUtils::getHeartBeatOrbit(rdhAny);
195 
196  if (mRdhHandler) {
197  mRdhHandler(&rdhAny);
198  }
199 
200  // add orbit to vector if not present yet
201  mOrbits.emplace(page);
202 
203  if (!mDecoder) {
204  DecodedDataHandlers handlers;
205  handlers.sampaChannelHandler = channelHandler;
206  mDecoder = mFee2Solar ? o2::mch::raw::createPageDecoder(page, handlers, mFee2Solar)
207  : o2::mch::raw::createPageDecoder(page, handlers);
208  }
209  mDecoder(page);
210 
211  if (mDebug) {
212  std::cout << "[decodeBuffer] mOutputDigits size: " << mOutputDigits.size() << std::endl;
213  dumpDigits(true);
214  dumpDigits(false);
215  }
216 };

255 void DataDecoder::init()
256 {
257  mNrdhs = 0;
258 
259  for (int i = 0; i < 64; i++) {
260  for (int j = 0; j < 64; j++) {
261  if (refManu2ds_st345[j] != i) {
262  continue;
263  }

265  break;
