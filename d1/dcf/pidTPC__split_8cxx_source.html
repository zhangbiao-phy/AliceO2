67  void init(o2::framework::InitContext&)
68  {
69  ccdb->setURL(url.value);
70  ccdb->setTimestamp(timestamp.value);
71  ccdb->setCaching(true);
72  ccdb->setLocalObjectValidityChecking();
73  // Not later than now objects
74  ccdb->setCreatedNotAfter(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
75  //
76  const std::string fname = paramfile.value;
77  if (!fname.empty()) { // Loading the parametrization from file
78  response.LoadParamFromFile(fname.data(), signalname.value, DetectorResponse::kSignal);
79  response.LoadParamFromFile(fname.data(), sigmaname.value, DetectorResponse::kSigma);
80  } else { // Loading it from CCDB
81  const std::string path = "Analysis/PID/TPC";
82  response.LoadParam(DetectorResponse::kSignal, ccdb->getForTimeStamp<Parametrization>(path + "/" + signalname.value, timestamp.value));
83  response.LoadParam(DetectorResponse::kSigma, ccdb->getForTimeStamp<Parametrization>(path + "/" + sigmaname.value, timestamp.value));
84  }
85  }

87  template <o2::track::PID::ID pid>
88  using ResponseImplementation = tpc::ELoss<Coll::iterator, Trks::iterator, pid>;
89  void process(Coll const& collisions, Trks const& tracks)
90  {
91  constexpr auto responseEl = ResponseImplementation<PID::Electron>();
92  constexpr auto responseMu = ResponseImplementation<PID::Muon>();
93  constexpr auto responsePi = ResponseImplementation<PID::Pion>();
94  constexpr auto responseKa = ResponseImplementation<PID::Kaon>();
95  constexpr auto responsePr = ResponseImplementation<PID::Proton>();
96  constexpr auto responseDe = ResponseImplementation<PID::Deuteron>();
97  constexpr auto responseTr = ResponseImplementation<PID::Triton>();
98  constexpr auto responseHe = ResponseImplementation<PID::Helium3>();
99  constexpr auto responseAl = ResponseImplementation<PID::Alpha>();
100 
101  // Check and fill enabled tables
102  auto makeTable = [&tracks](const Configurable<int>& flag, auto& table, const DetectorResponse& response, const auto& responsePID) {
103  if (flag.value) {
104  // Prepare memory for enabled tables
105  table.reserve(tracks.size());
106  for (auto const& trk : tracks) { // Loop on Tracks
107  table(responsePID.GetExpectedSigma(response, trk.collision(), trk),
108  responsePID.GetSeparation(response, trk.collision(), trk));
109  }
110  }
111  };
112  makeTable(pidEl, tablePIDEl, response, responseEl);
113  makeTable(pidMu, tablePIDMu, response, responseMu);
114  makeTable(pidPi, tablePIDPi, response, responsePi);
115  makeTable(pidKa, tablePIDKa, response, responseKa);
116  makeTable(pidPr, tablePIDPr, response, responsePr);
117  makeTable(pidDe, tablePIDDe, response, responseDe);
118  makeTable(pidTr, tablePIDTr, response, responseTr);
119  makeTable(pidHe, tablePIDHe, response, responseHe);
120  makeTable(pidAl, tablePIDAl, response, responseAl);
121  }
122 };
