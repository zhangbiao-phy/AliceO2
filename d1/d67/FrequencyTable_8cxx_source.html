23 void FrequencyTable::resizeFrequencyTable(value_t min, value_t max)
24 {
25  LOG(trace) << "start resizing frequency table";
26  internal::RANSTimer t;
27  t.start();
28  // calculate new dimensions
29  const value_t newMin = std::min(mMin, min);
30  const value_t newMax = std::max(mMax, max);
31  const size_t newSize = newMax - newMin + 1;
32 
33  // empty - init;
34  if (mMin == 0 && mMax == 0) {
35  mFrequencyTable.resize(newSize, 0);
36  }
37 
38  // if the new size is bigger than the old one we need to resize the frequency table
39  if ((newSize) > mFrequencyTable.size()) {
40  const size_t offset = newMin < mMin ? std::abs(min - mMin) : 0;
41  std::vector<uint32_t> tmpFrequencyTable;
42  tmpFrequencyTable.reserve(newSize);
43  // insert initial offset if applicable
44  tmpFrequencyTable.insert(std::begin(tmpFrequencyTable), offset, 0);
45  // append current frequency table
46  tmpFrequencyTable.insert(std::end(tmpFrequencyTable), mFrequencyTable.begin(), mFrequencyTable.end());
47  //fill tail with zeroes if applicable
48  const size_t tail = newMax > mMax ? std::abs(max - mMax) : 0;
49  tmpFrequencyTable.insert(std::end(tmpFrequencyTable), tail, 0);
50 
51  mFrequencyTable = std::move(tmpFrequencyTable);
52  }
53 
54  assert(mFrequencyTable.size() == newSize);
55  mMin = newMin;
56  mMax = newMax;
57 
58  t.stop();
59  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
60  LOG(trace) << "done resizing frequency table";
61 }
