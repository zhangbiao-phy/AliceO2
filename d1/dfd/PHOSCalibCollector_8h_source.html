57 // For real/mixed distribution calculation
58 class RungBuffer
59 {
60  public:
61  RungBuffer() = default;
62  ~RungBuffer() = default;
63 
64  short size()
65  {
66  if (mFilled) {
67  return kBufferSize;
68  } else {
69  return mCurrent;
70  }
71  }
72  void addEntry(TLorentzVector& v)
73  {
74  mBuffer[mCurrent] = v;
75  mCurrent++;
76  if (mCurrent >= kBufferSize) {
77  mFilled = true;
78  mCurrent -= kBufferSize;
79  }
80  }
81  const TLorentzVector getEntry(short index)
82  {
83  //get entry from (mCurrent-1) corresponding to index=size()-1 down to size
84  if (mFilled) {
85  index += mCurrent;
86  }
87  index = index % kBufferSize;
88  return mBuffer[index];
89  }
90  //mark that next added entry will be from next event
91  void startNewEvent() { mStartCurrentEvent = mCurrent; }
92 
93  bool isCurrentEvent(short index)
94  {
95  if (mCurrent >= mStartCurrentEvent) {
96  return (index >= mStartCurrentEvent && index < mCurrent);
97  } else {
98  return (index >= mStartCurrentEvent || index < mCurrent);
99  }
100  }
101 
102  private:
103  static constexpr short kBufferSize = 100;
104  std::array<TLorentzVector, kBufferSize> mBuffer;
105  bool mFilled = false;
106  short mCurrent = 0;
107  short mStartCurrentEvent = 0;
108 };
