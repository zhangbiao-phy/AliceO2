32 class TOFCalibCollectorDevice : public o2::framework::Task
33 {
34 
35  public:
36  void init(o2::framework::InitContext& ic) final
37  {
38 
39  bool isTFsendingPolicy = ic.options().get<bool>("tf-sending-policy");
40  int maxEnt = ic.options().get<int>("max-number-hits-to-fill-tree");
41  bool isTest = ic.options().get<bool>("running-in-test-mode");
42  bool absMaxEnt = ic.options().get<bool>("is-max-number-hits-to-fill-tree-absolute");
43  mCollector = std::make_unique<o2::tof::TOFCalibCollector>(isTFsendingPolicy, maxEnt);
44  mCollector->setIsTest(isTest);
45  mCollector->setIsMaxNumberOfHitsAbsolute(absMaxEnt);
46  mCollector->setSlotLength(1);
47  mCollector->setMaxSlotsDelay(0);
48  }
49 
50  void run(o2::framework::ProcessingContext& pc) final
51  {
52 
53  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime; // is this the timestamp of the current TF?
54  auto data = pc.inputs().get<gsl::span<o2::dataformats::CalibInfoTOF>>("input");
55  LOG(INFO) << "Processing TF " << tfcounter << " with " << data.size() << " tracks";
56  mCollector->process(tfcounter, data);
57  sendOutput(pc.outputs());
58  }
59 
60  void endOfStream(o2::framework::EndOfStreamContext& ec) final
61  {
62  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
63  mCollector->checkSlotsToFinalize(INFINITE_TF);
64  // we force finalizing slot zero (unless everything was already finalized), no matter how many entries we had
65  if (mCollector->getNSlots() != 0) {
66  mCollector->finalizeSlot(mCollector->getSlot(0));
67  }
68  sendOutput(ec.outputs());
69  }
70 
71  private:
72  std::unique_ptr<o2::tof::TOFCalibCollector> mCollector;
73  int mMaxNumOfHits = 0;
74 
75  //________________________________________________________________
76  void sendOutput(DataAllocator& output)
77  {
78  // in output we send the calibration tree
79  auto& collectedInfo = mCollector->getCollectedCalibInfo();
80  LOG(DEBUG) << "In CollectorSpec sendOutput: size = " << collectedInfo.size();
81  if (collectedInfo.size()) {
82  auto entries = collectedInfo.size();
83  // this means that we are ready to send the output
84  auto entriesPerChannel = mCollector->getEntriesPerChannel();
85  output.snapshot(Output{o2::header::gDataOriginTOF, "COLLECTEDINFO", 0, Lifetime::Timeframe}, collectedInfo);
86  output.snapshot(Output{o2::header::gDataOriginTOF, "ENTRIESCH", 0, Lifetime::Timeframe}, entriesPerChannel);
87  mCollector->initOutput(); // reset the output for the next round
88  }
89  }
90 };

119  {"running-in-test-mode", VariantType::Bool, false, {"to run in test mode for simplification"}}}};
