54 void PHOSPedestalCalibDevice::run(o2::framework::ProcessingContext& ctx)
55 {
56 
57  //Read current padestals for comarison
58  // if (mUseCCDB && mOldPed.get()==nullptr) {
59  // mOldPed.reset(std::move(ctx.inputs().get<o2::phos::Pedestals*>("oldPedestals")));
60  // }
61 
62  //
63  for (const auto& rawData : framework::InputRecordWalker(ctx.inputs())) {
64 
65  o2::phos::RawReaderMemory rawreader(o2::framework::DataRefUtils::as<const char>(rawData));
66 
67  // loop over all the DMA pages
68  while (rawreader.hasNext()) {
69  try {
70  rawreader.next();
71  } catch (RawDecodingError::ErrorType_t e) {
72  LOG(ERROR) << "Raw decoding error " << (int)e;
73  //if problem in header, abandon this page
74  if (e == RawDecodingError::ErrorType_t::PAGE_NOTFOUND ||
75  e == RawDecodingError::ErrorType_t::HEADER_DECODING ||
76  e == RawDecodingError::ErrorType_t::HEADER_INVALID) {
77  break;
78  }
79  //if problem in payload, try to continue
80  continue;
81  }
82  auto& header = rawreader.getRawHeader();
83  auto triggerBC = o2::raw::RDHUtils::getTriggerBC(header);
84  auto triggerOrbit = o2::raw::RDHUtils::getTriggerOrbit(header);
85  auto ddl = o2::raw::RDHUtils::getFEEID(header);
86 
87  if (mRunStartTime == 0) {
88  o2::InteractionRecord currentIR(triggerBC, triggerOrbit);
89  mRunStartTime = long(currentIR.bc2ns() * 1.e-9);
90  }
91 
92  if (ddl > o2::phos::Mapping::NDDL) { //only 14 correct DDLs
93  LOG(ERROR) << "DDL=" << ddl;
94  continue; //skip STU ddl
95  }
96  // use the altro decoder to decode the raw data, and extract the RCU trailer
97  o2::phos::AltroDecoder decoder(rawreader);
98  AltroDecoderError::ErrorType_t err = decoder.decode();
99 
100  if (err != AltroDecoderError::kOK) {
101  LOG(ERROR) << "Errror " << err << " in decoding DDL" << ddl;
102  }
103  auto& rcu = decoder.getRCUTrailer();
104  auto& channellist = decoder.getChannels();
105  // Loop over all the channels for this RCU
106  for (auto& chan : channellist) {
107  short absId;
108  Mapping::CaloFlag caloFlag;
109  short fee;
110  Mapping::ErrorStatus s = mMapping->hwToAbsId(ddl, chan.getHardwareAddress(), absId, caloFlag);
111  if (s != Mapping::ErrorStatus::kOK) {
112  LOG(ERROR) << "Error in mapping"
113  << " ddl=" << ddl << " hwaddress " << chan.getHardwareAddress();
114  continue;
115  }
116  if (caloFlag != Mapping::kTRU) { //HighGain or LowGain
117  CaloRawFitter::FitStatus fitResults = mRawFitter->evaluate(chan.getBunches());
118  if (fitResults == CaloRawFitter::FitStatus::kOK || fitResults == CaloRawFitter::FitStatus::kNoTime) {
119  //TODO: which results should be accepted? full configurable list
120  for (int is = 0; is < mRawFitter->getNsamples(); is++) {
121  if (caloFlag == Mapping::kHighGain) {
122  mMeanHG->Fill(absId, mRawFitter->getAmp(is));
123  mRMSHG->Fill(absId, mRawFitter->getTime(is));
124  } else {
125  mMeanLG->Fill(absId, mRawFitter->getAmp(is));
126  mRMSLG->Fill(absId, mRawFitter->getTime(is));
127  }
128  }
129  }
130  }
131  }
132  } //RawReader::hasNext
133  }
134 }

189 void PHOSPedestalCalibDevice::calculatePedestals()
190 {
191 
192  mPedestals.reset(new Pedestals());
193 
194  //Calculate mean of pedestal distributions
195  for (unsigned short i = mMeanHG->GetNbinsX(); i > 0; i--) {
196  TH1D* pr = mMeanHG->ProjectionY(Form("proj%d", i), i, i);
197  float a = pr->GetMean();
198  mPedestals->setHGPedestal(i - 1, std::min(255, int(a)));
199  pr->Delete();
200  pr = mMeanLG->ProjectionY(Form("projLG%d", i), i, i);
201  a = pr->GetMean();
202  mPedestals->setLGPedestal(i - 1, std::min(255, int(a)));
203  pr->Delete();
204  }
205 }

207 void PHOSPedestalCalibDevice::checkPedestals()
208 {
209  //Compare pedestals to current ones stored in CCDB
210  //and send difference to QC to check
211  if (!mUseCCDB) {
212  mUpdateCCDB = true;
213  return;
214  }
215 
216  if (mOldPed.get() == nullptr) { //was not read from CCDB, but expected
217  mUpdateCCDB = false;
218  return;
219  }
220 
221  //Compare to current
222  int nChanged = 0;
223  for (short i = o2::phos::Mapping::NCHANNELS; --i;) {
224  short dp = mPedestals->getHGPedestal(i) - mOldPed->getHGPedestal(i);
225  mPedHGDiff[i] = dp;
226  if (abs(dp) > 1) { //not a fluctuation
227  nChanged++;
228  }
229  dp = mPedestals->getLGPedestal(i) - mOldPed->getLGPedestal(i);
230  mPedLGDiff[i] = dp;
231  if (abs(dp) > 1) { //not a fluctuation
232  nChanged++;
233  }
234  }
235  if (nChanged > kMinorChange) { //serious change, do not update CCDB automatically, use "force" option to overwrite
236  mUpdateCCDB = false;
237  } else {
238  mUpdateCCDB = true;
239  }
240 }
