51 void TPCITSMatchingDPL::init(InitContext& ic)
52 {
53  mTimer.Stop();
54  mTimer.Reset();
55  //-------- init geometry and field --------//
56  o2::base::GeometryManager::loadGeometry();
57  o2::base::Propagator::initFieldFromGRP(o2::base::NameConf::getGRPFileName());
58  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom(o2::base::NameConf::getGRPFileName())};
59  mMatching.setITSTriggered(!grp->isDetContinuousReadOut(o2::detectors::DetID::ITS));
60  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
61  if (mMatching.isITSTriggered()) {
62  mMatching.setITSROFrameLengthMUS(alpParams.roFrameLengthTrig / 1.e3); // ITS ROFrame duration in \mus
63  } else {
64  mMatching.setITSROFrameLengthInBC(alpParams.roFrameLengthInBC); // ITS ROFrame duration in \mus
65  }
66  mMatching.setMCTruthOn(mUseMC);
67  mMatching.setUseFT0(mUseFT0);
68  mMatching.setVDriftCalib(mCalibMode);
69  //
70  std::string dictPath = ic.options().get<std::string>("its-dictionary-path");
71  std::string dictFile = o2::base::NameConf::getDictionaryFileName(o2::detectors::DetID::ITS, dictPath, ".bin");
72  if (o2::base::NameConf::pathExists(dictFile)) {
73  mITSDict.readBinaryFile(dictFile);
74  LOG(INFO) << "Matching is running with a provided ITS dictionary: " << dictFile;
75  } else {
76  LOG(INFO) << "Dictionary " << dictFile << " is absent, Matching expects ITS cluster patterns";
77  }
78 
79  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
80  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
81  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
82  if (o2::base::NameConf::pathExists(matLUTFile)) {
83  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
84  o2::base::Propagator::Instance()->setMatLUT(lut);
85  LOG(INFO) << "Loaded material LUT from " << matLUTFile;
86  } else {
87  LOG(INFO) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
88  }
89 
90  int dbgFlags = ic.options().get<int>("debug-tree-flags");
91  mMatching.setDebugFlag(dbgFlags);
92 
93  // set bunch filling. Eventually, this should come from CCDB
94  const auto* digctx = o2::steer::DigitizationContext::loadFromFile("collisioncontext.root");
95  const auto& bcfill = digctx->getBunchFilling();
96  mMatching.setBunchFilling(bcfill);
97 
98  mMatching.init();
99  //
100 }

102 void TPCITSMatchingDPL::run(ProcessingContext& pc)
103 {
104  mTimer.Start(false);
105  const auto tracksITS = pc.inputs().get<gsl::span<o2::its::TrackITS>>("trackITS");
106  const auto trackClIdxITS = pc.inputs().get<gsl::span<int>>("trackClIdx");
107  const auto tracksITSROF = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("trackITSROF");
108  const auto clusITS = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("clusITS");
109  const auto clusITSROF = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("clusITSROF");
110  const auto patterns = pc.inputs().get<gsl::span<unsigned char>>("clusITSPatt");
111  const auto tracksTPC = pc.inputs().get<gsl::span<o2::tpc::TrackTPC>>("trackTPC");
112  const auto tracksTPCClRefs = pc.inputs().get<gsl::span<o2::tpc::TPCClRefElem>>("trackTPCClRefs");
113 
114  const auto clusTPCShmap = pc.inputs().get<gsl::span<unsigned char>>("clusTPCshmap");
115 
116  const auto& inputsTPCclusters = o2::tpc::getWorkflowTPCInput(pc);
117 
118  //
119  MCLabelsTr lblITS;
120  MCLabelsTr lblTPC;
121 
122  const MCLabelsCl* lblClusITSPtr = nullptr;
123  std::unique_ptr<const MCLabelsCl> lblClusITS;
124 
125  if (mUseMC) {
126  lblITS = pc.inputs().get<gsl::span<o2::MCCompLabel>>("trackITSMCTR");
127  lblTPC = pc.inputs().get<gsl::span<o2::MCCompLabel>>("trackTPCMCTR");
128 
129  lblClusITS = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("clusITSMCTR");
130  lblClusITSPtr = lblClusITS.get();
131  }
132  //
133  // create ITS clusters as spacepoints in tracking frame
134  std::vector<o2::BaseCluster<float>> itsSP;
135  itsSP.reserve(clusITS.size());
136  auto pattIt = patterns.begin();
137  o2::its::ioutils::convertCompactClusters(clusITS, pattIt, itsSP, mITSDict);
138 
139  // pass input data to MatchTPCITS object
140  mMatching.setITSTracksInp(tracksITS);
141  mMatching.setITSTrackClusIdxInp(trackClIdxITS);
142  mMatching.setITSTrackROFRecInp(tracksITSROF);
143  mMatching.setITSClustersInp(itsSP);
144  mMatching.setITSClusterROFRecInp(clusITSROF);
145  mMatching.setTPCTracksInp(tracksTPC);
146  mMatching.setTPCTrackClusIdxInp(tracksTPCClRefs);
147  mMatching.setTPCClustersInp(&inputsTPCclusters->clusterIndex);
148  mMatching.setTPCClustersSharingMap(clusTPCShmap);
149 
150  if (mUseMC) {
151  mMatching.setITSTrkLabelsInp(lblITS);
152  mMatching.setTPCTrkLabelsInp(lblTPC);
153  mMatching.setITSClsLabelsInp(lblClusITSPtr);
154  }
155 
156  if (mUseFT0) {
157  // Note: the particular variable will go out of scope, but the span is passed by copy to the
158  // worker and the underlying memory is valid throughout the whole computation
159  auto fitInfo = pc.inputs().get<gsl::span<o2::ft0::RecPoints>>("fitInfo");
160  mMatching.setFITInfoInp(fitInfo);
161  }
162 
163  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().get("trackITSROF").header);
164  mMatching.setStartIR({0, dh->firstTForbit});
165 
166  //RSTODO: below is a hack, to remove once the framework will start propagating the header.firstTForbit
167  if (tracksITSROF.size()) {
168  mMatching.setStartIR(o2::raw::HBFUtils::Instance().getFirstIRofTF(tracksITSROF[0].getBCData()));
169  }
170 
171  mMatching.run();
172 
173  pc.outputs().snapshot(Output{"GLO", "TPCITS", 0, Lifetime::Timeframe}, mMatching.getMatchedTracks());
174  if (mUseMC) {
175  pc.outputs().snapshot(Output{"GLO", "TPCITS_MC", 0, Lifetime::Timeframe}, mMatching.getMatchLabels());
176  }
177 
178  if (mCalibMode) {
179  auto* hdtgl = mMatching.getHistoDTgl();
180  pc.outputs().snapshot(Output{"GLO", "TPCITS_VDHDTGL", 0, Lifetime::Timeframe}, (*hdtgl).getBase());
181  hdtgl->clear();
182  }
183 
184  mTimer.Stop();
185 }

193 DataProcessorSpec getTPCITSMatchingSpec(bool useFT0, bool calib, bool useMC)
194 {
195 
196  std::vector<InputSpec> inputs;
197  std::vector<OutputSpec> outputs;
198  inputs.emplace_back("trackITS", "ITS", "TRACKS", 0, Lifetime::Timeframe);
199  inputs.emplace_back("trackClIdx", "ITS", "TRACKCLSID", 0, Lifetime::Timeframe);
200  inputs.emplace_back("trackITSROF", "ITS", "ITSTrackROF", 0, Lifetime::Timeframe);
201  inputs.emplace_back("clusITS", "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe);
202  inputs.emplace_back("clusITSPatt", "ITS", "PATTERNS", 0, Lifetime::Timeframe);
203  inputs.emplace_back("clusITSROF", "ITS", "CLUSTERSROF", 0, Lifetime::Timeframe);
204  inputs.emplace_back("trackTPC", "TPC", "TRACKS", 0, Lifetime::Timeframe);
205  inputs.emplace_back("trackTPCClRefs", "TPC", "CLUSREFS", 0, Lifetime::Timeframe);
206 
207  inputs.emplace_back("clusTPC", ConcreteDataTypeMatcher{"TPC", "CLUSTERNATIVE"}, Lifetime::Timeframe);
208  inputs.emplace_back("clusTPCshmap", "TPC", "CLSHAREDMAP", 0, Lifetime::Timeframe);
209 
210  if (useFT0) {
211  inputs.emplace_back("fitInfo", "FT0", "RECPOINTS", 0, Lifetime::Timeframe);
212  }
213 
214  outputs.emplace_back("GLO", "TPCITS", 0, Lifetime::Timeframe);
215 
216  if (calib) {
217  outputs.emplace_back("GLO", "TPCITS_VDHDTGL", 0, Lifetime::Timeframe);
218  }
219 
220  if (useMC) {
221  inputs.emplace_back("trackITSMCTR", "ITS", "TRACKSMCTR", 0, Lifetime::Timeframe);
222  inputs.emplace_back("trackTPCMCTR", "TPC", "TRACKSMCLBL", 0, Lifetime::Timeframe);
223  inputs.emplace_back("clusITSMCTR", "ITS", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
224  //
225  outputs.emplace_back("GLO", "TPCITS_MC", 0, Lifetime::Timeframe);
226  }
227 
228  return DataProcessorSpec{
229  "itstpc-track-matcher",
230  inputs,
231  outputs,
232  AlgorithmSpec{adaptFromTask<TPCITSMatchingDPL>(useFT0, calib, useMC)},
233  Options{
234  {"its-dictionary-path", VariantType::String, "", {"Path of the cluster-topology dictionary file"}},
235  {"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}},
236  {"debug-tree-flags", VariantType::Int, 0, {"DebugFlagTypes bit-pattern for debug tree"}}}};
237 }
