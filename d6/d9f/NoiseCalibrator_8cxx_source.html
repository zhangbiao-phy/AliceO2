25 bool NoiseCalibrator::processTimeFrame(gsl::span<const o2::itsmft::CompClusterExt> const& clusters,
26  gsl::span<const unsigned char> const& patterns,
27  gsl::span<const o2::itsmft::ROFRecord> const& rofs)
28 {
29  static int nTF = 0;
30  LOG(INFO) << "Processing TF# " << nTF++;
31 
32  auto pattIt = patterns.begin();
33  for (const auto& rof : rofs) {
34  auto clustersInFrame = rof.getROFData(clusters);
35  for (const auto& c : clustersInFrame) {
36  if (c.getPatternID() != o2::itsmft::CompCluster::InvalidPatternID) {
37  // For the noise calibration, we use "pass1" clusters...
38  continue;
39  }
40  o2::itsmft::ClusterPattern patt(pattIt);
41 
42  auto id = c.getSensorID();
43  auto row = c.getRow();
44  auto col = c.getCol();
45  auto colSpan = patt.getColumnSpan();
46  auto rowSpan = patt.getRowSpan();
47 
48  // Fast 1-pixel calibration
49  if ((rowSpan == 1) && (colSpan == 1)) {
50  mNoiseMap.increaseNoiseCount(id, row, col);
51  continue;
52  }
53  if (m1pix) {
54  continue;
55  }
56 
57  // All-pixel calibration
58  auto nBits = rowSpan * colSpan;
59  int ic = 0, ir = 0;
60  for (unsigned int i = 2; i < patt.getUsedBytes() + 2; i++) {
61  unsigned char tempChar = patt.getByte(i);
62  int s = 128; // 0b10000000
63  while (s > 0) {
64  if ((tempChar & s) != 0) {
65  mNoiseMap.increaseNoiseCount(id, row + ir, col + ic);
66  }
67  ic++;
68  s >>= 1;
69  if ((ir + 1) * ic == nBits) {
70  break;
71  }
72  if (ic == colSpan) {
73  ic = 0;
74  ir++;
75  }
76  }
77  if ((ir + 1) * ic == nBits) {
78  break;
79  }
80  }
81  }
82  }
83  mNumberOfStrobes += rofs.size();
84  return (mNumberOfStrobes * mProbabilityThreshold >= mThreshold) ? true : false;
85 }

88 {
89  LOG(INFO) << "Number of processed strobes is " << mNumberOfStrobes;
90  mNoiseMap.applyProbThreshold(mProbabilityThreshold, mNumberOfStrobes);
91 }
