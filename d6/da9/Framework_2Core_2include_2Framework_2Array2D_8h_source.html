23 // matrix-like wrapper for C-array
24 // has no range checks
25 template <typename T>
26 struct Array2D {
27  using element_t = T;
28 
29  Array2D()
30  : data{nullptr},
31  rows{0},
32  cols{0}
33  {
34  }
35 
36  Array2D(T const* data_, uint32_t r, uint32_t c)
37  : rows{r}, cols{c}
38  {
39  data = new T[rows * cols];
40  for (auto i = 0U; i < rows; ++i) {
41  for (auto j = 0U; j < cols; ++j) {
42  data[i * cols + j] = data_[i * cols + j];
43  }
44  }
45  }
46  Array2D(std::vector<T> data_, uint32_t r, uint32_t c)
47  : rows{r}, cols{c}
48  {
49  data = new T[rows * cols];
50  for (auto i = 0U; i < rows; ++i) {
51  for (auto j = 0U; j < cols; ++j) {
52  data[i * cols + j] = data_[i * cols + j];
53  }
54  }
55  }
56 
57  Array2D(Array2D<T> const& other)
58  : rows{other.rows},
59  cols{other.cols}
60  {
61  data = new T[rows * cols];
62  for (auto i = 0U; i < rows; ++i) {
63  for (auto j = 0U; j < cols; ++j) {
64  data[i * cols + j] = other.data[i * cols + j];
65  }
66  }
67  }
68 
69  Array2D(Array2D<T>&& other)
70  : rows{other.rows},
71  cols{other.cols}
72  {
73  data = other.data;
74  other.data = nullptr;
75  other.rows = 0;
76  other.cols = 0;
77  }
78 
79  Array2D& operator=(Array2D<T> const& other)
80  {
81  this->rows = other.rows;
82  this->cols = other.cols;
83  data = new T[rows * cols];
84  for (auto i = 0U; i < rows; ++i) {
85  for (auto j = 0U; j < cols; ++j) {
86  data[i * cols + j] = *(other.data + (i * cols + j));
87  }
88  }
89  return *this;
90  }
91 
92  Array2D& operator=(Array2D<T>&& other)
93  {
94  this->rows = other.rows;
95  this->cols = other.cols;
96  data = other.data;
97  other.data = nullptr;
98  other.rows = 0;
99  other.cols = 0;
100  return *this;
101  }
102 
103  ~Array2D()
104  {
105  if (data != nullptr) {
106  delete[] data;
107  }
108  }
109 
110  T operator()(uint32_t y, uint32_t x) const
111  {
112  return data[y * cols + x];
113  }
114  T* operator[](uint32_t y) const
115  {
116  return data + y * cols;
117  }
118 
119  T* data;
120  uint32_t rows;
121  uint32_t cols;
122 };

127 using labelmap_t = std::unordered_map<std::string, uint32_t>;
128 struct LabelMap {
129  LabelMap()
130  : rowmap{},
131  colmap{},
132  labels_rows{},
133  labels_cols{}
134  {
135  }
136  LabelMap(uint32_t rows, uint32_t cols, std::vector<std::string> labels_rows_, std::vector<std::string> labels_cols_)
137  : rowmap{populate(rows, labels_rows_)},
138  colmap{populate(cols, labels_cols_)},
139  labels_rows{labels_rows_},
140  labels_cols{labels_cols_}
141  {
142  }
143 
144  LabelMap(uint32_t size, std::vector<std::string> labels)
145  : rowmap{},
146  colmap{populate(size, labels)},
147  labels_rows{},
148  labels_cols{labels}
149  {
150  }
151 
152  LabelMap(LabelMap const& other) = default;
153  LabelMap(LabelMap&& other) = default;
154  LabelMap& operator=(LabelMap const& other) = default;
155  LabelMap& operator=(LabelMap&& other) = default;
156 
157  labelmap_t rowmap;
158  labelmap_t colmap;
159 
160  std::vector<std::string> labels_rows;
161  std::vector<std::string> labels_cols;
162 
163  labelmap_t populate(uint32_t size, std::vector<std::string> labels)
164  {
165  labelmap_t map;
166  if (labels.empty() == false) {
167  if (labels.size() != size) {
168  throw runtime_error("labels array size != array dimension");
169  }
170  for (auto i = 0u; i < size; ++i) {
171  map.emplace(labels[i], (uint32_t)i);
172  }
173  }
174  return map;
175  }
176 
177  auto getLabelsRows()
178  {
179  return labels_rows;
180  }
181 
182  auto getLabelsCols()
183  {
184  return labels_cols;
185  }
186 };
