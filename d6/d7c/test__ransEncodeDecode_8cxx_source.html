28 template <typename CODER_T, typename STREAM_T, uint P>
29 struct Fixture {
30  // type of the coder: 32 bit or 64 bit
31  using coder_t = CODER_T;
32  // how many bytes do we stream out during normalization:
33  // 1 Byte for 32Bit coders, 4Byte for 64Bit coders
34  using stream_t = STREAM_T;
35  // what is the datatype of our source symbols?
36  using source_t = char;
37 
38  using encoder_t = o2::rans::Encoder<coder_t, stream_t, source_t>;
39  using decoder_t = o2::rans::Decoder<coder_t, stream_t, source_t>;
40  using literalEncoder_t = o2::rans::LiteralEncoder<coder_t, stream_t, source_t>;
41  using literalDecoder_t = o2::rans::LiteralDecoder<coder_t, stream_t, source_t>;
42  using dedupEncoder_t = o2::rans::DedupEncoder<coder_t, stream_t, source_t>;
43  using dedupDecoder_t = o2::rans::DedupDecoder<coder_t, stream_t, source_t>;
44 
45  //TUNIG parameters
46  // how many bits do we resample the symbol statistics to?
47  // this depends on the size of the alphabet (i.e. how big source_t is).
48  // See poster for more details
49  // https://indico.cern.ch/event/773049/contributions/3474364/attachments/1936180/3208584/Layout.pdf
50  // As a rule of thumb we need
51  // 10 Bits for 8Bit alphabets,
52  // 22Bits for 16Bit alphabets,
53  // 25Bits for 25Bit alphabets
54  const uint probabilityBits = P;
55 
56  const std::string source;
57 };

76 BOOST_FIXTURE_TEST_CASE_TEMPLATE(test_EncodeDecode, T, Fixtures, T)
77 {
78  fair::Logger::SetConsoleSeverity("trace");
79  // iterate over the message and create PDF and CDF for each symbol in the message
80  o2::rans::FrequencyTable frequencies;
81  frequencies.addSamples(std::begin(T::source), std::end(T::source));
82 
83  // buffer to write the rANS coded message into
84  std::vector<typename T::stream_t> encoderBuffer(1 << 20, 0);
85  //create a stateful encoder object that builds an encoding table from the given symbol statistics
86  const typename T::encoder_t encoder{frequencies, T::probabilityBits};
87  // encoder rANS encodes symbols from SOURCE array to encoder Buffer. Since coder and decoder
88  // are mathematical inverse functions on the ANS state, they operate as a stack -
89  // i.e. the decoder outputs the message in reverse order of the encoder.
90  // By convention the encoder runs backwards over the input so that the decoder can return
91  // the data in the expected order. The encoder will start from encoderBuffer.begin()
92  // and return an iterator one element past the last entry written.
93  // This means the encodeded message ranges from encoderBuffer.begin() to encodedMessageEnd -1, with (encodedMessageEnd -encoderBuffer.begin()) entries written.
94  auto encodedMessageEnd = encoder.process(encoderBuffer.begin(), encoderBuffer.end(), std::begin(T::source), std::end(T::source));
95 
96  // The decoded message will go into the decoder buffer which will have as many symbols as the original message
97  std::vector<typename T::source_t> decoderBuffer(std::distance(std::begin(T::source), std::end(T::source)), 0);
98  // create a stateful decoder object that build a decoding table from the given symbol statistics
99  const typename T::decoder_t decoder{frequencies, T::probabilityBits};
100  // the decoder unwinds the rANS state in the encoder buffer starting at ransBegin and decodes it into the decode buffer;
101  decoder.process(decoderBuffer.begin(), encodedMessageEnd, std::distance(std::begin(T::source), std::end(T::source)));
102 
103  //the decodeBuffer and the source message have to be identical afterwards.
104  BOOST_REQUIRE(std::memcmp(&(*T::source.begin()), decoderBuffer.data(),
105  decoderBuffer.size() * sizeof(typename T::source_t)) == 0);
106 }
