34 struct Topo {

36  int modCh = 0; // Channel within the Processing Module in range from 0-11
37  ClassDefNV(Topo, 1);

40 inline bool operator<(Topo const& a, Topo const& b)
41 {
42  return (a.modLink < b.modLink || (a.modLink == b.modLink && a.modCh < b.modCh));
43 }

45 class LookUpTable

55  explicit LookUpTable(bool fillLinearly)
56  : mTopoVector(Nmodules * NChPerMod, {0, 0}),
57  mInvTopo(mTopoVector.size(), 0)
58  {
59  if (fillLinearly) {
60  LOG(INFO) << "Mapping of global channel and (PM, PM channel) pair";
61  for (int link = 0; link < Nmodules; ++link) {
62  for (int ch = 0; ch < NChPerMod; ++ch) {
63  mTopoVector[link * NChPerMod + ch] = o2::fdd::Topo{link, ch};
64  }
65  }
66  } else {
67  // TODO: If needed: implement more realistic splitting: 1 ring -> 1 PM instead of linear
68  LOG(WARNING) << "Don't use it - not implemented yet.";
69  }
70 
71  // Fill inverted LUT - matters only if LUT is not linear
72  for (size_t channel = 0; channel < mTopoVector.size(); ++channel) {
73  mInvTopo[getIdx(mTopoVector[channel].modLink, mTopoVector[channel].modCh)] = channel;
74  }
75  }
76 
77  int getChannel(int link, int mcp) const { return mInvTopo[getIdx(link, mcp)]; }
78  int getLink(int channel) const { return mTopoVector[channel].modLink; }
79  int getModChannel(int channel) const { return mTopoVector[channel].modCh; }
80  int getTcmLink() const { return Nmodules; }
81  void printFullMap() const
82  {
83  std::cout << "o2::fdd::LookUpTable::printFullMap(): mTopoVector: [globalCh link modCh]" << std::endl;
84  for (size_t channel = 0; channel < mTopoVector.size(); ++channel) {
85  std::cout << " " << std::right << std::setw(2) << channel << " ";

88  }

91  std::cout << " " << std::right << std::setw(3) << idx << " ";

94  std::cout << std::right << std::setw(2) << getModChannelFromIdx(mInvTopo[idx]) << std::endl;
95  }
