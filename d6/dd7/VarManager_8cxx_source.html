49 void VarManager::ResetValues(int startValue, int endValue, float* values)
50 {
51  //
52  // reset all variables to an "innocent" value
53  // NOTE: here we use -9999.0 as a neutral value, but depending on situation, this may not be the case
54  if (!values) {
55  values = fgValues;
56  }
57  for (Int_t i = startValue; i < endValue; ++i) {
58  values[i] = -9999.;
59  }
60 }

63 void VarManager::SetRunNumbers(int n, int* runs)
64 {
65  //
66  // maps the list of runs such that one can plot the list of runs nicely in a histogram axis
67  //

86 //__________________________________________________________________
87 void VarManager::FillEventDerived(float* values)
88 {
89  //
90  // Fill event-wise derived quantities (these are all quantities which can be computed just based on the values already filled in the FillEvent() function)
91  //
92  if (fgUsedVars[kRunId]) {
93  values[kRunId] = (fgRunMap.size() > 0 ? fgRunMap[int(values[kRunNo])] : 0);
94  }
95 }
96 
97 //__________________________________________________________________
98 void VarManager::FillTrackDerived(float* values)
99 {
100  //
101  // Fill track-wise derived quantities (these are all quantities which can be computed just based on the values already filled in the FillTrack() function)
102  //
103  if (fgUsedVars[kP]) {
104  values[kP] = values[kPt] * TMath::CosH(values[kEta]);
105  }
106 }
