22 void Digits2Raw::readDigits(const std::string& outDir, const std::string& fileDigitsName)
23 {
24  LOG(INFO) << "==============FV0: Digits2Raw::convertDigits" << std::endl;
25  LookUpTable lut(true);
26 
27  std::string outd = outDir;
28  if (outd.back() != '/') {
29  outd += '/';
30  }
31 
32  // Register PM links linearly
33  for (int iPmLink = 0; iPmLink < Constants::nPms; ++iPmLink) {
34  uint64_t feeId = uint64_t(iPmLink);
35  uint32_t linkId = uint32_t(iPmLink);
36  std::string outFileLink = mOutputPerLink ? (outd + "fv0_link" + std::to_string(iPmLink) + ".raw") : (outd + "fv0.raw");
37  LOG(INFO) << " Register PM link: " << iPmLink << " to file: " << outFileLink;
38  mWriter.registerLink(feeId, sCruId, linkId, sEndPointId, outFileLink);
39  }
40 
41  // Register TCM link separately
42  std::string outFileLink = mOutputPerLink ? (outd + "fv0_link" + std::to_string(sTcmLink) + ".raw") : (outd + "fv0.raw");
43  LOG(INFO) << " Register TCM link: " << outFileLink;
44  mWriter.registerLink(uint64_t(sTcmLink), sCruId, sTcmLink, sEndPointId, outFileLink);
45 
46  TFile* fdig = TFile::Open(fileDigitsName.data());
47  assert(fdig != nullptr);
48  LOG(INFO) << "Open digits file: " << fileDigitsName.data();
49 
50  TTree* digTree = (TTree*)fdig->Get("o2sim");
51  std::vector<o2::fv0::BCData> digitsBC, *fv0BCDataPtr = &digitsBC;
52  std::vector<o2::fv0::ChannelData> digitsCh, *fv0ChDataPtr = &digitsCh;
53  digTree->SetBranchAddress("FV0DigitBC", &fv0BCDataPtr);
54  digTree->SetBranchAddress("FV0DigitCh", &fv0ChDataPtr);
55 
56  for (int ient = 0; ient < digTree->GetEntries(); ient++) {
57  digTree->GetEntry(ient);
58  int nbc = digitsBC.size();
59  for (int ibc = 0; ibc < nbc; ibc++) {
60  auto& bcd = digitsBC[ibc];
61  auto channels = bcd.getBunchChannelData(digitsCh);
62 
63  if (!channels.empty()) {
64  LOG(DEBUG) << "o2::fv0::Digits2Raw::readDigits(): Start to convertDigits() at ibc = " << ibc << " " << bcd.ir
65  << " iCh0:" << bcd.ref.getFirstEntry() << " nentries:" << bcd.ref.getEntries();
66  convertDigits(bcd, channels, lut);
67  }
68  }
69  }
70 }

72 void Digits2Raw::convertDigits(o2::fv0::BCData bcdigits, gsl::span<const ChannelData> pmchannels,
73  const o2::fv0::LookUpTable& lut)
74 {
75  const o2::InteractionRecord intRecord = bcdigits.getIntRecord();
76  int prevPmLink = -1;
77  int iChannelPerLink = 0;
78  int nch = pmchannels.size();
79 
80  std::stringstream ss;
81  ss << " Number of channels: " << nch << " (Ch, PMT, Q, T)\n";
82  for (int ich = 0; ich < nch; ich++) {
83  if (pmchannels[ich].chargeAdc != 0) {
84  ss << " " << std::setw(2) << ich
85  << std::setw(3) << pmchannels[ich].pmtNumber
86  << std::setw(5) << pmchannels[ich].chargeAdc
87  << std::setw(7) << std::setprecision(3) << pmchannels[ich].time << "\n";
88  }
89  }
90  LOG(DEBUG) << ss.str().substr(0, ss.str().size() - 1);
91 
92  for (int ich = 0; ich < nch; ich++) {
93  int nLinkPm = lut.getLink(pmchannels[ich].pmtNumber);
94  if (nLinkPm != prevPmLink) {
95  if (prevPmLink >= 0) {
96  fillSecondHalfWordAndAddData(iChannelPerLink, prevPmLink, intRecord);
97  }
98  makeGBTHeader(mRawEventData.mEventHeader, nLinkPm, intRecord);
99  iChannelPerLink = 0;
100  prevPmLink = nLinkPm;
101  }
102  if (pmchannels[ich].chargeAdc != 0) {
103  LOG(DEBUG) << " Store data for channel: " << ich << " PmLink = " << nLinkPm << " ";
104  auto& newData = mRawEventData.mEventData[iChannelPerLink];
105  newData.charge = pmchannels[ich].chargeAdc;
106  newData.time = pmchannels[ich].time;
107  newData.generateFlags();
108  newData.channelID = lut.getPmChannel(pmchannels[ich].pmtNumber);
109  iChannelPerLink++;
110  }
111  if (ich == nch - 1) {
112  fillSecondHalfWordAndAddData(iChannelPerLink, prevPmLink, intRecord);
113  }
114  }
115 
116  // TCM
117  makeGBTHeader(mRawEventData.mEventHeader, sTcmLink, intRecord);
118  mRawEventData.mEventHeader.nGBTWords = 1;
119  auto& tcmdata = mRawEventData.mTCMdata;
120  tcmdata.vertex = 1;
121  tcmdata.orA = 1;
122  tcmdata.orC = 0;
123  tcmdata.sCen = 0;
124  tcmdata.cen = 0;
125  tcmdata.nChanA = 0;
126  tcmdata.nChanC = 0;
127  tcmdata.amplA = 0;
128  tcmdata.amplC = 0;
129  tcmdata.timeA = 0;
130  tcmdata.timeC = 0;
131 
132  auto data = mRawEventData.to_vector(kTRUE); //for tcm module
133  uint32_t linkId = uint32_t(sTcmLink);
134  uint64_t feeId = uint64_t(sTcmLink);
135  mWriter.addData(feeId, sCruId, linkId, sEndPointId, intRecord, data);
136 
137  // fill mEventData[iChannelPerLink] with 0s to flag that this is a dummy data
138  uint nGBTWords = uint((iChannelPerLink + 1) / 2);
139  if ((iChannelPerLink % 2) == 1) {
140  mRawEventData.mEventData[iChannelPerLink] = {};
141  }
142  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
143  // LOG(DEBUG) << " last link: " << prevPmLink;
144 }

158 void Digits2Raw::fillSecondHalfWordAndAddData(int iChannelPerLink, int prevPmLink, const o2::InteractionRecord& ir)
159 {
160  uint nGBTWords = uint((iChannelPerLink + 1) / 2);
161  if ((iChannelPerLink % 2) == 1) {
162  mRawEventData.mEventData[iChannelPerLink] = {};
163  LOG(DEBUG) << " Fill up empty second half-word.";
164  }
165  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
166  auto data = mRawEventData.to_vector(false);
167  uint32_t linkId = uint32_t(prevPmLink);
168  uint64_t feeId = uint64_t(prevPmLink);
169  mWriter.addData(feeId, sCruId, linkId, sEndPointId, ir, data);
170  LOG(DEBUG) << " Switch prevPmLink: " << prevPmLink << ". Save data with nGBTWords="
171  << nGBTWords << " in header. Last channel: " << iChannelPerLink;
172 }
