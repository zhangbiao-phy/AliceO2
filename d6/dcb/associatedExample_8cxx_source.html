38 // This is a very simple example showing how to iterate over tracks
39 // and create a new collection for them.
40 // FIXME: this should really inherit from AnalysisTask but
41 // we need GCC 7.4+ for that
42 struct ATask {
43  Produces<aod::EtaPhi> etaphi;
44 
45  void process(aod::Tracks const& tracks)
46  {
47  for (auto& track : tracks) {
48  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
49  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
50  etaphi(eta, phi);
51  }
52  }
53 };

88 struct ZTask {
89  using myCol = soa::Join<aod::Collisions, aod::CollisionsExtra>;
90 
91  void process(myCol const& collisions, aod::Tracks const& tracks)
92  {
93  auto multbin0_10 = collisions.select(aod::collision::mult >= 0 && aod::collision::mult < 10);
94  auto multbin10_30 = collisions.select(aod::collision::mult >= 10 && aod::collision::mult < 30);
95  auto multbin30_100 = collisions.select(aod::collision::mult >= 30 && aod::collision::mult < 100);
96 
97  LOGF(INFO, "Bin 0-10");
98  for (auto& col : multbin0_10) {
99  auto groupedTracks = tracks.sliceBy(aod::track::collisionId, col.globalIndex());
100  LOGF(INFO, "Collision %d; Ntrk = %d vs %d", col.globalIndex(), col.mult(), groupedTracks.size());
101  if (groupedTracks.size() > 0) {
102  auto track = groupedTracks.begin();
103  LOGF(INFO, "Track 0 belongs to collision %d at Z = %f", track.collisionId(), track.collision_as<myCol>().posZ());
104  }
105  }
106 
107  LOGF(INFO, "Bin 10-30");
108  for (auto& col : multbin10_30) {
109  auto groupedTracks = tracks.sliceBy(aod::track::collisionId, col.globalIndex());
110  LOGF(INFO, "Collision %d; Ntrk = %d vs %d", col.globalIndex(), col.mult(), groupedTracks.size());
111  }
112 
113  LOGF(INFO, "Bin 30-100");
114  for (auto& col : multbin30_100) {
115  auto groupedTracks = tracks.sliceBy(aod::track::collisionId, col.globalIndex());
116  LOGF(INFO, "Collision %d; Ntrk = %d vs %d", col.globalIndex(), col.mult(), groupedTracks.size());
117  }
118  }
