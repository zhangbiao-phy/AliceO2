61  template <typename T>
62  float SpeciesNSigma(const T& track, int specie)
63  {
64  uint32_t pNsigma = 0xFFFFFF00; //15 bit precision for Nsigma
65  if (specie == 0) {
66  return track.tpcNSigmaEl();
67  } else if (specie == 1) {
68  return track.tpcNSigmaMu();
69  } else if (specie == 2) {
70  return track.tpcNSigmaPi();
71  } else if (specie == 3) {
72  return track.tpcNSigmaKa();
73  } else if (specie == 4) {
74  return track.tpcNSigmaPr();
75  } else if (specie == 5) {
76  return track.tpcNSigmaDe();
77  } else if (specie == 6) {
78  return truncateFloatFraction(track.tpcNSigmaTr(), pNsigma);
79  } else if (specie == 7) {
80  return truncateFloatFraction(track.tpcNSigmaHe(), pNsigma);
81  } else if (specie == 8) {
82  return truncateFloatFraction(track.tpcNSigmaAl(), pNsigma);
83  } else {
84  return -99.;
85  }
86  }
87 
88  Filter collisionFilter = nabs(aod::collision::posZ) < vertexZCut;
89  Filter trackFilter = (nabs(aod::track::eta) < trackEtaCut) && (aod::track::pt > trackPtCut) && (aod::track::isGlobalTrack == (uint8_t) true);

91  using TrackCandidates = soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra, aod::pidRespTPC, aod::TrackSelection>>;
92  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>::iterator const& collision, TrackCandidates const& tracks)
93  {
94  if (!collision.alias()[kINT7]) {
95  return;
96  }
97  if (!collision.sel7()) {
98  return;
99  }
100 
101  bool acceptedPID = true;
102  float species1NSigma = -99.;
103  float species2NSigma = -99.;
104  float species3NSigma = -99.;
105  float species4NSigma = -99.;
106 
107  for (auto& track : tracks) {
108 
109  if (saveSmallTracks || saveSingleTracks) {
110  species1NSigma = SpeciesNSigma(track, species1);
111  }
112  if (saveSmallTracks) {
113  species2NSigma = SpeciesNSigma(track, species2);
114  species3NSigma = SpeciesNSigma(track, species3);
115  species4NSigma = SpeciesNSigma(track, species4);
116  }
117  acceptedPID = false;
118  if (!species1NsigmaSelection && !species2NsigmaSelection && !species3NsigmaSelection && !species4NsigmaSelection) {
119  acceptedPID = true;
120  } else if (species1NsigmaSelection && species1NSigma > species1NsigmaLow && species1NSigma < species1NsigmaHigh) {
121  acceptedPID = true;
122  } else if (species2NsigmaSelection && species2NSigma > species2NsigmaLow && species2NSigma < species2NsigmaHigh) {
123  acceptedPID = true;
124  } else if (species3NsigmaSelection && species3NSigma > species3NsigmaLow && species3NSigma < species3NsigmaHigh) {
125  acceptedPID = true;
126  } else if (species4NsigmaSelection && species4NSigma > species4NsigmaLow && species4NSigma < species4NsigmaHigh) {
127  acceptedPID = true;
128  }
129 
130  if (!acceptedPID) {
131  continue;
132  }
133 
134  if (saveTracks) {
135  outputTracks(track.pt(), track.eta(), track.phi(),
136  SpeciesNSigma(track, 0), SpeciesNSigma(track, 1),
137  SpeciesNSigma(track, 2), SpeciesNSigma(track, 3),
138  SpeciesNSigma(track, 4), SpeciesNSigma(track, 5),
139  SpeciesNSigma(track, 6), SpeciesNSigma(track, 7),
140  SpeciesNSigma(track, 8));
141  }
142  if (saveSmallTracks) {
143  outputSmallTracks(track.pt(), track.eta(), track.phi(),
144  species1NSigma, species2NSigma,
145  species3NSigma, species4NSigma);
146  }
147  if (saveSingleTracks) {
148  outputSingleTracks(track.pt(), track.eta(), track.phi(),
149  species1NSigma);
150  }
151  }
152  }
153 };
