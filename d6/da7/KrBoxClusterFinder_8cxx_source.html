41 // You can pass a CalDet file to it so that the cluster finder can already correct for gain inhomogeneities
42 // The CalDet File should contain the relative Gain of each Pad.
43 void KrBoxClusterFinder::fillAndCorrectMap(std::vector<o2::tpc::Digit>& eventSector, const int sector)
44 {
45  // Reset whole map:
46  for (int iTime = 0; iTime < MaxTimes; iTime++) {
47  for (int iRow = 0; iRow < MaxRows; iRow++) {
48  for (int iPad = 0; iPad < MaxPads; iPad++) {
49  mMapOfAllDigits[iTime][iRow][iPad] = 0;
50  }
51  }
52  }
53 
54  if (eventSector.size() == 0) {
55  // prevents a segementation fault if the envent contains no data
56  // segementation fault would occure later when trying to dereference the
57  // max/min pointer (which are nullptr if data is empty) empty events should
58  // be catched in the main function, hence, this "if" is no longer necessary
59  LOGP(warning, "Sector size (amount of data points) in current run is 0!");
60  LOGP(warning, "mMapOfAllDigits with 0's is generated in order to prevent a segementation fault.");
61 
62  return;
63  }
64 
65  const auto correctionFactorCalDet = mGainMap.get();
66 
67  // Fill digits map
68  for (const auto& digit : eventSector) {
69  const int time = digit.getTimeStamp();
70  const int row = digit.getRow();
71  const int pad = digit.getPad();
72 
73  float correctionFactor = 1.0;
74 
75  if (correctionFactorCalDet) {
76  int padNum = mMapperInstance.globalPadNumber(PadPos(row, pad));
77  correctionFactor = correctionFactorCalDet->getValue(sector, padNum);
78  }
79  // Every row starts at pad zero. But the number of pads in a row is not a constant.
80  // If we would just fill the map naively, we would put pads next to each other, which are not neighbours on the pad plane.
81  // Hence, we need to correct for this:
82  const int pads = mMapperInstance.getNumberOfPadsInRowSector(row);
83  const int corPad = pad - (pads / 2) + (MaxPads / 2);
84 
85  if (correctionFactor == 0) {
86  LOGP(warning, "Encountered correction factor which is zero.");
87  LOGP(warning, "Digit will be set to 0!");
88  mMapOfAllDigits[time][row][corPad] = 0;
89  } else {
90  mMapOfAllDigits[time][row][corPad] = digit.getChargeFloat() / correctionFactor;
91  }
92  }
93 }

97 // Function to update the temporal cluster
98 void KrBoxClusterFinder::updateTempClusterFinal()
99 {
100  const float oneOverQtot = 1. / mTempCluster.totCharge;
101  mTempCluster.meanPad *= oneOverQtot;
102  mTempCluster.sigmaPad *= oneOverQtot;
103  mTempCluster.meanRow *= oneOverQtot;
104  mTempCluster.sigmaRow *= oneOverQtot;
105  mTempCluster.meanTime *= oneOverQtot;
106  mTempCluster.sigmaTime *= oneOverQtot;
107  mTempCluster.sigmaPad = std::sqrt(std::abs(mTempCluster.sigmaPad - mTempCluster.meanPad * mTempCluster.meanPad));
108  mTempCluster.sigmaRow = std::sqrt(std::abs(mTempCluster.sigmaRow - mTempCluster.meanRow * mTempCluster.meanRow));
109  mTempCluster.sigmaTime = std::sqrt(std::abs(mTempCluster.sigmaTime - mTempCluster.meanTime * mTempCluster.meanTime));
110 
111  const int corPadsMean = mMapperInstance.getNumberOfPadsInRowSector(int(mTempCluster.meanRow));
112  const int corPadsMaxCharge = mMapperInstance.getNumberOfPadsInRowSector(int(mTempCluster.maxChargeRow));
113 
114  // Since every padrow is shifted such that neighbouring pads are indeed neighbours, we have to shift once back:
115  mTempCluster.meanPad = mTempCluster.meanPad + (corPadsMean / 2.0) - (MaxPads / 2.0);
116  mTempCluster.maxChargePad = mTempCluster.maxChargePad + (corPadsMaxCharge / 2.0) - (MaxPads / 2.0);
117 }

119 // Function to update the temporal cluster.
120 void KrBoxClusterFinder::updateTempCluster(float tempCharge, int tempPad, int tempRow, int tempTime)
121 {
122  if (tempCharge < mQThreshold) {
123  LOGP(warning, "Update cluster was called but current charge is below mQThreshold");
124  return;
125  }
126 
127  mTempCluster.size += 1;
128  mTempCluster.totCharge += tempCharge;
129 
130  mTempCluster.meanPad += tempPad * tempCharge;
131  mTempCluster.sigmaPad += tempPad * tempPad * tempCharge;
132 
133  mTempCluster.meanRow += tempRow * tempCharge;
134  mTempCluster.sigmaRow += tempRow * tempRow * tempCharge;
135 
136  mTempCluster.meanTime += tempTime * tempCharge;
137  mTempCluster.sigmaTime += tempTime * tempTime * tempCharge;
138 
139  if (tempCharge > mTempCluster.maxCharge) {
140  mTempCluster.maxCharge = tempCharge;
141  mTempCluster.maxChargePad = tempPad;
142  mTempCluster.maxChargeRow = tempRow;
143  }
144 }

146 // This function finds and evaluates all clusters in a 3D mMapOfAllDigits generated by the
147 // mMapOfAllDigitsCreator function, this function also updates the cluster tree
148 std::vector<std::tuple<int, int, int>> KrBoxClusterFinder::findLocalMaxima()
149 {
150  std::vector<std::tuple<int, int, int>> localMaximaCoords;
151  // loop over whole mMapOfAllDigits the find clusters
152  for (int iTime = 0; iTime < MaxTimes; iTime++) { //mMapOfAllDigits.size()
153  const auto& mapRow = mMapOfAllDigits[iTime];
154  for (int iRow = 0; iRow < MaxRows; iRow++) { // mapRow.size()
155  // Since pad size is different for each ROC, we take this into account while looking for maxima:
156  setMaxClusterSize(iRow);
157 
158  const auto& mapPad = mapRow[iRow];
159  const int padsInRow = mMapperInstance.getNumberOfPadsInRowSector(iRow);
160 
161  // Only loop over existing pads:
162  for (int iPad = MaxPads / 2 - padsInRow / 2; iPad < MaxPads / 2 + padsInRow / 2; iPad++) { // mapPad.size()
163 
164  const float qMax = mapPad[iPad];
165 
166  // cluster Maximum must at least be larger than Threshold
167  if (qMax <= mQThresholdMax) {
168  continue;
169  }
170 
171  // Acceptance condition: Require at least mMinNumberOfNeighbours neigbours
172  // with signal in any direction!
173  int noNeighbours = 0;
174  if ((iPad + 1 < MaxPads) && (mMapOfAllDigits[iTime][iRow][iPad + 1] > mQThreshold)) {
175  noNeighbours++;
176  }
177  if ((iPad - 1 >= 0) && (mMapOfAllDigits[iTime][iRow][iPad - 1] > mQThreshold)) {
178  noNeighbours++;
179  }
180  if ((iTime + 1 < MaxTimes) && (mMapOfAllDigits[iTime + 1][iRow][iPad] > mQThreshold)) {
181  noNeighbours++;
182  }
183  if ((iTime - 1 >= 0) && (mMapOfAllDigits[iTime - 1][iRow][iPad] > mQThreshold)) {
184  noNeighbours++;
185  }
186  if ((iRow + 1 < MaxRows) && (mMapOfAllDigits[iTime][iRow + 1][iPad] > mQThreshold)) {
187  noNeighbours++;
188  }
189  if ((iRow - 1 >= 0) && (mMapOfAllDigits[iTime][iRow - 1][iPad] > mQThreshold)) {
190  noNeighbours++;
191  }
192  if (noNeighbours < mMinNumberOfNeighbours) {
193  continue;
194  }
195 
196  // Check that this is a local maximum
197  // Note that the checking is done so that if 2 charges have the same
198  // qMax then only 1 cluster is generated
199  // (that is why there is BOTH > and >=)
200  // -> only the maximum with the smalest indices will be accepted
201  bool thisIsMax = true;
202 
203  for (int i = -mMaxClusterSizePad; (i <= mMaxClusterSizePad) && thisIsMax; i++) {
204  if ((iPad + i >= MaxPads) || (iPad + i < 0)) {
205  continue;
206  }
207  for (int k = -mMaxClusterSizeRow; (k <= mMaxClusterSizeRow) && thisIsMax; k++) {
208  if ((iRow + k >= MaxRows) || (iRow + k < 0)) {
209  continue;
210  }
211  for (int j = -mMaxClusterSizeTime; (j <= mMaxClusterSizeTime) && thisIsMax; j++) {
212  if ((iTime + j >= MaxTimes) || (iTime + j < 0)) {
213  continue;
214  }
215  if (mMapOfAllDigits[iTime + j][iRow + k][iPad + i] > qMax) {
216  thisIsMax = false;
217  }
218  }
219  }
220  }
221 
222  if (!thisIsMax) {
223  continue;
224  } else {
225  localMaximaCoords.emplace_back(std::make_tuple(iPad, iRow, iTime));
226  }
227  }
228  }
229  }
230  return localMaximaCoords;
231 }

233 // Calculate the total charge as the sum over the region:
234 //
235 // o o o o o
236 // o i i i o
237 // o i C i o
238 // o i i i o
239 // o o o o o
240 //
241 // with qmax at the center C.
242 //
243 // The inner charge (i) we always add, but we only add the outer
244 // charge (o) if the neighboring inner bin (i) has a signal.

247 // for loop over whole cluster, to determine if a charge should be added
248 // conditions are extrapolation of the 5x5 cluster case to arbitrary
249 // cluster sizes in 3 dimensions
250 KrCluster KrBoxClusterFinder::buildCluster(int clusterCenterPad, int clusterCenterRow, int clusterCenterTime)
251 {
252  mTempCluster = KrCluster();
253  mTempCluster.reset();
254 
255  setMaxClusterSize(clusterCenterRow);
256 
257  // Loop over all neighbouring time bins:
258  for (int iTime = -mMaxClusterSizeTime; iTime <= mMaxClusterSizeTime; iTime++) {
259  // If we would look out of range, we skip
260  if (clusterCenterTime + iTime < 0) {
261  continue;
262  } else if (clusterCenterTime + iTime >= MaxTimes) {
263  break;
264  }
265 
266  // Loop over all neighbouring row bins:
267  for (int iRow = -mMaxClusterSizeRow; iRow <= mMaxClusterSizeRow; iRow++) {
268  // First: Check again if we look over array boundaries:
269  if (clusterCenterRow + iRow < 0) {
270  continue;
271  } else if (clusterCenterRow + iRow >= MaxRows) {
272  break;
273  }
274  // Second: Check if we might look over ROC boundaries:
275  else if (clusterCenterRow < MaxRowsIROC) {
276  if (clusterCenterRow + iRow > MaxRowsIROC) {
277  break;
278  }
279  } else if (clusterCenterRow < MaxRowsIROC + MaxRowsOROC1) {
280  if (clusterCenterRow + iRow < MaxRowsIROC || clusterCenterRow + iRow >= MaxRowsIROC + MaxRowsOROC1) {
281  continue;
282  }
283  } else if (clusterCenterRow < MaxRowsIROC + MaxRowsOROC1 + MaxRowsOROC2) {
284  if (clusterCenterRow + iRow < MaxRowsIROC + MaxRowsOROC1 || clusterCenterRow + iRow >= MaxRowsIROC + MaxRowsOROC1 + MaxRowsOROC2) {
285  continue;
286  }
287  } else if (clusterCenterRow < MaxRowsIROC + MaxRowsOROC1 + MaxRowsOROC2 + MaxRowsOROC3) {
288  if (clusterCenterRow + iRow < MaxRowsIROC + MaxRowsOROC1 + MaxRowsOROC2) {
289  continue;
290  }
291  }
292 
293  // Loop over all neighbouring pad bins:
294  for (int iPad = -mMaxClusterSizePad; iPad <= mMaxClusterSizePad; iPad++) {
295  // First: Check again if we might look outside of map:
296  if (clusterCenterPad + iPad < 0) {
297  continue;
298  } else if (clusterCenterPad + iPad >= MaxPads) {
299  break;
300  }
301 
302  // Second: Check if charge is above threshold
303  // Might be not necessary since we deal with pedestal subtracted data
304  if (mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad] <= mQThreshold) {
305  continue;
306  }
307 
308  // If not, there are several cases which were explained (for 2D) in the header of the code.
309  // The first one is for the diagonal. So, the digit we are investigating here is on the diagonal:
310  if (std::abs(iTime) == std::abs(iPad) && std::abs(iTime) == std::abs(iRow)) {
311  // Now we check, if the next inner digit has a signal above threshold:
312  if (mMapOfAllDigits[clusterCenterTime + iTime - signnum(iTime)][clusterCenterRow + iRow - signnum(iRow)][clusterCenterPad + iPad - signnum(iPad)] > mQThreshold) {
313  // If yes, the cluster gets updated with the digit on the diagonal.
314  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
315  }
316  }
317  // Basically, we go through every possible case in the next few if-else conditions:
318  else if (std::abs(iTime) == std::abs(iPad)) {
319  if (mMapOfAllDigits[clusterCenterTime + iTime - signnum(iTime)][clusterCenterRow + iRow][clusterCenterPad + iPad - signnum(iPad)] > mQThreshold) {
320  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
321  }
322  } else if (std::abs(iTime) == std::abs(iRow)) {
323  if (mMapOfAllDigits[clusterCenterTime + iTime - signnum(iTime)][clusterCenterRow + iRow - signnum(iRow)][clusterCenterPad + iPad] > mQThreshold) {
324  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
325  }
326  } else if (std::abs(iPad) == std::abs(iRow)) {
327  if (mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow - signnum(iRow)][clusterCenterPad + iPad - signnum(iPad)] > mQThreshold) {
328  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
329  }
330  } else if (std::abs(iTime) > std::abs(iPad) && std::abs(iTime) > std::abs(iRow)) {
331  if (mMapOfAllDigits[clusterCenterTime + iTime - signnum(iTime)][clusterCenterRow + iRow][clusterCenterPad + iPad] > mQThreshold) {
332  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
333  }
334  } else if (std::abs(iTime) < std::abs(iPad) && std::abs(iPad) > std::abs(iRow)) {
335  if (mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad - signnum(iPad)] > mQThreshold) {
336  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
337  }
338  } else if (std::abs(iTime) < std::abs(iRow) && std::abs(iPad) < std::abs(iRow)) {
339  if (mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow - signnum(iRow)][clusterCenterPad + iPad] > mQThreshold) {
340  updateTempCluster(mMapOfAllDigits[clusterCenterTime + iTime][clusterCenterRow + iRow][clusterCenterPad + iPad], clusterCenterPad + iPad, clusterCenterRow + iRow, clusterCenterTime + iTime);
341  }
342  }
343  }
344  }
345  }
346  // At the end, out mTempCluster should contain all digits that were assigned to the cluster.
347  // So before returning it, we update it one last time to calculate the correct means and sigmas.
348  updateTempClusterFinal();
349  return mTempCluster;
350 }
