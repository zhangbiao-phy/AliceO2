70  void process(soa::Filtered<aod::Collisions>::iterator const& collision, soa::Filtered<soa::Join<aod::Tracks, aod::TPhi>> const& tracks)
71  {
72  LOGF(INFO, "Collision: %d [N = %d out of %d], -10 < %.3f < 10",
73  collision.globalIndex(), tracks.size(), tracks.tableSize(), collision.posZ());
74  for (auto& track : tracks) {
75  LOGF(INFO, "id = %d; eta: %.3f < %.3f < %.3f; phi: %.3f < %.3f < %.3f; pt: %.3f < %.3f < %.3f",
76  track.collisionId(), etalow, track.eta(), etaup, philow, track.nphi(), phiup, ptlow, track.pt(), ptup);
77  }
78  }
