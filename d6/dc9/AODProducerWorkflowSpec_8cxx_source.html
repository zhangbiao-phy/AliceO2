50 void AODProducerWorkflowDPL::findMinMaxBc(gsl::span<const o2::ft0::RecPoints>& ft0RecPoints, gsl::span<const o2::vertexing::PVertex>& primVertices, const std::vector<o2::InteractionTimeRecord>& mcRecords)
51 {
52  for (auto& ft0RecPoint : ft0RecPoints) {
53  uint64_t bc = ft0RecPoint.getInteractionRecord().orbit * o2::constants::lhc::LHCMaxBunches + ft0RecPoint.getInteractionRecord().bc;
54  if (minGlBC > bc) {
55  minGlBC = bc;
56  }
57  if (maxGlBC < bc) {
58  maxGlBC = bc;
59  }
60  }
61 
62  for (auto& vertex : primVertices) {
63  const InteractionRecord& colIRMin = vertex.getIRMin();
64  const InteractionRecord& colIRMax = vertex.getIRMax();
65  uint64_t colBCMin = colIRMin.orbit * o2::constants::lhc::LHCMaxBunches + colIRMin.bc;
66  uint64_t colBCMax = colIRMax.orbit * o2::constants::lhc::LHCMaxBunches + colIRMax.bc;
67  if (minGlBC > colBCMin) {
68  minGlBC = colBCMin;
69  }
70  if (maxGlBC < colBCMax) {
71  maxGlBC = colBCMax;
72  }
73  }
74 
75  for (auto& rec : mcRecords) {
76  uint64_t bc = rec.bc + rec.orbit * o2::constants::lhc::LHCMaxBunches;
77  if (minGlBC > bc) {
78  minGlBC = bc;
79  }
80  if (maxGlBC < bc) {
81  maxGlBC = bc;
82  }
83  }
84 }

86 int64_t AODProducerWorkflowDPL::getTFNumber(uint64_t firstVtxGlBC, int runNumber)
87 {
88  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
89  o2::ccdb::CcdbApi ccdb_api;
90  const std::string rct_path = "RCT/RunInformation/";
91  const std::string start_orbit_path = "Trigger/StartOrbit";
92  const std::string url = "http://ccdb-test.cern.ch:8080";
93 
94  mgr.setURL(url);
95  ccdb_api.init(url);
96 
97  std::map<int, int>* mapStartOrbit = mgr.get<std::map<int, int>>(start_orbit_path);
98  int64_t ts = 0;
99  std::map<std::string, std::string> metadata;
100  std::map<std::string, std::string> headers;
101  const std::string run_path = Form("%s/%i", rct_path.data(), runNumber);
102  headers = ccdb_api.retrieveHeaders(run_path, metadata, -1);
103  ts = atol(headers["SOR"].c_str());
104 
105  // ccdb returns timestamp in mus
106  // mus to ms
107  ts = ts / 1000;
108 
109  // firstRec --> calculated using `minimal` global BC in the simulation (see AODProducerWorkflowDPL::findMinMaxBc)
110  // firstVtxGlBC --> calculated using global BC corresponding to the first prim. vertex
111 
112  uint32_t initialOrbit = mapStartOrbit->at(runNumber);
113  uint16_t firstRecBC = minGlBC % o2::constants::lhc::LHCMaxBunches;
114  uint32_t firstRecOrbit = minGlBC / o2::constants::lhc::LHCMaxBunches;
115  uint16_t firstVtxBC = firstVtxGlBC % o2::constants::lhc::LHCMaxBunches;
116  uint32_t firstVtxOrbit = firstVtxGlBC / o2::constants::lhc::LHCMaxBunches;
117  const o2::InteractionRecord firstRec(firstRecBC, firstRecOrbit);
118  const o2::InteractionRecord firstVtx(firstVtxBC, firstVtxOrbit);
119  ts += (firstVtx - firstRec).bc2ns() / 1000000;
120 
121  return ts;
122 };

204  truncateFloatFraction(tpcChi2NCl, mTrackCovOffDiag),
205  truncateFloatFraction(trdChi2, mTrackCovOffDiag),
206  truncateFloatFraction(tofChi2, mTrackCovOffDiag),
207  truncateFloatFraction(tpcSignal, mTrackSignal),
208  truncateFloatFraction(trdSignal, mTrackSignal),
209  truncateFloatFraction(tofSignal, mTrackSignal),
210  truncateFloatFraction(length, mTrackSignal),
211  truncateFloatFraction(tofExpMom, mTrack1Pt),

213  truncateFloatFraction(trackPhiEMCAL, mTrackPosEMCAL));

497  BCid = globalBC - minGlBC;
498  if (BCid < 0) {
499  BCid = 0;
500  } else if (BCid > maxGlBC) {
501  BCid = maxGlBC;
502  }
503  ft0Cursor(0,

511 
512  // initializing vectors for trackID --> collisionID connection
513  std::vector<int> vCollRefsITS(tracksITS.size(), -1);
514  std::vector<int> vCollRefsTPC(tracksTPC.size(), -1);
515  std::vector<int> vCollRefsTPCITS(tracksITSTPC.size(), -1);
516 
517  // TODO: determine the beginning of a TF in case when there are no reconstructed vertices
518  uint64_t firstVtxGlBC = minGlBC;
519  uint64_t startBCofTF = 0;
520  if (primVertices.size()) {
521  auto startIRofTF = o2::raw::HBFUtils::Instance().getFirstIRofTF(primVertices[0].getIRMin());
522  startBCofTF = startIRofTF.orbit * o2::constants::lhc::LHCMaxBunches + startIRofTF.bc;
523  firstVtxGlBC = std::round(startBCofTF + primVertices[0].getTimeStamp().getTimeStamp() / o2::constants::lhc::LHCBunchSpacingMS);
524  }
525 
526  // filling collisions table
527  int collisionID = 0;
528  for (auto& vertex : primVertices) {
529  auto& cov = vertex.getCov();
530  auto& timeStamp = vertex.getTimeStamp();
531  Double_t tsTimeStamp = timeStamp.getTimeStamp() * 1E3; // mus to ns
532  globalBC = std::round(startBCofTF + tsTimeStamp / o2::constants::lhc::LHCBunchSpacingNS);
533  LOG(DEBUG) << globalBC << " " << tsTimeStamp;
534  // collision timestamp in ns wrt the beginning of collision BC
535  tsTimeStamp = globalBC * o2::constants::lhc::LHCBunchSpacingNS - tsTimeStamp;
536  BCid = globalBC - minGlBC;
537  if (BCid < 0) {
538  BCid = 0;
539  } else if (BCid > maxGlBC) {
540  BCid = maxGlBC;
541  }
542  // TODO: get real collision time mask
543  int collisionTimeMask = 0;
544  collisionsCursor(0,
545  BCid,
546  truncateFloatFraction(vertex.getX(), mCollisionPosition),
547  truncateFloatFraction(vertex.getY(), mCollisionPosition),
548  truncateFloatFraction(vertex.getZ(), mCollisionPosition),
549  truncateFloatFraction(cov[0], mCollisionPositionCov),
550  truncateFloatFraction(cov[1], mCollisionPositionCov),
551  truncateFloatFraction(cov[2], mCollisionPositionCov),
552  truncateFloatFraction(cov[3], mCollisionPositionCov),
553  truncateFloatFraction(cov[4], mCollisionPositionCov),
554  truncateFloatFraction(cov[5], mCollisionPositionCov),
555  vertex.getFlags(),
556  truncateFloatFraction(vertex.getChi2(), mCollisionPositionCov),
557  vertex.getNContributors(),
558  truncateFloatFraction(tsTimeStamp, mCollisionPosition),
559  truncateFloatFraction(timeStamp.getTimeStampError() * 1E3, mCollisionPositionCov),
560  collisionTimeMask);
561 
562  auto trackRef = primVer2TRefs[collisionID];
563  int start = trackRef.getFirstEntryOfSource(0);
564  int ntracks = trackRef.getEntriesOfSource(0);
565  // FIXME: `track<-->vertex` ambiguity is not accounted for in this code
566  for (int ti = 0; ti < ntracks; ti++) {
567  auto trackIndex = primVerGIs[start + ti];
568  const auto source = trackIndex.getSource();
569  // setting collisionID for tracks attached to vertices
570  if (source == o2::vertexing::GIndex::Source::TPC) {
571  vCollRefsTPC[trackIndex.getIndex()] = collisionID;
572  } else if (source == o2::vertexing::GIndex::Source::ITS) {
573  vCollRefsITS[trackIndex.getIndex()] = collisionID;
574  } else if (source == o2::vertexing::GIndex::Source::ITSTPC) {
575  vCollRefsTPCITS[trackIndex.getIndex()] = collisionID;
576  } else {
577  LOG(WARNING) << "Unsupported track type!";
578  }
579  }
580  collisionID++;
581  }
582 
583  // filling tracks tables and track label table
584 
585  // labelMask (temporary) usage:
586  // bit 13 -- ITS and TPC labels are not equal
587  // bit 14 -- isNoise() == true
588  // bit 15 -- isFake() == true
589  // labelID = std::numeric_limits<uint32_t>::max() -- label is not set
590 
591  uint32_t labelID;
592  uint32_t labelITS;
593  uint32_t labelTPC;
594  uint16_t labelMask;
595 
596  if (mFillTracksITS) {
597  fillTracksTable(tracksITS, vCollRefsITS, tracksCursor, o2::vertexing::GIndex::Source::ITS);
598  for (auto& mcTruthITS : tracksITSMCTruth) {
599  labelID = std::numeric_limits<uint32_t>::max();
600  // TODO: fill label mask
601  labelMask = 0;
602  if (mcTruthITS.isValid()) {
603  labelID = mIDsToIndex.at(std::make_tuple(mcTruthITS.getSourceID(), mcTruthITS.getEventID(), mcTruthITS.getTrackID()));
604  }
605  if (mcTruthITS.isFake()) {
606  labelMask |= (0x1 << 15);
607  }
608  if (mcTruthITS.isNoise()) {
609  labelMask |= (0x1 << 14);
610  }
611  mcTrackLabelCursor(0,
612  labelID,
613  labelMask);
614  }
615  }
616 
617  if (mFillTracksTPC) {
618  fillTracksTable(tracksTPC, vCollRefsTPC, tracksCursor, o2::vertexing::GIndex::Source::TPC);
619  for (auto& mcTruthTPC : tracksTPCMCTruth) {
620  labelID = std::numeric_limits<uint32_t>::max();
621  // TODO: fill label mask
622  labelMask = 0;
623  if (mcTruthTPC.isValid()) {
624  labelID = mIDsToIndex.at(std::make_tuple(mcTruthTPC.getSourceID(), mcTruthTPC.getEventID(), mcTruthTPC.getTrackID()));
625  }
626  if (mcTruthTPC.isFake()) {
627  labelMask |= (0x1 << 15);
628  }
629  if (mcTruthTPC.isNoise()) {
630  labelMask |= (0x1 << 14);
631  }
632  mcTrackLabelCursor(0,
633  labelID,
634  labelMask);
635  }
636  }
637 
638  if (mFillTracksITSTPC) {
639  fillTracksTable(tracksITSTPC, vCollRefsTPCITS, tracksCursor, o2::vertexing::GIndex::Source::ITSTPC);
640  for (int i = 0; i < tracksITSTPC.size(); i++) {
641  const auto& trc = tracksITSTPC[i];
642  auto mcTruthITS = tracksITSMCTruth[trc.getRefITS()];
643  auto mcTruthTPC = tracksTPCMCTruth[trc.getRefTPC()];
644  labelID = std::numeric_limits<uint32_t>::max();
645  labelITS = std::numeric_limits<uint32_t>::max();
646  labelTPC = std::numeric_limits<uint32_t>::max();
647  // TODO: fill label mask
648  // currently using label mask to indicate labelITS != labelTPC
649  labelMask = 0;
650  if (mcTruthITS.isValid() && mcTruthTPC.isValid()) {
651  labelITS = mIDsToIndex.at(std::make_tuple(mcTruthITS.getSourceID(), mcTruthITS.getEventID(), mcTruthITS.getTrackID()));
652  labelTPC = mIDsToIndex.at(std::make_tuple(mcTruthTPC.getSourceID(), mcTruthTPC.getEventID(), mcTruthTPC.getTrackID()));
653  labelID = labelITS;
654  }
655  if (mcTruthITS.isFake() || mcTruthTPC.isFake()) {
656  labelMask |= (0x1 << 15);
657  }
658  if (mcTruthITS.isNoise() || mcTruthTPC.isNoise()) {
659  labelMask |= (0x1 << 14);
660  }
661  if (labelITS != labelTPC) {
662  LOG(DEBUG) << "ITS-TPC MCTruth: labelIDs do not match at " << i;
663  labelMask |= (0x1 << 13);
664  }
665  mcTrackLabelCursor(0,
666  labelID,
667  labelMask);
668  }
669  }
670 
671  if (mTFNumber == -1) {
672  timeFrameNumberBuilder = getTFNumber(firstVtxGlBC, runNumber);
673  } else {
674  timeFrameNumberBuilder = mTFNumber;
675  }
676 
677  mTimer.Stop();
678 }
