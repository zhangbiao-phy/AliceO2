23 void GPUO2InterfaceRefit::fillSharedClustersMap(const ClusterNativeAccess* cl, const gsl::span<const TrackTPC> trks, const TPCClRefElem* trackRef, unsigned char* shmap)
24 {
25  if (!cl || !shmap) {
26  throw std::runtime_error("Must provide clusters access and preallocated recepient for shared map");
27  }
28  memset(shmap, 0, sizeof(char) * cl->nClustersTotal);
29  for (unsigned int i = 0; i < trks.size(); i++) {
30  for (unsigned int j = 0; j < trks[i].getNClusterReferences(); j++) {
31  size_t idx = &trks[i].getCluster(trackRef, j, *cl) - cl->clustersLinear;
32  shmap[idx] = shmap[idx] ? 2 : 1;
33  }
34  }
35  for (unsigned int i = 0; i < cl->nClustersTotal; i++) {
36  shmap[i] = (shmap[i] > 1 ? GPUTPCGMMergedTrackHit::flagShared : 0) | cl->clustersLinear[i].getFlags();
37  }
38 }
