61 void Alice3Pipe::ConstructGeometry()
62 {
63  createMaterials();
64  //
65  // Class describing the beam Alice3Pipe geometry
66  //
67  float z, zsh, z0;
68  //
69  // Rotation Matrices
70  //
71  const float kDegRad = TMath::Pi() / 180.;
72  // Rotation by 180 deg
73  TGeoRotation* rot180 = new TGeoRotation("rot180", 90., 180., 90., 90., 180., 0.);
74  TGeoRotation* rotyz = new TGeoRotation("rotyz", 90., 180., 0., 180., 90., 90.);
75  TGeoRotation* rotxz = new TGeoRotation("rotxz", 0., 0., 90., 90., 90., 180.);
76  //
77 
78  //
79  // Media
80  auto& matmgr = o2::base::MaterialManager::Instance();
81 
82  const TGeoMedium* kMedBe = matmgr.getTGeoMedium("ALICE3PIPE_BE");
83  const TGeoMedium* kMedVac = matmgr.getTGeoMedium("ALICE3PIPE_VACUUM");
84  const TGeoMedium* kMedVacNF = matmgr.getTGeoMedium("ALICE3PIPE_VACUUM_NF");
85  const TGeoMedium* kMedVacHC = matmgr.getTGeoMedium("ALICE3PIPE_VACUUM_HC");
86  const TGeoMedium* kMedVacNFHC = matmgr.getTGeoMedium("ALICE3PIPE_VACUUM_NFHC");
87 
88  // Top volume
89  TGeoVolume* top = gGeoManager->GetVolume("cave");
90  TGeoVolume* barrel = gGeoManager->GetVolume("barrel");
91  if (!barrel) {
92  LOG(FATAL) << "Could not find the top volume";
93  }
94 
95  //---------------- Innermost Be pipe around the IP ----------
96  TGeoTube* innerBeTube =
97  new TGeoTube("INN_PIPEsh", mBeInnerPipeRmax - mBeInnerPipeThick, mBeInnerPipeRmax, mInnerIpHLength);
98  TGeoVolume* innerBeTubeVolume = new TGeoVolume("INN_PIPE", innerBeTube, kMedBe);
99  innerBeTubeVolume->SetLineColor(kRed);
100 
101  TGeoTube* berylliumTubeVacuum =
102  new TGeoTube("INN_PIPEVACUUMsh", 0., mBeInnerPipeRmax, mInnerIpHLength);
103  TGeoVolume* innerBerylliumTubeVacuumVolume = new TGeoVolume("INN_PIPEMOTHER", berylliumTubeVacuum, kMedVac);
104  innerBerylliumTubeVacuumVolume->AddNode(innerBeTubeVolume, 1, gGeoIdentity);
105  innerBerylliumTubeVacuumVolume->SetVisibility(0);
106  innerBerylliumTubeVacuumVolume->SetLineColor(kGreen);
107 
108  barrel->AddNode(innerBerylliumTubeVacuumVolume, 1, gGeoIdentity);
109 
110  //---------------- Outermost Be pipe around the IP ----------
111  TGeoTube* outerBeTube =
112  new TGeoTube("OUT_PIPEsh", mBeOuterPipeRmax - mBeOuterPipeThick, mBeOuterPipeRmax, mOuterIpHLength);
113  TGeoVolume* outerBeTubeVolume = new TGeoVolume("OUT_PIPE", outerBeTube, kMedBe);
114  outerBeTubeVolume->SetLineColor(kBlue);
115 
116  TGeoTube* outerBerylliumTubeVacuum =
117  new TGeoTube("OUT_PIPEVACUUMsh", 0., mBeOuterPipeRmax, mOuterIpHLength);
118  TGeoVolume* outerBerylliumTubeVacuumVolume = new TGeoVolume("OUT_PIPEMOTHER", outerBerylliumTubeVacuum, kMedVac);
119  outerBerylliumTubeVacuumVolume->AddNode(outerBeTubeVolume, 1, gGeoIdentity);
120  outerBerylliumTubeVacuumVolume->SetVisibility(0);
121  outerBerylliumTubeVacuumVolume->SetLineColor(kGreen);
122 
123  barrel->AddNode(outerBerylliumTubeVacuumVolume, 1, gGeoIdentity);
124 }

126 void Alice3Pipe::createMaterials()
127 {
128  //
129  // Define materials for beam Alice3Pipe
130  //
131  Int_t isxfld = 2.;
132  float sxmgmx = 10.;
133  o2::base::Detector::initFieldTrackingParams(isxfld, sxmgmx);
134 
135  //
136  // Air
137  //
138  float aAir[4] = {12.0107, 14.0067, 15.9994, 39.948};
139  float zAir[4] = {6., 7., 8., 18.};
140  float wAir[4] = {0.000124, 0.755267, 0.231781, 0.012827};
141  float dAir = 1.20479E-3;
142  float dAir1 = 1.20479E-11;
143 
144  // ****************
145  // Defines tracking media parameters.
146  //
147  float epsil = .1; // Tracking precision,
148  float stemax = -0.01; // Maximum displacement for multiple scat
149  float tmaxfd = -20.; // Maximum angle due to field deflection
150  float deemax = -.3; // Maximum fractional energy loss, DLS
151  float stmin = -.8;
152  // ***************
153 
154  auto& matmgr = o2::base::MaterialManager::Instance();
155 
156  // Beryllium
157  matmgr.Material("ALICE3PIPE", 5, "BERILLIUM$", 9.01, 4., 1.848, 35.3, 36.7);
158  matmgr.Medium("ALICE3PIPE", 5, "BE", 5, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
159 
160  // Vacuum
161  matmgr.Mixture("ALICE3PIPE", 16, "VACUUM$ ", aAir, zAir, dAir1, 4, wAir);
162  matmgr.Mixture("ALICE3PIPE", 36, "VACUUM$_NF", aAir, zAir, dAir1, 4, wAir);
163  matmgr.Mixture("ALICE3PIPE", 56, "VACUUM$_HC ", aAir, zAir, dAir1, 4, wAir);
164  matmgr.Mixture("ALICE3PIPE", 76, "VACUUM$_NFHC", aAir, zAir, dAir1, 4, wAir);
165 
166  matmgr.Medium("ALICE3PIPE", 16, "VACUUM", 16, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
167  matmgr.Medium("ALICE3PIPE", 36, "VACUUM_NF", 36, 0, 0, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
168  matmgr.Medium("ALICE3PIPE", 56, "VACUUM_HC", 56, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
169  matmgr.Medium("ALICE3PIPE", 76, "VACUUM_NFHC", 76, 0, 0, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
170 }
