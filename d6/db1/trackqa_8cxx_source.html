89  void process(soa::Filtered<soa::Join<aod::FullTracks, aod::TracksExtended, aod::TrackSelection>>::iterator const& track)
90  {
91  // fill kinematic variables
92  histos.fill(HIST("Kine/pt"), track.pt());
93  histos.fill(HIST("Kine/eta"), track.eta());
94  histos.fill(HIST("Kine/phi"), track.phi());
95 
96  // fill track parameters
97  histos.fill(HIST("TrackPar/alpha"), track.alpha());
98  histos.fill(HIST("TrackPar/x"), track.x());
99  histos.fill(HIST("TrackPar/y"), track.y());
100  histos.fill(HIST("TrackPar/z"), track.z());
101  histos.fill(HIST("TrackPar/signed1Pt"), track.signed1Pt());
102  histos.fill(HIST("TrackPar/snp"), track.snp());
103  histos.fill(HIST("TrackPar/tgl"), track.tgl());
104  for (unsigned int i = 0; i < 64; i++) {
105  if (track.flags() & (1 << i)) {
106  histos.fill(HIST("TrackPar/flags"), i);
107  }
108  }
109  histos.fill(HIST("TrackPar/dcaXY"), track.dcaXY());
110  histos.fill(HIST("TrackPar/dcaZ"), track.dcaZ());
111  histos.fill(HIST("TrackPar/length"), track.length());
112 
113  // fill ITS variables
114  histos.fill(HIST("ITS/itsNCls"), track.itsNCls());
115  histos.fill(HIST("ITS/itsChi2NCl"), track.itsChi2NCl());
116  for (unsigned int i = 0; i < 7; i++) {
117  if (track.itsClusterMap() & (1 << i)) {
118  histos.fill(HIST("ITS/itsHits"), i);
119  }
120  }
121 
122  // fill TPC variables
123  histos.fill(HIST("TPC/tpcNClsFindable"), track.tpcNClsFindable());
124  histos.fill(HIST("TPC/tpcNClsFound"), track.tpcNClsFound());
125  histos.fill(HIST("TPC/tpcNClsShared"), track.tpcNClsShared());
126  histos.fill(HIST("TPC/tpcNClsCrossedRows"), track.tpcNClsCrossedRows());
127  histos.fill(HIST("TPC/tpcCrossedRowsOverFindableCls"), track.tpcCrossedRowsOverFindableCls());
128  histos.fill(HIST("TPC/tpcFractionSharedCls"), track.tpcFractionSharedCls());
129  histos.fill(HIST("TPC/tpcChi2NCl"), track.tpcChi2NCl());
130  }
131 };

133 struct TrackCutQATask {
134  HistogramRegistry cuts{"Cuts", {}, OutputObjHandlingPolicy::QAObject};
135  TrackSelection selectedTracks = getGlobalTrackSelection();
136  static constexpr int ncuts = static_cast<int>(TrackSelection::TrackCuts::kNCuts);
137  void init(InitContext&)
138  {
139  cuts.add("single_cut", ";Cut;Tracks", kTH1D, {{ncuts, 0, ncuts}});
140  for (int i = 0; i < ncuts; i++) {
141  cuts.get<TH1>(HIST("single_cut"))->GetXaxis()->SetBinLabel(1 + i, TrackSelection::mCutNames[i].data());
142  }
143  }
144 
145  void process(soa::Join<aod::FullTracks, aod::TracksExtended>::iterator const& track)
146  {
147  for (int i = 0; i < ncuts; i++) {
148  if (selectedTracks.IsSelected(track, static_cast<TrackSelection::TrackCuts>(i))) {
149  cuts.fill(HIST("single_cut"), i);
150  }
151  }
152  }
153 };
