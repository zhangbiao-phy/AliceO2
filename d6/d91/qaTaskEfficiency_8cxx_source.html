47 template <typename T>
48 void makelogaxis(T h)
49 {
50  const int nbins = h->GetNbinsX();
51  double binp[nbins + 1];
52  double max = h->GetXaxis()->GetBinUpEdge(nbins);
53  double min = h->GetXaxis()->GetBinLowEdge(1);
54  if (min <= 0) {
55  min = 0.01;
56  }
57  double lmin = TMath::Log10(min);
58  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
59  for (int i = 0; i < nbins; i++) {
60  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
61  }
62  binp[nbins] = max + 1;
63  h->GetXaxis()->Set(nbins, binp);
64 }

75  o2fw::Configurable<float> phiMax{"phi-max", 2.f * M_PI, "Upper limit in phi"};
76  o2fw::Configurable<float> ptMin{"pt-min", 0.f, "Lower limit in pT"};
77  o2fw::Configurable<float> ptMax{"pt-max", 5.f, "Upper limit in pT"};
78  o2fw::Configurable<int> ptBins{"pt-bins", 500, "Number of pT bins"};
79  o2fw::Configurable<int> logPt{"log-pt", 0, "Flag to use a logarithmic pT axis"};
80  o2fw::Configurable<int> etaBins{"eta-bins", 500, "Number of eta bins"};
81  o2fw::Configurable<int> phiBins{"phi-bins", 500, "Number of phi bins"};
82  o2fw::Configurable<int> selPrim{"sel-prim", 1, "1 select primaries, 0 select all particles"};
83  o2fw::Configurable<int> makeEff{"make-eff", 0, "Flag to produce the efficiency with TEfficiency"};

100  ptMin.value, ptMax.value,

102  selPrim.value);

106  const TString tagPhi = Form("%s #eta [%.2f,%.2f] #it{p}_{T} [%.2f,%.2f] Prim %i",
107  o2::track::pid_constants::sNames[particle],
108  etaMin.value, etaMax.value,
109  ptMin.value, ptMax.value,
110  selPrim.value);

114  histos.add("pt/num", "Numerator " + tagPt + ";" + xPt,

124  o2fw::kTH1D, {axisEta});

128  histos.add("phi/num", "Numerator " + tagPhi + ";" + xPhi,
129  o2fw::kTH1D, {axisPhi});

131  o2fw::kTH1D, {axisPhi});
132 
133  list.setObject(new TList);
134  if (makeEff.value) {
135  auto makeEfficiency = [&](TString effname, TString efftitle, auto templateHisto) {
136  TAxis* axis = histos.get<TH1>(templateHisto)->GetXaxis();
137  if (axis->IsVariableBinSize()) {
138  list->Add(new TEfficiency(effname, efftitle,
139  axis->GetNbins(),
140  axis->GetXbins()->GetArray()));
141  } else {
142  list->Add(new TEfficiency(effname, efftitle,
143  axis->GetNbins(),
144  axis->GetXmin(),
145  axis->GetXmax()));
146  }
147  };
148  makeEfficiency("efficiencyVsPt", "Efficiency " + tagPt + ";" + xPt + ";Efficiency", HIST("pt/num"));
149  makeEfficiency("efficiencyVsEta", "Efficiency " + tagEta + ";" + xEta + ";Efficiency", HIST("eta/num"));
150  makeEfficiency("efficiencyVsPhi", "Efficiency " + tagPhi + ";" + xPhi + ";Efficiency", HIST("phi/num"));
151  }
152  }

154  void process(const o2::soa::Join<o2::aod::Tracks, o2::aod::McTrackLabels>& tracks,
155  const o2::aod::McParticles& mcParticles)
156  {
157  std::vector<int64_t> recoTracks(tracks.size());
158  int ntrks = 0;
159  for (const auto& track : tracks) {
160  const auto mcParticle = track.mcParticle();
161  if ((mcParticle.eta() < etaMin.value || mcParticle.eta() > etaMax.value)) { // Check eta
162  continue;
163  }
164  if ((mcParticle.phi() < phiMin.value || mcParticle.phi() > phiMax.value)) { // Check phi
165  continue;
166  }
167  if ((selPrim.value == 1) && (!MC::isPhysicalPrimary(mcParticles, mcParticle))) { // Requiring is physical primary
168  continue;
169  }
170  if (abs(mcParticle.pdgCode()) == particlePDG) { // Checking PDG code
171  histos.fill(HIST("pt/num"), mcParticle.pt());
172  histos.fill(HIST("eta/num"), mcParticle.eta());
173  histos.fill(HIST("phi/num"), mcParticle.phi());
174  recoTracks[ntrks++] = mcParticle.globalIndex();
175  }
176  }
177 
178  for (const auto& mcParticle : mcParticles) {
179  if ((mcParticle.eta() < etaMin.value || mcParticle.eta() > etaMax.value)) { // Check eta
180  continue;
181  }
182  if ((mcParticle.phi() < phiMin.value || mcParticle.phi() > phiMax.value)) { // Check phi
183  continue;
184  }
185  if ((selPrim.value == 1) && (!MC::isPhysicalPrimary(mcParticles, mcParticle))) { // Requiring is physical primary
186  continue;
187  }
188  if (abs(mcParticle.pdgCode()) == particlePDG) { // Checking PDG code
189  if (makeEff.value) {
190  const auto particleReconstructed = std::find(recoTracks.begin(), recoTracks.end(), mcParticle.globalIndex()) != recoTracks.end();
191  static_cast<TEfficiency*>(list->At(0))->Fill(particleReconstructed, mcParticle.pt());
192  static_cast<TEfficiency*>(list->At(1))->Fill(particleReconstructed, mcParticle.eta());
193  static_cast<TEfficiency*>(list->At(2))->Fill(particleReconstructed, mcParticle.phi());
194  }
195  histos.fill(HIST("pt/den"), mcParticle.pt());
196  histos.fill(HIST("eta/den"), mcParticle.eta());
197  histos.fill(HIST("phi/den"), mcParticle.phi());
198  }
199  }
200  }
