33 struct NucleiSpectraReferenceTask {
34 
35  HistogramRegistry spectra{"spectra", {}, OutputObjHandlingPolicy::AnalysisObject, true, true};
36 
37  void init(o2::framework::InitContext&)
38  {
39  std::vector<double> ptBinning = {0.5, 0.6, 0.7, 0.8, 0.9, 1.0, 1.1, 1.2, 1.3, 1.4, 1.5, 1.6, 1.8, 2.0, 2.2, 2.4, 2.8, 3.2, 3.6, 4., 5.};
40  std::vector<double> centBinning = {0., 1., 5., 10., 20., 30., 40., 50., 70., 100.};
41 
42  AxisSpec ptAxis = {ptBinning, "#it{p}_{T} (GeV/#it{c})"};
43  AxisSpec centAxis = {centBinning, "V0M (%)"};
44 
45  spectra.add("fCollZpos", "collision z position", HistType::kTH1F, {{600, -20., +20., "z position (cm)"}});
46  spectra.add("fKeepEvent", "skimming histogram", HistType::kTH1F, {{2, -0.5, +1.5, "true: keep event, false: reject event"}});
47  spectra.add("fTPCsignal", "Specific energy loss", HistType::kTH2F, {{600, 0., 3, "#it{p} (GeV/#it{c})"}, {1400, 0, 1400, "d#it{E} / d#it{X} (a. u.)"}});
48  spectra.add("fTPCcounts", "n-sigma TPC", HistType::kTH2F, {ptAxis, {200, -100., +100., "n#sigma_{He} (a. u.)"}});
49  spectra.add("fHePt", "He pT", HistType::kTH1F, {{60, 0.0, 30., "#it{p}_{T} (GeV/#it{c})"}});
50  }
51 
52  Configurable<float> yMin{"yMin", -0.8, "Maximum rapidity"};
53  Configurable<float> yMax{"yMax", 0.8, "Minimum rapidity"};
54  Configurable<float> yBeam{"yBeam", 0., "Beam rapidity"};
55 
56  Configurable<float> cfgCutVertex{"cfgCutVertex", 10.0f, "Accepted z-vertex range"};
57  Configurable<float> cfgCutEta{"cfgCutEta", 0.8f, "Eta range for tracks"};
58  Configurable<float> nsigmacutLow{"nsigmacutLow", -30.0, "Value of the Nsigma cut"};
59  Configurable<float> nsigmacutHigh{"nsigmacutHigh", +3., "Value of the Nsigma cut"};
60 
61  Filter collisionFilter = nabs(aod::collision::posZ) < cfgCutVertex;
62  Filter trackFilter = (nabs(aod::track::eta) < cfgCutEta) && (aod::track::isGlobalTrack == (uint8_t) true);
63 
64  using TrackCandidates = soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra, aod::pidRespTPC, aod::pidRespTOF, aod::pidRespTOFbeta, aod::TrackSelection>>;
65 
66  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>::iterator const& collision, aod::BCsWithTimestamps const&, TrackCandidates const& tracks)
67  {
68  //
69  // collision process loop
70  //
71  bool keepEvent = kFALSE;
72  if (!collision.alias()[kINT7]) {
73  return;
74  }
75  if (!collision.sel7()) {
76  return;
77  }
78  //
79  spectra.fill(HIST("fCollZpos"), collision.posZ());
80  //
81  for (auto track : tracks) { // start loop over tracks
82 
83  TLorentzVector cutVector{};
84  cutVector.SetPtEtaPhiM(track.pt() * 2.0, track.eta(), track.phi(), constants::physics::MassHelium3);
85  if (cutVector.Rapidity() < yMin + yBeam || cutVector.Rapidity() > yMax + yBeam) {
86  continue;
87  }
88  //
89  // fill QA histograms
90  //
91  spectra.fill(HIST("fTPCsignal"), track.tpcInnerParam(), track.tpcSignal());
92  spectra.fill(HIST("fTPCcounts"), track.tpcInnerParam(), track.tpcNSigmaHe());
93  //
94  // check offline-trigger (skimming) condidition
95  //
96  if (track.tpcNSigmaHe() > nsigmacutLow && track.tpcNSigmaHe() < nsigmacutHigh) {
97  keepEvent = kTRUE;
98  spectra.fill(HIST("fHePt"), track.pt());
99  }
100 
101  } // end loop over tracks
102  //
103  // fill trigger (skimming) results
104  //
105  spectra.fill(HIST("fKeepEvent"), keepEvent);
106  }
107 };
