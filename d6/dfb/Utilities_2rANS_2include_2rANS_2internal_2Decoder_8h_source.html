37 template <typename State_T, typename Stream_T>
38 class Decoder
39 {
40  // the Coder works either with a 64Bit state and 32 bit streaming or
41  //a 32 Bit state and 8 Bit streaming We need to make sure it gets initialized with
42  //the right template arguments at compile time.
43  static_assert((sizeof(State_T) == sizeof(uint32_t) && sizeof(Stream_T) == sizeof(uint8_t)) ||
44  (sizeof(State_T) == sizeof(uint64_t) && sizeof(Stream_T) == sizeof(uint32_t)),
45  "Coder can either be 32Bit with 8 Bit stream type or 64 Bit Type with 32 Bit stream type");
46 
47  public:
48  Decoder();
49 
50  // Initializes a rANS decoder.
51  // Unlike the encoder, the decoder works forwards as you'd expect.
52  template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool> = true>
53  Stream_IT init(Stream_IT iter);
54 
55  // Returns the current cumulative frequency (map it to a symbol yourself!)
56  uint32_t get(uint32_t scale_bits);
57 
58  // Equivalent to Rans32DecAdvance that takes a symbol.
59  template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool> = true>
60  Stream_IT advanceSymbol(Stream_IT iter, const DecoderSymbol& sym, uint32_t scale_bits);
61 
62  private:
63  State_T mState;
64 
65  // Renormalize.
66  template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool> = true>
67  std::tuple<State_T, Stream_IT> renorm(State_T x, Stream_IT iter);
68 
69  // L ('l' in the paper) is the lower bound of our normalization interval.
70  // Between this and our byte-aligned emission, we use 31 (not 32!) bits.
71  // This is done intentionally because exact reciprocals for 31-bit uints
72  // fit in 32-bit uints: this permits some optimizations during encoding.
73  inline static constexpr State_T LOWER_BOUND = needs64Bit<State_T>() ? (1u << 31) : (1u << 23); // lower bound of our normalization interval
74 
75  inline static constexpr State_T STREAM_BITS = sizeof(Stream_T) * 8; // lower bound of our normalization interval
76 };

81 template <typename State_T, typename Stream_T>
82 template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool>>
83 Stream_IT Decoder<State_T, Stream_T>::init(Stream_IT iter)
84 {
85 
86  State_T x = 0;
87  Stream_IT streamPos = iter;
88 
89  if constexpr (needs64Bit<State_T>()) {
90  x = static_cast<State_T>(*streamPos) << 0;
91  --streamPos;
92  x |= static_cast<State_T>(*streamPos) << 32;
93  --streamPos;
94  assert(std::distance(streamPos, iter) == 2);
95  } else {
96  x = static_cast<State_T>(*streamPos) << 0;
97  --streamPos;
98  x |= static_cast<State_T>(*streamPos) << 8;
99  --streamPos;
100  x |= static_cast<State_T>(*streamPos) << 16;
101  --streamPos;
102  x |= static_cast<State_T>(*streamPos) << 24;
103  --streamPos;
104  assert(std::distance(streamPos, iter) == 4);
105  }
106 
107  mState = x;
108  return streamPos;
109 };

117 template <typename State_T, typename Stream_T>
118 template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool>>
119 Stream_IT Decoder<State_T, Stream_T>::advanceSymbol(Stream_IT iter, const DecoderSymbol& sym, uint32_t scale_bits)
120 {
121  static_assert(std::is_same<typename std::iterator_traits<Stream_IT>::value_type, Stream_T>::value);
122 
123  State_T mask = (1ull << scale_bits) - 1;
124 
125  // s, x = D(x)
126  State_T x = mState;
127  x = sym.freq * (x >> scale_bits) + (x & mask) - sym.start;
128 
129  // renormalize
130  Stream_IT streamPos;
131  std::tie(mState, streamPos) = this->renorm(x, iter);
132  return streamPos;
133 };

135 template <typename State_T, typename Stream_T>
136 template <typename Stream_IT, std::enable_if_t<isCompatibleIter_v<Stream_T, Stream_IT>, bool>>
137 inline std::tuple<State_T, Stream_IT> Decoder<State_T, Stream_T>::renorm(State_T x, Stream_IT iter)
138 {
139  static_assert(std::is_same<typename std::iterator_traits<Stream_IT>::value_type, Stream_T>::value);
140 
141  Stream_IT streamPos = iter;
142 
143  // renormalize
144  if (x < LOWER_BOUND) {
145  if constexpr (needs64Bit<State_T>()) {
146  x = (x << STREAM_BITS) | *streamPos;
147  --streamPos;
148  assert(x >= LOWER_BOUND);
149  } else {
150 
151  do {
152  x = (x << STREAM_BITS) | *streamPos;
153  --streamPos;
154  } while (x < LOWER_BOUND);
155  }
156  }
157  return std::make_tuple(x, streamPos);
158 }
