130 struct MatchRecord {
131  float chi2 = -1.f;
132  int partnerID = MinusOne;
133  int nextRecID = MinusOne;
134  int matchedIC = MinusOne;
135  MatchRecord(int partID, float chi2match, int nxt = MinusOne, int candIC = MinusOne) : partnerID(partID), chi2(chi2match), nextRecID(nxt), matchedIC(candIC) {}
136  MatchRecord() = default;
137 
138  bool isBetter(float otherChi2, int otherIC = MinusOne) const
139  { // prefer record with matched IC candidate, otherwise, better chi2
140  if (otherIC == MinusOne) {
141  return matchedIC == MinusOne ? chi2 < otherChi2 : true;
142  } else {
143  return matchedIC == MinusOne ? false : chi2 < otherChi2;
144  }
145  }
146 
147  bool isBetter(const MatchRecord& other) const
148  {
149  return isBetter(other.chi2, other.matchedIC);
150  }
151 };
