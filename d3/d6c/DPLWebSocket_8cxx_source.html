49 void websocket_server_callback(uv_stream_t* stream, ssize_t nread, const uv_buf_t* buf)
50 {
51  WSDPLHandler* server = (WSDPLHandler*)stream->data;
52  assert(server);
53  if (nread == 0) {
54  return;
55  }
56  if (nread == UV_EOF) {
57  LOG(DEBUG) << "websocket_server_callback: communication with driver closed";
58  uv_close((uv_handle_t*)stream, websocket_server_close_callback);
59  return;
60  }
61  if (nread < 0) {
62  LOG(ERROR) << "websocket_server_callback: Error while reading from websocket";
63  uv_close((uv_handle_t*)stream, websocket_server_close_callback);
64  return;
65  }
66  try {
67  parse_http_request(buf->base, nread, server);
68  free(buf->base);
69  } catch (RuntimeErrorRef& ref) {
70  auto& err = o2::framework::error_from_ref(ref);
71  LOG(ERROR) << "Error while parsing request: " << err.what;
72  }
73 }

125 void WSDPLHandler::endHeaders()
126 {
128  if (mHeaders["upgrade"] != "websocket") {
129  throw WSError{400, "Bad Request: not a websocket upgrade"};
130  }
131  if (mHeaders["connection"] != "upgrade") {
132  throw WSError{400, "Bad Request: connection not for upgrade"};
133  }
134  if (mHeaders["sec-websocket-protocol"] != "dpl") {
135  throw WSError{400, "Bad Request: websocket protocol not \"dpl\"."};
136  }
137  if (mHeaders.count("sec-websocket-key") == 0) {
138  throw WSError{400, "Bad Request: sec-websocket-key missing"};
139  }
140  if (mHeaders["sec-websocket-version"] != "13") {
141  throw WSError{400, "Bad Request: wrong protocol version"};
142  }
143  mHandler->headers(mHeaders);
145  LOG(debug) << "Got upgrade request with nonce " << mHeaders["sec-websocket-key"].c_str();
146  std::string reply = encode_websocket_handshake_reply(mHeaders["sec-websocket-key"].c_str());
147  mHandshaken = true;
148 
149  uv_buf_t bfr = uv_buf_init(strdup(reply.data()), reply.size());
150  uv_write_t* info_req = (uv_write_t*)malloc(sizeof(uv_write_t));
151  uv_write(info_req, (uv_stream_t*)mStream, &bfr, 1, ws_handshake_done_callback);
152 }

177 void websocket_client_callback(uv_stream_t* stream, ssize_t nread, const uv_buf_t* buf)
178 {
179  WSDPLClient* client = (WSDPLClient*)stream->data;
180  assert(client);
181  if (nread == 0) {
182  return;
183  }
184  if (nread == UV_EOF) {
185  LOG(debug) << "EOF received from server, closing.";
186  uv_read_stop(stream);
187  uv_close((uv_handle_t*)stream, close_client_websocket);
188  return;
189  }
190  if (nread < 0) {
191  // FIXME: improve error message
192  // FIXME: should I close?
193  LOG(ERROR) << "Error while reading from websocket";
194  uv_read_stop(stream);
195  uv_close((uv_handle_t*)stream, close_client_websocket);
196  return;
197  }
198  try {
199  LOG(INFO) << "Data received from server";
200  parse_http_request(buf->base, nread, client);
201  } catch (RuntimeErrorRef& ref) {
202  auto& err = o2::framework::error_from_ref(ref);
203  LOG(ERROR) << "Error while parsing request: " << err.what;
204  }

254 void WSDPLClient::endHeaders()
255 {
257  if (mHeaders["upgrade"] != "websocket") {
258  throw runtime_error_f("No websocket upgrade");
259  }
260  if (mHeaders["connection"] != "upgrade") {
261  throw runtime_error_f("No connection upgrade");
262  }
263  if (mHeaders.count("sec-websocket-accept") == 0) {
264  throw runtime_error("sec-websocket-accept not found");
265  }
266 
267  std::string expectedAccept = HTTPParserHelpers::calculateAccept(mNonce.c_str());
268  if (mHeaders["sec-websocket-accept"] != expectedAccept) {
269  throw runtime_error_f(R"(Invalid accept received: "%s", expected "%s")", mHeaders["sec-websocket-accept"].c_str(), expectedAccept.c_str());
270  }
271 
272  LOG(INFO) << "Correctly handshaken websocket connection.";
274  mHandshaken = true;
275  mHandshake();
276 }
