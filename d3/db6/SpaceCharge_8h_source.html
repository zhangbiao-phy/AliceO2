873 template <typename DataT, size_t Nz, size_t Nr, size_t Nphi>
874 template <typename Fields>
875 void SpaceCharge<DataT, Nz, Nr, Nphi>::calcDistCorr(const DataT p1r, const DataT p1phi, const DataT p1z, const DataT p2z, DataT& ddR, DataT& ddPhi, DataT& ddZ, const Fields& formulaStruct, const bool localDistCorr) const
876 {
877  // see: https://edms.cern.ch/ui/file/1108138/1/ALICE-INT-2010-016.pdf
878  // needed for calculation of distortions/corrections
879  DataT localIntErOverEz = 0; // integral_p1z^p2z Er/Ez dz
880  DataT localIntEPhiOverEz = 0; // integral_p1z^p2z Ephi/Ez dz
881  DataT localIntDeltaEz = 0; // integral_p1z^p2z Ez dz
882 
883  // there are differentnumerical integration strategys implements. for details see each function.
884  switch (sNumericalIntegrationStrategy) {
885  case IntegrationStrategy::SimpsonIterative: // iterative simpson integration (should be more precise at least for the analytical E-Field case but takes alot more time than normal simpson integration)
886  for (int i = 0; i < sSimpsonNIteratives; ++i) { // TODO define a convergence criterion to abort the algorithm earlier for speed up.
887  const DataT tmpZ = localDistCorr ? (p2z + ddZ) : regulateZ(p2z + ddZ, formulaStruct.getSide()); // dont regulate for local distortions/corrections! (to get same result as using electric field at last/first bin)
888  integrateEFieldsSimpsonIterative(p1r, p1r + ddR, p1phi, p1phi + ddPhi, p1z, tmpZ, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
889  langevinCylindrical(ddR, ddPhi, ddZ, (p1r + 0.5 * ddR), localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz); // using the mean radius '(p1r + 0.5 * ddR)' for calculation of distortions/corections
890  }
891  break;
892  case IntegrationStrategy::Simpson: // simpson integration
893  integrateEFieldsSimpson(p1r, p1phi, p1z, p2z, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
894  langevinCylindrical(ddR, ddPhi, ddZ, p1r, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz);
895  break;
896  case IntegrationStrategy::Trapezoidal: // trapezoidal integration (fastest)
897  integrateEFieldsTrapezoidal(p1r, p1phi, p1z, p2z, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
898  langevinCylindrical(ddR, ddPhi, ddZ, p1r, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz);
899  break;
900  case IntegrationStrategy::Root: // using integration implemented in ROOT (slow)
901  integrateEFieldsRoot(p1r, p1phi, p1z, p2z, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
902  langevinCylindrical(ddR, ddPhi, ddZ, p1r, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz);
903  break;
904  default:
905  LOGP(INFO, "no matching case: Using Simpson");
906  integrateEFieldsSimpson(p1r, p1phi, p1z, p2z, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz, formulaStruct);
907  langevinCylindrical(ddR, ddPhi, ddZ, p1r, localIntErOverEz, localIntEPhiOverEz, localIntDeltaEz);
908  }
909 }
