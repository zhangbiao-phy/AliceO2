45 class RawReaderSpecs : public o2f::Task
46 {
47  public:
48  explicit RawReaderSpecs(const ReaderInp& rinp)
49  : mLoop(rinp.loop < 0 ? INT_MAX : (rinp.loop < 1 ? 1 : rinp.loop)), mDelayUSec(rinp.delay_us), mMinTFID(rinp.minTF), mMaxTFID(rinp.maxTF), mPartPerSP(rinp.partPerSP), mReader(std::make_unique<o2::raw::RawFileReader>(rinp.inifile, 0, rinp.bufferSize)), mRawChannelName(rinp.rawChannelConfig)
50  {
51  mReader->setCheckErrors(rinp.errMap);
52  mReader->setMaxTFToRead(rinp.maxTF);
53  mReader->setNominalSPageSize(rinp.spSize);
54  mReader->setCacheData(rinp.cache);
55  mReader->setTFAutodetect(rinp.autodetectTF0 ? RawFileReader::FirstTFDetection::Pending : RawFileReader::FirstTFDetection::Disabled);
56  mReader->setPreferCalculatedTFStart(rinp.preferCalcTF);
57  LOG(INFO) << "Will preprocess files with buffer size of " << rinp.bufferSize << " bytes";
58  LOG(INFO) << "Number of loops over whole data requested: " << mLoop;
59  for (int i = NTimers; i--;) {
60  mTimer[i].Stop();
61  mTimer[i].Reset();
62  }
63  }

65  void init(o2f::InitContext& ic) final
66  {
67  assert(mReader);
68  mTimer[TimerInit].Start();
69  mReader->init();
70  mTimer[TimerInit].Stop();
71  if (mMaxTFID >= mReader->getNTimeFrames()) {
72  mMaxTFID = mReader->getNTimeFrames() ? mReader->getNTimeFrames() - 1 : 0;
73  }

76  void run(o2f::ProcessingContext& ctx) final
77  {
78  assert(mReader);
79  if (mDone) {
80  return;
81  }
82  auto tTotStart = mTimer[TimerTotal].CpuTime(), tIOStart = mTimer[TimerIO].CpuTime();
83  mTimer[TimerTotal].Start(false);
84  auto device = ctx.services().get<o2f::RawDeviceService>().device();
85  assert(device);
86 
87  auto findOutputChannel = [&ctx, this](RawFileReader::LinkData& link, size_t timeslice) {
88  if (!this->mRawChannelName.empty()) {
89  link.fairMQChannel = this->mRawChannelName;
90  return true;
91  }
92  auto outputRoutes = ctx.services().get<o2f::RawDeviceService>().spec().outputs;
93  for (auto& oroute : outputRoutes) {
94  LOG(DEBUG) << "comparing with matcher to route " << oroute.matcher << " TSlice:" << oroute.timeslice;
95  if (o2f::DataSpecUtils::match(oroute.matcher, link.origin, link.description, link.subspec) && ((timeslice % oroute.maxTimeslices) == oroute.timeslice)) {
96  link.fairMQChannel = oroute.channel;
97  LOG(DEBUG) << "picking the route:" << o2f::DataSpecUtils::describe(oroute.matcher) << " channel " << oroute.channel;
98  return true;
99  }
100  }
101  LOGF(ERROR, "Failed to find output channel for %s/%s/0x%x", link.origin.as<std::string>(),
102  link.description.as<std::string>(), link.subspec);
103  return false;
104  };
105 
106  // clean-up before reading next TF
107  auto tfID = mReader->getNextTFToRead();
108  int nlinks = mReader->getNLinks();
109 
110  std::unordered_map<std::string, std::unique_ptr<FairMQParts>> messagesPerRoute;
111 
112  if (tfID > mMaxTFID) {
113  if (!mReader->isEmpty() && --mLoop) {
114  mLoopsDone++;
115  tfID = 0;
116  LOG(INFO) << "Starting new loop " << mLoopsDone << " from the beginning of data";
117  } else {
118  mTimer[TimerTotal].Stop();
119  LOGF(INFO, "Finished: payload of %zu bytes in %zu messages sent for %d TFs", mSentSize, mSentMessages, mTFCounter);
120  for (int i = 0; i < NTimers; i++) {
121  LOGF(INFO, "Timing for %15s: Cpu: %.3e Real: %.3e s in %d slots", TimerName[i], mTimer[i].CpuTime(), mTimer[i].RealTime(), mTimer[i].Counter() - 1);
122  }
123  ctx.services().get<o2f::ControlService>().endOfStream();
124  ctx.services().get<o2f::ControlService>().readyToQuit(o2f::QuitRequest::Me);
125  mDone = true;
126  return;
127  }
128  }
129 
130  if (tfID < mMinTFID) {
131  tfID = mMinTFID;
132  }
133  mReader->setNextTFToRead(tfID);
134  std::vector<RawFileReader::PartStat> partsSP;
135  const auto& hbfU = HBFUtils::Instance();
136 
137  // read next time frame
138  size_t tfNParts = 0, tfSize = 0;
139  LOG(INFO) << "Reading TF#" << mTFCounter << " (" << tfID << " at iteration " << mLoopsDone << ')';
140  o2::header::Stack dummyStack{o2h::DataHeader{}, o2::framework::DataProcessingHeader{0}}; // dummy stack to just to get stack size
141  auto hstackSize = dummyStack.size();
142 
143  for (int il = 0; il < nlinks; il++) {
144  auto& link = mReader->getLink(il);
145  if (!link.rewindToTF(tfID)) {
146  continue; // this link has no data for wanted TF
147  }
148  if (!findOutputChannel(link, mTFCounter)) { // no output channel
149  continue;
150  }
151 
152  o2h::DataHeader hdrTmpl(link.description, link.origin, link.subspec); // template with 0 size
153  int nParts = mPartPerSP ? link.getNextTFSuperPagesStat(partsSP) : link.getNHBFinTF();
154  hdrTmpl.payloadSerializationMethod = o2h::gSerializationMethodNone;
155  hdrTmpl.splitPayloadParts = nParts;
156 
157  while (hdrTmpl.splitPayloadIndex < hdrTmpl.splitPayloadParts) {
158 
159  tfSize += hdrTmpl.payloadSize = mPartPerSP ? partsSP[hdrTmpl.splitPayloadIndex].size : link.getNextHBFSize();
160  auto fmqFactory = device->GetChannel(link.fairMQChannel, 0).Transport();
161  auto hdMessage = fmqFactory->CreateMessage(hstackSize, fair::mq::Alignment{64});
162  auto plMessage = fmqFactory->CreateMessage(hdrTmpl.payloadSize, fair::mq::Alignment{64});
163  mTimer[TimerIO].Start(false);
164  auto bread = mPartPerSP ? link.readNextSuperPage(reinterpret_cast<char*>(plMessage->GetData()), &partsSP[hdrTmpl.splitPayloadIndex]) : link.readNextHBF(reinterpret_cast<char*>(plMessage->GetData()));
165  if (bread != hdrTmpl.payloadSize) {
166  LOG(ERROR) << "Link " << il << " read " << bread << " bytes instead of " << hdrTmpl.payloadSize
167  << " expected in TF=" << mTFCounter << " part=" << hdrTmpl.splitPayloadIndex;
168  }
169  mTimer[TimerIO].Stop();
170  // check if the RDH to send corresponds to expected orbit
171  if (hdrTmpl.splitPayloadIndex == 0) {
172  auto ir = o2::raw::RDHUtils::getHeartBeatIR(plMessage->GetData());
173  auto tfid = hbfU.getTF(ir);
174  hdrTmpl.firstTForbit = hbfU.getIRTF(tfid).orbit; // will be picked for the
175  hdrTmpl.tfCounter = mTFCounter; // following parts
176  // reinterpret_cast<o2::header::DataHeader*>(hdMessage->GetData())->firstTForbit = hdrTmpl.firstTForbit; // hack to fix already filled headers
177  // reinterpret_cast<o2::header::DataHeader*>(hdMessage->GetData())->tfCounter = mTFCounter; // at the moment don't use it
178  }
179  o2::header::Stack headerStack{hdrTmpl, o2::framework::DataProcessingHeader{mTFCounter}};
180  memcpy(hdMessage->GetData(), headerStack.data(), headerStack.size());
181 
182  FairMQParts* parts = nullptr;
183  parts = messagesPerRoute[link.fairMQChannel].get(); // FairMQParts*
184  if (!parts) {
185  messagesPerRoute[link.fairMQChannel] = std::make_unique<FairMQParts>();
186  parts = messagesPerRoute[link.fairMQChannel].get();
187  }
188  parts->AddPart(std::move(hdMessage));
189  parts->AddPart(std::move(plMessage));
190  hdrTmpl.splitPayloadIndex++; // prepare for next
191  tfNParts++;
192  }
193  LOGF(DEBUG, "Added %d parts for TF#%d(%d in iteration %d) of %s/%s/0x%u", hdrTmpl.splitPayloadParts, mTFCounter, tfID,
194  mLoopsDone, link.origin.as<std::string>(), link.description.as<std::string>(), link.subspec);
195  }
196 
197  if (mTFCounter) { // delay sending
198  usleep(mDelayUSec);
199  }
200  for (auto& msgIt : messagesPerRoute) {
201  LOG(INFO) << "Sending " << msgIt.second->Size() / 2 << " parts to channel " << msgIt.first;
202  device->Send(*msgIt.second.get(), msgIt.first);
203  }
204  mTimer[TimerTotal].Stop();
205 
206  LOGF(INFO, "Sent payload of %zu bytes in %zu parts in %zu messages for TF %d | Timing (total/IO): %.3e / %.3e", tfSize, tfNParts,
207  messagesPerRoute.size(), mTFCounter, mTimer[TimerTotal].CpuTime() - tTotStart, mTimer[TimerIO].CpuTime() - tIOStart);
208 
209  mSentSize += tfSize;
210  mSentMessages += tfNParts;
211 
212  mReader->setNextTFToRead(++tfID);
213  ++mTFCounter;
214  }

246 o2f::DataProcessorSpec getReaderSpec(const ReaderInp& rinp)
247 {
248  // check which inputs are present in files to read
249  o2f::DataProcessorSpec spec;
250  spec.name = "raw-file-reader";
251  std::string rawChannelName = "";
252  if (rinp.rawChannelConfig.empty()) {
253  if (!rinp.inifile.empty()) {
254  auto conf = o2::raw::RawFileReader::parseInput(rinp.inifile);
255  for (const auto& entry : conf) {
256  const auto& ordescard = entry.first;
257  if (!entry.second.empty()) { // origin and decription for files to process
258  spec.outputs.emplace_back(o2f::OutputSpec(o2f::ConcreteDataTypeMatcher{std::get<0>(ordescard), std::get<1>(ordescard)}));
259  }
260  }
261  }
262  } else {
263  auto nameStart = rinp.rawChannelConfig.find("name=");
264  if (nameStart == std::string::npos) {
265  throw std::runtime_error("raw channel name is not provided");
266  }
267  nameStart += strlen("name=");
268  auto nameEnd = rinp.rawChannelConfig.find(",", nameStart + 1);
269  if (nameEnd == std::string::npos) {
270  nameEnd = rinp.rawChannelConfig.size();
271  }
272  rawChannelName = rinp.rawChannelConfig.substr(nameStart, nameEnd - nameStart);
273  spec.options = {o2f::ConfigParamSpec{"channel-config", o2f::VariantType::String, rinp.rawChannelConfig, {"Out-of-band channel config"}}};
274  LOG(INFO) << "Will send output to non-DPL channel " << rinp.rawChannelConfig;
275  }
276 
277  spec.algorithm = o2f::adaptFromTask<RawReaderSpecs>(rinp);
278 
279  return spec;
280 }
