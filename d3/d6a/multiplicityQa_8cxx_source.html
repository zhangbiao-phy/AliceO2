13 // of vtx-Z dependencies or for the calibration of those).

16 // except for centrality calibration! The necessary task is the multiplicity

30 struct MultiplicityQaTask {

32  OutputObj<TH1F> hMultT0M{TH1F("hMultT0M", "", 10000, 0., 200000.)};
33  OutputObj<TH1F> hMultZNA{TH1F("hMultZNA", "", 600, 0., 240000.)};

40  OutputObj<TProfile> hVtxProfZNA{TProfile("hVtxProfZNA", "", 150, -15, 15)};
41  OutputObj<TProfile> hVtxProfZNC{TProfile("hVtxProfZNC", "", 150, -15, 15)};

45  Configurable<bool> isMC{"isMC", 0, "0 - data, 1 - MC"};
46  Configurable<int> selection{"sel", 7, "trigger: 7 - sel7, 8 - sel8"};
47 
48  void process(soa::Join<aod::Collisions, aod::EvSels, aod::Mults>::iterator const& col)
49  {
50  if (!isMC && !col.alias()[kINT7]) {
51  return;
52  }
53 
54  if (selection == 7 && !col.sel7()) {
55  return;
56  }
57 
58  if (selection == 8 && !col.sel8()) {
59  return;
60  }
61 
62  if (selection != 7 && selection != 8) {
63  LOGF(fatal, "Unknown selection type! Use `--sel 7` or `--sel 8`");
64  }
65 
66  LOGF(debug, "multV0A=%5.0f multV0C=%5.0f multV0M=%5.0f multT0A=%5.0f multT0C=%5.0f multT0M=%5.0f", col.multV0A(), col.multV0C(), col.multV0M(), col.multT0A(), col.multT0C(), col.multT0M());
67  // fill calibration histos
68  hMultV0M->Fill(col.multV0M());
69  hMultT0M->Fill(col.multT0M());
70  hMultZNA->Fill(col.multZNA());
71  hMultZNC->Fill(col.multZNC());
72  hMultV0MvsT0M->Fill(col.multV0M(), col.multT0M());
73  hMultNtrackletsVsV0M->Fill(col.multV0M(), col.multTracklets());
74 
75  //Vertex-Z dependencies
76  hVtxProfV0M->Fill(col.posZ(), col.multV0M());
77  hVtxProfT0M->Fill(col.posZ(), col.multT0M());
78  hVtxProfZNA->Fill(col.posZ(), col.multZNA());
79  hVtxProfZNC->Fill(col.posZ(), col.multZNC());
80  }
81 };
