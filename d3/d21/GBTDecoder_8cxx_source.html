42  {
44  mOutputHandler.set(orbit, data, rofs);
45  // for (auto& byte : payload) {
46  auto it = payload.begin();
47  auto end = payload.end();
48  while (it != end) {
49  if (mELinkDecoder.isZero(*it)) {
50  // The e-link decoder is empty, meaning that we expect a new board.
51  // The first byte of the board should have the STARTBIT on.
52  // If this is not the case, we move to the next byte.
53  // Notice that the payload has zeros until the end of the 256 bits word:
54  // a) when moving from the regional high to regional low
55  // b) at the end of the HBF
56  ++it;
57  continue;
58  }
59  if (mELinkDecoder.add(it, end)) {
60  if (raw::isLoc(mELinkDecoder.getStatusWord())) {
61  std::invoke(mOnDoneLoc, mOutputHandler, mELinkDecoder.getId() % 8, mELinkDecoder);
62  } else {
63  size_t ilink = 8 + mELinkDecoder.getId() % 8;
64  if (ilink <= 9) {
65  std::invoke(mOnDoneReg, mOutputHandler, ilink, mELinkDecoder);
66  }
67  }
68  mELinkDecoder.reset();
69  }
70  }
71  }

86  GBTBareDecoderImpl(uint16_t feeId, bool isDebugMode = false, uint8_t mask = 0xFF, const ElectronicsDelay& electronicsDelay = ElectronicsDelay()) : mIsDebugMode(isDebugMode), mMask(mask)
87  {
89  mOutputHandler.setFeeId(feeId);

92  mOnDoneLoc = &GBTOutputHandler::onDoneLocDebug;

96  void operator()(gsl::span<const uint8_t> payload, uint32_t orbit, std::vector<LocalBoardRO>& data, std::vector<ROFRecord>& rofs)
97  {
99  mOutputHandler.set(orbit, data, rofs);
100 
101  uint8_t byte = 0;
102  size_t ilink = 0, linkMask = 0, byteOffset = 0;
103 
104  for (int ireg = 0; ireg < 2; ++ireg) {
105  byteOffset = 5 * ireg;
106  if (mIsDebugMode) {
107  ilink = 8 + ireg;
108  linkMask = (1 << ilink);
109  for (size_t idx = byteOffset + 4; idx < payload.size(); idx += 16) {
110  byte = payload[idx];
111  if ((mIsFeeding & linkMask) || byte) {
112  processRegDebug(ilink, byte);
113  }
114  }
115  }
116  for (int ib = 0; ib < 4; ++ib) {
117  ilink = ib + 4 * ireg;
118  linkMask = (1 << ilink);
119  if (mMask & linkMask) {
120  for (size_t idx = byteOffset + ib; idx < payload.size(); idx += 16) {
121  byte = payload[idx];
122  if ((mIsFeeding & linkMask) || byte) {
123  processLoc(ilink, byte);
124  }
125  }
126  }
127  }
128  }
129  }
130 
131  private:
132  void processLoc(size_t ilink, uint8_t byte)
133  {
135  if (mELinkDecoders[ilink].getNBytes() > 0) {
136  mELinkDecoders[ilink].addAndComputeSize(byte);
137  if (mELinkDecoders[ilink].isComplete()) {
138  std::invoke(mOnDoneLoc, mOutputHandler, ilink, mELinkDecoders[ilink]);
139  mELinkDecoders[ilink].reset();
140  mIsFeeding &= (~(1 << ilink));
141  }
142  } else if ((byte & (raw::sSTARTBIT | raw::sCARDTYPE)) == (raw::sSTARTBIT | raw::sCARDTYPE)) {
143  mELinkDecoders[ilink].add(byte);
144  mIsFeeding |= (1 << ilink);
145  }
146  }
147 
148  void processRegDebug(size_t ilink, uint8_t byte)
149  {
151  if (mELinkDecoders[ilink].getNBytes() > 0) {
152  mELinkDecoders[ilink].add(byte);
153  if (mELinkDecoders[ilink].isComplete()) {
154  mOutputHandler.onDoneRegDebug(ilink, mELinkDecoders[ilink]);
155  mELinkDecoders[ilink].reset();
156  mIsFeeding &= (~(1 << ilink));
157  }
158  } else if (byte & raw::sSTARTBIT) {
159  mELinkDecoders[ilink].add(byte);
160  mIsFeeding |= (1 << ilink);
161  }
162  }
163 
164  bool mIsDebugMode{false};
165  uint8_t mMask{0xFF};
166  uint16_t mIsFeeding{0};
167  GBTOutputHandler mOutputHandler{};
168 
169  std::array<ELinkDecoder, crateparams::sNELinksPerGBT> mELinkDecoders{};
170 
171  typedef void (GBTOutputHandler::*OnDoneFunction)(size_t, const ELinkDecoder&);
172 
173  OnDoneFunction mOnDoneLoc{&GBTOutputHandler::onDoneLoc};
174 };

184  {
186  mOutputHandler.setFeeId(feeId);
187  mOutputHandler.setElectronicsDelay(electronicsDelay);
188  mIsDebugMode = isDebugMode;
189  if (isDebugMode) {
190  mOnDoneLoc = &GBTOutputHandler::onDoneLocDebug;
191  }

195  void operator()(gsl::span<const uint8_t> payload, uint32_t orbit, std::vector<LocalBoardRO>& data, std::vector<ROFRecord>& rofs)
196  {
198  mOutputHandler.set(orbit, data, rofs);
199 
200  for (int ireg = 0; ireg < 2; ++ireg) {
201  size_t byteOffset = 5 * ireg;
202 
203  if (mIsDebugMode) {
204  // Treat regional cards
205  size_t ilink = 8 + ireg;
206  mLinkPayload.clear();
207  for (size_t idx = byteOffset + 4, end = payload.size(); idx < end; idx += 16) {
208  mLinkPayload.emplace_back(payload[idx]);
209  }
210  auto it = mLinkPayload.begin();
211  auto end = mLinkPayload.end();
212  while (it != end) {
213  if (mELinkDecoders[ilink].isZero(*it)) {
214  ++it;
215  continue;
216  }
217  if (mELinkDecoders[ilink].addCore(it, end)) {
218  mOutputHandler.onDoneRegDebug(ilink, mELinkDecoders[ilink]);
219  mELinkDecoders[ilink].reset();
220  }
221  }
222  }
223 
224  // Treat local cards
225  for (int ib = 0; ib < 4; ++ib) {
226  size_t ilink = ib + 4 * ireg;
227  if (mMask & (1 << ilink)) {
228  mLinkPayload.clear();
229  for (size_t idx = byteOffset + ib, end = payload.size(); idx < end; idx += 16) {
230  mLinkPayload.emplace_back(payload[idx]);
231  }
232  auto it = mLinkPayload.begin();
233  auto end = mLinkPayload.end();
234  while (it != end) {
235  if (mELinkDecoders[ilink].isZero(*it)) {
236  ++it;
237  continue;
238  }
239  if (mELinkDecoders[ilink].add(it, end)) {
240  std::invoke(mOnDoneLoc, mOutputHandler, mELinkDecoders[ilink].getId() % 8, mELinkDecoders[ilink]);
241  mELinkDecoders[ilink].reset();
242  }
243  }
244  }
245  }
246  }
247  }
