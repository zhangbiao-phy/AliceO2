75 template <typename DataT, size_t N>
76 inline Vector<DataT, N> operator*(const Matrix<DataT, N>& a, const Vector<DataT, N>& b)
77 {
78  using V = Vc::Vector<DataT>;
79  // resulting vector c
80  Vector<DataT, N> c;
81  for (size_t i = 0; i < N; ++i) {
82  V c_ij{};
83  for (size_t j = 0; j < a[i].vectorsCount(); ++j) {
84  c_ij += a[i].vector(j) * b.getVector(j);
85  }
86  c[i] = c_ij.sum();
87  }
88  return c;
89 }
