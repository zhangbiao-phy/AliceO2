163 template <typename DataT, unsigned int Nx, unsigned int Ny, unsigned int Nz>
164 DataT RegularGrid3D<DataT, Nx, Ny, Nz>::clampToGrid(const DataT pos, const unsigned int dim) const
165 {
166  if (mMin[dim] < mMax[dim]) {
167  if (pos < mMin[dim]) {
168  return mMin[dim];
169  } else if (pos > mMax[dim]) {
170  return mMax[dim];
171  }
172  } else {
173  if (pos > mMin[dim]) {
174  return mMin[dim];
175  } else if (pos < mMax[dim]) {
176  return mMax[dim];
177  }
178  }
179  return pos;
180 }

182 template <typename DataT, unsigned int Nx, unsigned int Ny, unsigned int Nz>
183 DataT RegularGrid3D<DataT, Nx, Ny, Nz>::clampToGridRel(const DataT pos, const unsigned int dim) const
184 {
185  if (pos < 0) {
186  return 0;
187  } else if (pos >= sMaxIndex[dim]) {
188  return sMaxIndex[dim] - 1; // -1 return second last index. otherwise two additional points have to be extrapolated for tricubic interpolation
189  }
190  return pos;
191 }

193 template <typename DataT, unsigned int Nx, unsigned int Ny, unsigned int Nz>
194 DataT RegularGrid3D<DataT, Nx, Ny, Nz>::clampToGridCircular(DataT pos, const unsigned int dim) const
195 {
196  while (pos < mMin[dim]) {
197  pos += mMax[dim] - mMin[dim] + mSpacing[dim];
198  }
199  while (pos >= mMax[dim] + mSpacing[dim]) {
200  pos -= mMax[dim] + mSpacing[dim] - mMin[dim];
201  }
202  return pos;
203 }

205 template <typename DataT, unsigned int Nx, unsigned int Ny, unsigned int Nz>
206 DataT RegularGrid3D<DataT, Nx, Ny, Nz>::clampToGridCircularRel(DataT pos, const unsigned int dim) const
207 {
208  while (pos < 0) {
209  pos += sNdim[dim];
210  }
211  while (pos > sNdim[dim]) {
212  pos -= sNdim[dim];
213  }
214  if (pos == sNdim[dim]) {
215  pos = 0;
216  }
217  return pos;
218 }
