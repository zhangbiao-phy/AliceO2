34  float operator()(const float* x) const override
35  {
36  const float mom = abs(x[0]);
37  if (mom <= 0) {
38  return -999;
39  }
40  const float time = x[1];
41  const float evtimereso = x[2];
42  const float mass = x[3];
43  const float dpp = mParameters[0] + mParameters[1] * mom + mParameters[2] * mass / mom; // mean relative pt resolution;
44  const float sigma = dpp * time / (1. + mom * mom / (mass * mass));
45  return TMath::Sqrt(sigma * sigma + mParameters[3] * mParameters[3] / mom / mom + mParameters[4] * mParameters[4] + evtimereso * evtimereso);
46  }
