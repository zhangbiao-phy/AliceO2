69  template <uint8_t histID, typename T>
70  void Add(T&& hist)
71  {
72  if (mHistos.find(histID) != mHistos.end()) {
73  LOGF(WARNING, "HistContainer %s already holds a histogram at histID = %d. Overriding it now...", RootContainer::GetName(), histID);
74  TObject* oldPtr = nullptr;
75  std::visit([&](const auto& sharedPtr) { oldPtr = sharedPtr.get(); }, mHistos[histID]);
76  RootContainer::Remove(oldPtr);
77  }
78  // if shared pointers or rvalue raw pointers are provided as argument, the existing object is used
79  // otherwise the existing object is copied
80  std::optional<HistType> histVariant{};
81  if constexpr (is_shared_ptr<typename std::remove_reference<T>::type>::value) {
82  histVariant = GetHistVariant(hist);
83  } else if constexpr (std::is_pointer_v<T> && std::is_rvalue_reference_v<decltype(std::forward<T>(hist))>) {
84  histVariant = GetHistVariant(std::shared_ptr<std::remove_pointer_t<T>>(hist));
85  } else {
86  histVariant = GetHistVariant(std::make_shared<T>(hist));
87  }
88  if (histVariant) {
89  mHistos[histID] = *histVariant;
90  TObject* rawPtr = nullptr;
91  std::visit([&](const auto& sharedPtr) { rawPtr = sharedPtr.get(); }, mHistos[histID]);
92  RootContainer::Add(rawPtr);
93  } else {
94  LOGF(FATAL, "Could not create histogram.");
95  }
96  }

127  template <bool fillWeight = false, typename T, typename... Ts>
128  void GenericFill(std::shared_ptr<T> hist, const Ts&... position)
129  {
130  // filling with weights requires one additional argument
131  constexpr bool isTH3 = (std::is_same_v<TH3, T> && sizeof...(Ts) == 3 + fillWeight);
132  constexpr bool isTH2 = (std::is_same_v<TH2, T> && sizeof...(Ts) == 2 + fillWeight);
133  constexpr bool isTH1 = (std::is_same_v<TH1, T> && sizeof...(Ts) == 1 + fillWeight);
134  constexpr bool isTProfile3D = (std::is_same_v<TProfile3D, T> && sizeof...(Ts) == 4 + fillWeight);
135  constexpr bool isTProfile2D = (std::is_same_v<TProfile2D, T> && sizeof...(Ts) == 3 + fillWeight);
136  constexpr bool isTProfile = (std::is_same_v<TProfile, T> && sizeof...(Ts) == 2 + fillWeight);
137 
138  constexpr bool isValidPrimitive = isTH1 || isTH2 || isTH3 || isTProfile || isTProfile2D || isTProfile3D;
139  // unfortunately we dont know at compile the dimension of THn(Sparse)
140  constexpr bool isValidComplex = std::is_base_of_v<THnBase, T>;
141 
142  if constexpr (isValidPrimitive) {
143  hist->Fill(static_cast<double>(position)...);
144  } else if constexpr (isValidComplex) {
145  // savety check for n dimensional histograms (runtime overhead)
146  // if (hist->GetNdimensions() != sizeof...(position) - fillWeight) return;
147  double tempArray[] = {static_cast<double>(position)...};
148  if constexpr (fillWeight) {
149  hist->Fill(tempArray, tempArray[sizeof...(Ts) - 1]);
150  } else {
151  hist->Fill(tempArray);
152  }
153  }
154  }

156  template <typename T>
157  std::optional<HistType> GetHistVariant(std::shared_ptr<TObject> obj)
158  {
159  if (obj->InheritsFrom(T::Class())) {
160  return std::static_pointer_cast<T>(obj);
161  }
162  return std::nullopt;
163  }
164  template <typename T, typename Next, typename... Rest>
165  std::optional<HistType> GetHistVariant(std::shared_ptr<TObject> obj)
166  {
167  if (auto hist = GetHistVariant<T>(obj)) {
168  return hist;
169  }
170  return GetHistVariant<Next, Rest...>(obj);
171  }
172  std::optional<HistType> GetHistVariant(std::shared_ptr<TObject> obj)
173  {
174  if (obj) {
175  // TProfile3D is TH3, TProfile2D is TH2, TH3 is TH1, TH2 is TH1, TProfile is TH1
176  return GetHistVariant<THn, THnSparse, TProfile3D, TH3, TProfile2D, TH2, TProfile, TH1>(obj);
177  }
178  return std::nullopt;
179  }

240  // create histogram with the defined axes
241  template <typename RootHistType>
242  std::shared_ptr<RootHistType> Create(const std::string& name, bool useWeights = false)
243  {
244  const std::size_t MAX_DIM{10};
245  const std::size_t nAxes{mAxes.size()};
246  if (nAxes == 0 || nAxes > MAX_DIM) {
247  return nullptr;
248  }
249 
250  int nBins[MAX_DIM]{0};
251  double lowerBounds[MAX_DIM]{0.};
252  double upperBounds[MAX_DIM]{0.};
253 
254  // first figure out number of bins and dimensions
255  std::string title = "[ ";
256  for (std::size_t i = 0; i < nAxes; i++) {
257  nBins[i] = (mAxes[i].nBins) ? *mAxes[i].nBins : mAxes[i].binEdges.size() - 1;
258  lowerBounds[i] = mAxes[i].binEdges.front();
259  upperBounds[i] = mAxes[i].binEdges.back();
260  title += mAxes[i].name;
261  if (i < nAxes - 1) {
262  title += " : ";
263  } else {
264  title += " ]";
265  }
266  }
267 
268  // create histogram
269  std::shared_ptr<RootHistType> hist(HistFactory<RootHistType>(name, title, nAxes, nBins, lowerBounds, upperBounds));
270 
271  if (!hist) {
272  LOGF(FATAL, "The number of specified dimensions does not match the type.");
273  return nullptr;
274  }
275 
276  // set axis properties
277  for (std::size_t i = 0; i < nAxes; i++) {
278  TAxis* axis{GetAxis(i, hist)};
279  if (axis) {
280  axis->SetTitle(mAxes[i].title.data());
281  if constexpr (std::is_base_of_v<THnBase, RootHistType>) {
282  axis->SetName((std::to_string(i) + "-" + mAxes[i].name).data());
283  }
284 
285  // move the bin edges in case a variable binning was requested
286  if (!mAxes[i].nBins) {
287  if (!std::is_sorted(std::begin(mAxes[i].binEdges), std::end(mAxes[i].binEdges))) {
288  LOGF(FATAL, "The bin edges specified for axis %s in histogram %s are not in increasing order!", mAxes[i].name, name);
289  return nullptr;
290  }
291  axis->Set(nBins[i], mAxes[i].binEdges.data());
292  }
293  }
294  }
295  if (useWeights) {
296  hist->Sumw2();
297  }
298  return hist;
299  }

306  template <typename RootHistType>
307  RootHistType* HistFactory(const std::string& name, const std::string& title, const std::size_t nDim,
308  const int nBins[], const double lowerBounds[], const double upperBounds[])
309  {
310  if constexpr (std::is_base_of_v<THnBase, RootHistType>) {
311  return new RootHistType(name.data(), title.data(), nDim, nBins, lowerBounds, upperBounds);
312  } else if constexpr (std::is_base_of_v<TH3, RootHistType>) {
313  return (nDim != 3) ? nullptr
314  : new RootHistType(name.data(), title.data(), nBins[0], lowerBounds[0],
315  upperBounds[0], nBins[1], lowerBounds[1], upperBounds[1],
316  nBins[2], lowerBounds[2], upperBounds[2]);
317  } else if constexpr (std::is_base_of_v<TH2, RootHistType>) {
318  return (nDim != 2) ? nullptr
319  : new RootHistType(name.data(), title.data(), nBins[0], lowerBounds[0],
320  upperBounds[0], nBins[1], lowerBounds[1], upperBounds[1]);
321  } else if constexpr (std::is_base_of_v<TH1, RootHistType>) {
322  return (nDim != 1)
323  ? nullptr
324  : new RootHistType(name.data(), title.data(), nBins[0], lowerBounds[0], upperBounds[0]);
325  }
326  return nullptr;
327  }
