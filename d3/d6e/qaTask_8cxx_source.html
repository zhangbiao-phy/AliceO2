39 double ConvertPhiRange(double phi)
40 {
41  if (phi > M_PI) {
42  phi -= 2 * M_PI;
43  } else if (phi < -M_PI) {
44  phi += 2 * M_PI;
45  }
46 
47  return phi;
48 }

57 template <typename Track>
58 bool GetImpactParameterAndError(const Track& track, const o2df::VertexBase& primaryVertex, double& impactParameterRPhi,
59  double& impactParameterRPhiError, double& impactParameterZ,
60  double& impactParameterErrorZ)
61 {
62  impactParameterRPhi = -999.;
63  impactParameterRPhiError = -999.;
64  impactParameterZ = -999;
65  impactParameterErrorZ = -999;
66 
67  o2df::DCA dca;
68  // FIXME: get this from CCDB
69  constexpr float magneticField{5.0}; // in kG
70  auto trackParameter = getTrackParCov(track);
71  bool propagate = trackParameter.propagateToDCA(primaryVertex, magneticField, &dca);
72 
73  constexpr float conversion_to_micrometer = 1000;
74  if (propagate) {
75  impactParameterRPhi = conversion_to_micrometer * dca.getY();
76  impactParameterRPhiError = conversion_to_micrometer * std::sqrt(dca.getSigmaY2());
77  impactParameterZ = conversion_to_micrometer * dca.getZ();
78  impactParameterErrorZ = conversion_to_micrometer * std::sqrt(dca.getSigmaZ2());
79  }
80  return propagate;
81 }

433  void process(const o2::soa::Join<o2::aod::Collisions, o2::aod::McCollisionLabels>::iterator& collision,
434  const o2::soa::Join<o2::aod::Tracks, o2::aod::TracksCov, o2::aod::McTrackLabels>& tracks,
435  const o2::aod::McParticles& mcParticles, const o2::aod::McCollisions& mcCollisions)
436  {
437  const o2df::VertexBase primaryVertex = getPrimaryVertex(collision);
438 
439  for (const auto& track : tracks) {
440 
441  if (useOnlyPhysicsPrimary) {
442  const auto mcParticle = track.mcParticle();
443  if (!MC::isPhysicalPrimary(mcParticles, mcParticle)) {
444  continue;
445  }
446  }
447  const double deltaPt = track.mcParticle().pt() - track.pt();
448  histos.fill(HIST("pt/ptDiffMCRec"), deltaPt);
449 
450  const double deltaPtOverPt = deltaPt / track.pt();
451 
452  histos.fill(HIST("pt/ptResolution"), deltaPtOverPt);
453  histos.fill(HIST("pt/ptResolutionVsPt"), track.pt(), std::abs(deltaPtOverPt));
454  histos.fill(HIST("pt/ptResolutionVsEta"), track.eta(), std::abs(deltaPtOverPt));
455  histos.fill(HIST("pt/ptResolutionVsPhi"), track.phi(), std::abs(deltaPtOverPt));
456 
457  const double deltaEta = track.mcParticle().eta() - track.eta();
458  histos.fill(HIST("eta/etaDiffMCReco"), deltaEta);
459  histos.fill(HIST("eta/etaDiffMCRecoVsEtaMC"), deltaEta, track.mcParticle().eta());
460  histos.fill(HIST("eta/etaDiffMCRecoVsEtaReco"), deltaEta, track.eta());
461 
462  const double deltaPhi = track_utils::ConvertPhiRange(track.mcParticle().phi() - track.phi());
463  histos.fill(HIST("phi/phiDiffMCRec"), deltaPhi);
464 
465  double impactParameterRPhi{-999.}, impactParameterRPhiError{-999.};
466  double impactParameterZ{-999.}, impactParameterErrorZ{-999.};
467 
468  const bool propagate = track_utils::GetImpactParameterAndError(
469  track, primaryVertex, impactParameterRPhi, impactParameterRPhiError, impactParameterZ, impactParameterErrorZ);
470 
471  if (propagate) {
472  histos.fill(HIST("impactParameter/impactParameterRPhiVsPt"), track.pt(), impactParameterRPhi);
473  histos.fill(HIST("impactParameter/impactParameterRPhiVsEta"), track.eta(), impactParameterRPhi);
474  histos.fill(HIST("impactParameter/impactParameterRPhiVsPhi"), track.phi(), impactParameterRPhi);
475 
476  histos.fill(HIST("impactParameter/impactParameterZVsPt"), track.pt(), impactParameterZ);
477  histos.fill(HIST("impactParameter/impactParameterZVsEta"), track.eta(), impactParameterZ);
