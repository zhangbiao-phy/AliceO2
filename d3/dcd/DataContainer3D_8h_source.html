141  dataCont = reinterpret_cast<DataContainer3D<DataT, Nx, Ny, Nz>*>(inpf.GetObjectChecked(name, DataContainer3D<DataT, Nx, Ny, Nz>::Class()));
142  if (!dataCont) {
143  LOGP(ERROR, "Failed to load {} from {}", name, inpf.GetName());
144  return false;
145  }
