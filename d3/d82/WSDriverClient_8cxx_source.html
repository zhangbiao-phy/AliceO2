21 void on_connect(uv_connect_t* connection, int status)
22 {
23  if (status < 0) {
24  LOG(ERROR) << "Unable to connect to driver.";
25  return;
26  }
27  WSDriverClient* client = (WSDriverClient*)connection->data;
28  auto onHandshake = [client]() {
29  client->flushPending();
30  };
31  std::lock_guard<std::mutex> lock(client->mutex());
32  client->setDPLClient(std::make_unique<WSDPLClient>(connection->handle, client->spec(), onHandshake));
33  client->sendHandshake();
34 }

37 void connectToDriver(WSDriverClient* driver, uv_loop_t* loop, char const* address, short port)
38 {
39  uv_tcp_t* socket = (uv_tcp_t*)malloc(sizeof(uv_tcp_t));
40  uv_tcp_init(loop, socket);
41  uv_connect_t* connection = (uv_connect_t*)malloc(sizeof(uv_connect_t));
42  connection->data = driver;
43 
44  struct sockaddr_in dest;
45  uv_ip4_addr(strdup(address), port, &dest);
46 
47  uv_tcp_connect(connection, socket, (const struct sockaddr*)&dest, on_connect);
48 }

79 void WSDriverClient::tell(const char* msg, size_t s, bool flush)
80 {
81  static bool printed1 = false;
82  static bool printed2 = false;
83  if (mConnected && mClient->isHandshaken() && flush) {
84  flushPending();
85  std::lock_guard<std::mutex> lock(mClientMutex);
86  std::vector<uv_buf_t> outputs;
87  encode_websocket_frames(outputs, msg, s, WebSocketOpCode::Binary, 0);
88  mClient->write(outputs);
89  } else {
90  std::lock_guard<std::mutex> lock(mClientMutex);
91  encode_websocket_frames(mBacklog, msg, s, WebSocketOpCode::Binary, 0);
92  }
93 }

95 void WSDriverClient::flushPending()
96 {
97  std::lock_guard<std::mutex> lock(mClientMutex);
98  static bool printed1 = false;
99  static bool printed2 = false;
100  if (!mClient) {
101  if (mBacklog.size() > 2000) {
102  if (!printed1) {
103  LOG(WARNING) << "Unable to communicate with driver because client does not exist. Continuing connection attempts.";
104  printed1 = true;
105  }
106  }
107  return;
108  }
109  if (!(mClient->isHandshaken())) {
110  if (mBacklog.size() > 2000) {
111  if (!printed2) {
112  LOG(WARNING) << "Unable to communicate with driver because client is not connected. Continuing connection attempts.";
113  printed2 = true;
114  }
115  }
116  return;
117  }
118  if (printed1 || printed2) {
119  LOGP(warning, "DriverClient connected successfully. Flushing message backlog of {} messages. All is good.", mBacklog.size());
120  printed1 = false;
121  printed2 = false;
122  }
123  mClient->write(mBacklog);
124  mBacklog.resize(0);
125 }
