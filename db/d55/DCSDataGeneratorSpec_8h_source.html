93  void run(o2::framework::ProcessingContext& pc) final
94  {
95 
96  uint64_t tfid;
97  for (auto& input : pc.inputs()) {
98  tfid = header::get<o2::framework::DataProcessingHeader*>(input.header)->startTime;
99  LOG(DEBUG) << "tfid = " << tfid;
100  if (tfid >= mMaxTF) {
101  LOG(INFO) << "Data generator reached TF " << tfid << ", stopping";
102  pc.services().get<o2::framework::ControlService>().endOfStream();
103  pc.services().get<o2::framework::ControlService>().readyToQuit(o2::framework::QuitRequest::Me);
104  }
105  break; // we break because one input is enough to get the TF ID
106  }
107 
108  LOG(DEBUG) << "TF: " << tfid << " --> building binary blob...";
109  uint16_t flags = 0;
110  uint16_t milliseconds = 0;
111  TDatime currentTime;
112  uint32_t seconds = currentTime.Get();
113  uint64_t payload[7];
114  memset(payload, 0, sizeof(uint64_t) * 7);
115 
116  payload[0] = (uint64_t)tfid + 33; // adding 33 to have visible chars and strings
117 
118  DPVAL valchar(flags, milliseconds + tfid * 10, seconds + tfid, payload, mtypechar);
119  DPVAL valint(flags, milliseconds + tfid * 10, seconds + tfid, payload, mtypeint);
120  DPVAL valdouble(flags, milliseconds + tfid * 10, seconds + tfid, payload, mtypedouble);
121  DPVAL valstring(flags, milliseconds + tfid * 10, seconds + tfid, payload, mtypestring);
122 
123  LOG(DEBUG) << "Value used for char DPs:";
124  LOG(DEBUG) << valchar << " --> " << (char)valchar.payload_pt1;
125  LOG(DEBUG) << "Value used for int DPs:";
126  LOG(DEBUG) << valint << " --> " << (int)valint.payload_pt1;
127  LOG(DEBUG) << "Value used for double DPs:";
128  LOG(DEBUG) << valdouble << " --> " << (double)valdouble.payload_pt1;
129  char tt[56];
130  memcpy(&tt[0], &valstring.payload_pt1, 56);
131  LOG(DEBUG) << "Value used for string DPs:";
132  LOG(DEBUG) << valstring << " --> " << tt;
133 
134  // full map (all DPs)
135  mBuildingBinaryBlock.Start(mFirstTF);
136  std::vector<DPCOM> dpcomVectFull;
137  for (int i = 0; i < mNumDPscharFull; i++) {
138  dpcomVectFull.emplace_back(mDPIDvectFull[i], valchar);
139  }
140  for (int i = 0; i < mNumDPsintFull; i++) {
141  dpcomVectFull.emplace_back(mDPIDvectFull[mNumDPscharFull + i], valint);
142  }
143  for (int i = 0; i < mNumDPsdoubleFull; i++) {
144  dpcomVectFull.emplace_back(mDPIDvectFull[mNumDPscharFull + mNumDPsintFull + i], valdouble);
145  }
146  for (int i = 0; i < mNumDPsstringFull; i++) {
147  dpcomVectFull.emplace_back(mDPIDvectFull[mNumDPscharFull + mNumDPsintFull + mNumDPsdoubleFull + i], valstring);
148  }
149 
150  // delta map (only DPs that changed)
151  mDeltaBuildingBinaryBlock.Start(mFirstTF);
152  std::vector<DPCOM> dpcomVectDelta;
153  for (int i = 0; i < mNumDPscharDelta; i++) {
154  dpcomVectDelta.emplace_back(mDPIDvectDelta[i], valchar);
155  }
156  for (int i = 0; i < mNumDPsintDelta; i++) {
157  dpcomVectDelta.emplace_back(mDPIDvectDelta[mNumDPscharDelta + i], valint);
158  }
159  for (int i = 0; i < mNumDPsdoubleDelta; i++) {
160  dpcomVectDelta.emplace_back(mDPIDvectDelta[mNumDPscharDelta + mNumDPsintDelta + i], valdouble);
161  }
162  for (int i = 0; i < mNumDPsstringDelta; i++) {
163  dpcomVectDelta.emplace_back(mDPIDvectDelta[mNumDPscharDelta + mNumDPsintDelta + mNumDPsdoubleDelta + i],
164  valstring);
165  }
166 
167  // Full map
168  auto svect = dpcomVectFull.size();
169  LOG(DEBUG) << "dpcomVectFull has size " << svect;
170  for (int i = 0; i < svect; i++) {
171  LOG(DEBUG) << "i = " << i << ", DPCOM = " << dpcomVectFull[i];
172  }
173  std::vector<char> buff(mNumDPsFull * sizeof(DPCOM));
174  char* dptr = buff.data();
175  for (int i = 0; i < svect; i++) {
176  memcpy(dptr + i * sizeof(DPCOM), &dpcomVectFull[i], sizeof(DPCOM));
177  }
178  auto sbuff = buff.size();
179  LOG(DEBUG) << "size of output buffer = " << sbuff;
180  mBuildingBinaryBlock.Stop();
181  LOG(DEBUG) << "TF: " << tfid << " --> ...binary blob prepared: realTime = "
182  << mBuildingBinaryBlock.RealTime() << ", cpuTime = "
183  << mBuildingBinaryBlock.CpuTime();
184  LOG(DEBUG) << "TF: " << tfid << " --> sending snapshot...";
185  mSnapshotSending.Start(mFirstTF);
186  pc.outputs().snapshot(Output{"DCS", "DATAPOINTS", 0, Lifetime::Timeframe}, buff.data(), sbuff);
187  mSnapshotSending.Stop();
188  LOG(DEBUG) << "TF: " << tfid << " --> ...snapshot sent: realTime = " << mSnapshotSending.RealTime()
189  << ", cpuTime = " << mSnapshotSending.CpuTime();
190 
191  // Delta map
192  auto svectDelta = dpcomVectDelta.size();
193  LOG(DEBUG) << "dpcomVectDelta has size " << svect;
194  for (int i = 0; i < svectDelta; i++) {
195  LOG(DEBUG) << "i = " << i << ", DPCOM = " << dpcomVectDelta[i];
196  }
197  std::vector<char> buffDelta(mNumDPsDelta * sizeof(DPCOM));
198  char* dptrDelta = buffDelta.data();
199  for (int i = 0; i < svectDelta; i++) {
200  memcpy(dptrDelta + i * sizeof(DPCOM), &dpcomVectDelta[i], sizeof(DPCOM));
201  }
202  auto sbuffDelta = buffDelta.size();
203  LOG(DEBUG) << "size of output (delta) buffer = " << sbuffDelta;
204  mDeltaBuildingBinaryBlock.Stop();
205  LOG(DEBUG) << "TF: " << tfid << " --> ...binary (delta) blob prepared: realTime = "
206  << mDeltaBuildingBinaryBlock.RealTime() << ", cpuTime = " << mDeltaBuildingBinaryBlock.CpuTime();
207  LOG(DEBUG) << "TF: " << tfid << " --> sending (delta) snapshot...";
208  mDeltaSnapshotSending.Start(mFirstTF);
209  pc.outputs().snapshot(Output{"DCS", "DATAPOINTSdelta", 0, Lifetime::Timeframe}, buffDelta.data(), sbuffDelta);
210  mDeltaSnapshotSending.Stop();
211  LOG(DEBUG) << "TF: " << tfid << " --> ...snapshot (delta) sent: realTime = "
212  << mDeltaSnapshotSending.RealTime() << ", cpuTime = "
213  << mDeltaSnapshotSending.CpuTime();
214 
215  /*
216  LOG(INFO) << "Reading back";
217  DPCOM dptmp;
218  for (int i = 0; i < svect; i++) {
219  memcpy(&dptmp, dptr + i * sizeof(DPCOM), sizeof(DPCOM));
220  LOG(DEBUG) << "Check: Reading from generator: i = " << i << ", DPCOM = " << dptmp;
221  }
222  */
223  /*
224  auto& tmpDPmap = pc.outputs().make<std::unordered_map<DPID, DPVAL>>(o2::framework::OutputRef{"output", 0});
225  tmpDPmap[mcharVar] = valchar;
226  tmpDPmap[mintVar0] = valint;
227  tmpDPmap[mintVar1] = valint;
228  tmpDPmap[mintVar2] = valint;
229  tmpDPmap[mdoubleVar0] = valdouble;
230  tmpDPmap[mdoubleVar1] = valdouble;
231  tmpDPmap[mdoubleVar2] = valdouble;
232  if (tfid % 3 == 0)
233  tmpDPmap[mdoubleVar3] = valdouble; // to test the case when a DP is not updated, we skip some updates
234  tmpDPmap[mstringVar0] = valstring;
235  */
236  mFirstTF = false;
237  mTFs++;
238  }
