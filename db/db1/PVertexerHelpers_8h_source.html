84 struct TrackVF {
90  enum { kUsed,
91  kNoVtx = -1,
92  kDiscarded = kNoVtx - 1 };
93  float x;
94  float y;
95  float z;
96  float sig2YI;
97  float sig2ZI;
98  float sigYZI;
99  float tgP;
100  float tgL;
101  float cosAlp;
102  float sinAlp;
103 
104  TimeEst timeEst;
105  float wgh = 0.;
106  int entry;
107  int16_t bin = -1; // seeds histo bin
108  uint8_t flags = 0;
109  int vtxID = kNoVtx;
110  //
111  bool canAssign() const { return wgh > 0. && vtxID == kNoVtx; }
112  bool canUse() const { return vtxID == kNoVtx; }
113  bool canUse(float zmin, float zmax) const
114  {
115  return canUse() && (z > zmin && z < zmax);
116  }
117  bool operator<(const TrackVF& trc) const { return z < trc.z; }
118 
119  float getZForXY(float vx, float vy) const
120  {
121  return z + tgL * (vx * cosAlp + vy * sinAlp - x);
122  }
123 
124  // weighted distance^2 to other track (accounting for own errors only)
125  float getDist2(const TrackVF& o) const
126  {
127  auto dt = timeEst.getTimeStamp() - o.timeEst.getTimeStamp();
128  auto dte2 = timeEst.getTimeStampError() * timeEst.getTimeStampError() + o.timeEst.getTimeStampError() * o.timeEst.getTimeStampError();
129  auto dtnorm2 = dt * dt / dte2;
130  auto dz = z - o.z;
131  return dtnorm2 + dz * dz * sig2ZI;
132  }
133 
134  // weighted distance^2 to other track (accounting for both track errors errors only)
135  float getDist2Sym(const TrackVF& o) const
136  {
137  auto dt = timeEst.getTimeStamp() - o.timeEst.getTimeStamp();
138  auto dz = z - o.z;
139  float dte2 = o.timeEst.getTimeStampError() * o.timeEst.getTimeStampError() + timeEst.getTimeStampError() * timeEst.getTimeStampError();
140  return dt / dte2 + dz * dz / (1. / sig2ZI + 1. / o.sig2ZI);
141  }
142 
143  float getResiduals(const PVertex& vtx, float& dy, float& dz) const
144  {
145  // get residuals (Y and Z DCA in track frame) and calculate chi2
146  float dx = vtx.getX() * cosAlp + vtx.getY() * sinAlp - x; // VX rotated to track frame - trackX
147  dy = y + tgP * dx - (-vtx.getX() * sinAlp + vtx.getY() * cosAlp);
148  dz = z + tgL * dx - vtx.getZ();
149  return (dy * dy * sig2YI + dz * dz * sig2ZI) + 2. * dy * dz * sigYZI;
150  }
151 
152  TrackVF() = default;
153  TrackVF(const o2::track::TrackParCov& src, const TimeEst& t_est, int _entry)
154  : x(src.getX()), y(src.getY()), z(src.getZ()), tgL(src.getTgl()), tgP(src.getSnp() / std::sqrt(1. - src.getSnp()) * (1. + src.getSnp())), timeEst(t_est), entry(_entry)
155  {
156  o2::math_utils::sincos(src.getAlpha(), sinAlp, cosAlp);
157  auto det = src.getSigmaY2() * src.getSigmaZ2() - src.getSigmaZY() * src.getSigmaZY();
158  auto detI = 1. / det;
159  sig2YI = src.getSigmaZ2() * detI;
160  sig2ZI = src.getSigmaY2() * detI;
161  sigYZI = -src.getSigmaZY() * detI;
162  }
163 };

173 struct SeedHisto {
174  float range = 20;
175  float binSize = 0.5;
176  float binSizeInv = 0.;
177  int nFilled = 0;
178  std::vector<int> data;
179 
180  SeedHisto() = delete;
181  SeedHisto(float _range = 20., float _binsize = 0.5) : range(_range), binSize(_binsize)
182  {
183  auto zr = 2 * range;
184  int nzb = zr / binSize;
185  if (nzb * binSize < zr - 1e-9) {
186  nzb++;
187  }
188  binSizeInv = 1. / binSize;
189  range = nzb * binSize / 2.;
190  data.resize(nzb);
191  }
192 
193  int size() const { return data.size(); }
194 
195  void fill(float z)
196  {
197  incrementBin(findBin(z));
198  }
199 
200  void incrementBin(int bin)
201  {
202  data[bin]++;
203  nFilled++;
204  }
205 
206  void decrementBin(int bin)
207  {
208  data[bin]--;
209  nFilled--;
210  }
211 
212  int findBin(float z)
213  {
214  auto d = z + range;
215  if (d < 0.) {
216  return 0;
217  }
218  uint32_t n = d * binSizeInv;
219  return n < data.size() ? n : data.size() - 1;
220  }
221 
222  int findHighestPeakBin() const
223  {
224  if (nFilled < 2) {
225  return -1;
226  }
227  int n = data.size(), maxBin = -1, maxv = 0;
228  for (int i = 0; i < n; i++) {
229  if (data[i] > maxv) {
230  maxv = data[(maxBin = i)];
231  }
232  }
233  return maxBin;
234  }
235 
236  bool isValidBin(int ib) const
237  {
238  return static_cast<uint32_t>(ib) < data.size();
239  }
240 
241  float getBinCenter(int ib) const
242  {
243  return (ib + 0.5) * binSize - range; // no check for being in the range!!!
244  }
245 
246  void discardBin(int ib)
247  { // no check for being in the range!!!
248  nFilled -= data[ib];
249  data[ib] = 0;
250  }
251 };

253 struct TimeZCluster {
254  TimeEst timeEst;
255  int first = -1;
256  int last = -1;
257  int count = 0;
258 
259  void clear()
260  {
261  first = last = -1;
262  count = 0;
263  }
264 
265  void addTrack(int i, const TimeEst& trcT)
266  {
267  auto trcTErr2 = trcT.getTimeStampError() * trcT.getTimeStampError();
268  auto trcTErr2Inv = 1. / trcTErr2;
269  if (first < 0) {
270  first = last = i;
271  timeEst.setTimeStamp(trcT.getTimeStamp());
272  timeEst.setTimeStampError(trcT.getTimeStampError());
273  } else {
274  auto vtxTErr2Inv = 1. / (timeEst.getTimeStampError() * timeEst.getTimeStampError());
275  auto vtxTErr2UpdInv = trcTErr2Inv + vtxTErr2Inv;
276  auto vtxTErr2Upd = 1. / vtxTErr2UpdInv;
277  timeEst.setTimeStamp((timeEst.getTimeStamp() * vtxTErr2Inv + trcT.getTimeStamp() * trcTErr2Inv) * vtxTErr2Upd);
278  timeEst.setTimeStampError(std::sqrt(vtxTErr2Upd));
279  if (i > last) {
280  last = i;
281  }
282  }
283  count++;
284  }
285 
286  bool isCompatible(const TimeEst& c, float margin, float cut) const
287  {
288  if (first < 0) {
289  return true;
290  }
291  float dt = timeEst.getTimeStamp() - c.getTimeStamp();
292  if (c.getTimeStampError() && timeEst.getTimeStampError()) {
293  float trcTErr2 = c.getTimeStampError() * c.getTimeStampError();
294  float err = trcTErr2 + timeEst.getTimeStampError() + margin;
295  return dt * dt / err < cut;
296  } else {
297  return std::abs(dt) < cut;
298  }
299  }
300 
301  void merge(TimeZCluster& c)
302  {
303  if (c.first < last) {
304  first = c.first;
305  } else {
306  last = c.last;
307  }
308  if (timeEst.getTimeStampError() && c.timeEst.getTimeStampError()) { // weighted average
309  auto cTErr2 = c.timeEst.getTimeStampError() * c.timeEst.getTimeStampError();
310  auto cTErr2Inv = 1. / cTErr2;
311 
312  auto tErr2 = timeEst.getTimeStampError();
313  auto tErr2Inv = 1. / (tErr2 * tErr2);
314  auto tErr2UpdInv = cTErr2Inv + tErr2Inv;
315  auto tErr2Upd = 1. / tErr2UpdInv;
316  timeEst.setTimeStamp((timeEst.getTimeStamp() * tErr2Inv + c.timeEst.getTimeStamp() * cTErr2Inv) * tErr2Upd);
317  timeEst.setTimeStampError(std::sqrt(tErr2Upd));
318  } else {
319  timeEst.setTimeStamp((timeEst.getTimeStamp() * count + c.timeEst.getTimeStamp() * c.count) / (count + c.count));
320  count += c.count;
321  c.count = 0;
322  }
323  }
324 };
