31 GPUd() void TrackLTIntegral::addStep(float dL, const TrackPar& track)
32 {
34  mL += dL;
35  float p2 = track.getP2Inv();
36  float dTns = dL * 1000.f / o2::constants::physics::LightSpeedCm2NS; // time change in ps for beta = 1 particle
37  for (int id = 0; id < getNTOFs(); id++) {
38  float m2z = o2::track::PID::getMass2Z(id);
39  float betaInv = CAMath::Sqrt(1.f + m2z * m2z * p2);
40  mT[id] += dTns * betaInv;
41  }
42 }
