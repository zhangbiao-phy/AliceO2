39 void PrimaryVertexingSpec::init(InitContext& ic)
40 {
41  //-------- init geometry and field --------//
42  o2::base::GeometryManager::loadGeometry();
43  o2::base::Propagator::initFieldFromGRP("o2sim_grp.root");
44 
45  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom(o2::base::NameConf::getGRPFileName())};
46  const auto& alpParams = o2::itsmft::DPLAlpideParam<DetID::ITS>::Instance();
47  if (!grp->isDetContinuousReadOut(DetID::ITS)) {
48  mITSROFrameLengthMUS = alpParams.roFrameLengthTrig / 1.e3; // ITS ROFrame duration in \mus
49  } else {
50  mITSROFrameLengthMUS = alpParams.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingNS * 1e-3; // ITS ROFrame duration in \mus
51  }
52 
53  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
54  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
55  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
56  if (o2::base::NameConf::pathExists(matLUTFile)) {
57  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
58  o2::base::Propagator::Instance()->setMatLUT(lut);
59  LOG(INFO) << "Loaded material LUT from " << matLUTFile;
60  } else {
61  LOG(INFO) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
62  }
63  mTimer.Stop();
64  mTimer.Reset();
65  mVertexer.setValidateWithIR(mValidateWithIR);
66 
67  // set bunch filling. Eventually, this should come from CCDB
68  const auto* digctx = o2::steer::DigitizationContext::loadFromFile("collisioncontext.root");
69  const auto& bcfill = digctx->getBunchFilling();
70  mVertexer.setBunchFilling(bcfill);
71  mVertexer.init();
72 }

74 void PrimaryVertexingSpec::run(ProcessingContext& pc)
75 {
76  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
77  mTimer.Start(false);
78 
79  o2::globaltracking::RecoContainer recoData;
80  recoData.collectData(pc, dataRequest);
81  // select tracks of needed type, with minimal cuts, the real selected will be done in the vertexer
82 
83  std::vector<TrackWithTimeStamp> tracks;
84  std::vector<o2::MCCompLabel> tracksMCInfo;
85  std::vector<o2d::GlobalTrackID> gids;
86  auto maxTrackTimeError = PVertexerParams::Instance().maxTimeErrorMUS;
87  auto halfROFITS = 0.5 * mITSROFrameLengthMUS;
88  auto hw2ErrITS = 2.f / std::sqrt(12.f) * mITSROFrameLengthMUS; // conversion from half-width to error for ITS
89 
90  std::function<void(const o2::track::TrackParCov& _tr, float t0, float terr, GTrackID _origID)> creator =
91  [maxTrackTimeError, hw2ErrITS, halfROFITS, &tracks, &gids](const o2::track::TrackParCov& _tr, float t0, float terr, GTrackID _origID) {
92  if (!_origID.includesDet(DetID::ITS)) {
93  return; // just in case this selection was not done on RecoContainer filling level
94  }
95  if (_origID.getSource() == GTrackID::ITS) { // error is supplied a half-ROF duration, convert to \mus
96  t0 += halfROFITS; // ITS time is supplied as beginning of ROF
97  terr *= hw2ErrITS;
98  }
99  if (terr > maxTrackTimeError) {
100  return;
101  }
102  tracks.emplace_back(TrackWithTimeStamp{_tr, {t0, terr}});
103  gids.emplace_back(_origID);
104  };
105 
106  recoData.createTracks(creator); // create track sample considered for vertexing
107  if (mUseMC) {
108  recoData.fillTrackMCLabels(gids, tracksMCInfo);
109  }
110  std::vector<PVertex> vertices;
111  std::vector<GIndex> vertexTrackIDs;
112  std::vector<V2TRef> v2tRefs;
113  std::vector<o2::MCEventLabel> lblVtx;
114 
115  // RS FIXME this will not have effect until the 1st orbit is propagated, until that will work only for TF starting at orbit 0
116  mVertexer.setStartIR({0, DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getByPos(0))->firstTForbit});
117 
118  std::vector<o2::InteractionRecord> ft0Data;
119  if (mValidateWithIR) { // select BCs for validation
120  const o2::ft0::InteractionTag& ft0Params = o2::ft0::InteractionTag::Instance();
121  auto ft0all = recoData.getFT0RecPoints<o2::ft0::RecPoints>();
122  for (const auto& ftRP : ft0all) {
123  if (ft0Params.isSelected(ftRP)) {
124  ft0Data.push_back(ftRP.getInteractionRecord());
125  }
126  }
127  }
128 
129  mVertexer.process(tracks, gids, ft0Data, vertices, vertexTrackIDs, v2tRefs, tracksMCInfo, lblVtx);
130  pc.outputs().snapshot(Output{"GLO", "PVTX", 0, Lifetime::Timeframe}, vertices);
131  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe}, v2tRefs);
132  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTID", 0, Lifetime::Timeframe}, vertexTrackIDs);
133 
134  if (mUseMC) {
135  pc.outputs().snapshot(Output{"GLO", "PVTX_MCTR", 0, Lifetime::Timeframe}, lblVtx);
136  }
137 
138  mTimer.Stop();
139  LOG(INFO) << "Found " << vertices.size() << " primary vertices, timing: CPU: "
140  << mTimer.CpuTime() - timeCPU0 << " Real: " << mTimer.RealTime() - timeReal0 << " s";
141 }

149 DataProcessorSpec getPrimaryVertexingSpec(GTrackID::mask_t src, bool validateWithFT0, bool useMC)
150 {
151  std::vector<OutputSpec> outputs;
152  if (src[GTrackID::ITS]) {
153  dataRequest.requestITSTracks(useMC);
154  }
155  if (src[GTrackID::ITSTPC] || src[GTrackID::ITSTPCTOF]) { // ITSTPCTOF does not provide tracks, only matchInfo
156  dataRequest.requestITSTPCTracks(useMC);
157  }
158  if (src[GTrackID::ITSTPCTOF]) {
159  dataRequest.requestTOFMatches(useMC);
160  dataRequest.requestTOFClusters(false);
161  }
162  if (validateWithFT0 && src[GTrackID::FT0]) {
163  dataRequest.requestFT0RecPoints(false);
164  }
165 
166  outputs.emplace_back("GLO", "PVTX", 0, Lifetime::Timeframe);
167  outputs.emplace_back("GLO", "PVTX_CONTID", 0, Lifetime::Timeframe);
168  outputs.emplace_back("GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe);
169 
170  if (useMC) {
171  outputs.emplace_back("GLO", "PVTX_MCTR", 0, Lifetime::Timeframe);
172  }
173 
174  return DataProcessorSpec{
175  "primary-vertexing",
176  dataRequest.inputs,
177  outputs,
178  AlgorithmSpec{adaptFromTask<PrimaryVertexingSpec>(validateWithFT0, useMC)},
179  Options{{"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
180 }
