59  template <class DataBlockType>
60  void processDigits(DataBlockType& dataBlock, int linkID)
61  {
62  if constexpr (std::is_same<DataBlockType, DataBlockPM>::value) { //Filling data from PM
63  for (int iEventData = 0; iEventData < dataBlock.DataBlockWrapper<RawDataPM>::mNelements; iEventData++) {
64  mVecChannelData.emplace_back(int(sLookupTable.getChannel(linkID, dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].channelID)),
65  int(dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].time),
66  int(dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].charge),
67  dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].getFlagWord());
68  }
69  } else if constexpr (std::is_same<DataBlockType, DataBlockTCM>::value) { //Filling data from TCM
70  dataBlock.DataBlockWrapper<RawDataTCM>::mData[0].fillTrigger(mDigit.mTriggers);
71  }
72  }
73  void getDigits(std::vector<Digit>& vecDigits, std::vector<ChannelData>& vecChannelData)
74  {
75  //last digit filling
76  mDigit.ref.set(vecChannelData.size(), mVecChannelData.size());
77  //mDigit.mEventID = sEventID;
78  //

80  std::move(mVecChannelData.begin(), mVecChannelData.end(), std::back_inserter(vecChannelData));
81  mVecChannelData.clear();
82 
83  sEventID++; //Increasing static eventID. After each poping of the data, it will increase
84  }
