51 void BunchFilling::setBCTrain(int nBC, int bcSpacing, int firstBC)
52 {
53  // add interacting BC train with given spacing starting at given place, i.e.
54  // train with 25ns spacing should have bcSpacing = 1
55  for (int i = 0; i < nBC; i++) {
56  setBC(firstBC);
57  firstBC += bcSpacing;
58  }

62 void BunchFilling::setBCTrains(int nTrains, int trainSpacingInBC, int nBC, int bcSpacing, int firstBC)
63 {
64  // add nTrains trains of interacting BCs with bcSpacing within the train and trainSpacingInBC empty slots
65  // between the trains
66  for (int it = 0; it < nTrains; it++) {
67  setBCTrain(nBC, bcSpacing, firstBC);
68  firstBC += nBC * bcSpacing + trainSpacingInBC;
69  }
70 }
