48  void init(o2::framework::InitContext&)
49  {
50  ccdb->setURL(url.value);
51  ccdb->setTimestamp(timestamp.value);
52  ccdb->setCaching(true);
53  ccdb->setLocalObjectValidityChecking();
54  // Not later than now objects
55  ccdb->setCreatedNotAfter(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
56  //
57  const std::string fname = paramfile.value;
58  if (!fname.empty()) { // Loading the parametrization from file
59  response.LoadParamFromFile(fname.data(), signalname.value, DetectorResponse::kSignal);
60  response.LoadParamFromFile(fname.data(), sigmaname.value, DetectorResponse::kSigma);
61  } else { // Loading it from CCDB
62  const std::string path = "Analysis/PID/TPC";
63  response.LoadParam(DetectorResponse::kSignal, ccdb->getForTimeStamp<Parametrization>(path + "/" + signalname.value, timestamp.value));
64  response.LoadParam(DetectorResponse::kSigma, ccdb->getForTimeStamp<Parametrization>(path + "/" + sigmaname.value, timestamp.value));
65  }
66  }

124  template <typename T>
125  void makelogaxis(T h)
126  {
127  const int nbins = h->GetNbinsX();
128  double binp[nbins + 1];
129  double max = h->GetXaxis()->GetBinUpEdge(nbins);
130  double min = h->GetXaxis()->GetBinLowEdge(1);
131  if (min <= 0) {
132  min = 0.00001;
133  }
134  double lmin = TMath::Log10(min);
135  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
136  for (int i = 0; i < nbins; i++) {
137  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
138  }
139  binp[nbins] = max + 1;
140  h->GetXaxis()->Set(nbins, binp);
141  }

177  template <uint8_t i, typename T>
178  void fillParticleHistos(const T& t, const float mom, const float exp_diff, const float nsigma)
179  {

185  void process(aod::Collision const& collision, soa::Join<aod::Tracks, aod::TracksExtra, aod::pidRespTPC> const& tracks)
186  {
187  histos.fill(HIST("event/vertexz"), collision.posZ());
188 
189  for (auto t : tracks) {
190  // const float mom = t.p();
191  const float mom = t.tpcInnerParam();
192  histos.fill(HIST("event/tpcsignal"), mom, t.tpcSignal());
193  //
194  fillParticleHistos<0>(t, mom, t.tpcExpSignalDiffEl(), t.tpcNSigmaEl());
195  fillParticleHistos<1>(t, mom, t.tpcExpSignalDiffMu(), t.tpcNSigmaMu());
196  fillParticleHistos<2>(t, mom, t.tpcExpSignalDiffPi(), t.tpcNSigmaPi());
197  fillParticleHistos<3>(t, mom, t.tpcExpSignalDiffKa(), t.tpcNSigmaKa());
198  fillParticleHistos<4>(t, mom, t.tpcExpSignalDiffPr(), t.tpcNSigmaPr());
199  fillParticleHistos<5>(t, mom, t.tpcExpSignalDiffDe(), t.tpcNSigmaDe());
200  fillParticleHistos<6>(t, mom, t.tpcExpSignalDiffTr(), t.tpcNSigmaTr());
201  fillParticleHistos<7>(t, mom, t.tpcExpSignalDiffHe(), t.tpcNSigmaHe());
202  fillParticleHistos<8>(t, mom, t.tpcExpSignalDiffAl(), t.tpcNSigmaAl());
203  }
204  }
