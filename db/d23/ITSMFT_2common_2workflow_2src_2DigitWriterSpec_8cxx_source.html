40 DataProcessorSpec getDigitWriterSpec(bool mctruth, o2::header::DataOrigin detOrig, o2::detectors::DetID detId)
41 {
42  std::string detStr = o2::detectors::DetID::getName(detId);
43  std::string detStrL = detStr;
44  std::transform(detStrL.begin(), detStrL.end(), detStrL.begin(), ::tolower);
45  auto logger = [](std::vector<o2::itsmft::Digit> const& inDigits) {
46  LOG(INFO) << "RECEIVED DIGITS SIZE " << inDigits.size();
47  };
48 
49  // the callback to be set as hook for custom action when the writer is closed
50  auto finishWriting = [](TFile* outputfile, TTree* outputtree) {
51  const auto* brArr = outputtree->GetListOfBranches();
52  int64_t nent = 0;
53  for (const auto* brc : *brArr) {
54  int64_t n = ((const TBranch*)brc)->GetEntries();
55  if (nent && (nent != n)) {
56  LOG(ERROR) << "Branches have different number of entries";
57  }
58  nent = n;
59  }
60  outputtree->SetEntries(nent);
61  outputtree->Write("", TObject::kOverwrite);
62  outputfile->Close();
63  };
64 
65  // handler for labels
66  // This is necessary since we can't store the original label buffer in a ROOT entry -- as is -- if it exceeds a certain size.
67  // We therefore convert it to a special split class.
68  auto fillLabels = [](TBranch& branch, std::vector<char> const& labelbuffer, DataRef const& /*ref*/) {
69  o2::dataformats::ConstMCTruthContainerView<o2::MCCompLabel> labels(labelbuffer);
70  LOG(INFO) << "WRITING " << labels.getNElements() << " LABELS ";
71 
72  o2::dataformats::IOMCTruthContainerView outputcontainer;
73  auto ptr = &outputcontainer;
74  auto br = framework::RootTreeWriter::remapBranch(branch, &ptr);
75  outputcontainer.adopt(labelbuffer);
76  br->Fill();
77  br->ResetAddress();
78  };
79 
80  return MakeRootTreeWriterSpec((detStr + "DigitWriter").c_str(),
81  (detStrL + "digits.root").c_str(),
82  MakeRootTreeWriterSpec::TreeAttributes{"o2sim", "Digits tree"},
83  MakeRootTreeWriterSpec::CustomClose(finishWriting),
84  // in case of labels we first read them as std::vector<char> and process them correctly in the fillLabels hook
85  BranchDefinition<std::vector<char>>{InputSpec{"digitsMCTR", detOrig, "DIGITSMCTR", 0},
86  (detStr + "DigitMCTruth").c_str(),
87  (mctruth ? 1 : 0), fillLabels},
88  BranchDefinition<std::vector<itsmft::MC2ROFRecord>>{InputSpec{"digitsMC2ROF", detOrig, "DIGITSMC2ROF", 0},
89  (detStr + "DigitMC2ROF").c_str(),
90  (mctruth ? 1 : 0)},
91  BranchDefinition<std::vector<itsmft::Digit>>{InputSpec{"digits", detOrig, "DIGITS", 0},
92  (detStr + "Digit").c_str(),
93  logger},
94  BranchDefinition<std::vector<itsmft::ROFRecord>>{InputSpec{"digitsROF", detOrig, "DIGITSROF", 0},
95  (detStr + "DigitROF").c_str()})();
96 }
