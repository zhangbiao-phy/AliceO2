53 // This template magic allow users to customize the behavior of the process
54 // by (optionally) implementing a `configure` method which modifies one of the
55 // objects in question.
56 //
57 // For example it can be optionally implemented by the user to specify the
58 // channel policies for your setup. Use this if you want to customize the way
59 // your devices communicate between themself, e.g. if you want to use REQ/REP
60 // in place of PUB/SUB.
61 //
62 // The advantage of this approach is that we do not need to expose the
63 // configurability / configuration object to the user, unless he really wants to
64 // modify it. The drawback is that we need to declare the `customize` method
65 // before include this file.

125 int main(int argc, char** argv)
126 {
127  using namespace o2::framework;
128  using namespace boost::program_options;
129 
130  int result = 1;
131  try {
132  // The 0 here is an int, therefore having the template matching in the
133  // SFINAE expression above fit better the version which invokes user code over
134  // the default one.
135  // The default policy is a catch all pub/sub setup to be consistent with the past.
136  std::vector<o2::framework::ConfigParamSpec> workflowOptions;
137  UserCustomizationsHelper::userDefinedCustomization(workflowOptions, 0);
138  auto requiredWorkflowOptions = WorkflowCustomizationHelpers::requiredWorkflowOptions();
139  workflowOptions.insert(std::end(workflowOptions), std::begin(requiredWorkflowOptions), std::end(requiredWorkflowOptions));
140 
141  std::vector<CompletionPolicy> completionPolicies;
142  UserCustomizationsHelper::userDefinedCustomization(completionPolicies, 0);
143  auto defaultCompletionPolicies = CompletionPolicy::createDefaultPolicies();
144  completionPolicies.insert(std::end(completionPolicies), std::begin(defaultCompletionPolicies), std::end(defaultCompletionPolicies));
145 
146  std::vector<DispatchPolicy> dispatchPolicies;
147  UserCustomizationsHelper::userDefinedCustomization(dispatchPolicies, 0);
148  auto defaultDispatchPolicies = DispatchPolicy::createDefaultPolicies();
149  dispatchPolicies.insert(std::end(dispatchPolicies), std::begin(defaultDispatchPolicies), std::end(defaultDispatchPolicies));
150 
151  std::vector<std::unique_ptr<ParamRetriever>> retrievers;
152  std::unique_ptr<ParamRetriever> retriever{new BoostOptionsRetriever(true, argc, argv)};
153  retrievers.emplace_back(std::move(retriever));
154  auto workflowOptionsStore = std::make_unique<ConfigParamStore>(workflowOptions, std::move(retrievers));
155  workflowOptionsStore->preload();
156  workflowOptionsStore->activate();
157  ConfigParamRegistry workflowOptionsRegistry(std::move(workflowOptionsStore));
158  ConfigContext configContext(workflowOptionsRegistry, argc, argv);
159  o2::framework::WorkflowSpec specs = defineDataProcessing(configContext);
160  overrideCloning(configContext, specs);
161  overridePipeline(configContext, specs);
162  for (auto& spec : specs) {
163  UserCustomizationsHelper::userDefinedCustomization(spec.requiredServices, 0);
164  }
165  std::vector<ChannelConfigurationPolicy> channelPolicies;
166  UserCustomizationsHelper::userDefinedCustomization(channelPolicies, 0);
167  auto defaultChannelPolicies = ChannelConfigurationPolicy::createDefaultPolicies(configContext);
168  channelPolicies.insert(std::end(channelPolicies), std::begin(defaultChannelPolicies), std::end(defaultChannelPolicies));
169  result = doMain(argc, argv, specs, channelPolicies, completionPolicies, dispatchPolicies, workflowOptions, configContext);
170  } catch (boost::exception& e) {
171  doBoostException(e);
172  } catch (std::exception const& error) {
173  doUnknownException(error.what());
174  } catch (o2::framework::RuntimeErrorRef& ref) {
175  doDPLException(ref);
176  } catch (...) {
177  doUnknownException("");
178  }
179 
180  char* idstring = nullptr;
181  for (int argi = 0; argi < argc; argi++) {
182  if (strcmp(argv[argi], "--id") == 0 && argi + 1 < argc) {
183  idstring = argv[argi + 1];
184  break;
185  }
186  }
187  o2::framework::OnWorkflowTerminationHook onWorkflowTerminationHook;
188  UserCustomizationsHelper::userDefinedCustomization(onWorkflowTerminationHook, 0);
189  onWorkflowTerminationHook(idstring);
190  doDefaultWorkflowTerminationHook();
191  return result;
192 }
