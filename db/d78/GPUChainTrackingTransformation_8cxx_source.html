33 int GPUChainTracking::ConvertNativeToClusterData()
34 {
35 #ifdef HAVE_O2HEADERS
36  mRec->PushNonPersistentMemory(qStr2Tag("TPCTRANS"));
37  const auto& threadContext = GetThreadContext();
38  bool doGPU = GetRecoStepsGPU() & RecoStep::TPCConversion;
39  GPUTPCConvert& convert = processors()->tpcConverter;
40  GPUTPCConvert& convertShadow = doGPU ? processorsShadow()->tpcConverter : convert;
41 
42  SetupGPUProcessor(&convert, true);
43  if (doGPU) {
44  if (!(mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCClusterFinding)) {
45  mInputsHost->mNClusterNative = mInputsShadow->mNClusterNative = mIOPtrs.clustersNative->nClustersTotal;
46  AllocateRegisteredMemory(mInputsHost->mResourceClusterNativeBuffer);
47  processorsShadow()->ioPtrs.clustersNative = mInputsShadow->mPclusterNativeAccess;
48  WriteToConstantMemory(RecoStep::TPCConversion, (char*)&processors()->ioPtrs - (char*)processors(), &processorsShadow()->ioPtrs, sizeof(processorsShadow()->ioPtrs), 0);
49  *mInputsHost->mPclusterNativeAccess = *mIOPtrs.clustersNative;
50  mInputsHost->mPclusterNativeAccess->clustersLinear = mInputsShadow->mPclusterNativeBuffer;
51  mInputsHost->mPclusterNativeAccess->setOffsetPtrs();
52  GPUMemCpy(RecoStep::TPCConversion, mInputsShadow->mPclusterNativeBuffer, mIOPtrs.clustersNative->clustersLinear, sizeof(mIOPtrs.clustersNative->clustersLinear[0]) * mIOPtrs.clustersNative->nClustersTotal, 0, true);
53  TransferMemoryResourceLinkToGPU(RecoStep::TPCConversion, mInputsHost->mResourceClusterNativeAccess, 0);
54  }
55  }
56  if (!param().par.earlyTpcTransform) {
57  if (GetProcessingSettings().debugLevel >= 3) {
58  GPUInfo("Early transform inactive, skipping TPC Early transformation kernel, transformed on the fly during slice data creation / refit");
59  }
60  return 0;
61  }
62  for (unsigned int i = 0; i < NSLICES; i++) {
63  convert.mMemory->clusters[i] = convertShadow.mClusters + mIOPtrs.clustersNative->clusterOffset[i][0];
64  }
65 
66  WriteToConstantMemory(RecoStep::TPCConversion, (char*)&processors()->tpcConverter - (char*)processors(), &convertShadow, sizeof(convertShadow), 0);
67  TransferMemoryResourcesToGPU(RecoStep::TPCConversion, &convert, 0);
68  runKernel<GPUTPCConvertKernel>(GetGridBlk(NSLICES * GPUCA_ROW_COUNT, 0), krnlRunRangeNone, krnlEventNone);
69  TransferMemoryResourcesToHost(RecoStep::TPCConversion, &convert, 0);
70  SynchronizeStream(0);
71 
72  for (unsigned int i = 0; i < NSLICES; i++) {
73  mIOPtrs.nClusterData[i] = (i == NSLICES - 1 ? mIOPtrs.clustersNative->nClustersTotal : mIOPtrs.clustersNative->clusterOffset[i + 1][0]) - mIOPtrs.clustersNative->clusterOffset[i][0];
74  mIOPtrs.clusterData[i] = convert.mClusters + mIOPtrs.clustersNative->clusterOffset[i][0];
75  }
76  mRec->PopNonPersistentMemory(RecoStep::TPCConversion, qStr2Tag("TPCTRANS"));
77 #endif
78  return 0;
79 }

146 int GPUChainTracking::ForwardTPCDigits()
147 {
148 #ifdef HAVE_O2HEADERS
149  if (GetRecoStepsGPU() & RecoStep::TPCClusterFinding) {
150  throw std::runtime_error("Cannot forward TPC digits with Clusterizer on GPU");
151  }
152  std::vector<ClusterNative> tmp[NSLICES][GPUCA_ROW_COUNT];
153  unsigned int nTotal = 0;
154  const float zsThreshold = param().rec.tpcZSthreshold;
155  for (int i = 0; i < NSLICES; i++) {
156  for (unsigned int j = 0; j < mIOPtrs.tpcPackedDigits->nTPCDigits[i]; j++) {
157  const auto& d = mIOPtrs.tpcPackedDigits->tpcDigits[i][j];
158  if (d.getChargeFloat() >= zsThreshold) {
159  ClusterNative c;
160  c.setTimeFlags(d.getTimeStamp(), 0);
161  c.setPad(d.getPad());
162  c.setSigmaTime(1);
163  c.setSigmaPad(1);
164  c.qTot = c.qMax = d.getChargeFloat();
165  tmp[i][d.getRow()].emplace_back(c);
166  nTotal++;
167  }
168  }
169  }
170  mIOMem.clustersNative.reset(new ClusterNative[nTotal]);
171  nTotal = 0;
172  mClusterNativeAccess->clustersLinear = mIOMem.clustersNative.get();
173  for (int i = 0; i < NSLICES; i++) {
174  for (int j = 0; j < GPUCA_ROW_COUNT; j++) {
175  mClusterNativeAccess->nClusters[i][j] = tmp[i][j].size();
176  memcpy(&mIOMem.clustersNative[nTotal], tmp[i][j].data(), tmp[i][j].size() * sizeof(*mClusterNativeAccess->clustersLinear));
177  nTotal += tmp[i][j].size();
178  }
179  }
180  mClusterNativeAccess->setOffsetPtrs();
181  mIOPtrs.tpcPackedDigits = nullptr;
182  mIOPtrs.clustersNative = mClusterNativeAccess.get();
183  GPUInfo("Forwarded %u TPC clusters", nTotal);
184  mRec->MemoryScalers()->nTPCHits = nTotal;
185 #endif
186  return 0;
187 }
