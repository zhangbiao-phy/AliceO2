61  Configurable<int> d_selectionFlagD0{"d_selectionFlagD0", 1, "Selection Flag for D0"};
62  Configurable<int> d_selectionFlagD0bar{"d_selectionFlagD0bar", 1, "Selection Flag for D0bar"};

68  void process(aod::Collision const&, aod::BigTracks const&, soa::Filtered<soa::Join<aod::HfCandProng2, aod::HFSelD0Candidate, aod::Colls>> const& candidates)
69  {
70  for (auto& candidate : candidates) {
71  if (!(candidate.hfflag() & 1 << D0ToPiK)) {
72  continue;
73  }
74  if (cutEtaCandMax >= 0. && std::abs(candidate.eta()) > cutEtaCandMax) {
75  continue;
76  }
77  if (candidate.isSelD0bar() >= d_selectionFlagD0bar) {
78  registry.fill(HIST("hmassD0"), InvMassD0bar(candidate));
79  auto count = 0;
80  for (auto& track : positiveTracks) {
81  if (count % 100 == 0) {
82  LOGF(INFO, "Col: %d (cand); %d (track)", candidate.collisionId(), track.collisionId());
83  count++;
84  }
85  registry.fill(HIST("hptcand"), candidate.pt() + track.pt());
86  }
87  }
88  }
89  }
