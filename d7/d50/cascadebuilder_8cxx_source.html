11 // Cascade builder task
12 // =====================
13 //
14 // This task loops over an *existing* list of cascades (V0+bachelor track
15 // indices) and calculates the corresponding full cascade information
16 //
17 // Any analysis should loop over the "CascData"
18 // table as that table contains all information
19 //
20 // WARNING: adding filters to the builder IS NOT
21 // equivalent to re-running the finders. This will only
22 // ever produce *tighter* selection sections. It is your
23 // responsibility to check if, by setting a loose filter
24 // setting, you are going into a region in which no
25 // candidates exist because the original indices were generated
26 // using tigher selections.
27 //
28 // Comments, questions, complaints, suggestions?
29 // Please write to:
30 // david.dobrigkeit.chinellato@cern.ch
31 //

78 //This prefilter creates a skimmed list of good V0s to re-reconstruct so that
79 //CPU is saved in case there are specific selections that are to be done
80 //Note: more configurables, more options to be added as needed
81 struct cascadeprefilterpairs {
82  Configurable<int> mincrossedrows{"mincrossedrows", 70, "min crossed rows"};
83  Configurable<float> dcav0topv{"dcav0topv", .1, "DCA V0 To PV"};
84  Configurable<double> cospaV0{"cospaV0", .98, "CosPA V0"};
85  Configurable<float> lambdamasswindow{"lambdamasswindow", .006, "Distance from Lambda mass"};
86  Configurable<float> dcav0dau{"dcav0dau", .6, "DCA V0 Daughters"};
87  Configurable<float> dcanegtopv{"dcanegtopv", .1, "DCA Neg To PV"};
88  Configurable<float> dcapostopv{"dcapostopv", .1, "DCA Pos To PV"};
89  Configurable<float> dcabachtopv{"dcabachtopv", .1, "DCA Bach To PV"};
90  Configurable<bool> tpcrefit{"tpcrefit", 1, "demand TPC refit"};
91  Configurable<double> v0radius{"v0radius", 0.9, "v0radius"};
92 
93  Produces<aod::CascGood> cascgood;
94  void process(aod::Collision const& collision, aod::V0Datas const& V0s, aod::Cascades const& Cascades,
95  soa::Join<aod::FullTracks, aod::TracksExtended> const& tracks)
96  {
97  for (auto& casc : Cascades) {
98  //Single-track properties filter
99  auto v0 = casc.v0_as<aod::V0Datas>();
100  if (tpcrefit) {
101  if (!(v0.posTrack_as<FullTracksExt>().trackType() & o2::aod::track::TPCrefit)) {
102  continue; //TPC refit
103  }
104  if (!(v0.negTrack_as<FullTracksExt>().trackType() & o2::aod::track::TPCrefit)) {
105  continue; //TPC refit
106  }
107  if (!(casc.bachelor_as<FullTracksExt>().trackType() & o2::aod::track::TPCrefit)) {
108  continue; //TPC refit
109  }
110  }
111  if (v0.posTrack_as<FullTracksExt>().tpcNClsCrossedRows() < mincrossedrows) {
112  continue;
113  }
114  if (v0.negTrack_as<FullTracksExt>().tpcNClsCrossedRows() < mincrossedrows) {
115  continue;
116  }
117  if (casc.bachelor_as<FullTracksExt>().tpcNClsCrossedRows() < mincrossedrows) {
118  continue;
119  }
120  if (v0.posTrack_as<FullTracksExt>().dcaXY() < dcapostopv) {
121  continue;
122  }
123  if (v0.negTrack_as<FullTracksExt>().dcaXY() < dcanegtopv) {
124  continue;
125  }
126  if (casc.bachelor_as<FullTracksExt>().dcaXY() < dcabachtopv) {
127  continue;
128  }
129 
130  //V0 selections
131  if (fabs(v0.mLambda() - 1.116) > lambdamasswindow && fabs(v0.mAntiLambda() - 1.116) > lambdamasswindow) {
132  continue;
133  }
134  if (v0.dcaV0daughters() > dcav0dau) {
135  continue;
136  }
137  if (v0.v0radius() < v0radius) {
138  continue;
139  }
140  if (v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()) < cospaV0) {
141  continue;
142  }
143  if (v0.dcav0topv(collision.posX(), collision.posY(), collision.posZ()) < dcav0topv) {
144  continue;
145  }
146  cascgood(
147  v0.globalIndex(),
148  casc.bachelor_as<FullTracksExt>().globalIndex(),
149  casc.bachelor_as<FullTracksExt>().collisionId());
150  }
151  }
152 };

155 struct cascadebuilder {
156  Produces<aod::CascData> cascdata;
157 
158  OutputObj<TH1F> hEventCounter{TH1F("hEventCounter", "", 1, 0, 1)};
159  OutputObj<TH1F> hCascCandidate{TH1F("hCascCandidate", "", 10, 0, 10)};
160 
161  //Configurables
162  Configurable<double> d_bz{"d_bz", -5.0, "bz field"};
163  Configurable<double> d_UseAbsDCA{"d_UseAbsDCA", kTRUE, "Use Abs DCAs"};
164 
165  void process(aod::Collision const& collision, aod::V0Datas const& V0s, aod::CascGood const& Cascades, aod::FullTracks const& tracks)
166  {
167  //Define o2 fitter, 2-prong
168  o2::vertexing::DCAFitterN<2> fitterV0, fitterCasc;
169  fitterV0.setBz(d_bz);
170  fitterV0.setPropagateToPCA(true);
171  fitterV0.setMaxR(200.);
172  fitterV0.setMinParamChange(1e-3);
173  fitterV0.setMinRelChi2Change(0.9);
174  fitterV0.setMaxDZIni(1e9);
175  fitterV0.setMaxChi2(1e9);
176  fitterV0.setUseAbsDCA(d_UseAbsDCA);
177 
178  fitterCasc.setBz(d_bz);
179  fitterCasc.setPropagateToPCA(true);
180  fitterCasc.setMaxR(200.);
181  fitterCasc.setMinParamChange(1e-3);
182  fitterCasc.setMinRelChi2Change(0.9);
183  fitterCasc.setMaxDZIni(1e9);
184  fitterCasc.setMaxChi2(1e9);
185  fitterCasc.setUseAbsDCA(d_UseAbsDCA);
186 
187  hEventCounter->Fill(0.5);
188  std::array<float, 3> pVtx = {collision.posX(), collision.posY(), collision.posZ()};
189 
190  for (auto& casc : Cascades) {
191  auto charge = -1;
192  std::array<float, 3> pos = {0.};
193  std::array<float, 3> posXi = {0.};
194  std::array<float, 3> pvecpos = {0.};
195  std::array<float, 3> pvecneg = {0.};
196  std::array<float, 3> pvecbach = {0.};
197 
198  hCascCandidate->Fill(0.5);
199 
200  //Acquire basic tracks
201  auto pTrack = getTrackParCov(casc.v0Data().posTrack_as<FullTracksExt>());
202  auto nTrack = getTrackParCov(casc.v0Data().negTrack_as<FullTracksExt>());
203  auto bTrack = getTrackParCov(casc.bachelor_as<FullTracksExt>());
204  if (casc.bachelor().signed1Pt() > 0) {
205  charge = +1;
206  }
207 
208  int nCand = fitterV0.process(pTrack, nTrack);
209  if (nCand != 0) {
210  fitterV0.propagateTracksToVertex();
211  hCascCandidate->Fill(1.5);
212  const auto& v0vtx = fitterV0.getPCACandidate();
213  for (int i = 0; i < 3; i++) {
214  pos[i] = v0vtx[i];
215  }
216 
217  std::array<float, 21> cov0 = {0};
218  std::array<float, 21> cov1 = {0};
219  std::array<float, 21> covV0 = {0};
220 
221  //Covariance matrix calculation
222  const int momInd[6] = {9, 13, 14, 18, 19, 20}; // cov matrix elements for momentum component
223  fitterV0.getTrack(0).getPxPyPzGlo(pvecpos);
224  fitterV0.getTrack(1).getPxPyPzGlo(pvecneg);
225  fitterV0.getTrack(0).getCovXYZPxPyPzGlo(cov0);
226  fitterV0.getTrack(1).getCovXYZPxPyPzGlo(cov1);
227  for (int i = 0; i < 6; i++) {
228  int j = momInd[i];
229  covV0[j] = cov0[j] + cov1[j];
230  }
231  auto covVtxV0 = fitterV0.calcPCACovMatrix();
232  covV0[0] = covVtxV0(0, 0);
233  covV0[1] = covVtxV0(1, 0);
234  covV0[2] = covVtxV0(1, 1);
235  covV0[3] = covVtxV0(2, 0);
236  covV0[4] = covVtxV0(2, 1);
237  covV0[5] = covVtxV0(2, 2);
238 
239  const std::array<float, 3> vertex = {(float)v0vtx[0], (float)v0vtx[1], (float)v0vtx[2]};
240  const std::array<float, 3> momentum = {pvecpos[0] + pvecneg[0], pvecpos[1] + pvecneg[1], pvecpos[2] + pvecneg[2]};
241 
242  auto tV0 = o2::track::TrackParCov(vertex, momentum, covV0, 0);
243  tV0.setQ2Pt(0); //No bending, please
244  int nCand2 = fitterCasc.process(tV0, bTrack);
245  if (nCand2 != 0) {
246  fitterCasc.propagateTracksToVertex();
247  hCascCandidate->Fill(2.5);
248  const auto& cascvtx = fitterCasc.getPCACandidate();
249  for (int i = 0; i < 3; i++) {
250  posXi[i] = cascvtx[i];
251  }
252  fitterCasc.getTrack(1).getPxPyPzGlo(pvecbach);
253  } //end if cascade recoed
254  } //end if v0 recoed
255  //Fill table, please
256  cascdata(
257  casc.v0Data().globalIndex(),
258  casc.bachelor_as<FullTracksExt>().globalIndex(),
259  casc.bachelor_as<FullTracksExt>().collisionId(),
260  charge, posXi[0], posXi[1], posXi[2], pos[0], pos[1], pos[2],
261  pvecpos[0], pvecpos[1], pvecpos[2],
262  pvecneg[0], pvecneg[1], pvecneg[2],
263  pvecbach[0], pvecbach[1], pvecbach[2],
264  fitterV0.getChi2AtPCACandidate(), fitterCasc.getChi2AtPCACandidate(),
265  casc.v0Data().posTrack_as<FullTracksExt>().dcaXY(),
266  casc.v0Data().negTrack_as<FullTracksExt>().dcaXY(),
267  casc.bachelor_as<FullTracksExt>().dcaXY());
268  }
269  }
270 };
