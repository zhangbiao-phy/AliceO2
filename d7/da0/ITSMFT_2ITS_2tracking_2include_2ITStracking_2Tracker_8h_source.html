158 template <typename... T>
159 float Tracker::evaluateTask(void (Tracker::*task)(T...), const char* taskName, std::ostream& ostream,
160  T&&... args)
161 {
162  float diff{0.f};
163 
164  if (constants::DoTimeBenchmarks) {
165  auto start = std::chrono::high_resolution_clock::now();
166  (this->*task)(std::forward<T>(args)...);
167  auto end = std::chrono::high_resolution_clock::now();
168 
169  std::chrono::duration<double, std::milli> diff_t{end - start};
170  diff = diff_t.count();
171 
172  if (fair::Logger::Logging(fair::Severity::info)) {
173  if (taskName == nullptr) {
174  ostream << diff << "\t";
175  } else {
176  ostream << std::setw(2) << " - " << taskName << " completed in: " << diff << " ms" << std::endl;
177  }
178  }
179  } else {
180  (this->*task)(std::forward<T>(args)...);
181  }
182 
183  return diff;
184 }
