30 struct Triggers {
31  enum {
32  bitMinBias,
33  bitMinBiasInner, // experimental
34  bitMinBiasOuter, // experimental
35  bitHighMult,
36  bitDummy // non-defined yet, placeholder
37  };
38  uint8_t triggerSignals = 0; // V0 trigger signals
39  int8_t nChanA = 0; // number of fired channels [A side]
40  int32_t amplA = -1000; // sum amplitude [A side]
41  Triggers() = default;
42 
43  Triggers(uint8_t signals, int8_t chanA, int32_t amplASum)
44  {
45  triggerSignals = signals;
46  nChanA = chanA;
47  amplA = amplASum;
48  }
49 
50  bool getIsMinBias() const { return (triggerSignals & (1 << bitMinBias)) != 0; }
51  bool getIsMinBiasInner() const { return (triggerSignals & (1 << bitMinBiasInner)) != 0; }
52  bool getIsMinBiasOuter() const { return (triggerSignals & (1 << bitMinBiasOuter)) != 0; }
53  bool getIsHighMult() const { return (triggerSignals & (1 << bitHighMult)) != 0; }
54 
55  void setTriggers(Bool_t isMinBias, Bool_t isMinBiasInner, Bool_t isMinBiasOuter, Bool_t isHighMult, Bool_t isDummy, int8_t chanA, int32_t amplASum)
56  {
57  triggerSignals = (isMinBias << bitMinBias) | (isMinBiasInner << bitMinBiasInner) | (isMinBiasOuter << bitMinBiasOuter) | (isHighMult << bitHighMult) | (isDummy << bitDummy);
58  nChanA = chanA;
59  amplA = amplASum;
60  }
61 
62  ClassDefNV(Triggers, 1);
63 };
