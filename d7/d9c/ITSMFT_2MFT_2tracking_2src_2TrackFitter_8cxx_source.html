53 bool TrackFitter::fit(TrackLTF& track, bool outward)
54 {
55 
58 
59  auto& mftTrackingParam = MFTTrackingParam::Instance();
60  auto nClusters = track.getNumberOfPoints();
61 
62  if (mftTrackingParam.verbose) {
63  std::cout << "Seed covariances: \n"
64  << track.getCovariances() << std::endl
65  << std::endl;
66  }
67 
68  // recursively compute clusters, updating the track parameters
69  if (!outward) { // Inward for vertexing
70  nClusters--;
71  while (nClusters-- > 0) {
72  if (!computeCluster(track, nClusters)) {
73  return false;
74  }
75  }
76  } else { // Outward for MCH matching
77  int ncl = 1;
78  while (ncl < nClusters) {
79  if (!computeCluster(track, ncl)) {
80  return false;
81  }
82  ncl++;
83  }
84  }
85  if (mftTrackingParam.verbose) {
86  // Print final covariances? std::cout << "Track covariances:"; track->getCovariances().Print();
87  std::cout << "Track Chi2 = " << track.getTrackChi2() << std::endl;
88  std::cout << " ***************************** Done fitting *****************************\n";
89  }
90 
91  return true;
92 }

95 bool TrackFitter::initTrack(TrackLTF& track, bool outward)
96 {
97 
98  auto& mftTrackingParam = MFTTrackingParam::Instance();
99 
100  // initialize the starting track parameters and cluster
101  double sigmainvQPtsq;
102  double chi2invqptquad;
103  auto invQPt0 = invQPtFromFCF(track, mBZField, sigmainvQPtsq);
104  auto nPoints = track.getNumberOfPoints();
105  auto k = TMath::Abs(o2::constants::math::B2C * mBZField);
106  auto Hz = std::copysign(1, mBZField);
107 
108  if (mftTrackingParam.verbose) {
109  std::cout << "\n ***************************** Start Fitting new track ***************************** \n";
110  std::cout << "N Clusters = " << nPoints << std::endl;
111  }
112 
113  track.setInvQPtSeed(invQPt0);
114  track.setChi2QPtSeed(chi2invqptquad);
115  track.setInvQPt(invQPt0);
116 
118 
119  int first_cls, last_cls;
120  if (outward) { // MCH matching
121  first_cls = 0;
122  last_cls = nPoints - 1;
123  } else { // Vertexing
124  first_cls = nPoints - 1;
125  last_cls = 0;
126  }
127 
128  auto x0 = track.getXCoordinates()[first_cls];
129  auto y0 = track.getYCoordinates()[first_cls];
130  auto z0 = track.getZCoordinates()[first_cls];
131 
132  auto deltaX = track.getXCoordinates()[nPoints - 1] - track.getXCoordinates()[0];
133  auto deltaY = track.getYCoordinates()[nPoints - 1] - track.getYCoordinates()[0];
134  auto deltaZ = track.getZCoordinates()[nPoints - 1] - track.getZCoordinates()[0];
135  auto deltaR = TMath::Sqrt(deltaX * deltaX + deltaY * deltaY);
136  auto tanl0 = 0.5 * TMath::Sqrt2() * (deltaZ / deltaR) *
137  TMath::Sqrt(TMath::Sqrt((invQPt0 * deltaR * k) * (invQPt0 * deltaR * k) + 1) + 1);
138  auto phi0 = TMath::ATan2(deltaY, deltaX) - 0.5 * Hz * invQPt0 * deltaZ * k / tanl0;
139  auto sigmax0sq = track.getSigmasX2()[first_cls];
140  auto sigmay0sq = track.getSigmasY2()[first_cls];
141  auto sigmax1sq = track.getSigmasX2()[last_cls];
142  auto sigmay1sq = track.getSigmasY2()[last_cls];
143  auto sigmaDeltaXsq = sigmax0sq + sigmax1sq;
144  auto sigmaDeltaYsq = sigmay0sq + sigmay1sq;
145 
146  track.setX(x0);
147  track.setY(y0);
148  track.setZ(z0);
149  track.setPhi(phi0);
150  track.setTanl(tanl0);
151 
152  if (mftTrackingParam.verbose) {
153  std::cout << " Init " << (track.isCA() ? "CA Track " : "LTF Track") << std::endl;
154  auto model = (mftTrackingParam.trackmodel == Helix) ? "Helix" : (mftTrackingParam.trackmodel == Quadratic) ? "Quadratic" : "Linear";
155  std::cout << "Track Model: " << model << std::endl;
156  std::cout << " initTrack: X = " << x0 << " Y = " << y0 << " Z = " << z0 << " Tgl = " << tanl0 << " Phi = " << phi0 << " pz = " << track.getPz() << " qpt = " << 1.0 / track.getInvQPt() << std::endl;
157  std::cout << " Variances: sigma2_x0 = " << TMath::Sqrt(sigmax0sq) << " sigma2_y0 = " << TMath::Sqrt(sigmay0sq) << " sigma2_q/pt = " << TMath::Sqrt(sigmainvQPtsq) << std::endl;
158  }
159 
160  auto deltaR2 = deltaR * deltaR;
161  auto deltaR3 = deltaR2 * deltaR;
162  auto deltaR4 = deltaR2 * deltaR2;
163  auto k2 = k * k;
164  auto A = TMath::Sqrt(track.getInvQPt() * track.getInvQPt() * deltaR2 * k2 + 1);
165  auto A2 = A * A;
166  auto B = A + 1.0;
167  auto B2 = B * B;
168  auto B3 = B * B * B;
169  auto B12 = TMath::Sqrt(B);
170  auto B32 = B * B12;
171  auto B52 = B * B32;
172  auto C = invQPt0 * k;
173  auto C2 = C * C;
174  auto C3 = C * C2;
175  auto D = 1.0 / (A2 * B2 * B2 * deltaR4);
176  auto E = D * deltaZ / (B * deltaR);
177  auto F = deltaR * deltaX * C3 * Hz / (A * B32);
178  auto G = 0.5 * TMath::Sqrt2() * A * B32 * C * Hz * deltaR;
179  auto Gx = G * deltaX;
180  auto Gy = G * deltaY;
181  auto H = -0.25 * TMath::Sqrt2() * B12 * C3 * Hz * deltaR3;
182  auto Hx = H * deltaX;
183  auto Hy = H * deltaY;
184  auto I = A * B2;
185  auto Ix = I * deltaX;
186  auto Iy = I * deltaY;
187  auto J = 2 * B * deltaR3 * deltaR3 * k2;
188  auto K = 0.5 * A * B - 0.25 * C2 * deltaR2;
189  auto L0 = Gx + Hx + Iy;
190  auto M0 = -Gy - Hy + Ix;
191  auto N = -0.5 * B3 * C * Hz * deltaR3 * deltaR4 * k2;
192  auto O = 0.125 * C2 * deltaR4 * deltaR4 * k2;
193  auto P = -K * k * Hz * deltaR / A;
194  auto Q = deltaZ * deltaZ / (A2 * B * deltaR3 * deltaR3);
195  auto R = 0.25 * C * deltaZ * TMath::Sqrt2() * deltaR * k / (A * B12);
196 
197  SMatrix55 lastParamCov;
198  lastParamCov(0, 0) = sigmax0sq; // <X,X>
199  lastParamCov(0, 1) = 0; // <Y,X>
200  lastParamCov(0, 2) = 0; // <PHI,X>
201  lastParamCov(0, 3) = 0; // <TANL,X>
202  lastParamCov(0, 4) = 0; // <INVQPT,X>
203 
204  lastParamCov(1, 1) = sigmay0sq; // <Y,Y>
205  lastParamCov(1, 2) = 0; // <PHI,Y>
206  lastParamCov(1, 3) = 0; // <TANL,Y>
207  lastParamCov(1, 4) = 0; // <INVQPT,Y>
208 
209  lastParamCov(2, 2) = D * (J * K * K * sigmainvQPtsq + L0 * L0 * sigmaDeltaXsq + M0 * M0 * sigmaDeltaYsq); // <PHI,PHI>
210  lastParamCov(2, 3) = E * K * (TMath::Sqrt2() * B52 * (L0 * deltaX * sigmaDeltaXsq - deltaY * sigmaDeltaYsq * M0) + N * sigmainvQPtsq); // <TANL,PHI>
211  lastParamCov(2, 4) = P * sigmainvQPtsq * TMath::Sqrt2() / B32; // <INVQPT,PHI>
212 
213  lastParamCov(3, 3) = Q * (2 * K * K * (deltaX * deltaX * sigmaDeltaXsq + deltaY * deltaY * sigmaDeltaYsq) + O * sigmainvQPtsq); // <TANL,TANL>
214  lastParamCov(3, 4) = R * sigmainvQPtsq; // <INVQPT,TANL>
215 
216  lastParamCov(4, 4) = sigmainvQPtsq; // <INVQPT,INVQPT>
217 
218  track.setCovariances(lastParamCov);
219  track.setTrackChi2(0.);
220 
221  return true;
222 }

225 bool TrackFitter::computeCluster(TrackLTF& track, int cluster)
226 {
231 
232  auto& mftTrackingParam = MFTTrackingParam::Instance();
233  const auto& clx = track.getXCoordinates()[cluster];
234  const auto& cly = track.getYCoordinates()[cluster];
235  const auto& clz = track.getZCoordinates()[cluster];
236  const auto& sigmaX2 = track.getSigmasX2()[cluster];
237  const auto& sigmaY2 = track.getSigmasY2()[cluster];
238 
239  if (track.getZ() == clz) {
240  LOG(INFO) << "AddCluster ERROR: The new cluster must be upstream! Bug on TrackFinder. " << (track.isCA() ? " CATrack" : "LTFTrack");
241  LOG(INFO) << "track.getZ() = " << track.getZ() << " ; newClusterZ = " << clz << " ==> Skipping point.";
242  return true;
243  }
244  if (mftTrackingParam.verbose) {
245  std::cout << "computeCluster: X = " << clx << " Y = " << cly << " Z = " << clz << " nCluster = " << cluster << std::endl;
246  }
247 
248  // add MCS effects for the new cluster
249  using o2::mft::constants::LayerZPosition;
250  int startingLayerID, newLayerID;
251 
252  auto dZ = clz - track.getZ();
253  //LayerID of each cluster from ZPosition // TODO: Use ChipMapping
254  for (auto layer = 10; layer--;) {
255  if (track.getZ() < LayerZPosition[layer] + .3 & track.getZ() > LayerZPosition[layer] - .3) {
256  startingLayerID = layer;
257  }
258  }
259  for (auto layer = 10; layer--;) {
260  if (clz<LayerZPosition[layer] + .3 & clz> LayerZPosition[layer] - .3) {
261  newLayerID = layer;
262  }
263  }
264  // Number of disks crossed by this tracklet
265  int NDisksMS;
266  if (clz - track.getZ() > 0) {
267  NDisksMS = (startingLayerID % 2 == 0) ? (startingLayerID - newLayerID) / 2 : (startingLayerID - newLayerID + 1) / 2;
268  } else {
269  NDisksMS = (startingLayerID % 2 == 0) ? (newLayerID - startingLayerID + 1) / 2 : (newLayerID - startingLayerID) / 2;
270  }
271 
272  auto MFTDiskThicknessInX0 = mftTrackingParam.MFTRadLength / 5.0;
273  if (mftTrackingParam.verbose) {
274  std::cout << "startingLayerID = " << startingLayerID << " ; "
275  << "newLayerID = " << newLayerID << " ; ";
276  std::cout << "cl.getZ() = " << clz << " ; ";
277  std::cout << "startingParam.getZ() = " << track.getZ() << " ; ";
278  std::cout << "NDisksMS = " << NDisksMS << std::endl;
279  }
280 
281  if ((NDisksMS * MFTDiskThicknessInX0) != 0) {
282  track.addMCSEffect(-1, NDisksMS * MFTDiskThicknessInX0);
283  }
284 
285  if (mftTrackingParam.verbose) {
286  std::cout << " BeforeExtrap: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " qpt = " << 1.0 / track.getInvQPt() << std::endl;
287  }
288 
289  // Propagate track to the z position of the new cluster
290  switch (mftTrackingParam.trackmodel) {
291  case Linear:
292  track.propagateToZlinear(clz);
293  break;
294  case Quadratic:
295  track.propagateToZquadratic(clz, mBZField);
296  break;
297  case Helix:
298  track.propagateToZhelix(clz, mBZField);
299  break;
300  default:
301  std::cout << " Invalid track model.\n";
302  return false;
303  break;
304  }
305 
306  if (mftTrackingParam.verbose) {
307  std::cout << " AfterExtrap: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " qpt = " << 1.0 / track.getInvQPt() << std::endl;
308  }
309 
310  // recompute parameters
311  const std::array<float, 2>& pos = {clx, cly};
312  const std::array<float, 2>& cov = {sigmaX2, sigmaY2};
313 
314  if (track.update(pos, cov)) {
315  if (mftTrackingParam.verbose) {
316  std::cout << " New Cluster: X = " << clx << " Y = " << cly << " Z = " << clz << std::endl;
317  std::cout << " AfterKalman: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " qpt = " << 1.0 / track.getInvQPt() << std::endl;
318  std::cout << std::endl;
319  // Outputs track covariance matrix:
320  // param.getCovariances().Print();
321  }
322  return true;
323  }
324  return false;
325 }

328 Double_t invQPtFromFCF(const TrackLTF& track, Double_t bFieldZ, Double_t& sigmainvqptsq)
329 {
330 
331  const std::array<Float_t, constants::mft::LayersNumber>& xPositions = track.getXCoordinates();
332  const std::array<Float_t, constants::mft::LayersNumber>& yPositions = track.getYCoordinates();
333  const std::array<Float_t, constants::mft::LayersNumber>& zPositions = track.getZCoordinates();
334  const std::array<Float_t, constants::mft::LayersNumber>& SigmasX2 = track.getSigmasX2();
335  const std::array<Float_t, constants::mft::LayersNumber>& SigmasY2 = track.getSigmasY2();
336 
337  // Fast Circle Fit (Hansroul, Jeremie, Savard, 1987)
338  auto nPoints = track.getNumberOfPoints();
339  Double_t* xVal = new Double_t[nPoints];
340  Double_t* yVal = new Double_t[nPoints];
341  Double_t* zVal = new Double_t[nPoints];
342  Double_t* xErr = new Double_t[nPoints];
343  Double_t* yErr = new Double_t[nPoints];
344  Double_t* uVal = new Double_t[nPoints - 1];
345  Double_t* vVal = new Double_t[nPoints - 1];
346  Double_t* vErr = new Double_t[nPoints - 1];
347  Double_t* fweight = new Double_t[nPoints - 1];
348  Double_t* Rn = new Double_t[nPoints - 1];
349  Double_t* Pn = new Double_t[nPoints - 1];
350  Double_t A, Aerr, B, Berr, x2, y2, invx2y2, a, b, r, sigmaRsq, u2, sigma;
351  Double_t F0, F1, F2, F3, F4, SumSRn, SumSPn, SumRn, SumUPn, SumRP;
352 
353  SumSRn = SumSPn = SumRn = SumUPn = SumRP = 0.0;
354  F0 = F1 = F2 = F3 = F4 = 0.0;
355 
356  for (auto np = 0; np < nPoints; np++) {
357  xErr[np] = SigmasX2[np];
358  yErr[np] = SigmasY2[np];
359  if (np > 0) {
360  xVal[np] = xPositions[np] - xVal[0];
361  yVal[np] = yPositions[np] - yVal[0];
362  xErr[np] *= std::sqrt(2.);
363  yErr[np] *= std::sqrt(2.);
364  } else {
365  xVal[np] = 0.;
366  yVal[np] = 0.;
367  }
368  zVal[np] = zPositions[np];
369  }
370  for (int i = 0; i < (nPoints - 1); i++) {
371  x2 = xVal[i + 1] * xVal[i + 1];
372  y2 = yVal[i + 1] * yVal[i + 1];
373  invx2y2 = 1. / (x2 + y2);
374  uVal[i] = xVal[i + 1] * invx2y2;
375  vVal[i] = yVal[i + 1] * invx2y2;
376  vErr[i] = std::sqrt(8. * xErr[i + 1] * xErr[i + 1] * x2 * y2 + 2. * yErr[i + 1] * yErr[i + 1] * (x2 - y2) * (x2 - y2)) * invx2y2 * invx2y2;
377  u2 = uVal[i] * uVal[i];
378  fweight[i] = 1. / vErr[i];
379  F0 += fweight[i];
380  F1 += fweight[i] * uVal[i];
381  F2 += fweight[i] * u2;
382  F3 += fweight[i] * uVal[i] * u2;
383  F4 += fweight[i] * u2 * u2;
384  }
385 
386  double Rn_det1 = F2 * F4 - F3 * F3;
387  double Rn_det2 = F1 * F4 - F2 * F3;
388  double Rn_det3 = F1 * F3 - F2 * F2;
389  double Pn_det1 = Rn_det2;
390  double Pn_det2 = F0 * F4 - F2 * F2;
391  double Pn_det3 = F0 * F3 - F1 * F2;
392 
393  for (int j = 0; j < (nPoints - 1); j++) {
394  Rn[j] = fweight[j] * (Rn_det1 - uVal[j] * Rn_det2 + uVal[j] * uVal[j] * Rn_det3);
395  SumSRn += Rn[j] * Rn[j] * vErr[j] * vErr[j];
396  SumRn += Rn[j];
397 
398  Pn[j] = fweight[j] * (-Pn_det1 + uVal[j] * Pn_det2 - uVal[j] * uVal[j] * Pn_det3);
399  SumSPn += Pn[j] * Pn[j] * vErr[j] * vErr[j];
400  SumUPn += uVal[j] * Pn[j];
401 
402  SumRP += Rn[j] * Pn[j] * vErr[j] * vErr[j] * vErr[j];
403  }
404 
405  Double_t invqpt_fcf;
406  Int_t qfcf;
407  // chi2 = 0.;
408  if (LinearRegression((nPoints - 1), uVal, vVal, vErr, B, Berr, A, Aerr)) {
409  // v = a * u + b
410  // circle passing through (0,0):
411  // (x - rx)^2 + (y - ry)^2 = r^2
412  // ---> a = - rx / ry;
413  // ---> b = 1 / (2 * ry)
414  b = 1. / (2. * A);
415  a = -B * b;
416  r = std::sqrt(a * a + b * b);
417  double_t invR = 1. / r;
418 
419  // pt --->
420  Double_t invpt = 1. / (o2::constants::math::B2C * bFieldZ * r);
421 
422  // sign(q) --->
423  // rotate around the first point (0,0) to bring the last point
424  // on the x axis (y = 0) and check the y sign of the rotated
425  // center of the circle
426  Double_t x = xVal[nPoints - 1], y = yVal[nPoints - 1], z = zVal[nPoints - 1];
427  Double_t slope = TMath::ATan2(y, x);
428  Double_t cosSlope = TMath::Cos(slope);
429  Double_t sinSlope = TMath::Sin(slope);
430  Double_t rxRot = a * cosSlope + b * sinSlope;
431  Double_t ryRot = a * sinSlope - b * cosSlope;
432  qfcf = (ryRot > 0.) ? -1 : +1;
433 
434  Double_t alpha = 2.0 * std::abs(TMath::ATan2(rxRot, ryRot));
435  Double_t x0 = xVal[0], y0 = yVal[0], z0 = zVal[0];
436  Double_t dxyz2 = (x - x0) * (x - x0) + (y - y0) * (y - y0) + (z - z0) * (z - z0);
437  Double_t cst = 1000.;
438  Double_t c_alpha = cst * alpha;
439  Double_t p, pt, pz;
440  pt = 1. / invpt;
441  p = std::sqrt(dxyz2) / c_alpha;
442  pz = std::sqrt(p * p - pt * pt);
443 
444  invqpt_fcf = qfcf * invpt;
445 
446  //error calculations:
447  double invA2 = 1. / (A * A);
448 
449  double sigmaAsq = SumSRn / (SumRn * SumRn);
450  double sigmaBsq = SumSPn / (SumUPn * SumUPn);
451  double sigmaAB = SumRP / (SumRn * SumUPn);
452 
453  double sigmaasq_FCF = TMath::Abs(0.25 * invA2 * invA2 * (B * B * sigmaAsq + A * A * sigmaBsq - A * B * sigmaAB));
454  double sigmabsq_FCF = TMath::Abs(0.25 * invA2 * invA2 * sigmaAsq);
455  double sigma2R = invR * invR * (b * b * sigmaasq_FCF + a * a * sigmabsq_FCF + 2 * a * b * TMath::Sqrt(sigmaasq_FCF) * TMath::Sqrt(sigmabsq_FCF));
456 
457  sigmainvqptsq = sigma2R * invpt * invpt * invR * invR;
458 
459  } else { // the linear regression failed...
460  LOG(WARN) << "LinearRegression failed!";
461  invqpt_fcf = 1. / 100.;
462  }
463 
464  return invqpt_fcf;
465 }

468 Bool_t LinearRegression(Int_t nVal, Double_t* xVal, Double_t* yVal, Double_t* yErr, Double_t& B, Double_t& Berr, Double_t& A, Double_t& Aerr)
469 {
470  // linear regression y = B * x + A
471 
472  Double_t S1, SXY, SX, SY, SXX, SsXY, SsXX, SsYY, Xm, Ym, s, delta, difx;
473  Double_t invYErr2;
474 
475  S1 = SXY = SX = SY = SXX = 0.0;
476  SsXX = SsYY = SsXY = Xm = Ym = 0.;
477  difx = 0.;
478  for (Int_t i = 0; i < nVal; i++) {
479  invYErr2 = 1. / (yErr[i] * yErr[i]);
480  S1 += invYErr2;
481  SXY += xVal[i] * yVal[i] * invYErr2;
482  SX += xVal[i] * invYErr2;
483  SY += yVal[i] * invYErr2;
484  SXX += xVal[i] * xVal[i] * invYErr2;
485  if (i > 0) {
486  difx += TMath::Abs(xVal[i] - xVal[i - 1]);
487  }
488  Xm += xVal[i];

510  Berr = s / TMath::Sqrt(SsXX);
511  } else {
512  Aerr = 0.;
513  Berr = 0.;
514  }
