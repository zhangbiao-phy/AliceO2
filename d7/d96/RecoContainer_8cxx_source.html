131 void RecoContainer::collectData(ProcessingContext& pc, const DataRequest& requests)
132 {
133  auto& reqMap = requests.requestMap;
134 
136  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().getByPos(0).header);
137  startIR = {0, dh->firstTForbit};
138 
139  auto req = reqMap.find("trackITS");
140  if (req != reqMap.end()) {
141  addITSTracks(pc, req->second);
142  }
143 
144  req = reqMap.find("trackTPC");
145  if (req != reqMap.end()) {
146  addTPCTracks(pc, req->second);
147  }
148 
149  req = reqMap.find("trackITSTPC");
150  if (req != reqMap.end()) {
151  addITSTPCTracks(pc, req->second);
152  }
153 
154  req = reqMap.find("trackTPCTOF");
155  if (req != reqMap.end()) {
156  addTPCTOFTracks(pc, req->second);
157  }
158 
159  req = reqMap.find("matchTOF");
160  if (req != reqMap.end()) {
161  addTOFMatches(pc, req->second);
162  }
163 
164  req = reqMap.find("clusITS");
165  if (req != reqMap.end()) {
166  addITSClusters(pc, req->second);
167  }
168 
169  req = reqMap.find("clusTPC");
170  if (req != reqMap.end()) {
171  addTPCClusters(pc, req->second);
172  }
173 
174  req = reqMap.find("clusTOF");
175  if (req != reqMap.end()) {
176  addTOFClusters(pc, req->second);
177  }
178 
179  req = reqMap.find("FT0");
180  if (req != reqMap.end()) {
181  addFT0RecPoints(pc, req->second);
182  }
183 }

276 const o2::track::TrackParCov& RecoContainer::getTrackParamOut(GTrackID gidx) const
277 {
278  // get outer param of track
279  auto trSrc = gidx.getSource();
280  if (trSrc == GTrackID::ITSTPC) {
281  return tracksPool.get_as<o2d::TrackTPCITS>(gidx).getParamOut();
282  } else if (trSrc == GTrackID::ITSTPCTOF) { // the physical tracks are in ITS-TPC, need to get reference from match info
283  return tracksPool.get_as<o2d::TrackTPCITS>(getTOFMatch<o2d::MatchInfoTOF>(gidx).getEvIdxTrack().getIndex()).getParamOut();
284  } else if (trSrc == GTrackID::TPCTOF) {
285  return tracksPool.get_as<o2d::TrackTPCTOF>(gidx).getParamOut();
286  } else if (trSrc == GTrackID::ITS) {
287  return tracksPool.get_as<o2::its::TrackITS>(gidx).getParamOut();
288  } else if (trSrc == GTrackID::TPC) {
289  return tracksPool.get_as<o2::tpc::TrackTPC>(gidx).getParamOut();
290  } else {
291  throw std::runtime_error(fmt::format("not defined for tracks of source {:d}", int(trSrc)));
292  }
293 }

318 void RecoContainer::createTracks(std::function<void(const o2::track::TrackParCov&, float, float, GTrackID)> const& creator) const
319 {
320  // We go from most complete tracks to least complete ones, taking into account that some track times
321  // do not bear their own kinematics but just constrain the time
322  // As we get more track types functional, this method should be completed
323  constexpr float PS2MUS = 1e-6;
324  std::array<std::vector<uint8_t>, GTrackID::NSources> usedData;
325  auto flagUsed = [&usedData](const GTrackID gidx) { auto src = gidx.getSource();
326  if (!usedData[src].empty()) {
327  usedData[gidx.getSource()][gidx.getIndex()] = 1;
328  } };
329  auto isUsed = [&usedData](const GTrackID gidx) { return (!usedData[gidx.getSource()].empty()) && (usedData[gidx.getSource()][gidx.getIndex()] != 0); };
330  auto isUsed2 = [&usedData](int idx, int src) { return (!usedData[src].empty()) && (usedData[src][idx] != 0); };
331 
332  // create only for those data types which are used
333  const auto& tracksITS = getITSTracks<o2::its::TrackITS>();
334  const auto& tracksTPC = getTPCTracks<o2::tpc::TrackTPC>();
335  const auto& tracksTPCITS = getTPCITSTracks<o2d::TrackTPCITS>();
336  const auto& tracksTPCTOF = getTPCTOFTracks<o2d::TrackTPCTOF>();
337  const auto& matchesTPCTOF = getTPCTOFMatches<o2d::MatchInfoTOF>();
338 
339  usedData[GTrackID::ITS].resize(tracksITS.size()); // to flag used ITS tracks
340  usedData[GTrackID::TPC].resize(tracksTPC.size()); // to flag used TPC tracks
341  usedData[GTrackID::ITSTPC].resize(tracksTPCITS.size()); // to flag used ITSTPC tracks
342 
343  // ITS-TPC-TOF matches, may refer to ITS-TPC (TODO: something else?) tracks
344  {
345  auto matches = getTOFMatches<o2d::MatchInfoTOF>(); // thes are just MatchInfoTOF objects, pointing on ITS-TPC match and TOF cl.
346  auto tofClusters = getTOFClusters<o2::tof::Cluster>();
347  if (matches.size() && (!tofClusters.size() || !tracksTPCITS.size())) {
348  throw std::runtime_error(fmt::format("Global-TOF tracks ({}) require ITS-TPC tracks ({}) and TOF clusters ({})",
349  matches.size(), tracksTPCITS.size(), tofClusters.size()));
350  }
351  for (unsigned i = 0; i < matches.size(); i++) {
352  const auto& match = matches[i];
353  const auto& tofCl = tofClusters[match.getTOFClIndex()];
354  float timeTOFMUS = (tofCl.getTime() - match.getLTIntegralOut().getTOF(o2::track::PID::Pion)) * PS2MUS; // tof time in \mus, FIXME: account for time of flight to R TOF
355  const float timeErr = 0.010f; // assume 10 ns error FIXME
356 
357  auto gidx = match.getEvIdxTrack().getIndex(); // this should be corresponding ITS-TPC track
358  creator(tracksPool.get(gidx), timeTOFMUS, timeErr, {i, GTrackID::ITSTPCTOF});
359  flagUsed(gidx); // flag used ITS-TPC tracks
360  }
361  }
362 
363  // ITS-TPC matches, may refer to ITS, TPC (TODO: something else?) tracks
364  {
365  for (unsigned i = 0; i < tracksTPCITS.size(); i++) {
366  const auto& matchTr = tracksTPCITS[i];
367  flagUsed(matchTr.getRefITS()); // flag used ITS tracks
368  flagUsed(matchTr.getRefTPC()); // flag used TPC tracks
369  if (isUsed2(i, GTrackID::ITSTPC)) {
370  continue;
371  }
372  creator(matchTr, matchTr.getTimeMUS().getTimeStamp(), matchTr.getTimeMUS().getTimeStampError(), {i, GTrackID::ITSTPC});
373  }
374  }
375 
376  // TPC-TOF matches, may refer to TPC (TODO: something else?) tracks
377  {
378  if (matchesTPCTOF.size() && !tracksTPCTOF.size()) {
379  throw std::runtime_error(fmt::format("TPC-TOF matched tracks ({}) require TPCTOF matches ({}) and TPCTOF tracks ({})",
380  matchesTPCTOF.size(), tracksTPCTOF.size()));
381  }
382  for (unsigned i = 0; i < matchesTPCTOF.size(); i++) {
383  const auto& match = matchesTPCTOF[i];
384  const auto& gidx = match.getEvIdxTrack().getIndex(); // TPC (or other?) track global idx (FIXME: TOF has to git rid of EvIndex stuff)
385  if (isUsed(gidx)) { // is TPC track already used
386  continue;
387  }
388  flagUsed(gidx); // flag used TPC tracks
389  const auto& trc = tracksTPCTOF[i];
390  creator(trc, trc.getTimeMUS().getTimeStamp(), trc.getTimeMUS().getTimeStampError(), {i, GTrackID::TPCTOF});
391  }
392  }
393 
394  // TPC only tracks
395  {
396  for (unsigned i = 0; i < tracksTPC.size(); i++) {
397  if (isUsed2(i, GTrackID::TPC)) { // skip used tracks
398  continue;
399  }
400  const auto& trc = tracksTPC[i];
401  creator(trc, trc.getTime0(), 0.5 * (trc.getDeltaTBwd() + trc.getDeltaTFwd()), {i, GTrackID::TPC});
402  }
403  }
404 
405  // ITS only tracks
406  {
407  const auto& rofrs = getITSTracksROFRecords<o2::itsmft::ROFRecord>();
408  for (unsigned irof = 0; irof < rofrs.size(); irof++) {
409  const auto& rofRec = rofrs[irof];
410  float t0 = rofRec.getBCData().differenceInBC(startIR) * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
411  int trlim = rofRec.getFirstEntry() + rofRec.getNEntries();
412  for (int it = rofRec.getFirstEntry(); it < trlim; it++) {
413  if (isUsed2(it, GTrackID::ITS)) { // skip used tracks
414  continue;
415  }
416  GTrackID gidITS(it, GTrackID::ITS);
417  const auto& trc = getITSTrack<o2::its::TrackITS>(gidITS);
418  creator(trc, t0, 0.5, gidITS);
419  }
420  }
421  }
422 }

424 // get contributors from single detectors
425 RecoContainer::GlobalIDSet RecoContainer::getSingleDetectorRefs(GTrackID gidx) const
426 {
427  GlobalIDSet table;
428  auto src = gidx.getSource();
429  table[src] = gidx;
430  if (src == GTrackID::ITSTPCTOF) {
431  const auto& parent0 = getTOFMatch<o2d::MatchInfoTOF>(gidx); //ITS/TPC : TOF
432  const auto& parent1 = getTPCITSTrack<o2d::TrackTPCITS>(parent0.getEvIdxTrack().getIndex());
433  table[GTrackID::ITSTPC] = parent0.getEvIdxTrack().getIndex();
434  table[GTrackID::TOF] = {unsigned(parent0.getEvIdxTOFCl().getIndex()), GTrackID::TOF};
435  table[GTrackID::ITS] = parent1.getRefITS();
436  table[GTrackID::TPC] = parent1.getRefTPC();
437  } else if (src == GTrackID::TPCTOF) {
438  const auto& parent0 = getTPCTOFMatch<o2d::MatchInfoTOF>(gidx); //TPC : TOF
439  table[GTrackID::TOF] = {unsigned(parent0.getEvIdxTOFCl().getIndex()), GTrackID::TOF};
440  table[GTrackID::TPC] = parent0.getEvIdxTrack().getIndex();
441  } else if (src == GTrackID::ITSTPC) {
442  const auto& parent0 = getTPCITSTrack<o2d::TrackTPCITS>(gidx);
443  table[GTrackID::ITS] = parent0.getRefITS();
444  table[GTrackID::TPC] = parent0.getRefTPC();
445  }
446  return std::move(table);
447 }
