62 void AltroDecoder::readChannels()
63 {
64  mChannelsInitialized = false;
65  mChannels.clear();
66  int currentpos = 0;
67  auto& buffer = mRawReader.getPayload().getPayloadWords();
68 
69  int payloadend = mRCUTrailer.getPayloadSize();
70  while (currentpos < payloadend) {
71  auto currentword = buffer[currentpos++];
72  ChannelHeader header = {currentword};
73 
74  if (header.mMark != 1) {
75  LOG(ERROR) << "Channel header mark not found";
76  continue;
77  }
78  // starting a new channel
79  mChannels.emplace_back(int(header.mHardwareAddress), int(header.mPayloadSize));
80  auto& currentchannel = mChannels.back();
82  int numberofwords = (currentchannel.getPayloadSize() + 2) / 3;
83  if (numberofwords > payloadend - currentpos) {
84  LOG(ERROR) << "Channel payload " << numberofwords << " larger than left in total " << payloadend - currentpos;
85  continue;
86  }
87  std::vector<uint16_t> bunchwords;
88  for (int iword = 0; iword < numberofwords; iword++) {
89  currentword = buffer[currentpos++];
90  if ((currentword >> 30) != 0) {
91  LOG(ERROR) << "Unexpected end of payload in altro channel payload! FEE=" << o2::raw::RDHUtils::getFEEID(mRawReader.getRawHeader())
92  << ", Address=0x" << std::hex << currentchannel.getHardwareAddress() << ", word=0x" << currentword << std::dec;
93  currentpos--;
94  continue;
95  }
96  bunchwords.push_back((currentword >> 20) & 0x3FF);
97  bunchwords.push_back((currentword >> 10) & 0x3FF);
98  bunchwords.push_back(currentword & 0x3FF);
99  }
100 
101  // decode bunches
102  int currentsample = 0;
103  while (currentsample < currentchannel.getPayloadSize()) {
104  int bunchlength = bunchwords[currentsample] - 2, // remove words for bunchlength and starttime
105  starttime = bunchwords[currentsample + 1];
106  auto& currentbunch = currentchannel.createBunch(bunchlength, starttime);
107  currentbunch.initFromRange(gsl::span<uint16_t>(&bunchwords[currentsample + 2], std::min((unsigned long)bunchlength, bunchwords.size() - currentsample - 2)));
108  currentsample += bunchlength + 2;
109  }
110  }
111  mChannelsInitialized = true;
112 }
