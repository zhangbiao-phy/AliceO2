42 bool PadResponse::importPRF(std::string file, std::unique_ptr<TGraph2D>& grPRF) const
43 {
44  std::string inputDir;
45  const char* aliceO2env = std::getenv("O2_ROOT");
46  if (aliceO2env) {
47  inputDir = aliceO2env;
48  }
49  inputDir += "/share/Detectors/TPC/files/";
50 
51  float x, y, normalizedPadResponse;
52  int i = 0;
53  std::ifstream prfFile(inputDir + file, std::ifstream::in);
54  if (!prfFile) {
55  LOG(FATAL) << "tpc::PadResponse - Input file '" << inputDir + file << "' does not exist! No PRF loaded!";
56  return false;
57  }
58  for (std::string line; std::getline(prfFile, line);) {
59  std::istringstream is(line);
60  while (is >> x >> y >> normalizedPadResponse) {
61  grPRF->SetPoint(i++, x, y, normalizedPadResponse);
62  }
63  }
64  return true;
65 }

67 float PadResponse::getPadResponse(GlobalPosition3D posEle, DigitPos digiPadPos) const
68 {
70  const Mapper& mapper = Mapper::instance();
71  const PadCentre padCentre = mapper.padCentre(mapper.globalPadNumber(digiPadPos.getPadPos()));

73  const Sector sector(digiPadPos.getPadSecPos().getSector());
74  const LocalPosition3D padCentreLocal(padCentre.X(), padCentre.Y(), posEle.Z());
75  const GlobalPosition3D padCentrePos = mapper.LocalToGlobal(padCentreLocal, sector);
76 
78 
79  const int gemStack = int(cru.gemStack());
80  const float offsetX = std::abs(posEle.X() - padCentre.X()) * 10.f;
81  const float offsetY = std::abs(posEle.Y() - padCentre.Y()) * 10.f;
82  float normalizedPadResponse = 0;
83  if (gemStack == 0) {
84  normalizedPadResponse = mIROC->Interpolate(offsetX, offsetY);
85  } else if (gemStack == 1 || gemStack == 2) {
86  normalizedPadResponse = mOROC12->Interpolate(offsetX, offsetY);
87  } else {
88  normalizedPadResponse = mOROC3->Interpolate(offsetX, offsetY);
89  }
90  return normalizedPadResponse;
