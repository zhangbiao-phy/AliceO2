62  Double_t getInvPt() const { return TMath::Abs(mParameters(4)); }

64  Double_t getPx() const { return TMath::Cos(getPhi()) * getPt(); } // return px

71  Double_t getInvPz() const { return 1. / getPz(); } // return invpz
72 
73  Double_t getP() const { return getPt() * TMath::Sqrt(1. + getTanl() * getTanl()); } // return total momentum
74  Double_t getInverseMomentum() const { return 1.f / getP(); }
75 
76  Double_t getEta() const { return -TMath::Log(TMath::Tan((TMath::PiOver2() - TMath::ATan(getTanl())) / 2)); } // return total momentum
