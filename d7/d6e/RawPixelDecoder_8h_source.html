136 template <class Mapping>
137 template <class DigitContainer, class ROFContainer>
138 int RawPixelDecoder<Mapping>::fillDecodedDigits(DigitContainer& digits, ROFContainer& rofs)
139 {
140  if (mInteractionRecord.isDummy()) {
141  return 0; // nothing was decoded
142  }
143  mTimerFetchData.Start(false);
144  int ref = digits.size();
145  for (unsigned int iru = 0; iru < mRUDecodeVec.size(); iru++) {
146  for (int ic = 0; ic < mRUDecodeVec[iru].nChipsFired; ic++) {
147  const auto& chip = mRUDecodeVec[iru].chipsData[ic];
148  for (const auto& hit : mRUDecodeVec[iru].chipsData[ic].getData()) {
149  digits.emplace_back(chip.getChipID(), hit.getRow(), hit.getCol());
150  }
151  }
152  }
153  int nFilled = digits.size() - ref;
154  rofs.emplace_back(mInteractionRecord, mROFCounter, ref, nFilled);
155  mTimerFetchData.Stop();
156  return nFilled;
157 }
