43  void init(InitContext const&)
44  {
45  hJetPt.setObject(new TH1F("h_jet_pt", "jet p_{T};p_{T} (GeV/#it{c})",
46  100, 0., 100.));
47  hD0Pt.setObject(new TH1F("h_D0_pt", "jet p_{T,D};p_{T,D} (GeV/#it{c})",
48  60, 0., 60.));

51  Configurable<int> d_selectionFlagD0{"d_selectionFlagD0", 1, "Selection Flag for D0"};
52  Configurable<int> d_selectionFlagD0bar{"d_selectionFlagD0bar", 1, "Selection Flag for D0bar"};
53 
54  //need enum as configurable
55  enum pdgCode { pdgD0 = 421 };
56 
57  Filter trackCuts = (aod::track::pt > 0.15f && aod::track::eta > -0.9f && aod::track::eta < 0.9f);
58  Filter seltrack = (aod::hf_selcandidate_d0::isSelD0 >= d_selectionFlagD0 || aod::hf_selcandidate_d0::isSelD0bar >= d_selectionFlagD0bar);
59 
60  void process(aod::Collision const& collision,
61  soa::Filtered<aod::Tracks> const& tracks,
62  soa::Filtered<soa::Join<aod::HfCandProng2, aod::HFSelD0Candidate>> const& candidates)
63  {
64  std::cout << "Per Event" << std::endl;
65  // TODO: retrieve pion mass from somewhere
66  bool isHFJet;
67 
68  //this loop should be made more efficient
69  for (auto& candidate : candidates) {
70  jets.clear();
71  inputParticles.clear();
72  for (auto& track : tracks) {
73  auto energy = std::sqrt(track.p() * track.p() + JetFinder::mPion * JetFinder::mPion);
74  if (candidate.index0().globalIndex() == track.globalIndex() || candidate.index1().globalIndex() == track.globalIndex()) { //is it global index?
75  continue;
76  }
77  inputParticles.emplace_back(track.px(), track.py(), track.pz(), energy);
78  inputParticles.back().set_user_index(track.globalIndex());
79  }
80  inputParticles.emplace_back(candidate.px(), candidate.py(), candidate.pz(), candidate.e(RecoDecay::getMassPDG(pdgD0)));
81  inputParticles.back().set_user_index(1);
82 
83  fastjet::ClusterSequenceArea clusterSeq(jetFinder.findJets(inputParticles, jets));
84 
85  for (const auto& jet : jets) {
86  isHFJet = false;
87  for (const auto& constituent : jet.constituents()) {
88  if (constituent.user_index() == 1 && (candidate.isSelD0() == 1 || candidate.isSelD0bar() == 1)) {
89  isHFJet = true;
90  break;
91  }
92  }
93  if (isHFJet) {
94  jetsTable(collision, jet.eta(), jet.phi(), jet.pt(),
95  jet.area(), jet.E(), jet.m());
96  for (const auto& constituent : jet.constituents()) {
97  constituents(jetsTable.lastIndex(), constituent.user_index());
98  }
99  hJetPt->Fill(jet.pt());
100  std::cout << "Filling" << std::endl;
101  hD0Pt->Fill(candidate.pt());
102  break;
103  }
104  }
105  }
106  }
107 };
