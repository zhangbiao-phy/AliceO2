50  // Temporary function to check if track passes selection criteria. To be replaced by framework filters.
51  template <typename T>
52  bool IsSelected(T const& track)
53  {
54  if (track.trackType() == mTrackType &&
55  track.pt() >= mMinPt && track.pt() <= mMaxPt &&
56  track.eta() >= mMinEta && track.eta() <= mMaxEta &&
57  track.tpcNClsFound() >= mMinNClustersTPC &&
58  track.tpcNClsCrossedRows() >= mMinNCrossedRowsTPC &&
59  track.tpcCrossedRowsOverFindableCls() >= mMinNCrossedRowsOverFindableClustersTPC &&
60  (track.itsNCls() >= mMinNClustersITS) &&
61  (track.itsChi2NCl() <= mMaxChi2PerClusterITS) &&
62  (track.tpcChi2NCl() <= mMaxChi2PerClusterTPC) &&
63  ((mRequireITSRefit) ? (track.flags() & o2::aod::track::ITSrefit) : true) &&
64  ((mRequireTPCRefit) ? (track.flags() & o2::aod::track::TPCrefit) : true) &&
65  ((mRequireGoldenChi2) ? (track.flags() & o2::aod::track::GoldenChi2) : true) &&
66  FulfillsITSHitRequirements(track.itsClusterMap()) &&
67  abs(track.dcaXY()) <= ((mMaxDcaXYPtDep) ? mMaxDcaXYPtDep(track.pt()) : mMaxDcaXY) &&
68  abs(track.dcaZ()) <= mMaxDcaZ) {
69  return true;
70  } else {
71  return false;
72  }
73  }
74 
75  // Temporary function to check if track passes a given selection criteria. To be replaced by framework filters.
76  template <typename T>
77  bool IsSelected(T const& track, const TrackCuts& cut)
78  {
79  switch (cut) {
80  case TrackCuts::kTrackType:
81  return track.trackType() == mTrackType;
82  case TrackCuts::kPtRange:
83  return track.pt() >= mMinPt && track.pt() <= mMaxPt;
84  case TrackCuts::kEtaRange:
85  return track.eta() >= mMinEta && track.eta() <= mMaxEta;
86  case TrackCuts::kTPCNCls:
87  return track.tpcNClsFound() >= mMinNClustersTPC;
88  case TrackCuts::kTPCCrossedRows:
89  return track.tpcNClsCrossedRows() >= mMinNCrossedRowsTPC;
90  case TrackCuts::kTPCCrossedRowsOverNCls:
91  return track.tpcCrossedRowsOverFindableCls() >= mMinNCrossedRowsOverFindableClustersTPC;
92  case TrackCuts::kTPCChi2NDF:
93  return track.itsNCls() >= mMinNClustersITS;
94  case TrackCuts::kTPCRefit:
95  return track.itsChi2NCl() <= mMaxChi2PerClusterITS;
96  case TrackCuts::kITSNCls:
97  return track.tpcChi2NCl() <= mMaxChi2PerClusterTPC;
98  case TrackCuts::kITSChi2NDF:
99  return (mRequireITSRefit) ? (track.flags() & o2::aod::track::ITSrefit) : true;
100  case TrackCuts::kITSRefit:
101  return (mRequireTPCRefit) ? (track.flags() & o2::aod::track::TPCrefit) : true;
102  case TrackCuts::kITSHits:
103  return (mRequireGoldenChi2) ? (track.flags() & o2::aod::track::GoldenChi2) : true;
104  case TrackCuts::kGoldenChi2:
105  return FulfillsITSHitRequirements(track.itsClusterMap());
106  case TrackCuts::kDCAxy:
107  return abs(track.dcaXY()) <= ((mMaxDcaXYPtDep) ? mMaxDcaXYPtDep(track.pt()) : mMaxDcaXY);
108  case TrackCuts::kDCAz:
109  return abs(track.dcaZ()) <= mMaxDcaZ;
110  default:
111  return false;
112  }
113  }
