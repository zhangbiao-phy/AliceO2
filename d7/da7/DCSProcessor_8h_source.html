199 template <typename T>
200 int DCSProcessor::processArrayType(const std::vector<DPID>& array,
201  DeliveryType type, const std::unordered_map<DPID, DPVAL>& map,
202  std::vector<uint64_t>& latestTimeStamp,
203  std::unordered_map<DPID, std::deque<T>>& destmap)
204 {
205 
206  // processing the array of type T
207 
208  int found = 0;
209  //#ifdef WITH_OPENMP
210  //omp_set_num_threads(mNThreads);
211  //#pragma omp parallel for schedule(dynamic)
212  //#endif
213  for (size_t i = 0; i != array.size(); ++i) {
214  auto it = findAndCheckPid(array[i], type, map);
215  if (it == map.end()) {
216  if (!mIsDelta) {
217  LOG(ERROR) << "Element " << array[i] << " not found " << std::endl;
218  }
219  continue;
220  }
221  found++;
222  std::pair<DPID, DPVAL> pairIt = *it;
223  checkFlagsAndFill(pairIt, latestTimeStamp[i], destmap);
224  if (type == RAW_CHAR) {
225  processCharDP(array[i]);
226  } else if (type == RAW_INT) {
227  processIntDP(array[i]);
228  } else if (type == RAW_DOUBLE) {
229  processDoubleDP(array[i]);
230  } else if (type == RAW_UINT) {
231  processUIntDP(array[i]);
232  } else if (type == RAW_BOOL) {
233  processBoolDP(array[i]);
234  } else if (type == RAW_STRING) {
235  processStringDP(array[i]);
236  } else if (type == RAW_TIME) {
237  processTimeDP(array[i]);
238  } else if (type == RAW_BINARY) {
239  processBinaryDP(array[i]);
240  }
241  // todo: better to move the "found++" after the process, in case it fails?
242  }
243  return found;
244 }

248 template <typename T>
249 void DCSProcessor::checkFlagsAndFill(const std::pair<DPID, DPVAL>& dpcom, uint64_t& latestTimeStamp,
250  std::unordered_map<DPID, T>& destmap)
251 {
252 
253  // check the flags for the upcoming data, and if ok, fill the accumulator
254 
255  auto& dpid = dpcom.first;
256  auto& val = dpcom.second;
257  auto flags = val.get_flags();
258  if (processFlag(flags, dpid.get_alias()) == 0) {
259  auto etime = val.get_epoch_time();
260  // fill only if new value has a timestamp different from the timestamp of the previous one
261  LOG(DEBUG) << "destmap[pid].size() = " << destmap[dpid].size();
262  if (destmap[dpid].size() == 0 || etime != latestTimeStamp) {
263  LOG(DEBUG) << "adding new value";
264  destmap[dpid].push_back(val.payload_pt1);
265  latestTimeStamp = etime;
266  }
267  }
268 }

284 template <typename T>
285 void DCSProcessor::doSimpleMovingAverage(int nelements, std::deque<T>& vect, float& avg, bool& isSMA)
286 {
287 
288  // Do simple moving average on vector of type T
289 
290  if (vect.size() <= nelements) {
291  //avg = std::accumulate(vect.begin(), vect.end(), 0.0) / vect.size();
292  avg = (avg * (vect.size() - 1) + vect.back()) / vect.size();
293  return;
294  }
295 
296  avg += (vect[vect.size() - 1] - vect[0]) / nelements;
297  vect.pop_front();
298  isSMA = true;
299 }
