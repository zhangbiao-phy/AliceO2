32 // Digit class for TRD
33 // Notes:
34 // Shared pads:
35 // the lower mcm and rob is chosen for a given shared pad.
36 // this negates the need for need alternate indexing strategies.
37 // if you are trying to go from mcm/rob/adc to pad/row and back to mcm/rob/adc ,
38 // you may not end up in the same place, you need to remember to manually check for shared pads.
39 
40 class Digit
41 {
42  public:
43  Digit() = default;
44  ~Digit() = default;
45  Digit(const int det, const int row, const int pad, const ArrayADC adc);

54  // Modifiers
55  void setROB(int rob) { mROB = rob; }
56  void setROB(int row, int pad) { mROB = FeeParam::getROBfromPad(row, pad); }
57  void setMCM(int mcm) { mMCM = mcm; }
58  void setMCM(int row, int pad) { mMCM = FeeParam::getMCMfromPad(row, pad); }
59  void setChannel(int channel) { mChannel = channel; }
60  void setDetector(int det) { mDetector = det; }
61  void setADC(ArrayADC const& adc) { mADC = adc; }
62  // Get methods
63  int getDetector() const { return mDetector; }
64  int getRow() const { return FeeParam::getPadRowFromMCM(mROB, mMCM); }
65  int getPad() const { return FeeParam::getPadColFromADC(mROB, mMCM, mChannel); }
66  int getROB() const { return mROB; }
67  int getMCM() const { return mMCM; }
68  int getChannel() const { return mChannel; }
69  bool isSharedDigit() const;
70 
71  ArrayADC const& getADC() const { return mADC; }
72  ADC_t getADCsum() const { return std::accumulate(mADC.begin(), mADC.end(), (ADC_t)0); }
73 
74  private:
75  std::uint16_t mDetector{0}; // detector, the chamber [0-539]
76  std::uint8_t mROB{0}; // read out board within chamber [0-7] [0-5] depending on C0 or C1
77  std::uint8_t mMCM{0}; // MCM chip this digit is attached [0-15]
78  std::uint8_t mChannel{0}; // channel of this chip the digit is attached to, see TDP chapter ?? TODO fill in later the figure number of ROB to MCM mapping picture
79 
80  ArrayADC mADC{}; // ADC vector (30 time-bins)
