134 inline double SimParam::TimeResponse(double time) const
135 {
136  //
137  // Applies the preamp shaper time response
138  // (We assume a signal rise time of 0.2us = fTRFlo/2.
139  //
140 
141  double rt = (time - .5 * mTRFlo) * mInvTRFwid;
142  int iBin = (int)rt;
143  double dt = rt - iBin;
144  if ((iBin >= 0) && (iBin + 1 < mTRFbin)) {
145  return mTRFsmp[iBin] + (mTRFsmp[iBin + 1] - mTRFsmp[iBin]) * dt;
146  } else {
147  return 0.0;
148  }
149 }
150 
151 inline double SimParam::CrossTalk(double time) const
152 {
153  //
154  // Applies the pad-pad capacitive cross talk
155  //
156 
157  double rt = (time - mTRFlo) * mInvTRFwid;
158  int iBin = (int)rt;
159  double dt = rt - iBin;
160  if ((iBin >= 0) && (iBin + 1 < mTRFbin)) {
161  return mCTsmp[iBin] + (mCTsmp[iBin + 1] - mCTsmp[iBin]) * dt;
162  } else {
163  return 0.0;
164  }
165 }
