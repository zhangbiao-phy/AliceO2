90 void Digitizer::process(const std::vector<Hit> hits, std::vector<Digit>& digits, o2::dataformats::MCTruthContainer<o2::MCCompLabel>& mcContainer)
91 {
92  digits.clear();
93  mDigits.clear();
94  mTrackLabels.clear();
95  mcContainer.clear();
96  //array of MCH hits for a given simulated event
97  for (auto& hit : hits) {
98  int detID = hit.GetDetectorID();
99  int ndigits = processHit(hit, detID, mEventTime);
100  MCCompLabel label(hit.GetTrackID(), mEventID, mSrcID, false);
101  for (int i = 0; i < ndigits; ++i) {
102  int digitIndex = mDigits.size() - ndigits + i;
103  mMCTruthOutputContainer.addElement(digitIndex, label);
104  } //loop over digits to generate MCdigits
105  } //loop over hits
106 
107  //generate noise-only digits
108  if (mNoise) {
109  generateNoiseDigits();
110  }
111 
112  fillOutputContainer(digits);
113  provideMC(mcContainer);
114 }

116 int Digitizer::processHit(const Hit& hit, int detID, int event_time)
117 {
118  math_utils::Point3D<float> pos(hit.GetX(), hit.GetY(), hit.GetZ());
119 
120  Response& resp = response(isStation1(detID));
121 
122  //convert energy to charge
123  auto charge = resp.etocharge(hit.GetEnergyLoss());
124 
125  //convert float ns time to BC counts
126  auto time = event_time & int(hit.GetTime() / 25.);
127 
128  //transformation from global to local
129  auto transformation = o2::mch::geo::transformationFromTGeoManager(*gGeoManager);
130  auto t = transformation(detID);
131  math_utils::Point3D<float> lpos;
132  t.MasterToLocal(pos, lpos);
133 
134  auto anodpos = resp.getAnod(lpos.X());
135  auto fracplane = resp.chargeCorr();
136  auto chargebend = fracplane * charge;
137  auto chargenon = charge / fracplane;
138 
139  //borders of charge gen.
140  auto xMin = anodpos - resp.getQspreadX() * resp.getSigmaIntegration() * 0.5;
141  auto xMax = anodpos + resp.getQspreadX() * resp.getSigmaIntegration() * 0.5;
142  auto yMin = lpos.Y() - resp.getQspreadY() * resp.getSigmaIntegration() * 0.5;
143  auto yMax = lpos.Y() + resp.getQspreadY() * resp.getSigmaIntegration() * 0.5;
144 
145  //get segmentation for detector element
146  auto& seg = segmentation(detID);
147  auto localX = anodpos;
148  auto localY = lpos.Y();
149 
150  //get area for signal induction from segmentation
151  //single pad as check
152  int padidbendcent = 0;
153  int padidnoncent = 0;
154  int ndigits = 0;
155 
156  bool padexists = seg.findPadPairByPosition(localX, localY, padidbendcent, padidnoncent);
157  if (!padexists) {
158  LOG(WARNING) << "Did not find _any_ pad for localX,Y=" << localX << "," << localY << ", for detID: " << detID;
159  return 0;
160  }
161 
162  seg.forEachPadInArea(xMin, yMin, xMax, yMax, [&resp, &digits = this->mDigits, chargebend, chargenon, localX, localY, &seg, &ndigits, detID, time](int padid) {
163  auto dx = seg.padSizeX(padid) * 0.5;
164  auto dy = seg.padSizeY(padid) * 0.5;
165  auto xmin = (localX - seg.padPositionX(padid)) - dx;
166  auto xmax = xmin + 2 * dx;
167  auto ymin = (localY - seg.padPositionY(padid)) - dy;
168  auto ymax = ymin + 2 * dy;
169  auto q = resp.chargePadfraction(xmin, xmax, ymin, ymax);
170  if (resp.aboveThreshold(q)) {
171  if (seg.isBendingPad(padid)) {
172  q *= chargebend;
173  } else {
174  q *= chargenon;
175  }
176  auto signal = (unsigned long)q * resp.getInverseChargeThreshold();
177  if (signal > 0) {
178  Digit::Time dtime;
179  dtime.sampaTime = static_cast<uint16_t>(time) & 0x3FF;
180  digits.emplace_back(detID, padid, signal, dtime);
181  ++ndigits;
182  }
183  }
184  });
185  return ndigits;
186 }

188 void Digitizer::generateNoiseDigits()
189 {
190 
191  o2::mch::mapping::forEachDetectionElement([&digits = this->mDigits, &normProbNoise = this->mNormProbNoise,
192  &eventTime = this->mEventTime, &eventID = this->mEventID,
193  &srcID = this->mSrcID, &mcTruthOutputContainer = this->mMCTruthOutputContainer](int detID) {
194  auto& seg = segmentation(detID);
195  auto nPads = seg.nofPads();
196  auto nNoisyPadsAv = (float)nPads * normProbNoise;
197  int nNoisyPads = TMath::Nint(gRandom->Gaus(nNoisyPadsAv, TMath::Sqrt(nNoisyPadsAv)));
198  for (int i = 0; i < nNoisyPads; i++) {
199  int padid = gRandom->Integer(nNoisyPads + 1);
200  Digit::Time dtime;
201  dtime.sampaTime = static_cast<uint16_t>(eventTime) & 0x3FF;
202  digits.emplace_back(detID, padid, 0.6, dtime);
203  //just to roun adbove threshold when added
204  MCCompLabel label(-1, eventID, srcID, true);
205  mcTruthOutputContainer.addElement(digits.size() - 1, label);
206  }
207  });
208  //not clear how to normalise to time:
209  //assume that only "one" event equivalent,
210  //otherwise the probability will strongly depend on read-out-frame time length
211 }

213 void Digitizer::mergeDigits()
214 {
215  std::vector<int> indices(mDigits.size());
216  std::iota(begin(indices), end(indices), 0);
217  std::sort(indices.begin(), indices.end(), [&digits = this->mDigits, this](int a, int b) {
218  return (getGlobalDigit(digits[a].getDetID(), digits[a].getPadID()) < getGlobalDigit(digits[b].getDetID(), digits[b].getPadID()));
219  });
220 
221  auto sortedDigits = [digits = this->mDigits, &indices](int i) {
222  return digits[indices[i]];
223  };
224 
225  auto sortedLabels = [labels = this->mTrackLabels, &indices](int i) {
226  return labels[indices[i]];
227  };
228 
229  auto sizedigits = mDigits.size();
230 
231  mDigits.clear();
232  mDigits.reserve(sizedigits);
233  mMCTruthOutputContainer.clear();
234 
235  int count = 0;
236 
237  int i = 0;
238  while (i < indices.size()) {
239  int j = i + 1;
240  while (j < indices.size() && (getGlobalDigit(sortedDigits(i).getDetID(), sortedDigits(i).getPadID())) == (getGlobalDigit(sortedDigits(j).getDetID(), sortedDigits(j).getPadID())) && (std::fabs(sortedDigits(i).getTime().sampaTime - sortedDigits(j).getTime().sampaTime) < mDeltat)) {
241  j++;
242  }
243  unsigned long adc{0};
244  float padc{0};
245  Response& resp = response(isStation1(sortedDigits(i).getDetID()));
246 
247  for (int k = i; k < j; k++) {
248  adc += sortedDigits(k).getADC();
249  if (k == i) {
250  MCCompLabel label(sortedLabels(i).getTrackID(), sortedLabels(i).getEventID(), sortedLabels(i).getSourceID(), false);
251  mMCTruthOutputContainer.addElement(count, label);
252  } else {
253  if ((sortedLabels(k).getTrackID() != sortedLabels(k - 1).getTrackID()) || (sortedLabels(k).getSourceID() != sortedLabels(k - 1).getSourceID())) {
254  MCCompLabel label(sortedLabels(k).getTrackID(), sortedLabels(k).getEventID(), sortedLabels(k).getSourceID(), false);
255  mMCTruthOutputContainer.addElement(count, label);
256  }
257  }
258  }
259  padc = adc * resp.getChargeThreshold();
260  adc = TMath::Nint(padc);
261  adc = resp.response(adc);
262  mDigits.emplace_back(sortedDigits(i).getDetID(), sortedDigits(i).getPadID(), adc, sortedDigits(i).getTime());
263  i = j;
264  ++count;
265  }
266  mDigits.resize(mDigits.size());
267 }

269 void Digitizer::mergeDigits(std::vector<Digit>& digits, o2::dataformats::MCTruthContainer<o2::MCCompLabel>& mcContainer)
270 {
271  mDigits.clear();
272  mDigits.reserve(digits.size());
273  mTrackLabels.clear();
274 
275  for (int index = 0; index < digits.size(); ++index) {
276  auto digit = digits.at(index);
277  mDigits.emplace_back(digit.getDetID(), digit.getPadID(), digit.getADC(), digit.getTime());
278  }
279  for (int index = 0; index < mcContainer.getNElements(); ++index) {
280  auto label = mcContainer.getElement(index);
281  mTrackLabels.emplace_back(label.getTrackID(), label.getEventID(), label.getSourceID(), false);
282  }
283 
284  mergeDigits();
285  fillOutputContainer(digits);
286  provideMC(mcContainer);
287 }

302  digits.emplace_back(*iter);
