224 template <typename DataT, int YdimT>
225 class Spline2DSpec<DataT, YdimT, 0>
226  : public Spline2DContainer<DataT>
227 {
228  typedef Spline2DContainer<DataT> TBase;
229 
230  public:
231  typedef typename TBase::SafetyLevel SafetyLevel;
232  typedef typename TBase::Knot Knot;
233 
235 
237  GPUd() void interpolate(DataT x1, DataT x2, GPUgeneric() DataT S[/*mYdim*/]) const
238  {
239  interpolateU<SafetyLevel::kSafe>(mYdim, mParameters, mGridX1.convXtoU(x1), mGridX2.convXtoU(x2), S);
240  }
241 
243  template <SafetyLevel SafeT = SafetyLevel::kSafe>
244  GPUd() void interpolateU(int inpYdim, GPUgeneric() const DataT Parameters[],
245  DataT u1, DataT u2, GPUgeneric() DataT S[/*inpYdim*/]) const
246  {
247 
248  const auto nYdimTmp = SplineUtil::getNdim<YdimT>(inpYdim);
249  const int nYdim = nYdimTmp.get();
250 
251  const auto maxYdim = SplineUtil::getMaxNdim<YdimT>(inpYdim);
252  const int maxYdim4 = 4 * maxYdim.get();
253 
254  const auto nYdim2 = nYdim * 2;
255  const auto nYdim4 = nYdim * 4;
256 
257  const float& u = u1;
258  const float& v = u2;
259  int nu = mGridX1.getNumberOfKnots();
260  int iu = mGridX1.template getLeftKnotIndexForU<SafeT>(u);
261  int iv = mGridX2.template getLeftKnotIndexForU<SafeT>(v);
262 
263  const typename TBase::Knot& knotU = mGridX1.template getKnot<SafetyLevel::kNotSafe>(iu);
264  const typename TBase::Knot& knotV = mGridX2.template getKnot<SafetyLevel::kNotSafe>(iv);
265 
266  const DataT* par00 = Parameters + (nu * iv + iu) * nYdim4; // values { {Y1,Y2,Y3}, {Y1,Y2,Y3}'v, {Y1,Y2,Y3}'u, {Y1,Y2,Y3}''vu } at {u0, v0}
267  const DataT* par10 = par00 + nYdim4; // values { ... } at {u1, v0}
268  const DataT* par01 = par00 + nYdim4 * nu; // values { ... } at {u0, v1}
269  const DataT* par11 = par01 + nYdim4; // values { ... } at {u1, v1}
270 
271  DataT Su0[maxYdim4]; // values { {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v0), {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v1) }, at u0
272  DataT Du0[maxYdim4]; // derivatives {}'_u at u0
273  DataT Su1[maxYdim4]; // values { {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v0), {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v1) }, at u1
274  DataT Du1[maxYdim4]; // derivatives {}'_u at u1
275 
276  for (int i = 0; i < nYdim2; i++) {
277  Su0[i] = par00[i];
278  Su0[nYdim2 + i] = par01[i];
279 
280  Du0[i] = par00[nYdim2 + i];
281  Du0[nYdim2 + i] = par01[nYdim2 + i];
282 
283  Su1[i] = par10[i];
284  Su1[nYdim2 + i] = par11[i];
285 
286  Du1[i] = par10[nYdim2 + i];
287  Du1[nYdim2 + i] = par11[nYdim2 + i];
288  }
289 
290  DataT parU[maxYdim4]; // interpolated values { {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v0), {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v1) } at u
291 
292  typedef Spline1DSpec<DataT, 4 * YdimT, 0> TGridX1;
293  const TGridX1& gridX1 = reinterpret_cast<const TGridX1&>(mGridX1);
294 
295  gridX1.interpolateU(nYdim4, knotU, Su0, Du0, Su1, Du1, u, parU);
296 
297  const DataT* Sv0 = parU + 0;
298  const DataT* Dv0 = parU + nYdim;
299  const DataT* Sv1 = parU + nYdim2;
300  const DataT* Dv1 = parU + nYdim2 + nYdim;
301 
302  typedef Spline1DSpec<DataT, YdimT, 0> TGridX2;
303  const TGridX2& gridX2 = reinterpret_cast<const TGridX2&>(mGridX2);
304  gridX2.interpolateU(nYdim, knotV, Sv0, Dv0, Sv1, Dv1, v, S);
305  }
306 
307  protected:
308  using TBase::mGridX1;
309  using TBase::mGridX2;
310  using TBase::mParameters;
311  using TBase::mYdim;
312  using TBase::TBase; // inherit constructors and hide them
313 };
