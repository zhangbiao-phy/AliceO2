28 struct TimestampTask {

30  Service<o2::ccdb::BasicCCDBManager> ccdb;

32  std::map<int, int>* mapStartOrbit = nullptr;
33  std::pair<int, long> lastCall;
34  std::map<int, long> mapRunToTimestamp;

36  // Configurables
37  Configurable<bool> verbose{"verbose", false, "verbose mode"};
38  Configurable<std::string> rct_path{"rct-path", "RCT/RunInformation/", "path to the ccdb RCT objects for the SOR timestamps"};
39  Configurable<std::string> start_orbit_path{"start-orbit-path", "Trigger/StartOrbit", "path to the ccdb SOR orbit objects"};
40  Configurable<std::string> url{"ccdb-url", "http://ccdb-test.cern.ch:8080", "URL of the CCDB database"};
41  Configurable<bool> isMC{"isMC", 0, "0 - data, 1 - MC"};

43  void init(o2::framework::InitContext&)
44  {
45  LOGF(info, "Initializing TimestampTask");
46  ccdb->setURL(url.value); // Setting URL of CCDB manager from configuration
47  LOGF(debug, "Getting SOR orbit map from CCDB url '%s' path '%s'", url.value, start_orbit_path.value);
48  mapStartOrbit = ccdb->get<std::map<int, int>>(start_orbit_path.value);
49  if (!mapStartOrbit) {
50  LOGF(fatal, "Cannot find map of SOR orbits in CCDB in path %s", start_orbit_path.value.data());
51  }
52  ccdb_api.init(url.value);
53  if (!ccdb_api.isHostReachable()) {
54  LOGF(fatal, "CCDB host %s is not reacheable, cannot go forward", url.value.data());
55  }
56  }
57 
58  void process(aod::BC const& bc)
59  {
60  long ts = 0;
61  if (bc.runNumber() == lastCall.first) { // The run number coincides to the last run processed
62  LOGF(debug, "Getting timestamp from last call");
63  ts = lastCall.second;
64  } else if (mapRunToTimestamp.count(bc.runNumber())) { // The run number was already requested before: getting it from cache!
65  LOGF(debug, "Getting timestamp from cache");
66  ts = mapRunToTimestamp[bc.runNumber()];
67  } else { // The run was not requested before: need to acccess CCDB!
68  LOGF(debug, "Getting timestamp from CCDB");
69  std::map<std::string, std::string> metadata, headers;
70  const std::string run_path = Form("%s/%i", rct_path.value.data(), bc.runNumber());
71  headers = ccdb_api.retrieveHeaders(run_path, metadata, -1);
72  if (headers.count("SOR") == 0) {
73  LOGF(fatal, "Cannot find run-number to timestamp in path '%s'.", run_path.data());
74  }
75  ts = atol(headers["SOR"].c_str()); // timestamp of the SOR in ms
76 
77  // Adding the timestamp to the cache map
78  std::pair<std::map<int, long>::iterator, bool> check;
79  check = mapRunToTimestamp.insert(std::pair<int, long>(bc.runNumber(), ts));
80  if (!check.second) {
81  LOGF(fatal, "Run number %i already existed with a timestamp of %llu", bc.runNumber(), check.first->second);
82  }
83  LOGF(info, "Add new run %i with timestamp %llu to cache", bc.runNumber(), ts);
84  }
85 
86  // Setting latest run information
87  lastCall = std::make_pair(bc.runNumber(), ts);
88 
89  if (verbose.value) {
90  LOGF(info, "Run-number to timestamp found! %i %llu ms", bc.runNumber(), ts);
91  }
92  const uint16_t initialBC = 0; // exact bc number not relevant due to ms precision of timestamps
93  if (!mapStartOrbit->count(bc.runNumber())) {
94  LOGF(fatal, "Cannot find run %i in mapStartOrbit map", bc.runNumber());
95  }
96  const uint32_t initialOrbit = mapStartOrbit->at(bc.runNumber());
97  const uint16_t currentBC = isMC ? initialBC : bc.globalBC() % o2::constants::lhc::LHCMaxBunches;
98  const uint32_t currentOrbit = isMC ? initialOrbit : bc.globalBC() / o2::constants::lhc::LHCMaxBunches;
99  const InteractionRecord current(currentBC, currentOrbit);
100  const InteractionRecord initial(initialBC, initialOrbit);
101  ts += (current - initial).bc2ns() / 1000000;
102  ts_table(ts);
103  }
104 };
