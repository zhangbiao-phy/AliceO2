43  //decoding binary data into data blocks
44  template <class DataBlockType>
45  size_t decodeBlocks(const gsl::span<const uint8_t> binaryPayload, std::vector<DataBlockType>& vecDataBlocks)
46  {
47  size_t srcPos = 0;
48  while (srcPos < binaryPayload.size()) { //checking element
49  DataBlockType dataBlock{};
50  dataBlock.decodeBlock(binaryPayload, srcPos);
51  srcPos += dataBlock.mSize;
52  if (dataBlock.isCorrect()) {
53  vecDataBlocks.push_back(dataBlock); //change to in-place construction? TODO
54  } else {
55  LOG(WARNING) << "INCORRECT DATA BLOCK! Byte position: " << srcPos - dataBlock.mSize << " | " << binaryPayload.size() << " | " << dataBlock.mSize;
56  dataBlock.print();
57  }
58  }
59  return srcPos;
60  }

62  //processing data blocks into digits
63  template <class DataBlockType>
64  void processBinaryData(gsl::span<const uint8_t> payload, int linkID)
65  {
66  std::vector<DataBlockType> vecDataBlocks;
67  auto srcPos = decodeBlocks(payload, vecDataBlocks);
68 
69  for (auto& dataBlock : vecDataBlocks) {

71  auto [digitIter, isNew] = mMapDigits.try_emplace(intRec, intRec);
72  digitIter->second.template process<DataBlockType>(dataBlock, linkID);
73  }
74  }
