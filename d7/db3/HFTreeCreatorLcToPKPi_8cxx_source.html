176  Produces<o2::aod::HfCandProng3Full> rowCandidateFull;
177  Produces<o2::aod::HfCandProng3FullEvents> rowCandidateFullEvents;

182  void process(aod::Collisions const& collisions,
183  aod::McCollisions const& mccollisions,
184  soa::Join<aod::HfCandProng3, aod::HfCandProng3MCRec, aod::HFSelLcCandidate> const& candidates,
185  soa::Join<aod::McParticles, aod::HfCandProng3MCGen> const& particles,
186  aod::BigTracksPID const& tracks)
187  {
188 
189  // Filling event properties
190  rowCandidateFullEvents.reserve(collisions.size());
191  for (auto& collision : collisions) {
192  rowCandidateFullEvents(
193  collision.bcId(),
194  collision.numContrib(),
195  collision.posX(),
196  collision.posY(),
197  collision.posZ(),
198  0,
199  1);
200  }
201 
202  // Filling candidate properties
203  rowCandidateFull.reserve(candidates.size());
204  for (auto& candidate : candidates) {
205  auto fillTable = [&](int CandFlag,
206  int FunctionSelection,
207  float FunctionInvMass,
208  float FunctionCt,
209  float FunctionY,
210  float FunctionE) {
211  if (FunctionSelection >= 1) {
212  rowCandidateFull(
213  candidate.index0_as<aod::BigTracksPID>().collision().bcId(),
214  candidate.index0_as<aod::BigTracksPID>().collision().numContrib(),
215  candidate.posX(),
216  candidate.posY(),
217  candidate.posZ(),
218  candidate.xSecondaryVertex(),
219  candidate.ySecondaryVertex(),
220  candidate.zSecondaryVertex(),
221  candidate.errorDecayLength(),
222  candidate.errorDecayLengthXY(),
223  candidate.chi2PCA(),
224  candidate.rSecondaryVertex(),
225  candidate.decayLength(),
226  candidate.decayLengthXY(),
227  candidate.decayLengthNormalised(),
228  candidate.decayLengthXYNormalised(),
229  candidate.impactParameterNormalised0(),
230  candidate.ptProng0(),
231  RecoDecay::P(candidate.pxProng0(), candidate.pyProng0(), candidate.pzProng0()),
232  candidate.impactParameterNormalised1(),
233  candidate.ptProng1(),
234  RecoDecay::P(candidate.pxProng1(), candidate.pyProng1(), candidate.pzProng1()),
235  candidate.impactParameterNormalised2(),
236  candidate.ptProng2(),

242  candidate.pyProng1(),
243  candidate.pzProng1(),
244  candidate.pxProng2(),
245  candidate.pyProng2(),
246  candidate.pzProng2(),
247  candidate.impactParameter0(),
248  candidate.impactParameter1(),
249  candidate.impactParameter2(),
250  candidate.errorImpactParameter0(),
251  candidate.errorImpactParameter1(),
252  candidate.errorImpactParameter2(),
253  candidate.index0_as<aod::BigTracksPID>().tpcNSigmaPi(),
254  candidate.index0_as<aod::BigTracksPID>().tpcNSigmaKa(),
255  candidate.index0_as<aod::BigTracksPID>().tpcNSigmaPr(),
256  candidate.index0_as<aod::BigTracksPID>().tofNSigmaPi(),
257  candidate.index0_as<aod::BigTracksPID>().tofNSigmaKa(),
258  candidate.index0_as<aod::BigTracksPID>().tofNSigmaPr(),
259  candidate.index1_as<aod::BigTracksPID>().tpcNSigmaPi(),
260  candidate.index1_as<aod::BigTracksPID>().tpcNSigmaKa(),
261  candidate.index1_as<aod::BigTracksPID>().tpcNSigmaPr(),
262  candidate.index1_as<aod::BigTracksPID>().tofNSigmaPi(),
263  candidate.index1_as<aod::BigTracksPID>().tofNSigmaKa(),
264  candidate.index1_as<aod::BigTracksPID>().tofNSigmaPr(),
265  candidate.index2_as<aod::BigTracksPID>().tpcNSigmaPi(),
266  candidate.index2_as<aod::BigTracksPID>().tpcNSigmaKa(),
267  candidate.index2_as<aod::BigTracksPID>().tpcNSigmaPr(),
268  candidate.index2_as<aod::BigTracksPID>().tofNSigmaPi(),
269  candidate.index2_as<aod::BigTracksPID>().tofNSigmaKa(),
270  candidate.index2_as<aod::BigTracksPID>().tofNSigmaPr(),
271  1 << CandFlag,
272  FunctionInvMass,
273  candidate.pt(),
274  candidate.p(),
275  candidate.cpa(),
276  candidate.cpaXY(),
277  FunctionCt,
278  candidate.eta(),
279  candidate.phi(),
280  FunctionY,
281  FunctionE,
282  candidate.flagMCMatchRec());
283  }
284  };
285 
286  fillTable(0, candidate.isSelLcpKpi(), InvMassLcpKpi(candidate), CtLc(candidate), YLc(candidate), ELc(candidate));
287  fillTable(1, candidate.isSelLcpiKp(), InvMassLcpiKp(candidate), CtLc(candidate), YLc(candidate), ELc(candidate));
288  }
289 
290  // Filling particle properties
291  rowCandidateFullParticles.reserve(particles.size());
292  for (auto& particle : particles) {
293  if (std::abs(particle.flagMCMatchGen()) == 1 << LcToPKPi) {
294  rowCandidateFullParticles(
295  particle.mcCollision().bcId(),
296  particle.pt(),
297  particle.eta(),
298  particle.phi(),
299  RecoDecay::Y(array{particle.px(), particle.py(), particle.pz()}, RecoDecay::getMassPDG(particle.pdgCode())),
300  particle.flagMCMatchGen());
301  }
302  }
303  }
304 };
