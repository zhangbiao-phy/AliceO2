49 struct JetSkimmingTask1 {
50  Produces<o2::aod::JetSkim> skim;
51 
52  Filter trackCuts = aod::track::pt > 0.15f;
53  float mPionSquared = 0.139 * 0.139;
54 
55  void process(aod::Collision const& collision,
56  soa::Filtered<aod::Tracks> const& tracks)
57  {
58  for (auto& track : tracks) {
59  float energy = std::sqrt(track.p() * track.p() + mPionSquared);
60  skim(collision, track.pt(), track.eta(), track.phi(), energy);
61  }
62  }
63 };
