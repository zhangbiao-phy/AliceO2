31 int TOFFEElightReader::parseFEElightConfig()
32 {
33  // parse FEElight config
34  // loops over all FEE channels, checks whether they are enabled
35  // and sets channel enabled
36 
37  mFEElightInfo.resetAll();
38 
39  int nEnabled = 0, index;
40  TOFFEEchannelConfig* channelConfig = nullptr;
41  for (int crateId = 0; crateId < Geo::kNCrate; crateId++) {
42  for (int trmId = 0; trmId < Geo::kNTRM; trmId++) {
43  for (int chainId = 0; chainId < Geo::kNChain; chainId++) {
44  for (int tdcId = 0; tdcId < Geo::kNTdc; tdcId++) {
45  for (int channelId = 0; channelId < Geo::kNCh; channelId++) {
46  channelConfig = mFEElightConfig.getChannelConfig(crateId, trmId, chainId, tdcId, channelId);
47  LOG(INFO) << "Processing electronic channel with indices: crate = " << crateId << ", trm = " << trmId << ", chain = "
48  << chainId << ", tdc = " << tdcId << ", tdcChannel = " << channelId << " -> " << channelConfig;
49  if (channelConfig) {
50  if (!channelConfig->isEnabled()) {
51  continue;
52  }
53  // get index DO from crate, trm, chain, tdc, tdcchannel
54  index = Geo::getCHFromECH(Geo::getECHFromIndexes(crateId, trmId, chainId, tdcId, channelId));
55  if (index == -1) {
56  continue;
57  }
58  nEnabled++;
59  mFEElightInfo.mChannelEnabled[index] = channelConfig->isEnabled();
60  mFEElightInfo.mMatchingWindow[index] = channelConfig->mMatchingWindow;
61  mFEElightInfo.mLatencyWindow[index] = channelConfig->mLatencyWindow;
62  }
63  }
64  }
65  }
66  }
67  }
68 
69  TOFFEEtriggerConfig* triggerConfig = nullptr;
70  for (Int_t iddl = 0; iddl < TOFFEElightConfig::NTRIGGERMAPS; iddl++) {
71  triggerConfig = mFEElightConfig.getTriggerConfig(iddl);
72  LOG(INFO) << "Processing trigger config " << iddl << ": " << triggerConfig;
73  if (triggerConfig) {
74  mFEElightInfo.mTriggerMask[iddl] = triggerConfig->mStatusMap;
75  }
76  }
77 
78  return nEnabled;
79 }
