28 {
29  std::vector<BCData> bcdata;
30  std::vector<ChannelData> chandata;
31  std::vector<PedestalData> pedsdata;
32  // RS: don't understand why, but this library is not loaded automatically, although the dependencies are clearly
33  // indicated. What it more weird is that for similar tests of other detectors the library is loaded!
34  // Absence of the library leads to complains about the StreamerInfo and eventually segm.faul when appending the
35  // CTH to the tree. For this reason I am loading it here manually
36  // gSystem->Load("libO2DetectorsCommonDataFormats");
37  TStopwatch sw;
38  sw.Start();
39  o2::InteractionRecord ir(0, 0);
40  std::array<float, NTimeBinsPerBC> chanVals;
41 
42  // BCData and ChannelData
43  for (int irof = 0; irof < 1000; irof++) {
44  ir += 1 + gRandom->Integer(100); // randomly increaing BC
45 
46  uint32_t channPatt = 0, triggers = 0;
47  int8_t ich = -1;
48  int firstChEntry = chandata.size();
49  while ((ich += 1 + gRandom->Poisson(2.)) < NDigiChannels) {
50  channPatt |= 0x1 << ich;
51  for (int i = 0; i < NTimeBinsPerBC; i++) {
52  chanVals[i] = gRandom->Integer(0xffff);
53  }
54  if (gRandom->Rndm() > 0.4) {
55  triggers |= 0x1 << ich;
56  }
57  chandata.emplace_back(ich, chanVals);
58  }
59  auto& bcd = bcdata.emplace_back(firstChEntry, chandata.size() - firstChEntry, ir, channPatt, triggers, gRandom->Integer(0xff));
60  for (int im = 0; im < NModules; im++) {
61  bcd.moduleTriggers[im] = gRandom->Rndm() > 0.7 ? gRandom->Integer((0x1 << 10) - 1) : 0;
62  }
63  }
64 
65  // PedestalData
66  const auto &irFirst = bcdata.front().ir, irLast = bcdata.back().ir;
67  o2::InteractionRecord irPed(o2::constants::lhc::LHCMaxBunches - 1, irFirst.orbit);
68  int norbits = irLast.orbit - irFirst.orbit + 1;
69  pedsdata.resize(norbits);
70  for (int i = 0; i < norbits; i++) {
71  pedsdata[i].ir = irPed;
72  for (int ic = 0; ic < NChannels; ic++) {
73  pedsdata[i].data[ic] = gRandom->Integer(0xffff);
74  }
75  irPed.orbit++;
76  }
77 
78  sw.Start();
79  std::vector<o2::ctf::BufferType> vec;
80  {
81  CTFCoder coder;
82  coder.encode(vec, bcdata, chandata, pedsdata); // compress
83  }
84  sw.Stop();
85  LOG(INFO) << "Compressed in " << sw.CpuTime() << " s";
86 
87  // writing
88  {
89  sw.Start();
90  auto* ctfImage = o2::zdc::CTF::get(vec.data());
91  TFile flOut("test_ctf_zdc.root", "recreate");
92  TTree ctfTree(std::string(o2::base::NameConf::CTFTREENAME).c_str(), "O2 CTF tree");

98  }
99

108  sw.Stop();
109  LOG(INFO) << "Read back from tree in " << sw.CpuTime() << " s";
110  }
111

124

126  BOOST_CHECK(bcdataD.size() == bcdata.size());

131  bcdata[i].channels == bcdataD[i].channels &&

134
