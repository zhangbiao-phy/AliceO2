33 // convert all clusters from local to global reference frames
34 void local2global(geo::TransformationCreator transformation,
35  gsl::span<const ClusterStruct> localClusters,
36  std::vector<ClusterStruct, o2::pmr::polymorphic_allocator<ClusterStruct>>& globalClusters)
37 {
38  int i{0};
39  globalClusters.insert(globalClusters.end(), localClusters.begin(), localClusters.end());
40  for (auto& c : localClusters) {
41  auto deId = c.getDEId();
42  o2::math_utils::Point3D<float> local{c.x, c.y, c.z};
43  auto t = transformation(deId);
44  auto global = t(local);
45  auto& gcluster = globalClusters[i];
46  gcluster.x = global.x();
47  gcluster.y = global.y();
48  gcluster.z = global.z();
49  i++;
50  }
51 }

64  }

66  // read the clusters (assumed to be in local reference frame) and
67  // tranform them into master reference frame.
68  void run(ProcessingContext& pc)
69  {
70  // get the input clusters
71  auto localClusters = pc.inputs().get<gsl::span<ClusterStruct>>("clusters");
72 
73  // create the output message
74  auto& globalClusters = pc.outputs().make<std::vector<ClusterStruct>>(OutputRef{"globalclusters"});
75 
76  local2global(transformation, localClusters, globalClusters);
77  }
78 
79  public:
80  o2::mch::geo::TransformationCreator transformation;
