36  // Coordinates Conversion Functions
37  static inline uint32_t Abs(int ch, int pc, int x, int y) { return ch << 20 | pc << 16 | x << 8 | y; }
38  static inline int DDL2C(int ddl) { return ddl >> 1; } //ddl -> chamber
39  static inline int A2C(uint32_t pad) { return (pad & 0x00F00000) >> 20; } //abs pad -> chamber
40  static inline int A2P(uint32_t pad) { return (pad & 0x000F0000) >> 16; } //abs pad -> pc
41  static inline int A2X(uint32_t pad) { return (pad & 0x0000FF00) >> 8; } //abs pad -> pad X
42  static inline int A2Y(uint32_t pad) { return (pad & 0x000000FF); } //abs pad -> pad Y
43  static inline uint32_t Photo2Pad(int ch, int pc, int x, int y) { return Abs(ch, pc, x, y); }
