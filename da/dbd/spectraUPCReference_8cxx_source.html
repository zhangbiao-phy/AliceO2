40  void process(soa::Join<aod::Collisions, aod::EvSels>::iterator const& col, soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra, aod::TrackSelection>> const& tracks)
41  {
42  bool checkV0 = col.bbV0A() || col.bbV0C() || col.bgV0A() || col.bgV0C();
43  if (checkV0) {
44  return;
45  }
46  bool checkFDD = col.bbFDA() || col.bbFDC() || col.bgFDA() || col.bgFDC();
47  if (checkFDD) {
48  return;
49  }
50  if (!col.alias()[kCUP9]) {
51  return;
52  }
53  if (tracks.size() != 2) {
54  return;
55  }
56  auto first = tracks.begin();
57  auto second = first + 1;
58  if (first.charge() * second.charge() >= 0) {
59  return;
60  }
61  UChar_t clustermap1 = first.itsClusterMap();
62  UChar_t clustermap2 = second.itsClusterMap();
63  bool checkClusMap = TESTBIT(clustermap1, 0) && TESTBIT(clustermap1, 1) && TESTBIT(clustermap2, 0) && TESTBIT(clustermap2, 1);
64  if (!checkClusMap) {
65  return;
66  }
67  TLorentzVector p1, p2, p;
68  p1.SetXYZM(first.px(), first.py(), first.pz(), mPion);
69  p2.SetXYZM(second.px(), second.py(), second.pz(), mPion);
70  p = p1 + p2;
71  histos.fill(HIST("fhPt"), p.Pt());
72  float signalTPC1 = first.tpcSignal();
73  float signalTPC2 = second.tpcSignal();
74  //hSignalTPC1vsSignalTPC2->Fill(signalTPC1, signalTPC2);
75  if ((p.Pt() < 0.1) && (signalTPC1 + signalTPC2 < 140.)) {
76  histos.fill(HIST("fhMass"), p.M());
77  }
78  }
79 };
