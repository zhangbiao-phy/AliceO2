32 class TOFDigitWriterSplitter : public Task
33 {
34  using OutputType = std::vector<o2::tof::Digit>;
35  using ReadoutWinType = std::vector<o2::tof::ReadoutWindowData>;
36  using PatternType = std::vector<uint8_t>;
37  using ErrorType = std::vector<uint64_t>;
38  using HeaderType = o2::tof::DigitHeader;
39 
40  std::string mBaseName;
41 
42  public:
43  TOFDigitWriterSplitter(int nTF, bool storeErr = false) : mTFthr(nTF), mStoreErrors(storeErr) {}
44 
45  void createAndOpenFileAndTree()
46  {
47  TString filename = TString::Format("%s_%06d.root", mBaseName.c_str(), mCount);
48  LOG(DEBUG) << "opening file " << filename.Data();
49  mfileOut.reset(TFile::Open(TString::Format("%s", filename.Data()), "RECREATE"));
50  mOutputTree = std::make_unique<TTree>("o2sim", "Tree with TOF digits");
51  mOutputTree->Branch("TOFHeader", &mPHeader);
52  mOutputTree->Branch("TOFDigit", &mPDigits);
53  mOutputTree->Branch("TOFReadoutWindow", &mPROW);
54  mOutputTree->Branch("TOFPatterns", &mPDia);
55  if (mStoreErrors) {
56  mOutputTree->Branch("TOFErrors", &mPErr);
57  }
58 
59  mNTF = 0;
60  }
61 
62  void init(o2::framework::InitContext& ic) final
63  {
64  mBaseName = ic.options().get<std::string>("output-base-name");
65 
66  mCount = 0;
67  createAndOpenFileAndTree();
68  }
69 
70  void run(o2::framework::ProcessingContext& pc) final
71  {
72  auto digits = pc.inputs().get<OutputType>("digits");
73  mPDigits = &digits;
74  auto header = pc.inputs().get<HeaderType>("header");
75  mPHeader = &header;
76  auto row = pc.inputs().get<ReadoutWinType>("rows");
77  mPROW = &row;
78  auto dia = pc.inputs().get<PatternType>("patterns");
79  mPDia = &dia;
80  if (mStoreErrors) {
81  auto error = pc.inputs().get<ErrorType>("errors");
82  mPErr = &error;
83 
84  mOutputTree->Fill();
85  } else {
86  mOutputTree->Fill();
87  }
88  mNTF++;
89 
90  if (mNTF >= mTFthr) {
91  sendOutput();
92  }
93  }
94 
95  void endOfStream(o2::framework::EndOfStreamContext& ec) final
96  {
97  mIsEndOfStream = true;
98  sendOutput();
99  }
100 
101  private:
102  int mCount = 0; // how many times we filled the tree
103  int mNTF = 0;
104  int mTFthr = 1;
105  bool mStoreErrors = false;
106  bool mIsEndOfStream = false;
107  OutputType mDigits;
108  const OutputType* mPDigits = &mDigits;
109  ReadoutWinType mROW;
110  const ReadoutWinType* mPROW = &mROW;
111  PatternType mDia;
112  const PatternType* mPDia = &mDia;
113  ErrorType mErr;
114  const ErrorType* mPErr = &mErr;
115  HeaderType mHeader;
116  const HeaderType* mPHeader = &mHeader;
117  std::unique_ptr<TTree> mOutputTree;
118  std::unique_ptr<TFile> mfileOut = nullptr; // file in which to write the output
119 
120  //________________________________________________________________
121  void sendOutput()
122  {
123  // This is to fill the tree.
124  // One file with an empty tree will be created at the end, because we have to have a
125  // tree opened before processing, since we do not know a priori if something else
126  // will still come. The size of this extra file is ~6.5 kB
127 
128  mfileOut->cd();
129  mOutputTree->Write();
130  mOutputTree.reset();
131  mfileOut.reset();
132  mCount++;
133  if (!mIsEndOfStream) {
134  createAndOpenFileAndTree();
135  }
136  }
137 };

143 DataProcessorSpec getTOFDigitWriterSplitterSpec(int nTF, bool storeErr = false)
144 {

148  inputs.emplace_back("rows", o2::header::gDataOriginTOF, "READOUTWINDOW");
149  inputs.emplace_back("patterns", o2::header::gDataOriginTOF, "PATTERNS");
150

152  inputs.emplace_back("errors", o2::header::gDataOriginTOF, "ERRORS");
153  }
154 
155  std::vector<OutputSpec> outputs; // empty
156 
157  return DataProcessorSpec{
158  "tof-digit-splitter-writer",
159  inputs,
160  outputs,
161  AlgorithmSpec{adaptFromTask<o2::tof::TOFDigitWriterSplitter>(nTF, storeErr)},
162  Options{{"output-base-name", VariantType::String, "tofdigits", {"Name of the input file (root extension will be added)"}}}};
163 }
