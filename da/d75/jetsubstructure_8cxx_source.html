42 struct JetSubstructure {
43  Produces<aod::JetSubtructure> jetSubstructure;
44  OutputObj<TH1F> hZg{"h_jet_zg"};
45  OutputObj<TH1F> hRg{"h_jet_rg"};
46  OutputObj<TH1F> hNsd{"h_jet_nsd"};
47 
48  Configurable<float> f_jetPtMin{"f_jetPtMin", 0.0, "minimum jet pT cut"};
49  Configurable<float> f_zCut{"f_zCut", 0.1, "soft drop z cut"};
50  Configurable<float> f_beta{"f_beta", 0.0, "soft drop beta"};
51  Configurable<float> f_jetR{"f_jetR", 0.4, "jer resolution parameter"}; //possible to get configurable from another task? jetR
52  Configurable<bool> b_DoConstSub{"b_DoConstSub", false, "do constituent subtraction"};
53 
54  std::vector<fastjet::PseudoJet> jetConstituents;
55  std::vector<fastjet::PseudoJet> jetReclustered;
56  JetFinder jetReclusterer;
57 
58  void init(InitContext const&)
59  {
60  hZg.setObject(new TH1F("h_jet_zg", "zg ;zg",
61  10, 0.0, 0.5));
62  hRg.setObject(new TH1F("h_jet_rg", "rg ;rg",
63  10, 0.0, 0.5));
64  hNsd.setObject(new TH1F("h_jet_nsd", "nsd ;nsd",
65  7, -0.5, 6.5));
66  jetReclusterer.isReclustering = true;
67  jetReclusterer.algorithm = fastjet::JetAlgorithm::cambridge_algorithm;
68  jetReclusterer.jetR = f_jetR * 2.5;
69  }
70 
71  //Filter jetCuts = aod::jet::pt > f_jetPtMin; //how does this work?
72 
73  void process(aod::Jet const& jet,
74  aod::Tracks const& tracks,
75  aod::JetConstituents const& constituents,
76  aod::JetConstituentsSub const& constituentsSub)
77  {
78  jetConstituents.clear();
79  jetReclustered.clear();
80  if (b_DoConstSub) {
81  for (const auto constituent : constituentsSub) {
82  fillConstituents(constituent, jetConstituents);
83  }
84  } else {
85  for (const auto constituentIndex : constituents) {
86  auto constituent = constituentIndex.track();
87  fillConstituents(constituent, jetConstituents);
88  }
89  }
90  fastjet::ClusterSequenceArea clusterSeq(jetReclusterer.findJets(jetConstituents, jetReclustered));
91  jetReclustered = sorted_by_pt(jetReclustered);
92  fastjet::PseudoJet daughterSubJet = jetReclustered[0];
93  fastjet::PseudoJet parentSubJet1;
94  fastjet::PseudoJet parentSubJet2;
95  bool softDropped = false;
96  int nsd = 0.0;
97  auto zg = -1.0;
98  auto rg = -1.0;
99  while (daughterSubJet.has_parents(parentSubJet1, parentSubJet2)) {
100  if (parentSubJet1.perp() < parentSubJet2.perp()) {
101  std::swap(parentSubJet1, parentSubJet2);
102  }
103  auto z = parentSubJet2.perp() / (parentSubJet1.perp() + parentSubJet2.perp());
104  auto r = parentSubJet1.delta_R(parentSubJet2);
105  if (z >= f_zCut * TMath::Power(r / f_jetR, f_beta)) {
106  if (!softDropped) {
107  zg = z;
108  rg = r;
109  hZg->Fill(zg);
110  hRg->Fill(rg);
111  softDropped = true;
112  }
113  nsd++;
114  }
115  daughterSubJet = parentSubJet1;
116  }
117  hNsd->Fill(nsd);
118  jetSubstructure(zg, rg, nsd);
119  }
120 };
121 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
122 {
123  return WorkflowSpec{
124  adaptAnalysisTask<JetSubstructure>(cfgc, "jet-substructure")};
125 }
