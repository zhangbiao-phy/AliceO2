37  Produces<aod::HfCandProng3Base> rowCandidateBase;

39  Configurable<double> magneticField{"d_bz", 5., "magnetic field"};
40  Configurable<bool> b_propdca{"b_propdca", true, "create tracks version propagated to PCA"};
41  Configurable<double> d_maxr{"d_maxr", 200., "reject PCA's above this radius"};
42  Configurable<double> d_maxdzini{"d_maxdzini", 4., "reject (if>0) PCA candidate if tracks DZ exceeds threshold"};
43  Configurable<double> d_minparamchange{"d_minparamchange", 1.e-3, "stop iterations if largest change of any X is smaller than this"};
44  Configurable<double> d_minrelchi2change{"d_minrelchi2change", 0.9, "stop iterations is chi2/chi2old > this"};
45  Configurable<bool> b_dovalplots{"b_dovalplots", true, "do validation plots"};
46 
47  OutputObj<TH1F> hmass3{TH1F("hmass3", "3-prong candidates;inv. mass (#pi K #pi) (GeV/#it{c}^{2});entries", 500, 1.6, 2.1)};
48  OutputObj<TH1F> hCovPVXX{TH1F("hCovPVXX", "3-prong candidates;XX element of cov. matrix of prim. vtx position (cm^{2});entries", 100, 0., 1.e-4)};
49  OutputObj<TH1F> hCovSVXX{TH1F("hCovSVXX", "3-prong candidates;XX element of cov. matrix of sec. vtx position (cm^{2});entries", 100, 0., 0.2)};

51  double massPi = RecoDecay::getMassPDG(kPiPlus);
52  double massK = RecoDecay::getMassPDG(kKPlus);
53  double massPiKPi{0.};

55  void process(aod::Collisions const& collisions,
56  aod::HfTrackIndexProng3 const& rowsTrackIndexProng3,
57  aod::BigTracks const& tracks)
58  {
59  // 3-prong vertex fitter
60  o2::vertexing::DCAFitterN<3> df;
61  df.setBz(magneticField);
62  df.setPropagateToPCA(b_propdca);
63  df.setMaxR(d_maxr);
64  df.setMaxDZIni(d_maxdzini);
65  df.setMinParamChange(d_minparamchange);
66  df.setMinRelChi2Change(d_minrelchi2change);
67  df.setUseAbsDCA(true);
68 
69  // loop over triplets of track indices
70  for (const auto& rowTrackIndexProng3 : rowsTrackIndexProng3) {
71  auto trackParVar0 = getTrackParCov(rowTrackIndexProng3.index0_as<aod::BigTracks>());
72  auto trackParVar1 = getTrackParCov(rowTrackIndexProng3.index1_as<aod::BigTracks>());
73  auto trackParVar2 = getTrackParCov(rowTrackIndexProng3.index2_as<aod::BigTracks>());
74  auto collision = rowTrackIndexProng3.index0().collision();
75 
76  // reconstruct the 3-prong secondary vertex
77  if (df.process(trackParVar0, trackParVar1, trackParVar2) == 0) {
78  continue;
79  }
80  const auto& secondaryVertex = df.getPCACandidate();
81  auto chi2PCA = df.getChi2AtPCACandidate();
82  auto covMatrixPCA = df.calcPCACovMatrix().Array();
83  hCovSVXX->Fill(covMatrixPCA[0]); // FIXME: Calculation of errorDecayLength(XY) gives wrong values without this line.
84  trackParVar0 = df.getTrack(0);
85  trackParVar1 = df.getTrack(1);
86  trackParVar2 = df.getTrack(2);
87 
88  // get track momenta
89  array<float, 3> pvec0;
90  array<float, 3> pvec1;
91  array<float, 3> pvec2;
92  trackParVar0.getPxPyPzGlo(pvec0);
93  trackParVar1.getPxPyPzGlo(pvec1);
94  trackParVar2.getPxPyPzGlo(pvec2);
95 
96  // get track impact parameters
97  // This modifies track momenta!
98  auto primaryVertex = getPrimaryVertex(collision);
99  auto covMatrixPV = primaryVertex.getCov();
100  hCovPVXX->Fill(covMatrixPV[0]);
101  o2::dataformats::DCA impactParameter0;
102  o2::dataformats::DCA impactParameter1;
103  o2::dataformats::DCA impactParameter2;
104  trackParVar0.propagateToDCA(primaryVertex, magneticField, &impactParameter0);
105  trackParVar1.propagateToDCA(primaryVertex, magneticField, &impactParameter1);
106  trackParVar2.propagateToDCA(primaryVertex, magneticField, &impactParameter2);
107 
108  // get uncertainty of the decay length
109  double phi, theta;
110  getPointDirection(array{collision.posX(), collision.posY(), collision.posZ()}, secondaryVertex, phi, theta);
111  auto errorDecayLength = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, theta) + getRotatedCovMatrixXX(covMatrixPCA, phi, theta));
112  auto errorDecayLengthXY = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, 0.) + getRotatedCovMatrixXX(covMatrixPCA, phi, 0.));
113 
114  // fill candidate table rows
115  rowCandidateBase(collision.posX(), collision.posY(), collision.posZ(),
116  secondaryVertex[0], secondaryVertex[1], secondaryVertex[2],
117  errorDecayLength, errorDecayLengthXY,
118  chi2PCA,
119  pvec0[0], pvec0[1], pvec0[2],
120  pvec1[0], pvec1[1], pvec1[2],
121  pvec2[0], pvec2[1], pvec2[2],
122  impactParameter0.getY(), impactParameter1.getY(), impactParameter2.getY(),
123  std::sqrt(impactParameter0.getSigmaY2()), std::sqrt(impactParameter1.getSigmaY2()), std::sqrt(impactParameter2.getSigmaY2()),
124  rowTrackIndexProng3.index0Id(), rowTrackIndexProng3.index1Id(), rowTrackIndexProng3.index2Id(),
125  rowTrackIndexProng3.hfflag());
126 
127  // fill histograms
128  if (b_dovalplots) {
129  // calculate invariant mass
130  auto arrayMomenta = array{pvec0, pvec1, pvec2};
131  massPiKPi = RecoDecay::M(std::move(arrayMomenta), array{massPi, massK, massPi});
132  hmass3->Fill(massPiKPi);
133  }
134  }
135  }
136 };

145 struct HFCandidateCreator3ProngMC {
146  Produces<aod::HfCandProng3MCRec> rowMCMatchRec;
147  Produces<aod::HfCandProng3MCGen> rowMCMatchGen;
148 
149  void process(aod::HfCandProng3 const& candidates,
150  aod::BigTracksMC const& tracks,
151  aod::McParticles const& particlesMC)
152  {
153  int8_t sign = 0;
154  int8_t flag = 0;
155  int8_t DecayChannel = 0;
156  std::vector<int> arrDaughIndex;
157  std::array<int, 2> arrPDGDaugh;
158  std::array<int, 2> arrPDGResonant1 = {2212, 313}; // Λc± → p± K*
159  std::array<int, 2> arrPDGResonant2 = {2224, 321}; // Λc± → Δ(1232)±± K∓
160  std::array<int, 2> arrPDGResonant3 = {3124, 211}; // Λc± → Λ(1520) π±
161 
162  // Match reconstructed candidates.
163  for (auto& candidate : candidates) {
164  //Printf("New rec. candidate");
165  flag = 0;
166  DecayChannel = 0;
167  arrDaughIndex.clear();
168  auto arrayDaughters = array{candidate.index0_as<aod::BigTracksMC>(), candidate.index1_as<aod::BigTracksMC>(), candidate.index2_as<aod::BigTracksMC>()};
169 
170  // D± → π± K∓ π±
171  //Printf("Checking D± → π± K∓ π±");
172  if (RecoDecay::getMatchedMCRec(particlesMC, arrayDaughters, 411, array{+kPiPlus, -kKPlus, +kPiPlus}, true, &sign) > -1) {
173  flag = sign * (1 << DPlusToPiKPi);
174  }
175 
176  // Λc± → p± K∓ π±
177  if (flag == 0) {
178  //Printf("Checking Λc± → p± K∓ π±");
179  auto indexRecLc = RecoDecay::getMatchedMCRec(particlesMC, std::move(arrayDaughters), 4122, array{+kProton, -kKPlus, +kPiPlus}, true, &sign, 2);
180  if (indexRecLc > -1) {
181  flag = sign * (1 << LcToPKPi);
182 
183  //Printf("Flagging the different Λc± → p± K∓ π± decay channels");
184  RecoDecay::getDaughters(particlesMC, indexRecLc, &arrDaughIndex, array{0}, 1);
185  if (arrDaughIndex.size() == 2) {
186  for (auto iProng = 0; iProng < arrDaughIndex.size(); ++iProng) {
187  auto daughI = particlesMC.iteratorAt(arrDaughIndex[iProng]);
188  arrPDGDaugh[iProng] = std::abs(daughI.pdgCode());
189  }
190  if (arrPDGDaugh[0] == arrPDGResonant1[0] && arrPDGDaugh[1] == arrPDGResonant1[1]) {
191  DecayChannel = 1;
192  } else if (arrPDGDaugh[0] == arrPDGResonant2[0] && arrPDGDaugh[1] == arrPDGResonant2[1]) {
193  DecayChannel = 2;
194  } else if (arrPDGDaugh[0] == arrPDGResonant3[0] && arrPDGDaugh[1] == arrPDGResonant3[1]) {
195  DecayChannel = 3;
196  }
197  }
198  }
199  }
200 
201  rowMCMatchRec(flag, DecayChannel);
202  }
203 
204  // Match generated particles.
205  for (auto& particle : particlesMC) {
206  //Printf("New gen. candidate");
207  flag = 0;
208  DecayChannel = 0;
209  arrDaughIndex.clear();
210 
211  // D± → π± K∓ π±
212  //Printf("Checking D± → π± K∓ π±");
213  if (RecoDecay::isMatchedMCGen(particlesMC, particle, 411, array{+kPiPlus, -kKPlus, +kPiPlus}, true, &sign)) {
214  flag = sign * (1 << DPlusToPiKPi);
215  }
216 
217  // Λc± → p± K∓ π±
218  if (flag == 0) {
219  //Printf("Checking Λc± → p± K∓ π±");
220  auto isMatchedGenLc = RecoDecay::isMatchedMCGen(particlesMC, particle, 4122, array{+kProton, -kKPlus, +kPiPlus}, true, &sign, 2);
221  if (isMatchedGenLc) {
222  flag = sign * (1 << LcToPKPi);
223 
224  //Printf("Flagging the different Λc± → p± K∓ π± decay channels");
225  RecoDecay::getDaughters(particlesMC, particle.globalIndex(), &arrDaughIndex, array{0}, 1);
226  if (arrDaughIndex.size() == 2) {
227  for (auto jProng = 0; jProng < arrDaughIndex.size(); ++jProng) {
228  auto daughJ = particlesMC.iteratorAt(arrDaughIndex[jProng]);
229  arrPDGDaugh[jProng] = std::abs(daughJ.pdgCode());
230  }
231  if (arrPDGDaugh[0] == arrPDGResonant1[0] && arrPDGDaugh[1] == arrPDGResonant1[1]) {
232  DecayChannel = 1;
233  } else if (arrPDGDaugh[0] == arrPDGResonant2[0] && arrPDGDaugh[1] == arrPDGResonant2[1]) {
234  DecayChannel = 2;
235  } else if (arrPDGDaugh[0] == arrPDGResonant3[0] && arrPDGDaugh[1] == arrPDGResonant3[1]) {
236  DecayChannel = 3;
237  }
238  }
239  }
240  }
241 
242  rowMCMatchGen(flag, DecayChannel);
243  }
244  }
245 };
