53 void buildTrackletMCMData(TrackletMCMData& trackletword, const uint slope, const uint pos, const uint q0, const uint q1, const uint q2)

173 void printHalfCRUHeader(o2::trd::HalfCRUHeader const& halfcru)
174 {
175  std::array<uint32_t, 15> sizes;
176  std::array<uint32_t, 15> errorflags;
177  getlinkdatasizes(halfcru, sizes);
178  getlinkerrorflags(halfcru, errorflags);
179  LOGF(INFO, "V:%d BC:%d SB:%d EType:%d", halfcru.HeaderVersion, halfcru.BunchCrossing,
180  halfcru.StopBit, halfcru.EventType);
181  for (int i = 0; i < 15; i++) {
182  LOGF(INFO, "Link %d size: %ul eflag: 0x%02x", i, sizes[i], errorflags[i]);
183  }
184 }
