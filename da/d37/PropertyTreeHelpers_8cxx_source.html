24 void PropertyTreeHelpers::populateDefaults(std::vector<ConfigParamSpec> const& schema,
25  boost::property_tree::ptree& pt,
26  boost::property_tree::ptree& provenance)
27 {
28  for (auto const& spec : schema) {
29  std::string key = spec.name.substr(0, spec.name.find(','));
30  try {
31  if (spec.defaultValue.type() == VariantType::Empty) {
32  continue;
33  }
34  switch (spec.type) {
35  case VariantType::Int:
36  pt.put(key, spec.defaultValue.get<int>());
37  break;
38  case VariantType::Int64:
39  pt.put(key, spec.defaultValue.get<int64_t>());
40  break;
41  case VariantType::Float:
42  pt.put(key, spec.defaultValue.get<float>());
43  break;
44  case VariantType::Double:
45  pt.put(key, spec.defaultValue.get<double>());
46  break;
47  case VariantType::String:
48  pt.put(key, spec.defaultValue.get<std::string>());
49  break;
50  case VariantType::Bool:
51  pt.put(key, spec.defaultValue.get<bool>());
52  break;
53  case VariantType::ArrayInt:
54  pt.put_child(key, vectorToBranch(spec.defaultValue.get<int*>(), spec.defaultValue.size()));
55  break;
56  case VariantType::ArrayFloat:
57  pt.put_child(key, vectorToBranch(spec.defaultValue.get<float*>(), spec.defaultValue.size()));
58  break;
59  case VariantType::ArrayDouble:
60  pt.put_child(key, vectorToBranch(spec.defaultValue.get<double*>(), spec.defaultValue.size()));
61  break;
62  case VariantType::ArrayBool:
63  pt.put_child(key, vectorToBranch(spec.defaultValue.get<bool*>(), spec.defaultValue.size()));
64  break;
65  case VariantType::ArrayString:
66  pt.put_child(key, vectorToBranch(spec.defaultValue.get<std::string*>(), spec.defaultValue.size()));
67  break;
68  case VariantType::Array2DInt:
69  pt.put_child(key, array2DToBranch(spec.defaultValue.get<Array2D<int>>()));
70  break;
71  case VariantType::Array2DFloat:
72  pt.put_child(key, array2DToBranch(spec.defaultValue.get<Array2D<float>>()));
73  break;
74  case VariantType::Array2DDouble:
75  pt.put_child(key, array2DToBranch(spec.defaultValue.get<Array2D<double>>()));
76  break;
77  case VariantType::LabeledArrayInt:
78  pt.put_child(key, labeledArrayToBranch(spec.defaultValue.get<LabeledArray<int>>()));
79  break;
80  case VariantType::LabeledArrayFloat:
81  pt.put_child(key, labeledArrayToBranch(spec.defaultValue.get<LabeledArray<float>>()));
82  break;
83  case VariantType::LabeledArrayDouble:
84  pt.put_child(key, labeledArrayToBranch(spec.defaultValue.get<LabeledArray<double>>()));
85  break;
86  case VariantType::Unknown:
87  case VariantType::Empty:
88  default:
89  throw std::runtime_error("Unknown variant type");
90  }
91  provenance.put(key, "default");
92  } catch (std::runtime_error& re) {
93  throw;
94  } catch (std::exception& e) {
95  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
96  } catch (...) {
97  throw std::invalid_argument(std::string("missing option: ") + key);
98  }
99  }

102 void PropertyTreeHelpers::populate(std::vector<ConfigParamSpec> const& schema,
103  boost::property_tree::ptree& pt,
104  boost::program_options::variables_map const& vmap,
105  boost::property_tree::ptree& provenance)
106 {
107  for (auto const& spec : schema) {
108  // strip short version to get the correct key
109  std::string key = spec.name.substr(0, spec.name.find(','));
110  if (vmap.count(key) == 0) {
111  continue;
112  }
113  try {
114  switch (spec.type) {
115  case VariantType::Int:
116  pt.put(key, vmap[key].as<int>());
117  break;
118  case VariantType::Int64:
119  pt.put(key, vmap[key].as<int64_t>());
120  break;
121  case VariantType::Float:
122  pt.put(key, vmap[key].as<float>());
123  break;
124  case VariantType::Double:
125  pt.put(key, vmap[key].as<double>());
126  break;
127  case VariantType::String:
128  if (auto const* v = boost::any_cast<std::string>(&vmap[key].value())) {
129  pt.put(key, *v);
130  }
131  break;
132  case VariantType::Bool:
133  pt.put(key, vmap[key].as<bool>());
134  break;
135  case VariantType::ArrayInt:
136  pt.put_child(key, vectorToBranch<int>(stringToVector<int>(vmap[key].as<std::string>())));
137  break;
138  case VariantType::ArrayFloat:
139  pt.put_child(key, vectorToBranch<float>(stringToVector<float>(vmap[key].as<std::string>())));
140  break;
141  case VariantType::ArrayDouble:
142  pt.put_child(key, vectorToBranch<double>(stringToVector<double>(vmap[key].as<std::string>())));
143  break;
144  case VariantType::ArrayBool:
145  // pt.put_child(key, vectorToBranch<bool>(stringToVector<bool>(vmap[key].as<std::string>())));
146  break;
147  case VariantType::ArrayString:
148  pt.put_child(key, vectorToBranch<std::string>(stringToVector<std::string>(vmap[key].as<std::string>())));
149  break;
150  case VariantType::Array2DInt:
151  pt.put_child(key, array2DToBranch<int>(stringToArray2D<int>(vmap[key].as<std::string>())));
152  break;
153  case VariantType::Array2DFloat:
154  pt.put_child(key, array2DToBranch<float>(stringToArray2D<float>(vmap[key].as<std::string>())));
155  break;
156  case VariantType::Array2DDouble:
157  pt.put_child(key, array2DToBranch<double>(stringToArray2D<double>(vmap[key].as<std::string>())));
158  break;
159  case VariantType::Unknown:
160  case VariantType::Empty:
161  default:
162  throw std::runtime_error("Unknown variant type");
163  }
164  provenance.put(key, "fairmq");
165  } catch (std::runtime_error& re) {
166  throw;
167  } catch (std::exception& e) {
168  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
169  } catch (...) {
170  throw std::invalid_argument(std::string("missing option: ") + key);
171  }
172  }
173 }

175 void PropertyTreeHelpers::populate(std::vector<ConfigParamSpec> const& schema,
176  boost::property_tree::ptree& pt,
177  boost::property_tree::ptree const& in,
178  boost::property_tree::ptree& provenance,
179  std::string const& provenanceLabel)
180 {
181  for (auto const& spec : schema) {
182  // strip short version to get the correct key
183  std::string key = spec.name.substr(0, spec.name.find(','));
184  auto it = in.get_child_optional(key);
185  if (!it) {
186  continue;
187  }
188  try {
189  switch (spec.type) {
190  case VariantType::Int:
191  pt.put(key, (*it).get_value<int>());
192  break;
193  case VariantType::Int64:
194  pt.put(key, (*it).get_value<int64_t>());
195  break;
196  case VariantType::Float:
197  pt.put(key, (*it).get_value<float>());
198  break;
199  case VariantType::Double:
200  pt.put(key, (*it).get_value<double>());
201  break;
202  case VariantType::String:
203  pt.put(key, (*it).get_value<std::string>());
204  break;
205  case VariantType::Bool:
206  pt.put(key, (*it).get_value<bool>());
207  break;
208  case VariantType::ArrayInt:
209  case VariantType::ArrayFloat:
210  case VariantType::ArrayDouble:
211  case VariantType::ArrayBool:
212  case VariantType::ArrayString:
213  case VariantType::Array2DInt:
214  case VariantType::Array2DFloat:
215  case VariantType::Array2DDouble:
216  case VariantType::LabeledArrayInt:
217  case VariantType::LabeledArrayFloat:
218  case VariantType::LabeledArrayDouble:
219  pt.put_child(key, *it);
220  break;
221  case VariantType::Unknown:
222  case VariantType::Empty:
223  default:
224  throw std::runtime_error("Unknown variant type");
225  }
226  provenance.put(key, provenanceLabel);
227  } catch (std::runtime_error& re) {
228  throw;
229  } catch (std::exception& e) {
230  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
231  } catch (...) {
232  throw std::invalid_argument(std::string("missing option: ") + key);
233  }
234  }
235 }

239 void traverseRecursive(const boost::property_tree::ptree& parent,
240  const boost::property_tree::ptree::path_type& childPath,
241  const boost::property_tree::ptree& child,
242  PropertyTreeHelpers::WalkerFunction& method)
243 {
244  using boost::property_tree::ptree;
245 
246  method(parent, childPath, child);
247  for (ptree::const_iterator it = child.begin(); it != child.end(); ++it) {
248  ptree::path_type curPath = childPath / ptree::path_type(it->first);
249  traverseRecursive(parent, curPath, it->second, method);
250  }
251 }
