36 template <typename Track>
37 bool GetImpactParameterAndError(const Track& track, const o2df::VertexBase& primaryVertex, double& impactParameterRPhi,
38  double& impactParameterRPhiError, double& impactParameterZ,
39  double& impactParameterErrorZ)
40 {
41  impactParameterRPhi = -999.;
42  impactParameterRPhiError = -999.;
43  impactParameterZ = -999.;
44  impactParameterErrorZ = -999.;
45 
46  o2df::DCA dca;
47  // FIXME: get this from CCDB
48  constexpr float magneticField{5.0}; // in kG
49  auto trackParameter = getTrackParCov(track);
50  bool propagate = trackParameter.propagateToDCA(primaryVertex, magneticField, &dca);
51 
52  constexpr float conversion_to_micrometer = 10000.f; // From [cm] to [mum]
53  if (propagate) {
54  impactParameterRPhi = conversion_to_micrometer * dca.getY();
55  impactParameterRPhiError = conversion_to_micrometer * std::sqrt(dca.getSigmaY2());
56  impactParameterZ = conversion_to_micrometer * dca.getZ();
57  impactParameterErrorZ = conversion_to_micrometer * std::sqrt(dca.getSigmaZ2());
58  }
59  return propagate;
60 }

280  void process(const o2::soa::Join<o2::aod::Collisions, o2::aod::McCollisionLabels>::iterator& collision,
281  const o2::soa::Join<o2::aod::Tracks, o2::aod::TracksCov, o2::aod::McTrackLabels>& tracks,
282  const o2::aod::McParticles& mcParticles, const o2::aod::McCollisions& mcCollisions)
283  {
284  const o2df::VertexBase primaryVertex = getPrimaryVertex(collision);
285 
286  for (const auto& track : tracks) {
287 
288  const auto mcParticle = track.mcParticle();
289  if (pdgCode.value != 0 && abs(mcParticle.pdgCode()) != abs(pdgCode.value)) {
290  continue;
291  }
292  if (useOnlyPhysicsPrimary.value && !MC::isPhysicalPrimary(mcParticles, mcParticle)) {
293  continue;
294  }
295  const double deltaPt = mcParticle.pt() - track.pt();
296  histos.fill(HIST("pt/ptDiffMCRec"), deltaPt);
297 
298  const double deltaPtOverPt = deltaPt / track.pt();
299 
300  histos.fill(HIST("pt/ptResolution"), deltaPtOverPt);
301  histos.fill(HIST("pt/ptResolutionVsPt"), track.pt(), std::abs(deltaPtOverPt));
302  histos.fill(HIST("pt/ptResolutionVsEta"), track.eta(), std::abs(deltaPtOverPt));
303  histos.fill(HIST("pt/ptResolutionVsPhi"), track.phi(), std::abs(deltaPtOverPt));
304 
305  const double deltaEta = mcParticle.eta() - track.eta();
306  histos.fill(HIST("eta/etaDiffMCReco"), deltaEta);
307  histos.fill(HIST("eta/etaDiffMCRecoVsEtaMC"), deltaEta, mcParticle.eta());
308  histos.fill(HIST("eta/etaDiffMCRecoVsEtaReco"), deltaEta, track.eta());
309 
310  histos.fill(HIST("phi/phiDiffMCRec"), mcParticle.phi() - track.phi());
311 
312  double impactParameterRPhi{-999.}, impactParameterRPhiError{-999.};
313  double impactParameterZ{-999.}, impactParameterErrorZ{-999.};
314 
315  const bool propagate = GetImpactParameterAndError(
316  track, primaryVertex, impactParameterRPhi, impactParameterRPhiError, impactParameterZ, impactParameterErrorZ);
317 
318  if (propagate) {
319  histos.fill(HIST("impactParameter/impactParameterRPhiVsPt"), track.pt(), impactParameterRPhi);
320  histos.fill(HIST("impactParameter/impactParameterRPhiVsEta"), track.eta(), impactParameterRPhi);
321  histos.fill(HIST("impactParameter/impactParameterRPhiVsPhi"), track.phi(), impactParameterRPhi);
322 
323  histos.fill(HIST("impactParameter/impactParameterZVsPt"), track.pt(), impactParameterZ);
324  histos.fill(HIST("impactParameter/impactParameterZVsEta"), track.eta(), impactParameterZ);
325  histos.fill(HIST("impactParameter/impactParameterZVsPhi"), track.phi(), impactParameterZ);
326 
327  histos.fill(HIST("impactParameter/impactParameterErrorRPhiVsPt"), track.pt(), impactParameterRPhiError);
328  histos.fill(HIST("impactParameter/impactParameterErrorRPhiVsEta"), track.eta(), impactParameterRPhiError);
329  histos.fill(HIST("impactParameter/impactParameterErrorRPhiVsPhi"), track.phi(), impactParameterRPhiError);
330 
331  histos.fill(HIST("impactParameter/impactParameterErrorZVsPt"), track.pt(), impactParameterErrorZ);
332  histos.fill(HIST("impactParameter/impactParameterErrorZVsEta"), track.eta(), impactParameterErrorZ);
333  histos.fill(HIST("impactParameter/impactParameterErrorZVsPhi"), track.phi(), impactParameterErrorZ);
334  }
335  }
336  }
