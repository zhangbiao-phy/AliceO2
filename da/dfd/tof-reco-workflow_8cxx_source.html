46 void customize(std::vector<o2::framework::ConfigParamSpec>& workflowOptions)
47 {
48  workflowOptions.push_back(ConfigParamSpec{"input-type", o2::framework::VariantType::String, "digits", {"digits, raw, clusters"}});
49  workflowOptions.push_back(ConfigParamSpec{"output-type", o2::framework::VariantType::String, "clusters,matching-info,calib-info", {"digits, clusters, matching-info, calib-info, raw, ctf"}});
50  workflowOptions.push_back(ConfigParamSpec{"disable-mc", o2::framework::VariantType::Bool, false, {"disable sending of MC information, TBI"}});
51  workflowOptions.push_back(ConfigParamSpec{"tof-sectors", o2::framework::VariantType::String, "0-17", {"TOF sector range, e.g. 5-7,8,9 ,TBI"}});
52  workflowOptions.push_back(ConfigParamSpec{"tof-lanes", o2::framework::VariantType::Int, 1, {"number of parallel lanes up to the matcher, TBI"}});
53  workflowOptions.push_back(ConfigParamSpec{"use-ccdb", o2::framework::VariantType::Bool, false, {"enable access to ccdb tof calibration objects"}});
54  workflowOptions.push_back(ConfigParamSpec{"use-fit", o2::framework::VariantType::Bool, false, {"enable access to fit info for calibration"}});
55  workflowOptions.push_back(ConfigParamSpec{"input-desc", o2::framework::VariantType::String, "CRAWDATA", {"Input specs description string"}});
56  workflowOptions.push_back(ConfigParamSpec{"disable-root-input", o2::framework::VariantType::Bool, false, {"disable root-files input readers"}});
57  workflowOptions.push_back(ConfigParamSpec{"disable-root-output", o2::framework::VariantType::Bool, false, {"disable root-files output writers"}});
58  workflowOptions.push_back(ConfigParamSpec{"conet-mode", o2::framework::VariantType::Bool, false, {"enable conet mode"}});
59  workflowOptions.push_back(ConfigParamSpec{"configKeyValues", o2::framework::VariantType::String, "", {"Semicolon separated key=value strings ..."}});
60  workflowOptions.push_back(ConfigParamSpec{"disable-row-writing", o2::framework::VariantType::Bool, false, {"disable ROW in Digit writing"}});
61  workflowOptions.push_back(ConfigParamSpec{"write-decoding-errors", o2::framework::VariantType::Bool, false, {"trace errors in digits output when decoding"}});
62  workflowOptions.push_back(ConfigParamSpec{"calib-cluster", VariantType::Bool, false, {"to enable calib info production from clusters"}});
63 }

83 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
84 {
85  WorkflowSpec specs;
86 
87  if (!cfgc.helpOnCommandLine()) {
88  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
89  }
90  // the lane configuration defines the subspecification ids to be distributed among the lanes.
91  auto nLanes = cfgc.options().get<int>("tof-lanes");
92  auto inputType = cfgc.options().get<std::string>("input-type");
93  auto outputType = cfgc.options().get<std::string>("output-type");
94 
95  bool writecluster = 0;
96  bool writematching = 0;
97  bool writecalib = 0;
98  bool writedigit = 0;
99  bool writeraw = 0;
100  bool writectf = 0;
101  bool writeerr = 0;
102 
103  if (outputType.rfind("clusters") < outputType.size()) {
104  writecluster = 1;
105  }
106  if (outputType.rfind("matching-info") < outputType.size()) {
107  writematching = 1;
108  }
109  if (outputType.rfind("calib-info") < outputType.size()) {
110  writecalib = 1;
111  }
112  if (outputType.rfind("digits") < outputType.size()) {
113  writedigit = 1;
114  }
115  if (outputType.rfind("raw") < outputType.size()) {
116  writeraw = 1;
117  }
118  if (outputType.rfind("ctf") < outputType.size()) {
119  writectf = 1;
120  }
121 
122  bool dgtinput = 0;
123  bool clusterinput = 0;
124  bool rawinput = 0;
125  if (inputType == "digits") {
126  dgtinput = 1;
127  } else if (inputType == "clusters") {
128  clusterinput = 1;
129  } else if (inputType == "raw") {
130  rawinput = 1;
131  writeerr = cfgc.options().get<bool>("write-decoding-errors");
132  }
133 
134  auto useMC = !cfgc.options().get<bool>("disable-mc");
135  auto useCCDB = cfgc.options().get<bool>("use-ccdb");
136  auto useFIT = cfgc.options().get<bool>("use-fit");
137  bool disableRootInput = cfgc.options().get<bool>("disable-root-input") || rawinput;
138  bool disableRootOutput = cfgc.options().get<bool>("disable-root-output");
139  bool conetmode = cfgc.options().get<bool>("conet-mode");
140  bool disableROWwriting = cfgc.options().get<bool>("disable-row-writing");
141  auto isCalibFromCluster = cfgc.options().get<bool>("calib-cluster");
142 
143  LOG(INFO) << "TOF RECO WORKFLOW configuration";
144  LOG(INFO) << "TOF input = " << cfgc.options().get<std::string>("input-type");
145  LOG(INFO) << "TOF output = " << cfgc.options().get<std::string>("output-type");
146  LOG(INFO) << "TOF sectors = " << cfgc.options().get<std::string>("tof-sectors");
147  LOG(INFO) << "TOF disable-mc = " << cfgc.options().get<std::string>("disable-mc");
148  LOG(INFO) << "TOF lanes = " << cfgc.options().get<std::string>("tof-lanes");
149  LOG(INFO) << "TOF use-ccdb = " << cfgc.options().get<std::string>("use-ccdb");
150  LOG(INFO) << "TOF use-fit = " << cfgc.options().get<std::string>("use-fit");
151  LOG(INFO) << "TOF disable-root-input = " << disableRootInput;
152  LOG(INFO) << "TOF disable-root-output = " << disableRootOutput;
153  LOG(INFO) << "TOF conet-mode = " << conetmode;
154  LOG(INFO) << "TOF disable-row-writing = " << disableROWwriting;
155  LOG(INFO) << "TOF write-decoding-errors = " << writeerr;
156 
157  if (clusterinput && !disableRootInput) {
158  LOG(INFO) << "Insert TOF Cluster Reader";
159  specs.emplace_back(o2::tof::getClusterReaderSpec(useMC));
160  } else if (dgtinput) {
161  // TOF clusterizer
162  if (!disableRootInput) {
163  LOG(INFO) << "Insert TOF Digit reader from file";
164  specs.emplace_back(o2::tof::getDigitReaderSpec(useMC));
165  }
166  if (writeraw) {
167  LOG(INFO) << "Insert TOF Raw writer";
168  specs.emplace_back(o2::tof::getTOFRawWriterSpec());
169  }
170  } else if (rawinput) {
171  LOG(INFO) << "Insert TOF Compressed Raw Decoder";
172  auto inputDesc = cfgc.options().get<std::string>("input-desc");
173  specs.emplace_back(o2::tof::getCompressedDecodingSpec(inputDesc, conetmode));
174  useMC = 0;
175 
176  if (writedigit && !disableRootOutput) {
177  // add TOF digit writer without mc labels
178  LOG(INFO) << "Insert TOF Digit Writer";
179  specs.emplace_back(o2::tof::getTOFDigitWriterSpec(0, writeerr));
180  }
181  }
182 
183  if (!clusterinput && writecluster) {
184  LOG(INFO) << "Insert TOF Clusterizer";
185  specs.emplace_back(o2::tof::getTOFClusterizerSpec(useMC, useCCDB, isCalibFromCluster));
186  if (writecluster && !disableRootOutput) {
187  LOG(INFO) << "Insert TOF Cluster Writer";
188  specs.emplace_back(o2::tof::getTOFClusterWriterSpec(useMC));
189  }
190  }
191 
192  if (useFIT && !disableRootInput) {
193  specs.emplace_back(o2::ft0::getRecPointReaderSpec(useMC));
194  }
195 
196  if (writematching || writecalib) {
197  if (!disableRootInput) {
198  LOG(INFO) << "Insert ITS-TPC Track Reader";
199  specs.emplace_back(o2::globaltracking::getTrackTPCITSReaderSpec(useMC));
200  }
201  LOG(INFO) << "Insert TOF Matching";
202  specs.emplace_back(o2::tof::getTOFRecoWorkflowSpec(useMC, useFIT));
203 
204  if (writematching && !disableRootOutput) {
205  LOG(INFO) << "Insert TOF Matched Info Writer";
206  specs.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC));
207  }
208  if (writecalib && !disableRootOutput) {
209  LOG(INFO) << "Insert TOF Calib Info Writer";
210  specs.emplace_back(o2::tof::getTOFCalibWriterSpec());
211  }
212  }
213  if (writectf) {
214  LOG(INFO) << "Insert TOF CTF encoder";
215  specs.emplace_back(o2::tof::getEntropyEncoderSpec());
216  }
217 
218  LOG(INFO) << "Number of active devices = " << specs.size();
219 
220  return std::move(specs);
221 }
