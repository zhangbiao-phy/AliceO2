29 /*
30  Fast 1D histo class which can be messages as
31  FlatHisto1D<float> histo(nbins, xmin, xmax);
32  histo.fill(...);
33  pc.outputs().snapshot(Output{"Origin", "Desc", 0, Lifetime::Timeframe}, histo.getBase());
34 
35  and received (read only!) as
36  const auto hdata = pc.inputs().get<gsl::span<float>>("histodata");
37  FlatHisto1D<float> histoView;
38  histoView.adoptExternal(hdata);
39  or directly
40  FlatHisto1D<float> histoView(pc.inputs().get<gsl::span<float>>("histodata"));
41 */
42 
43 template <typename T = float>
44 class FlatHisto1D
45 {
46  static_assert(std::is_same<T, float>::value || std::is_same<T, double>::value, "T must be float or double");
47 
48  public:
49  enum { NBins,
50  XMin,
51  XMax,
52  BinSize,
53  BinSizeInv,
54  NServiceSlots };
55 
56  FlatHisto1D() = default;
57 
58  FlatHisto1D(int nb, T xmin, T xmax)
59  {
60  assert(nb > 0 && xmin < xmax);
61  mData.resize(nb + NServiceSlots, 0.);
62  mData[NBins] = nb;
63  mData[XMin] = xmin;
64  mData[XMax] = xmax;
65  mData[BinSize] = (xmax - xmin) / nb;
66  mData[BinSizeInv] = nb / (xmax - xmin);
67  init();
68  }
69 
70  FlatHisto1D(const gsl::span<const T> ext)
71  {
72  adoptExternal(ext);
73  }
74 
75  int getNBins() const { return mNBins; }
76  T getXMin() const { return mDataView[XMin]; }
77  T getXMax() const { return mDataView[XMax]; }
78  T getBinSize() const { return mDataView[BinSize]; }
79  T getBinContent(uint32_t ib) const { return ib < mNBins ? mDataView[ib + NServiceSlots] : 0.; }
80  T getBinContentForX(T x) const { getBinContent(getBin(x)); }
81 
82  T getBinStart(int i) const
83  {
84  assert(i < getNBins());
85  return getXMin() + i * getBinSize();
86  }
87 
88  T getBinCenter(int i) const
89  {
90  assert(i < getNBins());
91  return getXMin() + (i + 0.5) * getBinSize();
92  }
93 
94  T getBinEnd(int i) const
95  {
96  assert(i < getNBins());
97  return getXMin() + (i + 1) * getBinSize();
98  }
99 
100  void add(const FlatHisto1D& other)
101  {
102  assert(getNBins() == other.getNBins() && getXMin() == other.getXMin() && getXMax() == other.getXMax() && canFill());
103  int last = NServiceSlots + getNBins();
104  const auto& otherView = other.getView();
105  for (int i = NServiceSlots; i < last; i++) {
106  mData[i] += otherView[i];
107  }
108  }
109 
110  void subtract(const FlatHisto1D& other)
111  {
112  assert(getNBins() == other.getNBins() && getXMin() == other.getXMin() && getXMax() == other.getXMax() && canFill());
113  int last = NServiceSlots + getNBins();
114  const auto& otherView = other.getView();
115  for (int i = NServiceSlots; i < last; i++) {
116  mData[i] -= otherView[i];
117  }
118  }
119 
120  void setBinContent(uint32_t bin, T w)
121  {
122  assert(canFill() && bin < mNBins);
123  mData[bin + NServiceSlots] = w;
124  }
125 
126  void clear()
127  {
128  assert(canFill());
129  memset(mData.data() + NServiceSlots, 0, sizeof(T) * getNBins());
130  }
131 
132  T getSum() const
133  {
134  T sum = 0;
135  for (int i = getNBins(); i--;) {
136  sum += getBinContent(i);
137  }
138  return sum;
139  }
140 
141  void adoptExternal(const gsl::span<const T> ext)
142  {
143  assert(ext.size() > NServiceSlots);
144  mData.clear();
145  mDataView = ext;
146  mNBins = (int)mDataView[NBins];
147  }
148 
149  void init()
150  { // when reading from file, need to call this method to make it operational
151  assert(mData.size() > NServiceSlots);
152  mDataView = gsl::span<const T>(mData.data(), mData.size());
153  mNBins = (int)mData[NBins];
154  }
155 
156  void fill(T x)
157  {
158  uint32_t bin = getBin(x);
159  if (bin < mNBins) {
160  mData[NServiceSlots + bin]++;
161  }
162  }
163 
164  void fill(T x, T w)
165  {
166  uint32_t bin = getBin(x);
167  if (bin < mNBins) {
168  mData[NServiceSlots + bin] += w;
169  }
170  }
171 
172  uint32_t getBin(T x) const
173  {
174  auto dx = x - getXMin();
175  return dx < 0 ? 0xffffffff : uint32_t(dx * getBinSizeInv());
176  }
177 
178  bool canFill() const
179  {
180  // histo can be filled only if hase its own data, otherwise only query can be done on the view
181  return mData.size() > NServiceSlots;
182  }
183 
184  TH1F createTH1F(const std::string& name = "histo1d")
185  {
186  TH1F h(name.c_str(), name.c_str(), getNBins(), getXMin(), getXMax());
187  for (int i = getNBins(); i--;) {
188  auto w = getBinContent(i);
189  if (w) {
190  h.SetBinContent(i + 1, w);
191  }
192  }
193  return h;
194  }
195 
196  const std::vector<T>& getBase() const { return mData; }
197  gsl::span<const T> getView() const { return mDataView; }
198 
199  protected:
200  T getBinSizeInv() const { return mDataView[BinSizeInv]; }
201 
202  std::vector<T> mData; // data to fill
203  gsl::span<const T> mDataView;
204  int mNBins = 0;
205 
206  ClassDefNV(FlatHisto1D, 1);
207 };
