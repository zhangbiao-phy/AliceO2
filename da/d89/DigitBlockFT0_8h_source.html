58  template <class DataBlockType>
59  void processDigits(DataBlockType& dataBlock, int linkID)
60  {
61  if constexpr (std::is_same<DataBlockType, DataBlockPM>::value) { //Filling data from PM
62  for (int iEventData = 0; iEventData < dataBlock.DataBlockWrapper<RawDataPM>::mNelements; iEventData++) {
63  mVecChannelData.emplace_back(uint8_t(o2::ft0::SingleLUT::Instance().getChannel(linkID, dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].channelID)),
64  int(dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].time),
65  int(dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].charge),
66  dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].getFlagWord());
67  }
68  } else if constexpr (std::is_same<DataBlockType, DataBlockTCM>::value) { //Filling data from TCM (normal/extended mode)
69  dataBlock.DataBlockWrapper<RawDataTCM>::mData[0].fillTrigger(mDigit.mTriggers);
70  }
71  }
72  void getDigits(std::vector<Digit>& vecDigits, std::vector<ChannelData>& vecChannelData)
73  {
74  //last digit filling
75  mDigit.ref.set(vecChannelData.size(), mVecChannelData.size());
76  mDigit.mEventID = sEventID;
77  //
78  vecDigits.push_back(std::move(mDigit));
79  std::move(mVecChannelData.begin(), mVecChannelData.end(), std::back_inserter(vecChannelData));
80  mVecChannelData.clear();
81 
82  sEventID++; //Increasing static eventID. After each poping of the data, it will increase
83  }
