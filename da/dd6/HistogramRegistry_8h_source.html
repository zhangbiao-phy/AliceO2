221  // create histogram of type T with the axes defined in HistogramSpec
222  template <typename T>
223  static std::shared_ptr<T> createHist(const HistogramSpec& histSpec)
224  {
225  constexpr std::size_t MAX_DIM{10};
226  const std::size_t nAxes{histSpec.config.axes.size()};
227  if (nAxes == 0 || nAxes > MAX_DIM) {
228  LOGF(FATAL, "The histogram specification contains no (or too many) axes.");
229  return nullptr;
230  }
231 
232  int nBins[MAX_DIM]{0};
233  double lowerBounds[MAX_DIM]{0.};
234  double upperBounds[MAX_DIM]{0.};
235 
236  // first figure out number of bins and dimensions
237  for (std::size_t i = 0; i < nAxes; i++) {
238  nBins[i] = (histSpec.config.axes[i].nBins) ? *histSpec.config.axes[i].nBins : histSpec.config.axes[i].binEdges.size() - 1;
239  lowerBounds[i] = histSpec.config.axes[i].binEdges.front();
240  upperBounds[i] = histSpec.config.axes[i].binEdges.back();
241  }
242 
243  // create histogram
244  std::shared_ptr<T> hist{generateHist<T>(histSpec.name, histSpec.title, nAxes, nBins, lowerBounds, upperBounds, histSpec.config.nSteps)};
245  if (!hist) {
246  LOGF(FATAL, "The number of dimensions specified for histogram %s does not match the type.", histSpec.name);
247  return nullptr;
248  }
249 
250  // set axis properties
251  for (std::size_t i = 0; i < nAxes; i++) {
252  TAxis* axis{getAxis(i, hist)};
253  if (axis) {
254  if (histSpec.config.axes[i].title) {
255  axis->SetTitle((*histSpec.config.axes[i].title).data());
256  }
257 
258  // this helps to have axes not only called 0,1,2... in ndim histos
259  if constexpr (std::is_base_of_v<THnBase, T>) {
260  if (histSpec.config.axes[i].name) {
261  axis->SetName((std::string(axis->GetName()) + "-" + *histSpec.config.axes[i].name).data());
262  }
263  }
264 
265  // move the bin edges in case a variable binning was requested
266  if (!histSpec.config.axes[i].nBins) {
267  if (!std::is_sorted(std::begin(histSpec.config.axes[i].binEdges), std::end(histSpec.config.axes[i].binEdges))) {
268  LOGF(FATAL, "The bin edges in histogram %s are not in increasing order!", histSpec.name);
269  return nullptr;
270  }
271  axis->Set(nBins[i], histSpec.config.axes[i].binEdges.data());
272  }
273  }
274  }
275  if (histSpec.callSumw2) {
276  hist->Sumw2();
277  }
278  return hist;
279  }

302  // helper function to get the axis via index for any type of root histogram
303  template <typename T>
304  static TAxis* getAxis(const int i, std::shared_ptr<T>& hist)
305  {
306  if constexpr (std::is_base_of_v<THnBase, T> || std::is_base_of_v<StepTHn, T>) {
307  return hist->GetAxis(i);
308  } else {
309  if (i == 0) {
310  return hist->GetXaxis();
311  } else if (i == 1) {
312  return hist->GetYaxis();
313  } else if (i == 2) {
314  return hist->GetZaxis();
315  } else {
316  return nullptr;
317  }

324  // helper function to generate the actual histograms
325  template <typename T>
326  static T* generateHist(const std::string& name, const std::string& title, const std::size_t nDim,
327  const int nBins[], const double lowerBounds[], const double upperBounds[], const int nSteps = 1)
328  {
329  if constexpr (std::is_base_of_v<StepTHn, T>) {
330  return new T(name.data(), title.data(), nSteps, nDim, nBins, lowerBounds, upperBounds);
331  } else if constexpr (std::is_base_of_v<THnBase, T>) {
332  return new T(name.data(), title.data(), nDim, nBins, lowerBounds, upperBounds);
333  } else if constexpr (std::is_base_of_v<TH3, T>) {
334  return (nDim != 3) ? nullptr
335  : new T(name.data(), title.data(), nBins[0], lowerBounds[0],
336  upperBounds[0], nBins[1], lowerBounds[1], upperBounds[1],
337  nBins[2], lowerBounds[2], upperBounds[2]);
338  } else if constexpr (std::is_base_of_v<TH2, T>) {
339  return (nDim != 2) ? nullptr
340  : new T(name.data(), title.data(), nBins[0], lowerBounds[0],
341  upperBounds[0], nBins[1], lowerBounds[1], upperBounds[1]);
342  } else if constexpr (std::is_base_of_v<TH1, T>) {
343  return (nDim != 1)
344  ? nullptr
345  : new T(name.data(), title.data(), nBins[0], lowerBounds[0], upperBounds[0]);
346  }
347  return nullptr;
348  }

350  // helper function to cast the actual histogram type (e.g. TH2F) to the correct interface type (e.g. TH2) that is stored in the HistPtr variant
351  template <typename T>
352  static std::optional<HistPtr> castToVariant(std::shared_ptr<TObject> obj)
353  {
354  if (obj->InheritsFrom(T::Class())) {
355  return std::static_pointer_cast<T>(obj);
356  }
357  return std::nullopt;
358  }
359 
360  template <typename T, typename Next, typename... Rest>
361  static std::optional<HistPtr> castToVariant(std::shared_ptr<TObject> obj)
362  {
363  if (auto hist = castToVariant<T>(obj)) {
364  return hist;
365  }
366  return castToVariant<Next, Rest...>(obj);
367  }
368 
369  static std::optional<HistPtr> castToVariant(std::shared_ptr<TObject> obj)
370  {
371  if (obj) {
372  // TProfile3D is TH3, TProfile2D is TH2, TH3 is TH1, TH2 is TH1, TProfile is TH1
373  return castToVariant<THn, THnSparse, TProfile3D, TH3, TProfile2D, TH2, TProfile, TH1, StepTHn>(obj);
374  }
375  return std::nullopt;
376  }

385 struct HistFiller {
386  // fill any type of histogram (if weight was requested it must be the last argument)
387  template <typename T, typename... Ts>
388  static void fillHistAny(std::shared_ptr<T>& hist, const Ts&... positionAndWeight)
389  {
390  constexpr int nArgs = sizeof...(Ts);
391 
392  constexpr bool validTH3 = (std::is_same_v<TH3, T> && (nArgs == 3 || nArgs == 4));
393  constexpr bool validTH2 = (std::is_same_v<TH2, T> && (nArgs == 2 || nArgs == 3));
394  constexpr bool validTH1 = (std::is_same_v<TH1, T> && (nArgs == 1 || nArgs == 2));
395  constexpr bool validTProfile3D = (std::is_same_v<TProfile3D, T> && (nArgs == 4 || nArgs == 5));
396  constexpr bool validTProfile2D = (std::is_same_v<TProfile2D, T> && (nArgs == 3 || nArgs == 4));
397  constexpr bool validTProfile = (std::is_same_v<TProfile, T> && (nArgs == 2 || nArgs == 3));
398 
399  constexpr bool validSimpleFill = validTH1 || validTH2 || validTH3 || validTProfile || validTProfile2D || validTProfile3D;
400  // unfortunately we dont know at compile the dimension of THn(Sparse)
401  constexpr bool validComplexFill = std::is_base_of_v<THnBase, T>;
402  constexpr bool validComplexFillStep = std::is_base_of_v<StepTHn, T>;
403 
404  if constexpr (validSimpleFill) {
405  hist->Fill(static_cast<double>(positionAndWeight)...);
406  } else if constexpr (validComplexFillStep) {
407  hist->Fill(positionAndWeight...); // first argument in pack is iStep, dimension check is done in StepTHn itself
408  } else if constexpr (validComplexFill) {
409  double tempArray[] = {static_cast<double>(positionAndWeight)...};
410  double weight{1.};
411  constexpr int nArgsMinusOne = nArgs - 1;
412  if (hist->GetNdimensions() == nArgsMinusOne) {
413  weight = tempArray[nArgsMinusOne];
414  } else if (hist->GetNdimensions() != nArgs) {
415  LOGF(FATAL, "The number of arguments in fill function called for histogram %s is incompatible with histogram dimensions.", hist->GetName());
416  }
417  hist->Fill(tempArray, weight);
418  } else {
419  LOGF(FATAL, "The number of arguments in fill function called for histogram %s is incompatible with histogram dimensions.", hist->GetName());
420  }
421  }
422 
423  // fill any type of histogram with columns (Cs) of a filtered table (if weight is requested it must reside the last specified column)
424  template <typename... Cs, typename R, typename T>
425  static void fillHistAny(std::shared_ptr<R>& hist, const T& table, const o2::framework::expressions::Filter& filter)
426  {
427  if constexpr (std::is_base_of_v<StepTHn, T>) {
428  LOGF(FATAL, "Table filling is not (yet?) supported for StepTHn.");
429  return;
430  }
431  auto filtered = o2::soa::Filtered<T>{{table.asArrowTable()}, o2::framework::expressions::createSelection(table.asArrowTable(), filter)};
432  for (auto& t : filtered) {
433  fillHistAny(hist, (*(static_cast<Cs>(t).getIterator()))...);
434  }
435  }

437  // function that returns rough estimate for the size of a histogram in MB
438  template <typename T>
439  static double getSize(std::shared_ptr<T>& hist, double fillFraction = 1.)
440  {
441  double size{0.};
442  if constexpr (std::is_base_of_v<TH1, T>) {
443  size = hist->GetNcells() * (HistFiller::getBaseElementSize(hist.get()) + ((hist->GetSumw2()->fN) ? sizeof(double) : 0.));
444  } else if constexpr (std::is_base_of_v<THn, T>) {
445  size = hist->GetNbins() * (HistFiller::getBaseElementSize(hist.get()) + ((hist->GetSumw2() != -1.) ? sizeof(double) : 0.));
446  } else if constexpr (std::is_base_of_v<THnSparse, T>) {
447  // THnSparse has massive overhead and should only be used when histogram is large and a very small fraction of bins is filled
448  double nBinsTotal = 1.;
449  int compCoordSize = 0; // size required to store a compact coordinate representation
450  for (int d = 0; d < hist->GetNdimensions(); ++d) {
451  int nBins = hist->GetAxis(d)->GetNbins() + 2;
452  nBinsTotal *= nBins;
453 
454  // number of bits needed to store compact coordinates
455  int b = 1;
456  while (nBins /= 2) {
457  ++b;
458  }
459  compCoordSize += b;
460  }
461  compCoordSize = (compCoordSize + 7) / 8; // turn bits into bytes
462 
463  // THnSparse stores the data in an array of chunks (THnSparseArrayChunk), each containing a fixed number of bins (e.g. 1024 * 16)
464  double nBinsFilled = fillFraction * nBinsTotal;
465  int nCunks = ceil(nBinsFilled / hist->GetChunkSize());
466  int chunkOverhead = sizeof(THnSparseArrayChunk);
467 
468  // each chunk holds array of compact bin-coordinates and an array of bin content (+ one of bin error if requested)
469  double binSize = compCoordSize + HistFiller::getBaseElementSize(hist.get()) + ((hist->GetSumw2() != -1.) ? sizeof(double) : 0.);
470  size = nCunks * (chunkOverhead + hist->GetChunkSize() * binSize);
471  // since THnSparse must keep track of all the stored bins, it stores a map that
472  // relates the compact bin coordinates (or a hash thereof) to a linear index
473  // this index determines in which chunk and therein at which position to find / store bin coordinate and content
474  size += nBinsFilled * 3 * sizeof(Long64_t); // hash, key, value; not sure why 3 are needed here...
475  }
476  return size / 1048576.;
477  }
478 
479  private:
480  // helper function to determine base element size of histograms (in bytes)
481  // the complicated casting gymnastics are needed here since we only store the interface types in the registry
482  template <typename T>
483  static int getBaseElementSize(T* ptr)
484  {
485  if constexpr (std::is_base_of_v<TH1, T> || std::is_base_of_v<THnSparse, T>) {
486  return getBaseElementSize<TArrayD, TArrayF, TArrayC, TArrayI, TArrayC, TArrayL>(ptr);
487  } else {
488  return getBaseElementSize<double, float, int, short, char, long>(ptr);
489  }
490  }
491 
492  template <typename T, typename Next, typename... Rest, typename P>
493  static int getBaseElementSize(P* ptr)
494  {
495  if (auto size = getBaseElementSize<T>(ptr)) {
496  return size;
497  }
498  return getBaseElementSize<Next, Rest...>(ptr);
499  }
500 
501  template <typename B, typename T>
502  static int getBaseElementSize(T* ptr)
503  {
504  if constexpr (std::is_base_of_v<THn, T>) {
505  if (dynamic_cast<THnT<B>*>(ptr)) {
506  return sizeof(B);
507  }
508  } else if constexpr (std::is_base_of_v<THnSparse, T>) {
509  if (dynamic_cast<THnSparseT<B>*>(ptr)) {
510  TDataMember* dm = B::Class()->GetDataMember("fArray");
511  return dm ? dm->GetDataType()->Size() : 0;
512  }
513  } else if constexpr (std::is_base_of_v<TH1, T>) {
514  if (auto arrayPtr = dynamic_cast<B*>(ptr)) {
515  return sizeof(arrayPtr->At(0));
516  }
517  }
518  return 0;
519  };

651  // clone an existing histogram and insert it into the registry
652  template <typename T>
653  void insertClone(const HistName& histName, const std::shared_ptr<T>& originalHist)
654  {
655  validateHistName(histName.str, histName.hash);
656  for (auto i = 0u; i < MAX_REGISTRY_SIZE; ++i) {
657  TObject* rawPtr = nullptr;
658  std::visit([&](const auto& sharedPtr) { rawPtr = sharedPtr.get(); }, mRegistryValue[imask(histName.idx + i)]);
659  if (!rawPtr) {
660  registerName(histName.str);
661  mRegistryKey[imask(histName.idx + i)] = histName.hash;
662  mRegistryValue[imask(histName.idx + i)] = std::shared_ptr<T>(static_cast<T*>(originalHist->Clone(histName.str)));
663  lookup += i;
664  return;
665  }
666  }
667  LOGF(FATAL, R"(Internal array of HistogramRegistry "%s" is full.)", mName);
668  }

676  {

685  throw runtime_error_f(R"(Could not find histogram "%s" in HistogramRegistry "%s"!)", histName.str, mName.data());
