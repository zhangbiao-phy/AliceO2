55  void process(aod::McCollision const& mccollision, aod::McParticles const& particlesMC)
56  {
57  int cPerCollision = 0;
58  int cBarPerCollision = 0;
59  int bPerCollision = 0;
60  int bBarPerCollision = 0;
61  double sumPxDau, sumPyDau, sumPzDau;
62  bool momentumCheck;
63  double pxDiff, pyDiff, pzDiff;
64 
65  //Particles and their decay checked in the second part of the task
66  std::array<int, 4> PDGArrayParticle = {411, 413, 421, 4122};
67  std::array<std::array<int, 3>, 4> arrPDGFinal = {{{211, 211, -321}, {211, 211, -321}, {-321, 211, 0}, {2212, -321, 211}}};
68  int counter[4] = {0, 0, 0, 0};
69  std::vector<int> listDaughters;
70 
71  for (auto& particle : particlesMC) {
72  int particlePdgCode = particle.pdgCode();
73  if (particle.mother0() < 0) {
74  continue;
75  }
76  auto mother = particlesMC.iteratorAt(particle.mother0());
77  if (particlePdgCode != mother.pdgCode()) {
78  switch (particlePdgCode) {
79  case 4:
80  cPerCollision++;
81  break;
82  case -4:
83  cBarPerCollision++;
84  break;
85  case 5:
86  bPerCollision++;
87  break;
88  case -5:
89  bBarPerCollision++;
90  break;
91  }
92  }
93 
94  sumPxDau = 0;
95  sumPyDau = 0;
96  sumPzDau = 0;
97  momentumCheck = 1;
98  listDaughters.clear();
99 
100  // Checking the decay of the particles and the momentum conservation
101  for (int iD = 0; iD < PDGArrayParticle.size(); iD++) {
102  if (std::abs(particlePdgCode) == PDGArrayParticle[iD]) {
103  RecoDecay::getDaughters(particlesMC, particle.globalIndex(), &listDaughters, arrPDGFinal[iD], -1);
104  int arrayPDGsize = arrPDGFinal[iD].size() - std::count(arrPDGFinal[iD].begin(), arrPDGFinal[iD].end(), 0);
105  if (listDaughters.size() == arrayPDGsize) {
106  counter[iD]++;
107  }
108  for (int i = 0; i < listDaughters.size(); i++) {
109  auto daughter = particlesMC.iteratorAt(listDaughters.at(i));
110  sumPxDau += daughter.px();
111  sumPyDau += daughter.py();
112  sumPzDau += daughter.pz();
113  }
114  pxDiff = particle.px() - sumPxDau;
115  pyDiff = particle.py() - sumPyDau;
116  pzDiff = particle.pz() - sumPzDau;
117  if (std::abs(pxDiff) > 0.001 || std::abs(pyDiff) > 0.001 || std::abs(pzDiff) > 0.001) {
118  momentumCheck = 0;
119  }
120  double pDiff = RecoDecay::P(pxDiff, pyDiff, pzDiff);
121  double ptDiff = RecoDecay::Pt(pxDiff, pyDiff);
122  //Filling histograms with per-component momentum conservation
123  registry.fill(HIST("hMomentumCheck"), momentumCheck);
124  registry.fill(HIST("hPxDiffMotherDaughterGen"), pxDiff);
125  registry.fill(HIST("hPyDiffMotherDaughterGen"), pyDiff);
126  registry.fill(HIST("hPzDiffMotherDaughterGen"), pzDiff);
127  registry.fill(HIST("hPdiffMotherDaughterGen"), pDiff);
128  registry.fill(HIST("hPtDiffMotherDaughterGen"), ptDiff);
129  }
130  }
131  } //end particles
132  registry.fill(HIST("hCountAverageC"), cPerCollision);
133  registry.fill(HIST("hCountAverageB"), bPerCollision);
134  registry.fill(HIST("hCountAverageCbar"), cBarPerCollision);
135  registry.fill(HIST("hCountAverageBbar"), bBarPerCollision);
136  registry.fill(HIST("hCouterPerCollisionDplus"), counter[0]);
137  registry.fill(HIST("hCouterPerCollisionDstar"), counter[1]);
138  registry.fill(HIST("hCouterPerCollisionDzero"), counter[2]);
139  registry.fill(HIST("hCouterPerCollisionLambdaC"), counter[3]);
140  }
141 };

165  void process(soa::Filtered<soa::Join<aod::HfCandProng2, aod::HFSelD0Candidate, aod::HfCandProng2MCRec>> const& candidates, aod::BigTracksMC const& tracks, aod::McParticles const& particlesMC)
166  {
167  int indexParticle = 0;
168  double pxDiff, pyDiff, pzDiff, pDiff;
169  double decayLength;
170  for (auto& candidate : candidates) {
171  if (!(candidate.hfflag() & 1 << D0ToPiK)) {
172  continue;
173  }
174  if (std::abs(candidate.flagMCMatchRec()) == 1 << D0ToPiK) {
175  indexParticle = RecoDecay::getMother(particlesMC, candidate.index0_as<aod::BigTracksMC>().mcParticle(), 421, true);
176  auto mother = particlesMC.iteratorAt(indexParticle);
177  registry.fill(HIST("histPt"), candidate.pt() - mother.pt());
178  registry.fill(HIST("histPx"), candidate.px() - mother.px());
179  registry.fill(HIST("histPy"), candidate.py() - mother.py());
180  registry.fill(HIST("histPz"), candidate.pz() - mother.pz());
181  //Compare Secondary vertex and decay length with MC
182  auto daughter0 = particlesMC.iteratorAt(mother.daughter0());
183  double vertexDau[3] = {daughter0.vx(), daughter0.vy(), daughter0.vz()};
184  double vertexMoth[3] = {mother.vx(), mother.vy(), mother.vz()};
185  decayLength = RecoDecay::distance(vertexMoth, vertexDau);
186 
187  registry.fill(HIST("histSecVx"), candidate.xSecondaryVertex() - vertexDau[0]);
188  registry.fill(HIST("histSecVy"), candidate.ySecondaryVertex() - vertexDau[1]);
189  registry.fill(HIST("histSecVz"), candidate.zSecondaryVertex() - vertexDau[2]);
190  registry.fill(HIST("histDecLen"), candidate.decayLength() - decayLength);
191  }
192  } //end loop on candidates
193  } //end process
194 };
