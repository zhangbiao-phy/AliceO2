23 std::vector<double> helpers::makeLogBinning(const int nbins, const double min, const double max)
24 {
25  assert(min > 0);
26  assert(min < max);
27 
28  std::vector<double> binLim(nbins + 1);
29 
30  const double expMax = std::log(max / min);
31  const double binWidth = expMax / nbins;
32 
33  binLim[0] = min;
34  binLim[nbins] = max;
35 
36  for (Int_t i = 1; i < nbins; ++i) {
37  binLim[i] = min * std::exp(i * binWidth);
38  }
39 
40  return binLim;
41 }
