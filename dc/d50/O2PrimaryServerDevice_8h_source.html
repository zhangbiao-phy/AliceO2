91  void InitTask() final
92  {
93  LOG(INFO) << "Init Server device ";
94 
95  // init sim config
96  auto& conf = o2::conf::SimConfig::Instance();
97  auto& vm = GetConfig()->GetVarMap();
98  conf.resetFromParsedMap(vm);
99  // output varmap
100  for (auto& keyvalue : vm) {
101  LOG(INFO) << "///// " << keyvalue.first << " " << keyvalue.second.value().type().name();
102  }
103  // update the parameters from an INI/JSON file, if given (overrides code-based version)
104  o2::conf::ConfigurableParam::updateFromFile(conf.getConfigFile());
105  // update the parameters from stuff given at command line (overrides file-based version)
106  o2::conf::ConfigurableParam::updateFromString(conf.getKeyValueString());
107 
108  mStack = new o2::data::Stack();
109  mStack->setExternalMode(true);
110 
111  // MC ENGINE
112  LOG(INFO) << "ENGINE SET TO " << vm["mcEngine"].as<std::string>();
113  // CHUNK SIZE
114  mChunkGranularity = vm["chunkSize"].as<unsigned int>();
115  LOG(INFO) << "CHUNK SIZE SET TO " << mChunkGranularity;
116 
117  // initial initial seed --> we should store this somewhere
118  mInitialSeed = vm["seed"].as<int>();
119  mInitialSeed = o2::utils::RngHelper::setGRandomSeed(mInitialSeed);
120  LOG(INFO) << "RNG INITIAL SEED " << mInitialSeed;
121 
122  mMaxEvents = conf.getNEvents();
123 
124  // need to make ROOT thread-safe since we use ROOT services in all places
125  ROOT::EnableThreadSafety();
126 
127  // lunch initialization of particle generator asynchronously
128  // so that we reach the RUNNING state of the server quickly
129  // and do not block here
130  mGeneratorThread = std::thread(&O2PrimaryServerDevice::initGenerator, this);
131 
132  // init pipe
133  auto pipeenv = getenv("ALICE_O2SIMSERVERTODRIVER_PIPE");
134  if (pipeenv) {
135  mPipeToDriver = atoi(pipeenv);
136  LOG(INFO) << "ASSIGNED PIPE HANDLE " << mPipeToDriver;
137  } else {
138  LOG(INFO) << "DID NOT FIND ENVIRONMENT VARIABLE TO INIT PIPE";
139  }

166  bool ConditionalRun() override
167  {
168  auto& channel = fChannels.at("primary-get").at(0);
169  std::unique_ptr<FairMQMessage> request(channel.NewMessage());
170  auto bytes = channel.Receive(request);
171  if (bytes < 0) {
172  LOG(ERROR) << "Some error occurred on socket during receive";
173  return true; // keep going
174  }
175  return HandleRequest(request, 0);
176  }

179  bool HandleRequest(FairMQMessagePtr& request, int /*index*/)
180  {
181  LOG(INFO) << "GOT A REQUEST WITH SIZE " << request->GetSize();
182  std::string requeststring(static_cast<char*>(request->GetData()), request->GetSize());
183 
184  if (requeststring.compare("configrequest") == 0) {
185  return HandleConfigRequest(request);
186  }
187 
188  else if (requeststring.compare("primrequest") != 0) {
189  LOG(INFO) << "unknown request\n";
190  return true;
191  }
192 
193  static int counter = 0;
194  if (counter >= mMaxEvents && mNeedNewEvent) {
195  return false;
196  }
197 
198  LOG(INFO) << "Received request for work ";
199  if (mNeedNewEvent) {
200  // we need a newly generated event now
201  if (mGeneratorThread.joinable()) {
202  mGeneratorThread.join();
203  }
204  mNeedNewEvent = false;
205  mPartCounter = 0;
206  counter++;
207  }
208 
209  auto& prims = mStack->getPrimaries();
210  auto numberofparts = (int)std::ceil(prims.size() / (1. * mChunkGranularity));
211  // number of parts should be at least 1 (even if empty)
212  numberofparts = std::max(1, numberofparts);
213 
214  o2::data::PrimaryChunk m;
215  o2::data::SubEventInfo i;
216  i.eventID = counter;
217  i.maxEvents = mMaxEvents;
218  i.part = mPartCounter + 1;
219  i.nparts = numberofparts;
220  i.seed = counter + mInitialSeed;
221  i.index = m.mParticles.size();
222  i.mMCEventHeader = mEventHeader;
223  m.mSubEventInfo = i;
224 
225  int endindex = prims.size() - mPartCounter * mChunkGranularity;
226  int startindex = prims.size() - (mPartCounter + 1) * mChunkGranularity;
227  if (startindex < 0) {
228  startindex = 0;
229  }
230  if (endindex < 0) {
231  endindex = 0;
232  }
233 
234  for (int index = startindex; index < endindex; ++index) {
235  m.mParticles.emplace_back(prims[index]);
236  }
237 
238  LOG(INFO) << "Sending " << m.mParticles.size() << " particles";
239  LOG(INFO) << "treating ev " << counter << " part " << i.part << " out of " << i.nparts;
240 
241  // feedback to driver if new event started
242  if (mPipeToDriver != -1 && i.part == 1) {
243  if (write(mPipeToDriver, &counter, sizeof(counter))) {
244  }
245  }
246 
247  mPartCounter++;
248  if (mPartCounter == numberofparts) {
249  mNeedNewEvent = true;
250  // start generation of a new event
251  mGeneratorThread = std::thread(&O2PrimaryServerDevice::generateEvent, this);
252  }
253 
254  TMessage* tmsg = new TMessage(kMESS_OBJECT);
255  tmsg->WriteObjectAny((void*)&m, TClass::GetClass("o2::data::PrimaryChunk"));
256 
257  auto free_tmessage = [](void* data, void* hint) { delete static_cast<TMessage*>(hint); };
258 
259  std::unique_ptr<FairMQMessage> message(
260  fTransportFactory->CreateMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
261 
262  // send answer
263  TStopwatch timer;
264  timer.Start();
265  auto code = Send(message, "primary-get", 0, 5000); // we introduce timeout in order not to block other requests
266  timer.Stop();
267  auto time = timer.CpuTime();
268  if (code > 0) {
269  LOG(INFO) << "Reply send in " << time << "s";
270  return true;
271  } else {
272  LOG(WARN) << "Sending process had problems. Return code : " << code << " time " << time << "s";
273  }
274  return true;
275  }
