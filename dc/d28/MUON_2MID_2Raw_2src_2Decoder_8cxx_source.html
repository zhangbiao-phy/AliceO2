57 void Decoder::process(gsl::span<const uint8_t> bytes)
58 {
60  clear();
61  o2::framework::RawParser parser(bytes.data(), bytes.size());
62  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
63  if (it.size() == 0) {
64  continue;
65  }
66  gsl::span<const uint8_t> payload(it.data(), it.size());
67  auto const* rdhPtr = reinterpret_cast<const o2::header::RDHAny*>(it.raw());
68  process(payload, *rdhPtr);
69  }
70 }
