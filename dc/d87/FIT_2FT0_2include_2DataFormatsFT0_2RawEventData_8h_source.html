187  std::vector<char> to_vector(bool tcm)
188  {
189  constexpr int CRUWordSize = 16;
190  const char padding[CRUWordSize] = {0};
191 
192  std::vector<char> result(size() * CRUWordSize);
193  char* out = result.data();
194  if (!tcm) {
195  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
196  out += EventHeader::PayloadSize;
197  LOG(DEBUG) << "write header words " << (int)mEventHeader.nGBTWords << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size();
198  if (mIsPadded) {
199  out += CRUWordSize - EventHeader::PayloadSize;
200  }
201  for (int i = 0; i < mEventHeader.nGBTWords; ++i) {
202  std::memcpy(out, &mEventData[2 * i], EventData::PayloadSizeFirstWord);
203  LOG(DEBUG) << " 1st word " << mEventData[2 * i].channelID << " charge " << mEventData[2 * i].charge << " time " << mEventData[2 * i].time << " out " << result.size();
204  out += EventData::PayloadSizeFirstWord;
205  std::memcpy(out, &mEventData[2 * i + 1], EventData::PayloadSizeSecondWord);
206  out += EventData::PayloadSizeSecondWord;
207  LOG(DEBUG) << " 2nd word " << mEventData[2 * i + 1].channelID << " charge " << mEventData[2 * i + 1].charge << " time " << mEventData[2 * i + 1].time << " out " << result.size();
208  if (mIsPadded) {
209  out += CRUWordSize - EventData::PayloadSizeSecondWord - EventData::PayloadSizeFirstWord;
210  }
211  }
212  } else {
213  // TCM data
214  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
215  out += EventHeader::PayloadSize;
216  LOG(DEBUG) << "write TCM header words " << (int)mEventHeader.nGBTWords << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size();
217  std::memcpy(out, &mTCMdata, sizeof(TCMdata));
218  out += sizeof(TCMdata);
219  LOG(DEBUG) << "write TCM words " << sizeof(mTCMdata) << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size() << " sum time A " << mTCMdata.timeA;
220  }
221 
222  return result;
223  }
224  void setIsPadded(bool isPadding128)
225  {
226  mIsPadded = isPadding128;
227  }
228 
229  public:
230  EventHeader mEventHeader;
231  EventData mEventData[Nchannels_PM];
232  TCMdata mTCMdata;
233  bool mIsPadded = true;
235  ClassDefNV(RawEventData, 2);
236 };

239 class DataPageWriter
240 {
241  std::vector<char> mBuffer;
242  int mNpacketsInBuffer = 0;
243  std::vector<std::vector<char>> mPages;
244  std::vector<int> mNpackets;
245  static constexpr int MAX_Page_size = 8192;
246 
247  public:
248  o2::header::RAWDataHeader mRDH;
249  void flush(std::ostream& str)
250  {
251  writePage();
252  mRDH.stop = 0;
253  for (int page = 0; page < int(mPages.size()); ++page) {
254  mRDH.memorySize = mPages[page].size() + mRDH.headerSize;
255  mRDH.offsetToNext = mRDH.memorySize;
256  mRDH.packetCounter = mNpackets[page];
257  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
258  str.write(mPages[page].data(), mPages[page].size());
259  mRDH.pageCnt++;
260  }
261  if (!mPages.empty()) {
262  mRDH.memorySize = mRDH.headerSize;
263  mRDH.offsetToNext = mRDH.memorySize;
264  mRDH.stop = 1;
265  mRDH.pageCnt++;
266  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
267  mPages.clear();
268  mNpackets.clear();
269  }
270  }
271 
272  void writePage()
273  {
274  if (mBuffer.size() == 0) {
275  return;
276  }
277  mPages.emplace_back(std::move(mBuffer));
278  LOG(DEBUG) << " writePage " << mBuffer.size();
279  mNpackets.push_back(mNpacketsInBuffer);
280  mNpacketsInBuffer = 0;
281  mBuffer.clear();
282  }
283 
284  void write(std::vector<char> const& new_data)
285  {
286  if (mBuffer.size() + new_data.size() + mRDH.headerSize > MAX_Page_size) {
287  LOG(DEBUG) << " write rest " << mBuffer.size() << " " << new_data.size() << " " << mRDH.headerSize;
288  writePage();
289  }
290  LOG(DEBUG) << " write vector " << new_data.size() << " buffer " << mBuffer.size() << " RDH " << mRDH.headerSize << " new data " << new_data.data();
291  mBuffer.insert(mBuffer.end(), new_data.begin(), new_data.end());
292  mNpacketsInBuffer++;
293  LOG(DEBUG) << " write vector end mBuffer.size " << mBuffer.size() << " mNpacketsInBuffer " << mNpacketsInBuffer << " newdtata " << new_data.size();
294  }
295 };
