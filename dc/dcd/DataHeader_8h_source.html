167 template <typename T, typename... Targs>
168 constexpr T String2(char c, Targs... Fargs)
169 {
170  // number of arguments has either to match the size of the type, or the last element is treated
171  // as '0' if missing
172  // TODO: this is a bit of a question, do we allow the number of characters to
173  // be smaller than the size of the type and just pad with 0 like in the case
174  // of a char array argument?
175  static_assert(sizeof...(Targs) == sizeof(T) + 1 ||
176  sizeof...(Targs) == sizeof(T),
177  "number of arguments does not match the uint type width");
178  return internal::String2__<T>(c, Fargs...);
179 }

185 template <typename T, std::size_t N, std::size_t pos = 0, bool suppressAssert = false>
186 constexpr T String2(const char (&str)[N])
187 {
188  static_assert(std::is_integral<T>::value, "Non integral types not compatible with String2<type>");
189  static_assert(N >= pos, "Position is larger than the length of string");
190  static_assert(suppressAssert || N - pos <= sizeof(T) + 1,
191  "String parameter is longer than the size of the data type");
192  // clang-format off
193  return ((T)str[0 + pos] |
194  (str[0 + pos] && sizeof(T) >= 2 ? ((T)str[1 + pos] << (sizeof(T) >= 2 ? 8 : 0) |
195  (str[1 + pos] && sizeof(T) >= 4 ? ((T)str[2 + pos] << (sizeof(T) >= 4 ? 16 : 0) |
196  (str[2 + pos] && sizeof(T) >= 4 ? ((T)str[3 + pos] << (sizeof(T) >= 4 ? 24 : 0) |
197  (str[3 + pos] && sizeof(T) >= 8 ? ((T)str[4 + pos] << (sizeof(T) >= 8 ? 32 : 0) |
198  (str[4 + pos] && sizeof(T) >= 8 ? ((T)str[5 + pos] << (sizeof(T) >= 8 ? 40 : 0) |
199  (str[5 + pos] && sizeof(T) >= 8 ? ((T)str[6 + pos] << (sizeof(T) >= 8 ? 48 : 0) |
200  (str[6 + pos] && sizeof(T) >= 8 ? ((T)str[7 + pos] << (sizeof(T) >= 8 ? 56 : 0))
201  : 0)) : 0)) : 0)) : 0)) : 0)) : 0)) : 0));
202  // clang-format on
203 }

210 struct DescriptorCompareTraits {
211  template <typename T, typename Length>
212  static bool compare(const T& lh, const T& rh, Length N)
213  {
214  return std::memcmp(lh.str, rh.str, N) == 0;
215  }
216  template <typename T, typename Length>
217  static bool lessThen(const T& lh, const T& rh, Length N)
218  {
219  return std::memcmp(lh.str, rh.str, N) < 0;
220  }
221 };
222 template <>
223 struct DescriptorCompareTraits<1> {
224  template <typename T, typename Length>
225  static bool compare(const T& lh, const T& rh, Length)
226  {
227  return lh.itg[0] == rh.itg[0];
228  }
229  template <typename T, typename Length>
230  static bool lessThen(const T& lh, const T& rh, Length)
231  {
232  return lh.itg[0] < rh.itg[0];

236 struct DescriptorCompareTraits<2> {
237  template <typename T, typename Length>
238  static bool compare(const T& lh, const T& rh, Length)
239  {
240  return (lh.itg[0] == rh.itg[0]) && (lh.itg[1] == rh.itg[1]);
241  }
242  template <typename T, typename Length>
243  static bool lessThen(const T& lh, const T& rh, Length)
244  {
245  return std::tie(lh.itg[0], lh.itg[1]) < std::tie(rh.itg[0], rh.itg[1]);

304  void runtimeInit(const char* string, short length = -1)
305  {
306  char* target = str;
307  char* targetEnd = target;
308  if (length >= 0 && length < (int)N) {
309  targetEnd += length;
310  } else {
311  targetEnd += N;
312  }
313  const char* source = string;
314  for (; source != nullptr && target < targetEnd && *source != 0; ++target, ++source) {
315  *target = *source;
316  }
317  targetEnd = str + N;
318  for (; target < targetEnd; ++target) {
319  *target = 0;
320  }
321  // require the string to be not longer than the descriptor size
322  if (source != nullptr && (*source == 0 || (length >= 0 && length <= (int)N))) {
323  } else {
324  throw std::invalid_argument("argument must not be longer than descriptor size");
325  }
326  }

328  bool operator==(const Descriptor& other) const { return DescriptorCompareTraits<arraySize>::compare(*this, other, N); }
329  bool operator<(const Descriptor& other) const { return DescriptorCompareTraits<arraySize>::lessThen(*this, other, N); }
330  bool operator!=(const Descriptor& other) const { return not this->operator==(other); }
331

341  template <typename T>
342  std::enable_if_t<std::is_same<T, std::string>::value == true, T> as() const
343  {
344  // backward search to find first non-zero char
345  // FIXME: can optimize this by using the int value to start at e.g. size/2
346  // if the upper part of the string is just zeros.
347  size_t len = size;
348  while (len > 1 && str[len - 1] == 0) {
349  --len;
350  }
351  std::string ret(str, len);
352  return std::move(ret);
353  }
354  // print function needs to be implemented for every derivation
355  void print() const
356  {
357  // eventually terminate string before printing
358  PrinterPolicy printer;
359  printer(str);
360  };

499 template <typename HeaderType, typename std::enable_if_t<std::is_pointer<HeaderType>::value, int> = 0>
500 auto get(const o2::byte* buffer, size_t /*len*/ = 0)
501 {
502  using HeaderConstPtrType = const typename std::remove_pointer<HeaderType>::type*;
503  using HeaderValueType = typename std::remove_pointer<HeaderType>::type;
504 
505  const BaseHeader* current = BaseHeader::get(buffer);
506  if (!current) {
507  return HeaderConstPtrType{nullptr};
508  }
509  if (current->description == HeaderValueType::sHeaderType) {
510  // FIXME: We need to think how to handle multiple versions.
511  // For the moment we require the version to exactly match the expected version, which
512  // is part of the header definition. The check function 'sanityCheck' will throw
513  // otherwise, we keep the code related to the exception outside the header file.
514  // Note: Can not check on size because the O2 data model requires variable size headers
515  // to be supported.
516  if (current->sanityCheck(HeaderValueType::sVersion)) {
517  return reinterpret_cast<HeaderConstPtrType>(current);
518  }
519  }
520  auto* prev = current;
521  while ((current = current->next())) {
522  prev = current;
523  if (current->description == HeaderValueType::sHeaderType) {
524  if (current->sanityCheck(HeaderValueType::sVersion)) {
525  return reinterpret_cast<HeaderConstPtrType>(current);
526  }
527  }
528  }
529  // if a next header was expected but not found, the header stack is inconsistent
530  // either the flag is wrongly set, or the size of the current header is wrong so
531  // that the BaseHeader information is not found at the expected offset.
532  if (prev->flagsNextHeader) {
533  prev->throwInconsistentStackError();
534  }
535  return HeaderConstPtrType{nullptr};
536 }

545 // utilities for converting strings to ints - tries to mimick the stadard API, but the result is
546 // strongly typed - can be used for fixed width and short unsigned integer types unlike the standard
547 // utilities.
548 template <typename T>
549 T strtoui(const char* str, char** str_end, int base) noexcept
550 {
551  static_assert(std::numeric_limits<T>::is_integer && !std::numeric_limits<T>::is_signed,
552  "Target must be an unsigned integer type");
553  // standard lib gives us 2 specializations of strtouX, so let's use them both where appropriate
554  if constexpr (sizeof(T) <= sizeof(uint32_t)) {
555  unsigned long res = strtoul(str, str_end, base);
556  if (res > std::numeric_limits<T>::max()) {
557  errno = ERANGE;
558  return std::numeric_limits<T>::max();
559  } else {
560  return static_cast<T>(res);
561  };
562  } else {
563  unsigned long long res = strtoull(str, str_end, base);
564  if (res > std::numeric_limits<T>::max()) {
565  errno = ERANGE;
566  return std::numeric_limits<T>::max();
567  } else {
568  return static_cast<T>(res);
569  };
570  }
571 };

573 template <typename T>
574 T stoui(const std::string& str, size_t* pos = nullptr, int base = 10)
575 {
576  static_assert(std::numeric_limits<T>::is_integer && !std::numeric_limits<T>::is_signed,
577  "Target must be an unsigned integer type");
578  // standard lib gives us 2 specializations of stouX, so let's use them both where appropriate
579  if constexpr (sizeof(T) <= sizeof(uint32_t)) {
580  unsigned long res = std::stoul(str, pos, base);
581  if (res > std::numeric_limits<T>::max()) {
582  throw std::out_of_range("result does not fit in target type");
583  } else {
584  return static_cast<T>(res);
585  };
586  } else {
587  unsigned long long res = std::stoull(str, pos, base);
588  if (res > std::numeric_limits<T>::max()) {
589  throw std::out_of_range("result does not fit in target type");
590  } else {
591  return static_cast<T>(res);
592  };
593  }
