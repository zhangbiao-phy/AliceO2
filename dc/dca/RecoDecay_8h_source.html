227  template <typename T, typename U, typename V>
228  static double CPA(const T& posPV, const U& posSV, const array<V, 3>& mom)
229  {
230  // CPA = (l . p)/(|l| |p|)
231  auto lineDecay = array{posSV[0] - posPV[0], posSV[1] - posPV[1], posSV[2] - posPV[2]};
232  auto cos = dotProd(lineDecay, mom) / std::sqrt(mag2(lineDecay) * mag2(mom));
233  if (cos < -1.) {
234  return -1.;
235  }
236  if (cos > 1.) {
237  return 1.;
238  }
239  return cos;
240  }

283  template <typename T, typename U, typename V>
284  static double CosThetaStar(const array<array<T, 3>, 2>& arrMom, const array<U, 2>& arrMass, V mTot, int iProng)
285  {
286  auto pVecTot = PVec(arrMom[0], arrMom[1]); // momentum of the mother particle
287  auto pTot = P(pVecTot); // magnitude of the momentum of the mother particle
288  auto eTot = E(pTot, mTot); // energy of the mother particle
289  auto gamma = eTot / mTot; // γ, Lorentz gamma factor of the mother particle
290  auto beta = pTot / eTot; // β, velocity of the mother particle
291  auto pStar = std::sqrt(sq(sq(mTot) - sq(arrMass[0]) - sq(arrMass[1])) - sq(2 * arrMass[0] * arrMass[1])) / (2 * mTot); // p*, prong momentum in the rest frame of the mother particle
292  // p* = √[(M^2 - m1^2 - m2^2)^2 - 4 m1^2 m2^2]/2M
293  // Lorentz transformation of the longitudinal momentum of the prong into the detector frame:
294  // p_L,i = γ (p*_L,i + β E*_i)
295  // p*_L,i = p_L,i/γ - β E*_i
296  // cos(θ*_i) = (p_L,i/γ - β E*_i)/p*
297  return (dotProd(arrMom[iProng], pVecTot) / (pTot * gamma) - beta * E(pStar, arrMass[iProng])) / pStar;
298  }

416  template <std::size_t N, typename T, typename U>
417  static double M2(const array<array<T, 3>, N>& arrMom, const array<U, N>& arrMass)
418  {
419  array<double, 3> momTotal{0., 0., 0.}; // candidate momentum vector
420  double energyTot{0.}; // candidate energy
421  for (auto iProng = 0; iProng < N; ++iProng) {
422  for (auto iMom = 0; iMom < 3; ++iMom) {
423  momTotal[iMom] += arrMom[iProng][iMom];
424  } // loop over momentum components
425  energyTot += E(arrMom[iProng], arrMass[iProng]);
426  } // loop over prongs
427  return energyTot * energyTot - P2(momTotal);
428  }

448  template <typename T, typename U, typename V>
449  static double ImpParXY(const T& point, const U& posSV, const array<V, 3>& mom)
450  {
451  // Ported from AliAODRecoDecay::ImpParXY
452  auto flightLineXY = array{posSV[0] - point[0], posSV[1] - point[1]};
453  auto k = dotProd(flightLineXY, array{mom[0], mom[1]}) / Pt2(mom);
454  auto dx = flightLineXY[0] - k * (double)mom[0];
455  auto dy = flightLineXY[1] - k * (double)mom[1];
456  auto absImpPar = sqrtSumOfSquares(dx, dy);
457  auto flightLine = array{posSV[0] - point[0], posSV[1] - point[1], posSV[2] - point[2]};
458  auto cross = crossProd(mom, flightLine);
459  return (cross[2] > 0. ? absImpPar : -1. * absImpPar);
460  }

471  template <std::size_t N, std::size_t M, typename T, typename U, typename V, typename W, typename X, typename Y>
472  static double normImpParMeasMinusExpProng(T decLenXY, U errDecLenXY, const array<V, N>& momMother, W impParProng,
473  X errImpParProng, const array<Y, M>& momProng)
474  {
475  // Ported from AliAODRecoDecayHF::Getd0MeasMinusExpProng adding normalization directly in the function
476  auto sinThetaP = ((double)momProng[0] * (double)momMother[1] - (double)momProng[1] * (double)momMother[0]) /
477  (Pt(momProng) * Pt(momMother));
478  auto diff = impParProng - (double)decLenXY * sinThetaP;
479  auto errImpParExpProng = (double)errDecLenXY * sinThetaP;
480  auto errDiff = sqrtSumOfSquares(errImpParProng, errImpParExpProng);
481  return (errDiff > 0. ? diff / errDiff : 0.);
482  }

540  template <typename T>
541  static int getMother(const T& particlesMC, const typename T::iterator& particle, int PDGMother, bool acceptAntiParticles = false, int8_t* sign = nullptr)
542  {
543  int8_t sgn = 0; // 1 if the expected mother is particle, -1 if antiparticle (w.r.t. PDGMother)
544  int indexMother = -1; // index of the final matched mother, if found
545  auto particleMother = particle; // Initialise loop over mothers.
546  int stage = 0; // mother tree level (just for debugging)
547  if (sign) {
548  *sign = sgn;
549  }
550  while (particleMother.mother0() > -1) {
551  auto indexMotherTmp = particleMother.mother0();
552  particleMother = particlesMC.iteratorAt(indexMotherTmp);
553  // Check mother's PDG code.
554  auto PDGParticleIMother = particleMother.pdgCode(); // PDG code of the mother
555  //printf("getMother: ");
556  //for (int i = stage; i < 0; i++) // Indent to make the tree look nice.
557  // printf(" ");
558  //printf("Stage %d: Mother PDG: %d\n", stage, PDGParticleIMother);
559  if (PDGParticleIMother == PDGMother) { // exact PDG match
560  sgn = 1;
561  indexMother = indexMotherTmp;
562  break;
563  } else if (acceptAntiParticles && PDGParticleIMother == -PDGMother) { // antiparticle PDG match
564  sgn = -1;
565  indexMother = indexMotherTmp;
566  break;
567  }
568  stage--;
569  }
570  if (sign) {
571  *sign = sgn;
572  }
573  return indexMother;
574  }

585  template <std::size_t N, typename T>
586  static void getDaughters(const T& particlesMC,
587  int index,
588  std::vector<int>* list,
589  const array<int, N>& arrPDGFinal,
590  int8_t depthMax = -1,
591  int8_t stage = 0)
592  {
593  if (index <= -1) {
594  //Printf("getDaughters: Error: No particle: index %d", index);
595  return;
596  }
597  if (!list) {
598  //Printf("getDaughters: Error: No list!");
599  return;
600  }
601  // Get the particle.
602  auto particle = particlesMC.iteratorAt(index);
603  bool isFinal = false; // Flag to indicate the end of recursion
604  if (depthMax > -1 && stage >= depthMax) { // Maximum depth has been reached (or exceeded).
605  isFinal = true;
606  }
607  // Get the range of daughter indices.
608  int indexDaughterFirst = particle.daughter0();
609  int indexDaughterLast = particle.daughter1();
610  // Check whether there are any daughters.
611  if (!isFinal && indexDaughterFirst <= -1 && indexDaughterLast <= -1) {
612  // If the original particle has no daughters, we do nothing and exit.
613  if (stage == 0) {
614  //Printf("getDaughters: No daughters of %d", index);
615  return;
616  }
617  // If this is not the original particle, we are at the end of this branch and this particle is final.
618  isFinal = true;
619  }
620  auto PDGParticle = std::abs(particle.pdgCode());
621  // If this is not the original particle, check its PDG code.
622  if (!isFinal && stage > 0) {
623  // If the particle has daughters but is considered to be final, we label it as final.
624  for (auto PDGi : arrPDGFinal) {
625  if (PDGParticle == std::abs(PDGi)) { // Accept antiparticles.
626  isFinal = true;
627  break;
628  }
629  }
630  }
631  // If the particle is labelled as final, we add this particle in the list of final daughters and exit.
632  if (isFinal) {
633  //printf("getDaughters: ");
634  //for (int i = 0; i < stage; i++) // Indent to make the tree look nice.
635  // printf(" ");
636  //printf("Stage %d: Adding %d (PDG %d) as final daughter.\n", stage, index, PDGParticle);
637  list->push_back(index);
638  return;
639  }
640  // If we are here, we have to follow the daughter tree.
641  //printf("getDaughters: ");
642  //for (int i = 0; i < stage; i++) // Indent to make the tree look nice.
643  // printf(" ");
644  //printf("Stage %d: %d (PDG %d) -> %d-%d\n", stage, index, PDGParticle, indexDaughterFirst, indexDaughterLast);
645  // Call itself to get daughters of daughters recursively.
646  // Get daughters of the first daughter.
647  if (indexDaughterFirst > -1) {
648  getDaughters(particlesMC, indexDaughterFirst, list, arrPDGFinal, depthMax, stage + 1);
649  }
650  // Get daughters of the daughters in between if any.
651  // Daughter indices are supposed to be consecutive and in increasing order.
652  // Reverse order means two daughters.
653  if (indexDaughterFirst > -1 && indexDaughterLast > -1) {
654  for (auto iD = indexDaughterFirst + 1; iD < indexDaughterLast; ++iD) {
655  getDaughters(particlesMC, iD, list, arrPDGFinal, depthMax, stage + 1);
656  }
657  }
658  // Get daughters of the last daughter if different from the first one.
659  // Same indices indicate a single daughter.
660  if (indexDaughterLast > -1 && indexDaughterLast != indexDaughterFirst) {
661  getDaughters(particlesMC, indexDaughterLast, list, arrPDGFinal, depthMax, stage + 1);
662  }
663  }

674  template <std::size_t N, typename T, typename U>
675  static int getMatchedMCRec(const T& particlesMC, const array<U, N>& arrDaughters, int PDGMother, array<int, N> arrPDGDaughters, bool acceptAntiParticles = false, int8_t* sign = nullptr, int depthMax = 1)
676  {
677  //Printf("MC Rec: Expected mother PDG: %d", PDGMother);
678  int8_t sgn = 0; // 1 if the expected mother is particle, -1 if antiparticle (w.r.t. PDGMother)
679  int indexMother = -1; // index of the mother particle
680  std::vector<int> arrAllDaughtersIndex; // vector of indices of all daughters of the mother of the first provided daughter
681  array<int, N> arrDaughtersIndex; // array of indices of provided daughters
682  if (sign) {
683  *sign = sgn;
684  }
685  // Loop over decay candidate prongs
686  for (auto iProng = 0; iProng < N; ++iProng) {
687  auto particleI = arrDaughters[iProng].mcParticle(); // ith daughter particle
688  arrDaughtersIndex[iProng] = particleI.globalIndex();
689  // Get the list of daughter indices from the mother of the first prong.
690  if (iProng == 0) {
691  // Get the mother index and its sign.
692  // PDG code of the first daughter's mother determines whether the expected mother is a particle or antiparticle.
693  indexMother = getMother(particlesMC, particleI, PDGMother, acceptAntiParticles, &sgn);
694  // Check whether mother was found.
695  if (indexMother <= -1) {
696  //Printf("MC Rec: Rejected: bad mother index or PDG");
697  return -1;
698  }
699  //Printf("MC Rec: Good mother: %d", indexMother);
700  auto particleMother = particlesMC.iteratorAt(indexMother);
701  int indexDaughterFirst = particleMother.daughter0(); // index of the first direct daughter
702  int indexDaughterLast = particleMother.daughter1(); // index of the last direct daughter
703  // Check the daughter indices.
704  if (indexDaughterFirst <= -1 && indexDaughterLast <= -1) {
705  //Printf("MC Rec: Rejected: bad daughter index range: %d-%d", indexDaughterFirst, indexDaughterLast);
706  return -1;
707  }
708  // Check that the number of direct daughters is not larger than the number of expected final daughters.
709  if (indexDaughterFirst > -1 && indexDaughterLast > -1 && indexDaughterLast - indexDaughterFirst + 1 > N) {
710  //Printf("MC Rec: Rejected: too many direct daughters: %d (expected %d final)", indexDaughterLast - indexDaughterFirst + 1, N);
711  return -1;
712  }
713  // Get the list of actual final daughters.
714  getDaughters(particlesMC, indexMother, &arrAllDaughtersIndex, arrPDGDaughters, depthMax);
715  //printf("MC Rec: Mother %d has %d final daughters:", indexMother, arrAllDaughtersIndex.size());
716  //for (auto i : arrAllDaughtersIndex) {
717  // printf(" %d", i);
718  //}
719  //printf("\n");
720  // Check whether the number of actual final daughters is equal to the number of expected final daughters (i.e. the number of provided prongs).
721  if (arrAllDaughtersIndex.size() != N) {
722  //Printf("MC Rec: Rejected: incorrect number of final daughters: %d (expected %d)", arrAllDaughtersIndex.size(), N);
723  return -1;
724  }
725  }
726  // Check that the daughter is in the list of final daughters.
727  // (Check that the daughter is not a stepdaughter, i.e. particle pointing to the mother while not being its daughter.)
728  bool isDaughterFound = false; // Is the index of this prong among the remaining expected indices of daughters?
729  for (auto iD = 0; iD < arrAllDaughtersIndex.size(); ++iD) {
730  if (arrDaughtersIndex[iProng] == arrAllDaughtersIndex[iD]) {
731  arrAllDaughtersIndex[iD] = -1; // Remove this index from the array of expected daughters. (Rejects twin daughters, i.e. particle considered twice as a daughter.)
732  isDaughterFound = true;
733  break;
734  }
735  }
736  if (!isDaughterFound) {
737  //Printf("MC Rec: Rejected: bad daughter index: %d not in the list of final daughters", arrDaughtersIndex[iProng]);
738  return -1;
739  }
740  // Check daughter's PDG code.
741  auto PDGParticleI = particleI.pdgCode(); // PDG code of the ith daughter
742  //Printf("MC Rec: Daughter %d PDG: %d", iProng, PDGParticleI);
743  bool isPDGFound = false; // Is the PDG code of this daughter among the remaining expected PDG codes?
744  for (auto iProngCp = 0; iProngCp < N; ++iProngCp) {
745  if (PDGParticleI == sgn * arrPDGDaughters[iProngCp]) {
746  arrPDGDaughters[iProngCp] = 0; // Remove this PDG code from the array of expected ones.
747  isPDGFound = true;
748  break;
749  }
750  }
751  if (!isPDGFound) {
752  //Printf("MC Rec: Rejected: bad daughter PDG: %d", PDGParticleI);
753  return -1;
754  }
755  }
756  //Printf("MC Rec: Accepted: m: %d", indexMother);

786  template <std::size_t N, typename T, typename U>
787  static bool isMatchedMCGen(const T& particlesMC, const U& candidate, int PDGParticle, array<int, N> arrPDGDaughters, bool acceptAntiParticles = false, int8_t* sign = nullptr, int depthMax = 1)
788  {
789  //Printf("MC Gen: Expected particle PDG: %d", PDGParticle);
790  int8_t sgn = 0; // 1 if the expected mother is particle, -1 if antiparticle (w.r.t. PDGParticle)
791  if (sign) {
792  *sign = sgn;
793  }
794  // Check the PDG code of the particle.
795  auto PDGCandidate = candidate.pdgCode();
796  //Printf("MC Gen: Candidate PDG: %d", PDGCandidate);
797  if (PDGCandidate == PDGParticle) { // exact PDG match
798  sgn = 1;
799  } else if (acceptAntiParticles && PDGCandidate == -PDGParticle) { // antiparticle PDG match
800  sgn = -1;
801  } else {
802  //Printf("MC Gen: Rejected: bad particle PDG: %s%d != %d", acceptAntiParticles ? "abs " : "", PDGCandidate, std::abs(PDGParticle));
803  return false;
804  }
805  // Check the PDG codes of the decay products.
806  if (N > 0) {
807  //Printf("MC Gen: Checking %d daughters", N);
808  std::vector<int> arrAllDaughtersIndex; // vector of indices of all daughters
809  int indexDaughterFirst = candidate.daughter0(); // index of the first direct daughter
810  int indexDaughterLast = candidate.daughter1(); // index of the last direct daughter
811  // Check the daughter indices.
812  if (indexDaughterFirst <= -1 && indexDaughterLast <= -1) {
813  //Printf("MC Gen: Rejected: bad daughter index range: %d-%d", indexDaughterFirst, indexDaughterLast);
814  return false;
815  }
816  // Check that the number of direct daughters is not larger than the number of expected final daughters.
817  if (indexDaughterFirst > -1 && indexDaughterLast > -1 && indexDaughterLast - indexDaughterFirst + 1 > N) {
818  //Printf("MC Gen: Rejected: too many direct daughters: %d (expected %d final)", indexDaughterLast - indexDaughterFirst + 1, N);
819  return false;
820  }
821  // Get the list of actual final daughters.
822  getDaughters(particlesMC, candidate.globalIndex(), &arrAllDaughtersIndex, arrPDGDaughters, depthMax);
823  //printf("MC Gen: Mother %d has %d final daughters:", candidate.globalIndex(), arrAllDaughtersIndex.size());
824  //for (auto i : arrAllDaughtersIndex) {
825  // printf(" %d", i);
826  //}
827  //printf("\n");
828  // Check whether the number of final daughters is equal to the required number.
829  if (arrAllDaughtersIndex.size() != N) {
830  //Printf("MC Gen: Rejected: incorrect number of final daughters: %d (expected %d)", arrAllDaughtersIndex.size(), N);
831  return false;
832  }
833  // Check daughters' PDG codes.
834  for (auto indexDaughterI : arrAllDaughtersIndex) {
835  auto candidateDaughterI = particlesMC.iteratorAt(indexDaughterI); // ith daughter particle
836  auto PDGCandidateDaughterI = candidateDaughterI.pdgCode(); // PDG code of the ith daughter
837  //Printf("MC Gen: Daughter %d PDG: %d", indexDaughterI, PDGCandidateDaughterI);
838  bool isPDGFound = false; // Is the PDG code of this daughter among the remaining expected PDG codes?
839  for (auto iProngCp = 0; iProngCp < N; ++iProngCp) {
840  if (PDGCandidateDaughterI == sgn * arrPDGDaughters[iProngCp]) {
841  arrPDGDaughters[iProngCp] = 0; // Remove this PDG code from the array of expected ones.
842  isPDGFound = true;
843  break;
844  }
845  }
846  if (!isPDGFound) {
847  //Printf("MC Gen: Rejected: bad daughter PDG: %d", PDGCandidateDaughterI);
848  return false;
849  }
850  }
851  }
852  //Printf("MC Gen: Accepted: m: %d", candidate.globalIndex());
853  if (sign) {
854  *sign = sgn;
855  }
856  return true;
857  }
