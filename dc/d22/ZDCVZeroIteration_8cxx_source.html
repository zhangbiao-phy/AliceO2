21 // This example accesses the collisions and the related FV0A information.
22 // Note that one has to subscribe to aod::FV0As const& to load
23 // the relevant data even if you access the data itself through collisionMatched.fv0a()
24 // Here the "sparse" matcher is used which means, there can be collisions without FV0A information
25 // To find out, collisionMatched.has_fv0a() has to be called. Otherwise collisionMatched.fv0a() will fail.
26 // NOTE: subscribing to Collisions separately will lead to a circular dependency due to forwarding
27 
28 struct IterateV0 {
29  void process(aod::CollisionMatchedRun2Sparse const& collisionMatched, aod::FV0As const&)
30  {
31  LOGF(INFO, "Vertex = %f", collisionMatched.posZ());
32  if (collisionMatched.has_fv0a()) {
33  auto v0a = collisionMatched.fv0a();
34  LOGF(info, "V0A: %f %f", v0a.amplitude()[0], v0a.amplitude()[1]);
35  } else {
36  LOGF(INFO, "No V0A info");
37  }
38  }
39 };

41 // This example is identical to IterateV0, but uses the exclusive match. This means that collisions where any
42 // of the tables asked for in Run2MatchedExclusive (see AnalysisDataModel.h) are missing are not there.
43 // Therefore, the syntax is more complicated because we cannot join against Collision
44 // (the tables have different number of entries)
45 // Only to be used if one is sure that all your events have the desired information
46 struct IterateV0Exclusive {
47  void process(aod::Run2MatchedExclusive::iterator const& matcher, aod::Collisions const&, aod::FV0As const&)
48  {
49  LOGF(INFO, "Vertex = %f", matcher.collision().posZ());
50  auto fv0a = matcher.fv0a();
51  LOGF(info, "V0: %f %f", fv0a.amplitude()[0], fv0a.amplitude()[1]);
52  }
53 };

55 // This example builds on IterateV0 and in addition accesses also the tracks grouped to the specific collision.
