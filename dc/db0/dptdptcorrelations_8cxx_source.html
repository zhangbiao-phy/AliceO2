229 inline void AcceptTrack(aod::TrackData const& track, bool& asone, bool& astwo)
230 {
231 
232  asone = false;
233  astwo = false;
234 
235  /* TODO: incorporate a mask in the scanned tracks table for the rejecting track reason */
236  if (matchTrackType(track)) {
237  if (ptlow < track.pt() and track.pt() < ptup and etalow < track.eta() and track.eta() < etaup) {
238  if (((track.charge() > 0) and (trackonecharge > 0)) or ((track.charge() < 0) and (trackonecharge < 0))) {
239  asone = true;
240  }
241  if (((track.charge() > 0) and (tracktwocharge > 0)) or ((track.charge() < 0) and (tracktwocharge < 0))) {
242  astwo = true;
243  }
244  }
245  }

272 inline int GetEtaPhiIndex(aod::FilteredTrackData const& t)
273 {
274  int etaix = int((t.eta() - etalow) / etabinwidth);
275  /* consider a potential phi origin shift */
276  float phi = GetShiftedPhi(t.phi());
277  int phiix = int((phi - philow) / phibinwidth);
278  return etaix * phibins + phiix;

393  void process(aod::CollisionEvSelCent const& collision, soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra, aod::TracksExtended, aod::TrackSelection> const& ftracks)
394  {
395  using namespace filteranalyistask;
396 
397  // LOGF(INFO,"New collision with %d filtered tracks", ftracks.size());
398  fhCentMultB->Fill(collision.centV0M());
399  fhVertexZB->Fill(collision.posZ());
400  bool acceptedevent = false;
401  float centormult = -100.0;
402  if (IsEvtSelected(collision, centormult)) {
403  acceptedevent = true;
404  fhCentMultA->Fill(collision.centV0M());
405  fhVertexZA->Fill(collision.posZ());
406  // LOGF(INFO,"New accepted collision with %d filtered tracks", ftracks.size());
407 
408  for (auto& track : ftracks) {
409  /* before track selection */
410  fhPtB->Fill(track.pt());
411  fhEtaB->Fill(track.eta());
412  fhPhiB->Fill(track.phi());
413  fhEtaVsPhiB->Fill(track.phi(), track.eta());
414  fhPtVsEtaB->Fill(track.eta(), track.pt());
415  if (track.charge() > 0) {
416  fhPtPosB->Fill(track.pt());
417  } else {
418  fhPtNegB->Fill(track.pt());
419  }
420 
421  /* track selection */
422  /* tricky because the boolean columns issue */
423  bool asone, astwo;
424  AcceptTrack(track, asone, astwo);
425  if (asone or astwo) {
426  /* the track has been accepted */
427  fhPtA->Fill(track.pt());
428  fhEtaA->Fill(track.eta());
429  fhPhiA->Fill(track.phi());
430  fhEtaVsPhiA->Fill(track.phi(), track.eta());
431  fhPtVsEtaA->Fill(track.eta(), track.pt());
432  if (track.charge() > 0) {
433  fhPtPosA->Fill(track.pt());
434  } else {
435  fhPtNegA->Fill(track.pt());
436  }
437  }
438  scannedtracks((uint8_t)asone, (uint8_t)astwo);
439  }
440  } else {
441  for (auto& track : ftracks) {
442  scannedtracks((uint8_t) false, (uint8_t) false);
443  }
444  }
445  acceptedevents((uint8_t)acceptedevent, centormult);
446  }
