103 struct TCMdata {
104  static constexpr size_t PayloadSize = 16; //should be equal to 10

106  static constexpr int MinNelements = 1;

112  orA : 1,

126  void fillTrigger(Triggers& trg)
127  {
128  trg.triggersignals = ((bool)orA << Triggers::bitA) |
129  ((bool)orC << Triggers::bitC) |
130  ((bool)vertex << Triggers::bitVertex) |
131  ((bool)cen << Triggers::bitCen) |
132  ((bool)sCen << Triggers::bitSCen);
133  trg.nChanA = (int8_t)nChanA;
134  trg.nChanC = (int8_t)nChanC;
135  trg.amplA = (int32_t)amplA;
136  trg.amplC = (int32_t)amplC;
137  trg.timeA = (int16_t)timeA;
138  trg.timeC = (int16_t)timeC;
139  }
140  void print() const;
141 };

168  std::vector<char> to_vector(bool tcm)
169  {
170  constexpr int CRUWordSize = 16;
171 
172  std::vector<char> result(size() * CRUWordSize);
173  char* out = result.data();
174  if (!tcm) {
175  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
176  out += EventHeader::PayloadSize;
177  LOG(DEBUG) << " Write PM header: nWords: " << (int)mEventHeader.nGBTWords
178  << " orbit: " << int(mEventHeader.orbit)
179  << " BC: " << int(mEventHeader.bc)
180  << " size: " << result.size();
181  printHexEventHeader();
182  out += CRUWordSize - EventHeader::PayloadSize; // Padding enabled
183 
184  for (uint64_t i = 0; i < mEventHeader.nGBTWords; ++i) {
185  std::memcpy(out, &mEventData[2 * i], EventData::PayloadSizeFirstWord);
186  out += EventData::PayloadSizeFirstWord;
187  LOG(DEBUG) << " 1st word: Ch: " << std::setw(2) << mEventData[2 * i].channelID
188  << " charge: " << std::setw(4) << mEventData[2 * i].charge
189  << " time: " << std::setw(4) << mEventData[2 * i].time;
190  std::memcpy(out, &mEventData[2 * i + 1], EventData::PayloadSizeSecondWord);
191  out += EventData::PayloadSizeSecondWord;
192  LOG(DEBUG) << " 2nd word: Ch: " << std::setw(2) << mEventData[2 * i + 1].channelID
193  << " charge: " << std::setw(4) << mEventData[2 * i + 1].charge
194  << " time: " << std::setw(4) << mEventData[2 * i + 1].time;
195 
196  out += CRUWordSize - EventData::PayloadSizeSecondWord - EventData::PayloadSizeFirstWord;
197  printHexEventData(i);
198  }
199  } else {
200  // TCM data
201  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
202  out += EventHeader::PayloadSize;
203  LOG(DEBUG) << " Write TCM header: nWords: " << (int)mEventHeader.nGBTWords
204  << " orbit: " << int(mEventHeader.orbit)
205  << " BC: " << int(mEventHeader.bc)
206  << " size: " << result.size();
207  std::memcpy(out, &mTCMdata, sizeof(TCMdata));
208  out += sizeof(TCMdata);
209  // TODO: No TCM payload printing until the meaning of trigger bits and other flags is clarified
210  }
211  return result;
212  }

217  TCMdata mTCMdata;
