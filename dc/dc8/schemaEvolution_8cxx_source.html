34 // Producer of EtaPhiV1
35 struct ATask {
36  Produces<aod::EtaPhiV1> etaphi;
37 
38  void process(aod::Tracks const& tracks)
39  {
40  for (auto& track : tracks) {
41  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
42  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
43 
44  etaphi(eta, phi);
45  }
46  }
47 };

64 // Consumper of both EtaPhiV1 and EtaPhiV2
65 // InputTable is a template which is then specified below when the workflow is defined
66 template <typename InputTable>
67 struct CTask {
68  void process(InputTable const& etaPhis)
69  {
70  constexpr bool isV2 = std::is_same<InputTable, aod::EtaPhiV2>::value;
71  for (auto& etaPhi : etaPhis) {
72  LOGF(info, "(%f, %f)", etaPhi.eta(), etaPhi.phi());
73  if constexpr (isV2) {
74  // This line is only compiled if this is templated with EtaPhiV2
75  LOGF(info, "We have the new data model (%f)", etaPhi.absEta());
76  }
77  }
78  }
79 };
