11 // V0 builder task
12 // ================
13 //
14 // This task loops over an *existing* list of V0s (neg/pos track
15 // indices) and calculates the corresponding full V0 information
16 //
17 // Any analysis should loop over the "V0Data"
18 // table as that table contains all information
19 //
20 // WARNING: adding filters to the builder IS NOT
21 // equivalent to re-running the finders. This will only
22 // ever produce *tighter* selection sections. It is your
23 // responsibility to check if, by setting a loose filter
24 // setting, you are going into a region in which no
25 // candidates exist because the original indices were generated
26 // using tigher selections.
27 //
28 // Comments, questions, complaints, suggestions?
29 // Please write to:
30 // david.dobrigkeit.chinellato@cern.ch

77 //This prefilter creates a skimmed list of good V0s to re-reconstruct so that
78 //CPU is saved in case there are specific selections that are to be done
79 //Note: more configurables, more options to be added as needed
80 struct lambdakzeroprefilterpairs {
81  Configurable<float> dcanegtopv{"dcanegtopv", .1, "DCA Neg To PV"};
82  Configurable<float> dcapostopv{"dcapostopv", .1, "DCA Pos To PV"};
83  Configurable<int> mincrossedrows{"mincrossedrows", 70, "min crossed rows"};
84  Configurable<int> tpcrefit{"tpcrefit", 1, "demand TPC refit"};
85 
86  HistogramRegistry registry{
87  "registry",
88  {
89  {"hGoodIndices", "hGoodIndices", {HistType::kTH1F, {{4, 0.0f, 4.0f}}}},
90  },
91  };

93  Produces<aod::V0GoodIndices> v0goodindices;

95  void process(aod::Collision const& collision, aod::V0s const& V0s,
96  soa::Join<aod::FullTracks, aod::TracksExtended> const& tracks)
97  {
98  for (auto& V0 : V0s) {
99  registry.fill(HIST("hGoodIndices"), 0.5);
100  if (tpcrefit) {
101  if (!(V0.posTrack_as<FullTracksExt>().trackType() & o2::aod::track::TPCrefit)) {
102  continue; //TPC refit
103  }
104  if (!(V0.negTrack_as<FullTracksExt>().trackType() & o2::aod::track::TPCrefit)) {
105  continue; //TPC refit
106  }
107  }
108  registry.fill(HIST("hGoodIndices"), 1.5);
109  if (V0.posTrack_as<FullTracksExt>().tpcNClsCrossedRows() < mincrossedrows) {
110  continue;
111  }
112  if (V0.negTrack_as<FullTracksExt>().tpcNClsCrossedRows() < mincrossedrows) {
113  continue;
114  }
115  registry.fill(HIST("hGoodIndices"), 2.5);
116  if (fabs(V0.posTrack_as<FullTracksExt>().dcaXY()) < dcapostopv) {
117  continue;
118  }
119  if (fabs(V0.negTrack_as<FullTracksExt>().dcaXY()) < dcanegtopv) {
120  continue;
121  }
122  registry.fill(HIST("hGoodIndices"), 3.5);
123  v0goodindices(V0.posTrack_as<FullTracksExt>().globalIndex(),
124  V0.negTrack_as<FullTracksExt>().globalIndex(),
125  V0.posTrack_as<FullTracksExt>().collisionId());
126  }
127  }
128 };

143  //Configurables
144  Configurable<double> d_bz{"d_bz", -5.0, "bz field"};
145  // Configurable<int> d_UseAbsDCA{"d_UseAbsDCA", 1, "Use Abs DCAs"}; uncomment this once we want to use the weighted DCA
146 
147  //Selection criteria
148  Configurable<double> v0cospa{"v0cospa", 0.995, "V0 CosPA"}; //double -> N.B. dcos(x)/dx = 0 at x=0)
149  Configurable<float> dcav0dau{"dcav0dau", 1.0, "DCA V0 Daughters"};
150  Configurable<float> v0radius{"v0radius", 5.0, "v0radius"};
151 
152  double massPi = TDatabasePDG::Instance()->GetParticle(kPiPlus)->Mass();
153  double massKa = TDatabasePDG::Instance()->GetParticle(kKPlus)->Mass();
154  double massPr = TDatabasePDG::Instance()->GetParticle(kProton)->Mass();

157 
158  void process(aod::Collision const& collision, aod::V0GoodIndices const& V0s, soa::Join<aod::FullTracks, aod::TracksExtended> const& tracks)
159  {
160  //Define o2 fitter, 2-prong
161  o2::vertexing::DCAFitterN<2> fitter;
162  fitter.setBz(d_bz);
163  fitter.setPropagateToPCA(true);
164  fitter.setMaxR(200.);
165  fitter.setMinParamChange(1e-3);
166  fitter.setMinRelChi2Change(0.9);
167  fitter.setMaxDZIni(1e9);
168  fitter.setMaxChi2(1e9);
169  fitter.setUseAbsDCA(true); // use d_UseAbsDCA once we want to use the weighted DCA
170 
171  registry.fill(HIST("hEventCounter"), 0.5);
172  std::array<float, 3> pVtx = {collision.posX(), collision.posY(), collision.posZ()};
173 
174  for (auto& V0 : V0s) {
175  std::array<float, 3> pos = {0.};
176  std::array<float, 3> pvec0 = {0.};
177  std::array<float, 3> pvec1 = {0.};
178 
179  registry.fill(HIST("hV0Candidate"), 0.5);
180 
181  auto pTrack = getTrackParCov(V0.posTrack_as<FullTracksExt>());
182  auto nTrack = getTrackParCov(V0.negTrack_as<FullTracksExt>());
183  int nCand = fitter.process(pTrack, nTrack);
184  if (nCand != 0) {
185  fitter.propagateTracksToVertex();
186  const auto& vtx = fitter.getPCACandidate();
187  for (int i = 0; i < 3; i++) {
188  pos[i] = vtx[i];
189  }
190  fitter.getTrack(0).getPxPyPzGlo(pvec0);
191  fitter.getTrack(1).getPxPyPzGlo(pvec1);
192  } else {
193  continue;
194  }
195 
196  //Apply selections so a skimmed table is created only
197  if (fitter.getChi2AtPCACandidate() > dcav0dau) {
198  continue;
199  }
200 
201  auto V0CosinePA = RecoDecay::CPA(array{collision.posX(), collision.posY(), collision.posZ()}, array{pos[0], pos[1], pos[2]}, array{pvec0[0] + pvec1[0], pvec0[1] + pvec1[1], pvec0[2] + pvec1[2]});
202  if (V0CosinePA < v0cospa) {
203  continue;
204  }
205 
206  auto V0radius = RecoDecay::sqrtSumOfSquares(pos[0], pos[1]); //probably find better name to differentiate the cut from the variable
207  if (V0radius < v0radius) {
208  continue;
209  }
210 
211  registry.fill(HIST("hV0Candidate"), 1.5);
212  v0data(
213  V0.posTrack_as<FullTracksExt>().globalIndex(),
214  V0.negTrack_as<FullTracksExt>().globalIndex(),
215  V0.negTrack_as<FullTracksExt>().collisionId(),
216  fitter.getTrack(0).getX(), fitter.getTrack(1).getX(),
217  pos[0], pos[1], pos[2],
218  pvec0[0], pvec0[1], pvec0[2],
219  pvec1[0], pvec1[1], pvec1[2],
220  fitter.getChi2AtPCACandidate(),
221  V0.posTrack_as<FullTracksExt>().dcaXY(),
222  V0.negTrack_as<FullTracksExt>().dcaXY());
223  }
224  }
225 };
