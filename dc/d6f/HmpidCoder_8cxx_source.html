150 void HmpidCoder::fillTheOutputBuffer(uint32_t* padMap)
151 {
152  uint32_t rowMarker, segMarker, eoeMarker, padWord;
153  uint32_t rowSize;
154  uint32_t ptr = 0;
155  int pads[Geo::MAXEQUIPMENTS];
156  int padsCount;
157  int segSize;
158 
159  for (int i = 0; i < Geo::MAXEQUIPMENTS; i++) {
160  mEventSizePerEquipment[i] = 0;
161  }
162 
163  for (int eq = 0; eq < mNumberOfEquipments; eq++) {
164  int startPtr = ptr;
165  padsCount = 0;
166  for (int s = 1; s <= Geo::N_SEGMENTS; s++) {
167  segSize = 0;
168  for (int c = 1; c <= Geo::N_COLXSEGMENT; c++) {
169 
170  // ---- Pre calculate the size of each column
171  for (int j = 0; j < Geo::N_DILOGICS; j++) {
172  pads[j] = 0;
173  }
174  rowSize = 0;
175  for (int j = 0; j < Geo::N_DILOGICS; j++) {
176  for (int k = 0; k < Geo::N_CHANNELS; k++) {
177  int idx = getEquipmentPadIndex(eq, ((s - 1) * Geo::N_COLXSEGMENT + (c - 1)), j, k);
178  if (padMap[idx] > 0) {
179  pads[j]++;
180  rowSize++;
181  padsCount++;
182  }
183  }
184  }
185  rowSize += Geo::N_DILOGICS;
186  segSize += (rowSize + 1);
187  rowMarker = 0x000036A8 | ((rowSize << 16) & 0x03ff0000);
188 
189  // ---- fills the Payload Buffer
190  mPayloadBufferPtr[ptr++] = rowMarker;
191  int col = (s - 1) * Geo::N_COLXSEGMENT + c;
192  for (int d = 1; d <= Geo::N_DILOGICS; d++) {
193  for (int p = 0; p < Geo::N_CHANNELS; p++) {
194  int idx = getEquipmentPadIndex(eq, ((s - 1) * Geo::N_COLXSEGMENT + (c - 1)), (d - 1), p);
195  if (padMap[idx] > 0) {
196  padWord = ((col << 22) & 0x07c00000) | ((d << 18) & 0x003C0000) | ((p << 12) & 0x0003F000) | (padMap[idx] & 0x00000FFF);
197  mPayloadBufferPtr[ptr++] = padWord;
198  }
199  }
200  eoeMarker = 0x08000080 | ((col << 22) & 0x07c00000) | (d << 18 & 0x003C0000) | (pads[d - 1] & 0x0000007F);
201  mPayloadBufferPtr[ptr++] = eoeMarker;
202  }
203  }
204  segSize += 1;
205  segMarker = 0xAB000000 | ((segSize << 8) & 0x000fff00) | (s & 0x0000000F);
206  mPayloadBufferPtr[ptr++] = segMarker;
207  }
208 
209  if (mSkipEmptyEvents && padsCount == 0) { // Sets the lenght of Events
210  mEventSizePerEquipment[eq] = 0;
211  } else {
212  mEventSizePerEquipment[eq] = ptr - startPtr;
213  }
214  }
215  return;
216 }

218 void HmpidCoder::writePaginatedEvent(uint32_t orbit, uint16_t bc)
219 {
220  int nWordToRead;
221  int count;
222  int payloatPtr = 0;
223  if (orbit == 0 || bc == 0) {
224  std::cerr << "HmpidCoder [ERROR] : Bad Orbit/BCid number (ORBIT=" << orbit << " BCID=" << bc << ")" << std::endl;
225  return;
226  }
227 
228  for (int eq = 0; eq < mNumberOfEquipments; eq++) {
229  int EventSize = mEventSizePerEquipment[eq];
230  if (EventSize == 0) {
231  continue; // Skips the Events sized with 0
232  }
233  int EquipmentCounter = 0;
234  int numOfPages = EventSize / PAYLOADMAXSPACE_W + 1;
235 
236  for (uint32_t PageNum = 1; PageNum <= numOfPages; PageNum++) {
237  count = 0;
238  while (count < PAYLOADMAXSPACE_W && EquipmentCounter < EventSize) {
239  mEventBufferPtr[HEADERDIMENSION_W + count++] = mPayloadBufferPtr[payloatPtr++];
240  EquipmentCounter++;
241  }
242  nWordToRead = count;
243 
244  int nWordsTail = (mFixedPacketLenght) ? PAYLOADDIMENSION_W : ((count / 8 + 1) * 8);
245  while (count < nWordsTail) {
246  mEventBufferPtr[HEADERDIMENSION_W + count++] = 0;
247  }
248  uint32_t PackNum = mPacketCounterPerEquipment[eq]++;
249  int PacketSize = writeHeader(mEventBufferPtr, nWordToRead, nWordsTail, mEqIds[eq], PackNum, bc, orbit, PageNum);
250  savePacket(mFlpIds[eq], PacketSize);
251  }
252  }
253  return;
254 }

256 int HmpidCoder::writeHeader(uint32_t* Buffer, uint32_t WordsToRead, uint32_t PayloadWords,
257  int Equip, uint32_t PackNum, uint32_t BCDI, uint32_t ORBIT, uint32_t PageNum)
258 {
259  uint32_t CruId, LinkId;
260  uint32_t TType = 0;
261  uint32_t HeStop = 0;
262  uint32_t FirmVers = 9;
263  uint32_t HeError = 0;
264  uint32_t Busy = 3000;
265  uint32_t PAR = 0;
266 
267  uint32_t MemSize = WordsToRead * sizeof(uint32_t) + HEADERDIMENSION_W * sizeof(uint32_t);
268  uint32_t OffsetNext = PayloadWords * sizeof(uint32_t) + HEADERDIMENSION_W * sizeof(uint32_t);
269  getEquipCoord(Equip, &CruId, &LinkId);
270 
271  // FEEID Header Size Header version
272  Buffer[0] = 0xFFF00000 | ((Equip & 0x0F) << 16) | 0x00004000 | 0x00000006;
273  // Priority System ID
274  Buffer[1] = 0x00000100 | 0x00000006;
275  // .....Memory Size HeOffsetNewPack;
276  Buffer[2] = (MemSize << 16) | (OffsetNext & 0x0000FFFF);
277  // DW CruId PacNum Link Num
278  Buffer[3] = 0x10000000 | ((CruId & 0x00FF) << 16) | ((PackNum & 0x0FF) << 8) | (LinkId & 0x0FF);
279  Buffer[4] = 0x00000FFF & BCDI;
280  Buffer[5] = ORBIT;
281  Buffer[6] = 0;
282  Buffer[7] = 0;
283  Buffer[8] = TType;
284  Buffer[9] = ((HeStop & 0x00ff0000) << 16) | (PageNum & 0x0000FFFF);
285  Buffer[10] = 0;
286  Buffer[11] = 0;
287  Buffer[12] = (Busy << 9) | ((HeError & 0x000001F0) << 4) | (FirmVers & 0x0000000f);
288  Buffer[13] = PAR;
289  Buffer[14] = 0xAAAA0001;
290  Buffer[15] = 0xAAAA0001;
291  return (OffsetNext);
292 }

312 void HmpidCoder::codeEventChunkDigits(std::vector<Digit>& digits, bool flushVector)
313 {
314  int eq, col, dil, cha, mo, x, y, idx;
315  uint32_t orbit = 0;
316  uint16_t bc = 0;
317 
318  int padsCount = 0;
319  int lastEventDigit = -1;
320  mLastProcessedDigit = -1;
321 
322  for (o2::hmpid::Digit d : digits) {
323  orbit = d.getOrbit();
324  bc = d.getBC();
325  lastEventDigit++;
326 
327  if (orbit != mPreviousOrbit || bc != mPreviousBc) { //the event is changed
328  if (mPreviousOrbit != 0 || mPreviousBc != 0) { // isn't the first !
329  fillTheOutputBuffer(mPadMap);
330  writePaginatedEvent(mPreviousOrbit, mPreviousBc);
331  mLastProcessedDigit = lastEventDigit;
332  }
333  memset(mPadMap, 0, sizeof(uint32_t) * Geo::N_HMPIDTOTALPADS); // Update for the new event
334  mPreviousOrbit = orbit;
335  mPreviousBc = bc;
336  }
337  Digit::Pad2Equipment(d.getPadID(), &eq, &col, &dil, &cha); // From Digit to Hardware coords
338  eq = mEqIds[eq]; // converts the Equipment Id in Cru/Link position ref
339  idx = getEquipmentPadIndex(eq, col, dil, cha); // finally to the unique padmap index
340 
341  if (mPadMap[idx] != 0) { // We already have the pad set
342  std::cerr << "HmpidCoder [ERROR] : Duplicated DIGIT =" << d << " (" << eq << "," << col << "," << dil << "," << cha << ")" << std::endl;
343  } else {
344  mPadMap[idx] = d.getCharge();
345  padsCount++;
346  }
347  }
348  mPreviousOrbit = 0;
349  mPreviousBc = 0;
350  mPadsCoded += padsCount;
351 
352  if (flushVector) {
353  fillTheOutputBuffer(mPadMap); // Finalize the last event
354  writePaginatedEvent(orbit, bc);
355  digits.clear();
356  } else {
357  if (mLastProcessedDigit >= 0) {
358  digits.erase(digits.begin(), digits.begin() + mLastProcessedDigit);
359  }
360  }
361  memset(mPadMap, 0, sizeof(uint32_t) * Geo::N_HMPIDTOTALPADS); // Update for the new event
362  return;
363 }
