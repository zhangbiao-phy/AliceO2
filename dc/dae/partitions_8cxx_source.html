18 // This is a very simple example showing how to iterate over tracks
19 // and create a new collection for them.
20 // FIXME: this should really inherit from AnalysisTask but
21 // we need GCC 7.4+ for that
22 struct ATask {
23  float fPI = static_cast<float>(M_PI);
24  Configurable<float> ptlow{"pTlow", 0.5f, "Lowest pT"};
25  Configurable<float> ptup{"pTup", 2.0f, "highest pT"};
26  Filter ptFilter_a = aod::track::pt > ptlow;
27  Filter ptFilter_b = aod::track::pt < ptup;
28 
29  Configurable<float> etalow{"etaLow", -1.0f, "lowest eta"};
30  Configurable<float> etaup{"etaUp", 1.0f, "highest eta"};
31  Filter etafilter = (aod::track::eta < etaup) && (aod::track::eta > etalow);
32 
33  Configurable<float> philow{"phiLow", 1.0f, "lowest phi"};
34  Configurable<float> phiup{"phiUp", 2.0f, "highest phi"};
35 
36  using myTracks = soa::Filtered<aod::Tracks>;
37 
38  Partition<myTracks> leftPhi = aod::track::phiraw < philow;
39  Partition<myTracks> midPhi = aod::track::phiraw >= philow && aod::track::phiraw < phiup;
40  Partition<myTracks> rightPhi = aod::track::phiraw >= phiup;
41 
42  void process(aod::Collision const& collision, myTracks const& tracks)
43  {
44  LOGF(INFO, "Collision: %d [N = %d] [left phis = %d] [mid phis = %d] [right phis = %d]",
45  collision.globalIndex(), tracks.size(), leftPhi.size(), midPhi.size(), rightPhi.size());
46 
47  for (auto& track : leftPhi) {
48  LOGF(INFO, "id = %d, from collision: %d, collision: %d; eta: %.3f < %.3f < %.3f; phi: %.3f < %.3f; pt: %.3f < %.3f < %.3f",
49  track.collisionId(), track.collision().globalIndex(), collision.globalIndex(), (float)etalow, track.eta(), (float)etaup, track.phiraw(), (float)philow, (float)ptlow, track.pt(), (float)ptup);
50  }
51  for (auto& track : midPhi) {
52  LOGF(INFO, "id = %d, from collision: %d, collision: %d; eta: %.3f < %.3f < %.3f; phi: %.3f <= %.3f < %.3f; pt: %.3f < %.3f < %.3f",
53  track.collisionId(), track.collision().globalIndex(), collision.globalIndex(), (float)etalow, track.eta(), (float)etaup, (float)philow, track.phiraw(), (float)phiup, (float)ptlow, track.pt(), (float)ptup);
54  }
55  for (auto& track : rightPhi) {
56  LOGF(INFO, "id = %d, from collision: %d, collision: %d; eta: %.3f < %.3f < %.3f; phi: %.3f < %.3f; pt: %.3f < %.3f < %.3f",
57  track.collisionId(), track.collision().globalIndex(), collision.globalIndex(), (float)etalow, track.eta(), (float)etaup, (float)phiup, track.phiraw(), (float)ptlow, track.pt(), (float)ptup);
58  }
59  }
60 };

62 // Partition inside process
63 // Caveat: partitioned table cannot be passed as const& to process()
64 struct BTask {
65  void process(aod::Collisions const& collisions, aod::Tracks& tracks)
66  {
67  for (auto& c : collisions) {
68  Partition<aod::Tracks> groupedTracks = aod::track::collisionId == c.globalIndex();
69  groupedTracks.bindTable(tracks);
70  for (auto& t : groupedTracks) {
71  LOGF(INFO, "collision global index: %d grouped track collision id: %d", c.globalIndex(), t.collisionId());
72  }
73  }
74  }
75 };
