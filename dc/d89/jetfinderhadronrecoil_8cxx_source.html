48  template <typename T>
49  T relativePhi(T phi1, T phi2)
50  {
51  if (phi1 < -TMath::Pi()) {
52  phi1 += (2 * TMath::Pi());
53  } else if (phi1 > TMath::Pi()) {
54  phi1 -= (2 * TMath::Pi());
55  }
56  if (phi2 < -TMath::Pi()) {
57  phi2 += (2 * TMath::Pi());
58  } else if (phi2 > TMath::Pi()) {
59  phi2 -= (2 * TMath::Pi());
60  }
61  T deltaPhi = phi2 - phi1;
62  if (deltaPhi < -TMath::Pi()) {
63  deltaPhi += (2 * TMath::Pi());
64  } else if (deltaPhi > TMath::Pi()) {
65  deltaPhi -= (2 * TMath::Pi());
66  }
67  return deltaPhi;
68  }

80  void process(aod::Collision const& collision,
81  soa::Filtered<aod::Tracks> const& tracks)
82  {
83 
84  jets.clear();
85  inputParticles.clear();
86  auto trackTTPhi = 0.0;
87  auto trackTTPt = 0.0;
88  auto comparisonPt = 0.0;
89  bool isTT = false;
90  for (auto& track : tracks) {
91  if (track.pt() >= f_trackTTMin && track.pt() < f_trackTTMax) { //can this also go into a partition?
92  isTT = true;
93  if (track.pt() >= comparisonPt) { //currently take highest pT but later to randomise
94  comparisonPt = track.pt();
95  trackTTPt = track.pt();
96  trackTTPhi = track.phi();
97  }
98  }
99  auto energy = std::sqrt(track.p() * track.p() + JetFinder::mPion * JetFinder::mPion);
100  inputParticles.emplace_back(track.px(), track.py(), track.pz(), energy);
101  inputParticles.back().set_user_index(track.globalIndex());
102  }
103  if (!isTT) {
104  return;
105  }
106  hHadronPt->Fill(trackTTPt);
107 
108  // you can set phi selector here for jets
109  fastjet::ClusterSequenceArea clusterSeq(jetFinder.findJets(inputParticles, jets));
110 
111  for (const auto& jet : jets) {
112  auto deltaPhi = TMath::Abs(relativePhi(jet.phi(), trackTTPhi));
113  if (deltaPhi >= (TMath::Pi() - f_recoilWindow)) {
114  hJetPt->Fill(jet.pt());
115  }
116  if (deltaPhi >= TMath::Pi() / 2.0 && deltaPhi <= TMath::Pi()) {
117  hJetHadronDeltaPhi->Fill(deltaPhi);
118  }
119  }
120  }
121 };
