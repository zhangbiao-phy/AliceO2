94  void process(soa::Join<aod::Collisions, aod::EvSels>::iterator const& col, aod::BCs const& bcs, aod::Zdcs const& zdcs, aod::FV0As const& fv0as, aod::FV0Cs const& fv0cs, aod::FT0s ft0s, aod::FDDs fdds)
95  {

99 
100  auto fv0a = getVZeroA(col.bc(), fv0as);
101  auto fv0c = getVZeroC(col.bc(), fv0cs);
102  hTimeV0Aall->Fill(fv0a.time());
103  hTimeV0Call->Fill(fv0c.time());
104 
105  auto zdc = getZdc(col.bc(), zdcs);
106  hTimeZNAall->Fill(zdc.timeZNA());
107  hTimeZNCall->Fill(zdc.timeZNC());
108 
109  auto ft0 = getFT0(col.bc(), ft0s);
110  hTimeT0Aall->Fill(ft0.timeA());
111  hTimeT0Call->Fill(ft0.timeC());
112 
113  auto fdd = getFDD(col.bc(), fdds);
114  hTimeFDAall->Fill(fdd.timeA());
115  hTimeFDCall->Fill(fdd.timeC());
116 
117  if (selection == 7 && !col.sel7()) {
118  return;
119  }
120 
121  if (selection == 8 && !col.sel8()) {
122  return;
123  }
124 
125  if (selection != 7 && selection != 8) {
126  LOGF(fatal, "Unknown selection type! Use `--sel 7` or `--sel 8`");
127  }
128 
129  hTimeV0Aacc->Fill(fv0a.time());
130  hTimeV0Cacc->Fill(fv0c.time());
131  hTimeZNAacc->Fill(zdc.timeZNA());
132  hTimeZNCacc->Fill(zdc.timeZNC());
133  hTimeT0Aacc->Fill(ft0.timeA());
134  hTimeT0Cacc->Fill(ft0.timeC());
135  hTimeFDAacc->Fill(fdd.timeA());
136  hTimeFDCacc->Fill(fdd.timeC());
137  }
