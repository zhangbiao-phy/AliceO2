49  Filter collisionFilter = nabs(aod::collision::posZ) < vertexZCut;

52  using TrackCandidates = soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra, aod::pidRespTPC, aod::pidRespTOF, aod::pidRespTOFbeta, aod::TrackSelection>>;
53  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>::iterator const& collision, TrackCandidates const& tracks)
54  {
55  bool keepEvent = false;
56  if (!collision.alias()[kINT7]) {
57  return; //is this correct?
58  }
59  if (!collision.sel7()) {
60  return;
61  }
62  for (auto& track : tracks) {
63  TLorentzVector cutVector{};
64  cutVector.SetPtEtaPhiM(track.pt() * 2.0, track.eta(), track.phi(), constants::physics::MassHelium3);
65  if (cutVector.Rapidity() < yMin + yBeam || cutVector.Rapidity() > yMax + yBeam) {
66  continue;
67  }
68  if (track.tpcNSigmaHe() > nSigmaCutLow && track.tpcNSigmaHe() < nSigmaCutHigh) {
69  keepEvent = true;
70  break;
71  }
72  }
73  if (keepEvent) {
74  outputCollisions(collision.posZ());
75  uint32_t pNsigma = 0xFFFFFF00; //15 bit precision for Nsigma - does this respect the sign?
76  outputTracks.reserve(tracks.size());
77  for (auto track : tracks) {
78  outputTracks(outputCollisions.lastIndex(), track.pt(), track.eta(), track.phi(),
79  //truncateFloatFraction(track.tpcNSigmaEl(), pNsigma), truncateFloatFraction(track.tpcNSigmaMu(), pNsigma),
80  truncateFloatFraction(track.tpcNSigmaPi(), pNsigma), truncateFloatFraction(track.tpcNSigmaKa(), pNsigma),
81  truncateFloatFraction(track.tpcNSigmaPr(), pNsigma), //truncateFloatFraction(track.tpcNSigmaDe(), pNsigma),
82  //truncateFloatFraction(track.tpcNSigmaTr(), pNsigma),
83  truncateFloatFraction(track.tpcNSigmaHe(), pNsigma),
84  //truncateFloatFraction(track.tpcNSigmaAl(), pNsigma),
85  //truncateFloatFraction(track.tofNSigmaEl(), pNsigma), truncateFloatFraction(track.tofNSigmaMu(), pNsigma),
86  truncateFloatFraction(track.tofNSigmaPi(), pNsigma), truncateFloatFraction(track.tofNSigmaKa(), pNsigma),
87  truncateFloatFraction(track.tofNSigmaPr(), pNsigma), //truncateFloatFraction(track.tofNSigmaDe(), pNsigma),
88  //truncateFloatFraction(track.tofNSigmaTr(), pNsigma),
89  truncateFloatFraction(track.tofNSigmaHe(), pNsigma));
90  //truncateFloatFraction(track.tofNSigmaAl(), pNsigma));
91  }
92  }
93  }
94 };
