54  int ch, ax, ay;
55  if (ax > Geo::MAXHALFXROWS) {
56  *Equi = ch * Geo::EQUIPMENTSPERMODULE + 1;
57  ax = ax - Geo::HALFXROWS;

74 {
75  if (Equi < 0 || Equi >= Geo::MAXEQUIPMENTS || Colu < 0 || Colu >= Geo::N_COLUMNS ||

77  return;

80  int a2y[6] = {3, 2, 4, 1, 5, 0}; //pady for a given padress (for single DILOGIC chip)

82  int tmp = (23 - Colu) / Geo::N_COLXSEGMENT;
83  int pc = (Equi % Geo::EQUIPMENTSPERMODULE) ? 5 - 2 * tmp : 2 * tmp; // The PhotoCatode
84  int px = (Geo::N_DILOGICS - Dilo) * Geo::DILOPADSROWS - Chan / Geo::DILOPADSCOLS - 1;
85  tmp = (Equi % Geo::EQUIPMENTSPERMODULE) ? Colu : (23 - Colu);
86  int py = Geo::DILOPADSCOLS * (tmp % Geo::DILOPADSROWS) + a2y[Chan % Geo::DILOPADSCOLS];
87  *Mod = ch;
88  *Row = px;
