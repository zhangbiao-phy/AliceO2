40 void DataProcessor::doSend(FairMQDevice& device, MessageContext& context, ServiceRegistry&)
41 {
42  std::unordered_map<std::string const*, FairMQParts> outputs;
43  auto contextMessages = context.getMessagesForSending();
44  for (auto& message : contextMessages) {
45  // monitoringService.send({ message->parts.Size(), "outputs/total" });
46  FairMQParts parts = std::move(message->finalize());
47  assert(message->empty());
48  assert(parts.Size() == 2);
49  for (auto& part : parts) {
50  outputs[&(message->channel())].AddPart(std::move(part));
51  }
52  }
53  for (auto& [channel, parts] : outputs) {
54  device.Send(parts, *channel, 0);
55  }
56 }

58 void DataProcessor::doSend(FairMQDevice& device, StringContext& context, ServiceRegistry&)
59 {
60  for (auto& messageRef : context) {
61  FairMQParts parts;
62  FairMQMessagePtr payload(device.NewMessage());
63  auto a = messageRef.payload.get();
64  // Rebuild the message using the string as input. For now it involves a copy.
65  payload->Rebuild(reinterpret_cast<void*>(const_cast<char*>(strdup(a->data()))), a->size(), nullptr, nullptr);
66  const DataHeader* cdh = o2::header::get<DataHeader*>(messageRef.header->GetData());
67  // sigh... See if we can avoid having it const by not
68  // exposing it to the user in the first place.
69  DataHeader* dh = const_cast<DataHeader*>(cdh);
70  dh->payloadSize = payload->GetSize();
71  parts.AddPart(std::move(messageRef.header));
72  parts.AddPart(std::move(payload));
73  device.Send(parts, messageRef.channel, 0);

77 void DataProcessor::doSend(FairMQDevice& device, ArrowContext& context, ServiceRegistry& registry)
78 {
79  using o2::monitoring::Metric;
80  using o2::monitoring::Monitoring;
81  using o2::monitoring::tags::Key;
82  using o2::monitoring::tags::Value;
83 
84  for (auto& messageRef : context) {
85  FairMQParts parts;
86  // Depending on how the arrow table is constructed, we finalize
87  // the writing here.
88  messageRef.finalize(messageRef.buffer);
89 
90  std::unique_ptr<FairMQMessage> payload = messageRef.buffer->Finalise();
91  // FIXME: for the moment we simply send empty bodies.
92  const DataHeader* cdh = o2::header::get<DataHeader*>(messageRef.header->GetData());
93  // sigh... See if we can avoid having it const by not
94  // exposing it to the user in the first place.
95  DataHeader* dh = const_cast<DataHeader*>(cdh);
96  dh->payloadSize = payload->GetSize();
97  dh->serialization = o2::header::gSerializationMethodArrow;
98  context.updateBytesSent(payload->GetSize());
99  context.updateMessagesSent(1);
100  parts.AddPart(std::move(messageRef.header));
101  parts.AddPart(std::move(payload));
102  device.Send(parts, messageRef.channel, 0);
103  }
104  auto& monitoring = registry.get<Monitoring>();
105  monitoring.send(Metric{(uint64_t)context.bytesSent(), "arrow-bytes-created"}.addTag(Key::Subsystem, Value::DPL));
106  monitoring.send(Metric{(uint64_t)context.messagesCreated(), "arrow-messages-created"}.addTag(Key::Subsystem, Value::DPL));
107  monitoring.flushBuffer();
108 }

110 void DataProcessor::doSend(FairMQDevice& device, RawBufferContext& context, ServiceRegistry& registry)
111 {
112  for (auto& messageRef : context) {
113  FairMQParts parts;

115  auto buffer = messageRef.serializeMsg().str();
116  // Rebuild the message using the serialized ostringstream as input. For now it involves a copy.

120  const DataHeader* cdh = o2::header::get<DataHeader*>(messageRef.header->GetData());
121  // sigh... See if we can avoid having it const by not
