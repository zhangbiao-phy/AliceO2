34 void doQueryHVLV(const std::string ccdbUrl, uint64_t timestamp, bool hv, bool lv)
35 {
36  std::vector<std::string> what;
37  if (hv) {
38  what.emplace_back("MCH/HV");
39  }
40  if (lv) {
41  what.emplace_back("MCH/LV");
42  }
43 
44  auto sum =
45  [](float s, o2::dcs::DataPointValue v) {
46  union Converter {
47  uint64_t raw_data;
48  double value;
49  } converter;
50  converter.raw_data = v.payload_pt1;
51  return s + converter.value;
52  };
53 
54  o2::ccdb::CcdbApi api;
55  api.init(ccdbUrl);
56  for (auto w : what) {
57  std::map<std::string, std::string> metadata;
58  auto* m = api.retrieveFromTFileAny<DPMAP>(w, metadata, timestamp);
59  std::cout << "size of " << w << " map = " << m->size() << std::endl;
60  if (verbose) {
61  for (auto& i : *m) {
62  auto v = i.second;
63  auto mean = std::accumulate(v.begin(), v.end(), 0.0, sum);
64  if (v.size()) {
65  mean /= v.size();
66  }
67  std::cout << fmt::format("{:64s} {:4d} values of mean {:7.2f}\n", i.first.get_alias(), v.size(),
68  mean);
69  }
70  }
71  }
72 }

89 void makeCCDBEntryForDCS(const std::string ccdbUrl, uint64_t timestamp)
90 {
91  std::unordered_map<DPID, std::string> dpid2DataDesc;
92  auto aliases = o2::mch::dcs::aliases();
93 
94  DPID dpidtmp;
95  for (const auto& a : aliases) {
96  DPID::FILL(dpidtmp, a, o2::dcs::DeliveryType::RAW_DOUBLE);
97  dpid2DataDesc[dpidtmp] = "MCHDATAPOINTS";
98  }
99 
100  o2::ccdb::CcdbApi api;
101  api.init(ccdbUrl);
102  std::map<std::string, std::string> md;
103  std::cout << "storing config of " << dpid2DataDesc.size() << " MCH data points to " << CCDB_DPCONF_NAME << "\n";
104 
105  api.storeAsTFileAny(&dpid2DataDesc, CCDB_DPCONF_NAME, md, timestamp);
106 }

108 bool match(const std::vector<std::string>& queries, const char* pattern)
109 {
110  return std::find_if(queries.begin(), queries.end(), [pattern](std::string s) { return std::regex_match(s, std::regex(pattern, std::regex::extended | std::regex::icase)); }) != queries.end();

130  ("ccdb,c",po::value<std::string>(&ccdbUrl)->default_value("http://localhost:6464"),"ccdb url")
131  ("query,q",po::value<std::vector<std::string>>(),"what to query (if anything)")
132  ("timestamp,t",po::value<uint64_t>(&timestamp)->default_value(now),"timestamp for query or put")
133  ("put-datapoint-config,p",po::bool_switch(&put),"upload datapoint configuration")
134  ("verbose,v",po::bool_switch(&verbose),"verbose output")
135  ;
136  // clang-format on
137 
138  po::options_description cmdline;
139  cmdline.add(usage);
140 
141  po::store(po::command_line_parser(argc, argv).options(cmdline).run(), vm);
142 
143  if (vm.count("help")) {
144  std::cout << "This program printout summary information from MCH DCS entries.\n";
145  std::cout << usage << "\n";
146  return 2;
147  }
148 
149  try {
150  po::notify(vm);
151  } catch (boost::program_options::error& e) {
152  std::cout << "Error: " << e.what() << "\n";
153  exit(1);
154  }
155 
156  if (vm.count("query")) {
157  auto query = vm["query"].as<std::vector<std::string>>();
158 
159  hv = match(query, ".*(hv)");
160  lv = match(query, ".*(lv)");
161  dpconf = match(query, ".*(dpconf)");
162 
163  if (!hv && !lv && !dpconf) {
164  std::cout << "Must specify at least one of --hv or --lv\n";
165  std::cout << usage << "\n";
166  return 3;
167  }
168 
169  if (hv || lv) {
170  doQueryHVLV(ccdbUrl, timestamp, hv, lv);
171  }
172 
173  if (dpconf) {
174  doQueryDataPointConfig(ccdbUrl, timestamp);
175  }
176  }
177 
178  if (put) {
179  makeCCDBEntryForDCS(ccdbUrl, timestamp);
180  }
181  return 0;
182 }
