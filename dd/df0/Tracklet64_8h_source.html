40  -------------------------------------------------------------------------------------------------
41 Word 0 | slope | Q2 | Q1 | Q0 |
42  -------------------------------------------------------------------------------------------------
43 */
44 class Tracklet64
45 {
46 
47  public:
48  Tracklet64() = default;
49  Tracklet64(uint64_t trackletword) { mtrackletWord = trackletword; }
50  Tracklet64(const Tracklet64&) = default;
51  Tracklet64(uint64_t format, uint64_t hcid, uint64_t padrow, uint64_t col, uint64_t position,
52  uint64_t slope, uint64_t Q0, uint64_t Q1, uint64_t Q2)
53  {
54  mtrackletWord = ((format << formatbs) & formatmask) |
55  ((hcid << hcidbs) & hcidmask) |
56  ((padrow << padrowbs) & padrowmask) |
57  ((col << colbs) & colmask) |
58  ((position << posbs) & posmask) |
59  ((slope << slopebs) & slopemask) |
60  ((Q2 << Q2bs) & Q2mask) |
61  ((Q1 << Q1bs) & Q1mask) |
62  ((Q0 << Q0bs) & Q0mask);
63  }
64 
65  ~Tracklet64() = default;
66  Tracklet64& operator=(const Tracklet64& rhs) = default;
67 
68  // ----- Getters for contents of tracklet word -----
69  uint64_t getHCID() const { return ((mtrackletWord & hcidmask) >> hcidbs); }; // no units 0..1077
70  uint64_t getPadRow() const { return ((mtrackletWord & padrowmask) >> padrowbs); }; // pad row number [0..15]
71  uint64_t getColumn() const { return ((mtrackletWord & colmask) >> colbs); }; // column refers to MCM position in column direction on readout board [0..3]
72  uint64_t getPosition() const { return ((mtrackletWord & posmask) >> posbs); }; // in units of 1/80 pads, 11 bit granularity [-12.8..12.8] relative to MCM center
73  uint64_t getSlope() const { return ((mtrackletWord & slopemask) >> slopebs); }; // in units of 1/1000 pads/timebin, 8 bit granularity [-0.128 to 0.128]
74  uint64_t getPID() const { return ((mtrackletWord & PIDmask)); }; // no unit, all 3 charge windows combined
75  uint64_t getQ0() const { return ((mtrackletWord & Q0mask) >> Q0bs); }; // no unit
76  uint64_t getQ1() const { return ((mtrackletWord & Q1mask) >> Q1bs); }; // no unit
77  uint64_t getQ2() const { return ((mtrackletWord & Q2mask) >> Q2bs); }; // no unit
78 
79  void setTrackletWord(uint64_t trackletword) { mtrackletWord = trackletword; }
80 
81  // ----- Getters for tracklet information -----
82  int getMCM() const { return 4 * (getPadRow() % 4) + getColumn(); } // returns MCM position on ROB [0..15]
83  int getROB() const { return (getHCID() % 2) ? (getPadRow() / 4) * 2 + 1 : (getPadRow() / 4) * 2; } // returns ROB number [0..5] for C0 chamber and [0..7] for C1 chamber
84  float getUncalibratedY() const; // translate local position into global y (in cm) not taking into account calibrations (ExB, vDrift, t0)
85  float getUncalibratedDy(float nTbDrift = 19.4f) const; // translate local slope into dy/dx with dx=3m (drift length) and default drift time in time bins (19.4 timebins / 3cm)
86 
87  // ----- Getters for offline corresponding values -----
88  int getDetector() const { return getHCID() / 2; }
89 
90  uint64_t getTrackletWord() const { return mtrackletWord; }
91 
92  void setQ0(int charge)
93  {
94  mtrackletWord &= ~Q0mask;
95  mtrackletWord |= ((charge << Q0bs) & Q0mask);
96  }
97  void setQ1(int charge)
98  {
99  mtrackletWord &= ~Q1mask;
100  mtrackletWord |= ((charge << Q1bs) & Q1mask);
101  }
102  void setQ2(int charge)
103  {
104  mtrackletWord &= ~Q2mask;
105  mtrackletWord |= ((charge << Q2bs) & Q2mask);
106  }
107  void setPID(uint64_t pid)
108  {
109  // set the entire pid area of the trackletword, all the 3 Q's
110  mtrackletWord &= ~PIDmask;
111  mtrackletWord |= ((pid << PIDbs) & PIDmask);
112  }
113  void setPosition(uint64_t position)
114  {
115  mtrackletWord &= ~posmask;
116  mtrackletWord |= ((position << posbs) & posmask);
117  }
118  void setSlope(uint64_t slope)
119  {
120  mtrackletWord &= ~slopemask;
121  mtrackletWord |= ((slope << slopebs) & slopemask);
122  }
123 
124  void printStream(std::ostream& stream) const;

126  // bit masks for the above raw data;
127  static constexpr uint64_t formatmask = 0xf000000000000000;
128  static constexpr uint64_t hcidmask = 0x0ffe000000000000;
129  static constexpr uint64_t padrowmask = 0x0001e00000000000;
130  static constexpr uint64_t colmask = 0x0000180000000000;
131  static constexpr uint64_t posmask = 0x000007ff00000000;
132  static constexpr uint64_t slopemask = 0x00000000ff000000;
133  static constexpr uint64_t Q2mask = 0x0000000000ff0000;
134  static constexpr uint64_t Q1mask = 0x000000000000ff00;
135  static constexpr uint64_t Q0mask = 0x00000000000000ff;
136  static constexpr uint64_t PIDmask = 0x0000000000ffffff;
137  //bit shifts for the above raw data
138  static constexpr uint64_t formatbs = 60;
139  static constexpr uint64_t hcidbs = 49;
140  static constexpr uint64_t padrowbs = 45;
141  static constexpr uint64_t colbs = 43;
142  static constexpr uint64_t posbs = 32;
143  static constexpr uint64_t slopebs = 24;
144  static constexpr uint64_t PIDbs = 0;
145  static constexpr uint64_t Q2bs = 16;
146  static constexpr uint64_t Q1bs = 8;
147  static constexpr uint64_t Q0bs = 0;
148 
149  protected:
150  uint64_t mtrackletWord; // the 64 bit word holding all the tracklet information for run3.
151  private:
152  ClassDefNV(Tracklet64, 1);
153 };
