55 RawErrorType_t RawDecoder::readChannels()
56 {
57  mChannelsInitialized = false;
58  auto& header = mRawReader.getRawHeader();
59  // short ddl = o2::raw::RDHUtils::getLinkID(header) ; //Current module
60 
61  auto& payloadwords = mRawReader.getPayload();
62  uint32_t wc = 0;
63  auto b = payloadwords.cbegin();
64  auto e = payloadwords.cend();
65  while (b != e) {
66  cpvword w(b, e);
67  if (w.isOK()) {
68  for (int i = 0; i < 3; i++) {
69  PadWord pw = {w.cpvPadWord(i)};
70  if (pw.zero == 0) {
71  addDigit(pw.mDataWord, w.ccId());
72  } else {
73  if (pw.mDataWord != 0xffffff) { //not empty pad
74  LOG(ERROR) << "no zero bit in data word " << pw.mDataWord;
75  mErrors.emplace_back(w.ccId(), 0, 0, 0, kPADERROR); //add error for non-existing row
76  }
77  }
78  }
79  } else { //this may be trailer
80  cpvtrailer tr(b, e);
81  if (tr.isOK()) {
82  if (tr.wordCounter() != wc) {
83  //some words lost?
84  LOG(ERROR) << "Read " << wc << " words, expected " << tr.wordCounter();
85  mErrors.emplace_back(w.ccId(), 0, 0, 0, kPAYLOAD_DECODING);
86  //TODO! should we continuew or brake?
87  }
88  } else {
89  //error
90  LOG(ERROR) << "Read neither data nor trailer word";
91  mErrors.emplace_back(w.ccId(), 0, 0, 0, kPADERROR); //add error for non-existing row
92  }
93  }
94  b += 16;
95  wc++;
96  }
97 
98  mChannelsInitialized = true;
99  return kOK;
100 }
