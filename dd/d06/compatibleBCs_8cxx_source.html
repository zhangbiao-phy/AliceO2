19 // In Run 3 the association between collisions and bunch crossings is not unique as the time of a collision vertex
20 // is derived from the track information themselves.
21 
22 template <typename T>
23 T getCompatibleBCs(aod::Collision const& collision, T const& bcs)
24 {
25  auto bcIter = collision.bc_as<T>();
26 
27  // due to the filling scheme the most probably BC may not be the one estimated from the collision time
28  uint64_t mostProbableBC = bcIter.globalBC();
29  uint64_t meanBC = mostProbableBC - std::lround(collision.collisionTime() / (o2::constants::lhc::LHCBunchSpacingNS / 1000));
30  int deltaBC = std::ceil(collision.collisionTimeRes() / (o2::constants::lhc::LHCBunchSpacingNS / 1000) * 4);
31 
32  LOGF(INFO, "BC range: %llu - %llu", meanBC - deltaBC, meanBC + deltaBC);
33 
34  // find slice of BCs table with BC in [meanBC - deltaBC, meanBC + deltaBC]
35  int64_t maxBCId = bcIter.globalIndex();
36  int moveCount = 0; // optimize to avoid to re-create the iterator
37  while (bcIter != bcs.end() && bcIter.globalBC() <= meanBC + deltaBC && bcIter.globalBC() >= meanBC - deltaBC) {
38  LOGF(DEBUG, "Table id %d BC %llu", bcIter.globalIndex(), bcIter.globalBC());
39  maxBCId = bcIter.globalIndex();
40  ++bcIter;
41  ++moveCount;
42  }
43 
44  bcIter.moveByIndex(-moveCount); // Move back to original position
45  int64_t minBCId = collision.bcId();
46  while (bcIter != bcs.begin() && bcIter.globalBC() <= meanBC + deltaBC && bcIter.globalBC() >= meanBC - deltaBC) {
47  LOGF(DEBUG, "Table id %d BC %llu", bcIter.globalIndex(), bcIter.globalBC());
48  minBCId = bcIter.globalIndex();
49  --bcIter;
50  }
51 
52  LOGF(INFO, "Will consider BC entries from %d to %d", minBCId, maxBCId);
53 
54  T slice{{bcs.asArrowTable()->Slice(minBCId, maxBCId - minBCId + 1)}, (uint64_t)minBCId};
55  bcs.copyIndexBindings(slice);
56  return slice;
57 }

59 // Example 1 (academic, because it is not enough to just access the BC table):
60 // This task shows how to loop over the bunch crossings which are compatible with the estimated collision time.
61 struct CompatibleBCs {
62  void process(aod::Collision const& collision, aod::BCs const& bcs)
63  {
64  LOGF(INFO, "Vertex with most probably BC %llu and collision time %f +- %f ps", collision.bc().globalBC(), collision.collisionTime(), collision.collisionTimeRes());
65 
66  auto bcSlice = getCompatibleBCs(collision, bcs);
67 
68  for (auto& bc : bcSlice) {
69  LOGF(info, "This collision may belong to BC %lld", bc.globalBC());
70  }
71  }
72 };

76 // and making sure that one has the direct association between them

81  {
82  // NOTE collision.bc() causes SEGV here because we have only subscribed to BCs joined, therefore:

85 
86  auto bcSlice = getCompatibleBCs(collision, bct0s);
87 
88  for (auto& bc : bcSlice) {
89  if (bc.has_ft0() && bc.has_fv0a()) {

93  }
