115 template <typename coder_T, typename stream_T, typename source_T>
116 template <typename stream_IT, typename source_IT, std::enable_if_t<internal::isCompatibleIter_v<stream_T, stream_IT> && internal::isCompatibleIter_v<source_T, source_IT>, bool>>
117 const stream_IT Encoder<coder_T, stream_T, source_T>::Encoder::process(const stream_IT outputBegin, const stream_IT outputEnd, const source_IT inputBegin, const source_IT inputEnd) const
118 {
119  using namespace internal;
120  LOG(trace) << "start encoding";
121  RANSTimer t;
122  t.start();
123 
124  static_assert(std::is_same<typename std::iterator_traits<source_IT>::value_type, source_T>::value);
125  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
126 
127  if (inputBegin == inputEnd) {
128  LOG(warning) << "passed empty message to encoder, skip encoding";
129  return outputEnd;
130  }
131 
132  if (outputBegin == outputEnd) {
133  const std::string errorMessage("Unallocated encode buffer passed to encoder. Aborting");
134  LOG(error) << errorMessage;
135  throw std::runtime_error(errorMessage);
136  }
137 
138  ransCoder rans0, rans1;
139 
140  stream_IT outputIter = outputBegin;
141  source_IT inputIT = inputEnd;
142 
143  const auto inputBufferSize = std::distance(inputBegin, inputEnd);
144 
145  auto encode = [this](source_IT symbolIter, stream_IT outputIter, ransCoder& coder) {
146  const source_T symbol = *symbolIter;
147  const auto& encoderSymbol = (*this->mSymbolTable)[symbol];
148  return std::tuple(symbolIter, coder.putSymbol(outputIter, encoderSymbol, this->mProbabilityBits));
149  };
150 
151  // odd number of bytes?
152  if (inputBufferSize & 1) {
153  std::tie(inputIT, outputIter) = encode(--inputIT, outputIter, rans0);
154  assert(outputIter < outputEnd);
155  }
156 
157  while (inputIT != inputBegin) { // NB: working in reverse!
158  std::tie(inputIT, outputIter) = encode(--inputIT, outputIter, rans1);
159  std::tie(inputIT, outputIter) = encode(--inputIT, outputIter, rans0);
160  assert(outputIter < outputEnd);
161  }
162  outputIter = rans1.flush(outputIter);
163  outputIter = rans0.flush(outputIter);
164  // first iterator past the range so that sizes, distances and iterators work correctly.
165  ++outputIter;
166 
167  assert(!(outputIter > outputEnd));
168 
169  // deal with overflow
170  if (outputIter > outputEnd) {
171  const std::string exceptionText = [&]() {
172  std::stringstream ss;
173  ss << __func__ << " detected overflow in encode buffer: allocated:" << std::distance(outputBegin, outputEnd) << ", used:" << std::distance(outputBegin, outputIter);
174  return ss.str();
175  }();
176 
177  LOG(error) << exceptionText;
178  throw std::runtime_error(exceptionText);
179  }
180 
181  t.stop();
182  LOG(debug1) << "Encoder::" << __func__ << " {ProcessedBytes: " << inputBufferSize * sizeof(source_T) << ","
183  << " inclusiveTimeMS: " << t.getDurationMS() << ","
184  << " BandwidthMiBPS: " << std::fixed << std::setprecision(2) << (inputBufferSize * sizeof(source_T) * 1.0) / (t.getDurationS() * 1.0 * (1 << 20)) << "}";
185 
186 // advanced diagnostics for debug builds
187 #if !defined(NDEBUG)
188 
189  const auto inputBufferSizeB = inputBufferSize * sizeof(source_T);
190  const auto outputBufferSizeB = std::distance(outputBegin, outputIter) * sizeof(stream_T);
191 
192  LOG(debug2) << "EncoderProperties: {"
193  << "sourceTypeB: " << sizeof(source_T) << ", "
194  << "streamTypeB: " << sizeof(stream_T) << ", "
195  << "coderTypeB: " << sizeof(coder_T) << ", "
196  << "probabilityBits: " << mProbabilityBits << ", "
197  << "inputBufferSizeB: " << inputBufferSizeB << ", "
198  << "outputBufferSizeB: " << outputBufferSizeB << ", "
199  << "compressionFactor: " << std::fixed << std::setprecision(2) << static_cast<double>(inputBufferSizeB) / static_cast<double>(outputBufferSizeB) << "}";
200 #endif
201 
202  LOG(trace) << "done encoding";
203 
204  return outputIter;
205 };
