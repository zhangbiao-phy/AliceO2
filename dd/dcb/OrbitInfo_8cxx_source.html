21 {
22  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(rdhBuffer[0])));
23  auto orbit = o2::raw::RDHUtils::getHeartBeatOrbit(rdhAny);

30 }
