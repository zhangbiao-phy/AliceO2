101 template <typename Source_IT>
102 void FrequencyTable::addSamples(Source_IT begin, Source_IT end, value_t min, value_t max)
103 {
104  static_assert(std::is_integral<typename std::iterator_traits<Source_IT>::value_type>::value);
105 
106  LOG(trace) << "start adding samples";
107  internal::RANSTimer t;
108  t.start();
109 
110  if (begin == end) {
111  LOG(debug) << "Passed empty message to " << __func__; // RS this is ok for empty columns
112  return;
113  }
114 
115  if (min == max) {
116  const auto& [minIter, maxIter] = std::minmax_element(begin, end);
117  resizeFrequencyTable(*minIter, *maxIter);
118 
119  } else {
120  resizeFrequencyTable(min, max);
121  }
122 
123  for (auto it = begin; it != end; ++it) {
124  assert((*it - mMin) < mFrequencyTable.size());
125  mFrequencyTable[*it - mMin]++;
126  }
127  mNumSamples = std::distance(begin, end);
128 
129  t.stop();
130  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
131 
132 #if !defined(NDEBUG)
133  LOG(debug2) << *this;
134 #endif
135  LOG(trace) << "done adding samples";
136 }

138 template <typename Freq_IT>
139 void FrequencyTable::addFrequencies(Freq_IT begin, Freq_IT end, value_t min, value_t max)
140 {
141  static_assert(std::is_integral<typename std::iterator_traits<Freq_IT>::value_type>::value);
142 
143  LOG(trace) << "start adding frequencies";
144  internal::RANSTimer t;
145  t.start();
146 
147  if (begin == end) {
148  LOG(debug) << "Passed empty FrequencyTable to " << __func__; // RS this is ok for empty columns
149  return;
150  }
151 
152  // ensure correct size of frequency table and grow it if necessary
153  resizeFrequencyTable(min, max);
154 
155  // either 0 or offset from array start.
156  const size_t offset = std::abs(mMin - min);
157 
158  // ftableA[i] += fTableB[i+offset]
159  std::transform(begin, end,
160  mFrequencyTable.begin() + offset,
161  mFrequencyTable.begin() + offset,
162  [this](typename std::iterator_traits<Freq_IT>::value_type first, uint32_t second) {
163  mNumSamples += first;
164  return first + second; });
165 
166  t.stop();
167  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
168 
169 #if !defined(NDEBUG)
170  LOG(debug2) << *this;
171 #endif
172 
173  LOG(trace) << "done adding frequencies";
174 }
