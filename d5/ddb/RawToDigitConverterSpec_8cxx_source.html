34 void RawToDigitConverterSpec::run(framework::ProcessingContext& ctx)
35 {
36  // Cache digits from bunch crossings as the component reads timeframes from many links consecutively
37  std::map<o2::InteractionRecord, std::shared_ptr<std::vector<o2::cpv::Digit>>> digitBuffer; // Internal digit buffer
38  int firstEntry = 0;
39  mOutputHWErrors.clear();
40 
41  if (!mCalibParams) {
42  if (o2::cpv::CPVSimParams::Instance().mCCDBPath.compare("localtest") == 0) {
43  mCalibParams = std::make_unique<o2::cpv::CalibParams>(1); // test default calibration
44  LOG(INFO) << "No reading calibration from ccdb requested, set default";
45  } else {
46  LOG(INFO) << "Getting calibration object from ccdb";
47  //TODO: configuring ccdb address from config file, readign proper calibration/BadMap and updateing if necessary
48  o2::ccdb::CcdbApi ccdb;
49  std::map<std::string, std::string> metadata;
50  ccdb.init("http://ccdb-test.cern.ch:8080"); // or http://localhost:8080 for a local installation
51  // auto tr = triggerbranch.begin();
52  // double eventTime = -1;
53  // if(tr!=triggerbranch.end()){
54  // eventTime = (*tr).getBCData().getTimeNS() ;
55  // }
56  // mCalibParams = ccdb.retrieveFromTFileAny<o2::cpv::CalibParams>("CPV/Calib", metadata, eventTime);
57  // if (!mCalibParams) {
58  // LOG(FATAL) << "Can not get calibration object from ccdb";
59  // }
60  }
61  }
62 
63  if (!mBadMap) {
64  if (o2::cpv::CPVSimParams::Instance().mCCDBPath.compare("localtest") == 0) {
65  mBadMap = std::make_unique<o2::cpv::BadChannelMap>(1); // test default calibration
66  LOG(INFO) << "No reading bad map from ccdb requested, set default";
67  } else {
68  LOG(INFO) << "Getting bad map object from ccdb";
69  o2::ccdb::CcdbApi ccdb;
70  std::map<std::string, std::string> metadata;
71  ccdb.init("http://ccdb-test.cern.ch:8080"); // or http://localhost:8080 for a local installation
72  // auto tr = triggerbranch.begin();
73  // double eventTime = -1;
74  // if(tr!=triggerbranch.end()){
75  // eventTime = (*tr).getBCData().getTimeNS() ;
76  // }
77  // mBadMap = ccdb.retrieveFromTFileAny<o2::cpv::BadChannelMap>("CPV/BadMap", metadata, eventTime);
78  // if (!mBadMap) {
79  // LOG(FATAL) << "Can not get bad map object from ccdb";
80  // }
81  }
82  }
83 
84  for (const auto& rawData : framework::InputRecordWalker(ctx.inputs())) {
85 
86  // enum RawErrorType_t {
87  // kOK, ///< NoError
88  // kNO_PAYLOAD, ///< No payload per ddl
89  // kHEADER_DECODING,
90  // kPAGE_NOTFOUND,
91  // kPAYLOAD_DECODING,
92  // kHEADER_INVALID,
93  // kRCU_TRAILER_ERROR, ///< RCU trailer cannot be decoded or invalid
94  // kRCU_VERSION_ERROR, ///< RCU trailer version not matching with the version in the raw header
95  // kRCU_TRAILER_SIZE_ERROR, ///< RCU trailer size length
96  // kSEGMENT_HEADER_ERROR,
97  // kROW_HEADER_ERROR,
98  // kEOE_HEADER_ERROR,
99  // kPADERROR,
100  // kPadAddress
101  // };
102 
103  o2::cpv::RawReaderMemory rawreader(o2::framework::DataRefUtils::as<const char>(rawData));
104  // loop over all the DMA pages
105  while (rawreader.hasNext()) {
106  try {
107  rawreader.next();
108  } catch (RawErrorType_t e) {
109  LOG(ERROR) << "Raw decoding error " << (int)e;
110  //add error list
111  mOutputHWErrors.emplace_back(5, 0, 0, 0, e); //Put general errors to non-existing DDL5
112  //if problem in header, abandon this page
113  if (e == RawErrorType_t::kPAGE_NOTFOUND ||
114  e == RawErrorType_t::kHEADER_DECODING ||
115  e == RawErrorType_t::kHEADER_INVALID) {
116  break;
117  }
118  //if problem in payload, try to continue
119  continue;
120  }
121  auto& header = rawreader.getRawHeader();
122  auto triggerBC = o2::raw::RDHUtils::getTriggerBC(header);
123  auto triggerOrbit = o2::raw::RDHUtils::getTriggerOrbit(header);
124  // auto ddl = o2::raw::RDHUtils::getFEEID(header);
125  auto mod = o2::raw::RDHUtils::getLinkID(header) + 2; //ddl=0,1,2 -> mod=2,3,4
126  // if(ddl != mDDL){
127  // LOG(ERROR) << "DDL from header "<< ddl << " != configured DDL=" << mDDL;
128  // }
129 
130  o2::InteractionRecord currentIR(triggerBC, triggerOrbit);
131  std::shared_ptr<std::vector<o2::cpv::Digit>> currentDigitContainer;
132  auto found = digitBuffer.find(currentIR);
133  if (found == digitBuffer.end()) {
134  currentDigitContainer = std::make_shared<std::vector<o2::cpv::Digit>>();
135  digitBuffer[currentIR] = currentDigitContainer;
136  } else {
137  currentDigitContainer = found->second;
138  }
139  //
140  if (mod > o2::cpv::Geometry::kNMod) { //only 3 correct modules:2,3,4
141  LOG(ERROR) << "module=" << mod << "do not exist";
142  mOutputHWErrors.emplace_back(6, mod, 0, 0, kHEADER_INVALID); //Add non-existing DDL as DDL 5
143  continue; //skip STU mod
144  }
145  // use the altro decoder to decode the raw data, and extract the RCU trailer
146  o2::cpv::RawDecoder decoder(rawreader);
147  RawErrorType_t err = decoder.decode();
148 
149  if (err != kOK) {
150  //TODO handle severe errors
151  //TODO: probably careful conversion of decoder errors to Fitter errors?
152  mOutputHWErrors.emplace_back(mod, 1, 0, 0, err); //assign general header errors to non-existing FEE 16
153  }
154  // Loop over all the channels
155  for (uint32_t adch : decoder.getDigits()) {
156  AddressCharge ac = {adch};
157  unsigned short absId = ac.Address;
158  //test bad map
159  if (mBadMap->isChannelGood(absId)) {
160  if (ac.Charge > o2::cpv::CPVSimParams::Instance().mZSthreshold) {
161  float amp = mCalibParams->getGain(absId) * ac.Charge;
162  currentDigitContainer->emplace_back(absId, amp, -1);
163  }
164  }
165  }
166  //Check and send list of hwErrors
167  for (auto& er : decoder.getErrors()) {
168  mOutputHWErrors.push_back(er);
169  }
170  } //RawReader::hasNext
171  }
172 
173  // Loop over BCs, sort digits with increasing digit ID and write to output containers
174  mOutputDigits.clear();
175  mOutputTriggerRecords.clear();
176  for (auto [bc, digits] : digitBuffer) {
177  int prevDigitSize = mOutputDigits.size();
178  if (digits->size()) {
179  // Sort digits according to digit ID
180  std::sort(digits->begin(), digits->end(), [](o2::cpv::Digit& lhs, o2::cpv::Digit& rhs) { return lhs.getAbsId() < rhs.getAbsId(); });
181 
182  for (auto digit : *digits) {
183  mOutputDigits.push_back(digit);
184  }
185  }
186 
187  mOutputTriggerRecords.emplace_back(bc, prevDigitSize, mOutputDigits.size() - prevDigitSize);
188  }
189  digitBuffer.clear();
190 
191  LOG(INFO) << "[CPVRawToDigitConverter - run] Writing " << mOutputDigits.size() << " digits ...";
192  ctx.outputs().snapshot(o2::framework::Output{"CPV", "DIGITS", 0, o2::framework::Lifetime::Timeframe}, mOutputDigits);
193  ctx.outputs().snapshot(o2::framework::Output{"CPV", "DIGITTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggerRecords);
194  ctx.outputs().snapshot(o2::framework::Output{"CPV", "RAWHWERRORS", 0, o2::framework::Lifetime::Timeframe}, mOutputHWErrors);
195 }
