44 void TRDGlobalTracking::init(InitContext& ic)
45 {
46  //-------- init geometry and field --------//
47  o2::base::GeometryManager::loadGeometry();
48  o2::base::Propagator::initFieldFromGRP(o2::base::NameConf::getGRPFileName());
49  auto geo = Geometry::instance();
50  geo->createPadPlaneArray();
51  geo->createClusterMatrixArray();
52  mFlatGeo = std::make_unique<GeometryFlat>(*geo);
53 
54  //-------- init GPU reconstruction --------//
55  GPURecoStepConfiguration cfgRecoStep;
56  cfgRecoStep.steps = GPUDataTypes::RecoStep::NoRecoStep;
57  cfgRecoStep.inputs.clear();
58  cfgRecoStep.outputs.clear();
59  mRec = GPUReconstruction::CreateInstance("CPU", true);
60  mRec->SetSettings(o2::base::Propagator::Instance()->getNominalBz(), &cfgRecoStep);
61 
62  mChainTracking = mRec->AddChain<GPUChainTracking>();
63 
64  mTracker = new GPUTRDTracker();
65  mTracker->SetNCandidates(mRec->GetProcessingSettings().trdNCandidates); // must be set before initialization
66  mTracker->SetProcessPerTimeFrame();
67  mTracker->SetNMaxCollisions(mRec->GetProcessingSettings().trdNMaxCollisions);
68 
69  mRec->RegisterGPUProcessor(mTracker, false);
70  mChainTracking->SetTRDGeometry(std::move(mFlatGeo));
71  if (mRec->Init()) {
72  LOG(FATAL) << "GPUReconstruction could not be initialized";
73  }
74 
75  mTracker->PrintSettings();
76 
77  mTimer.Stop();
78  mTimer.Reset();
79 }

81 void TRDGlobalTracking::run(ProcessingContext& pc)
82 {
83  mTimer.Start(false);
84  const auto tracksITSTPC = pc.inputs().get<gsl::span<o2::dataformats::TrackTPCITS>>("tpcitstrack");
85  const auto trackletsTRD = pc.inputs().get<gsl::span<o2::trd::Tracklet64>>("trdtracklets");
86  const auto triggerRecords = pc.inputs().get<gsl::span<o2::trd::TriggerRecord>>("trdtriggerrec");
87 
88  int nTracks = tracksITSTPC.size();
89  int nCollisions = triggerRecords.size();
90  int nTracklets = trackletsTRD.size();
91  LOGF(INFO, "There are %i tracklets in total from %i trigger records", nTracklets, nCollisions);
92 
93  std::vector<float> trdTriggerTimes;
94  std::vector<int> trdTriggerIndices;
95 
96  for (int iEv = 0; iEv < nCollisions; ++iEv) {
97 #ifdef MS_GSL_V3
98  const auto& trg = triggerRecords[iEv];
99 #else
100  const auto& trg = triggerRecords.at(iEv);
101 #endif
102  int nTrackletsCurrent = trg.getNumberOfObjects();
103  int iFirstTracklet = trg.getFirstEntry();
104  int64_t evTime = trg.getBCData().toLong() * o2::constants::lhc::LHCBunchSpacingNS; // event time in ns
105  trdTriggerTimes.push_back(evTime / 1000.);
106  trdTriggerIndices.push_back(iFirstTracklet);
107  LOGF(DEBUG, "Event %i: Occured at %li us after SOR, contains %i tracklets, index of first tracklet is %i", iEv, evTime / 1000, nTrackletsCurrent, iFirstTracklet);
108  }
109 
110  mTracker->Reset();
111 
112  mChainTracking->mIOPtrs.nMergedTracks = nTracks;
113  mChainTracking->mIOPtrs.nTRDTracklets = nTracklets;
114  mChainTracking->AllocateIOMemory();
115  mRec->PrepareEvent();
116  mRec->SetupGPUProcessor(mTracker, true);
117 
118  LOG(DEBUG) << "Start loading input into TRD tracker";
119  // load everything into the tracker
120  int nTracksLoaded = 0;
121  for (int iTrk = 0; iTrk < nTracks; ++iTrk) {
122  const auto& match = tracksITSTPC[iTrk];
123  const auto& trk = match.getParamOut();
124  GPUTRDTrack trkLoad;
125  trkLoad.setX(trk.getX());
126  trkLoad.setAlpha(trk.getAlpha());
127  for (int i = 0; i < 5; ++i) {
128  trkLoad.setParam(trk.getParam(i), i);
129  }
130  for (int i = 0; i < 15; ++i) {
131  trkLoad.setCov(trk.getCov()[i], i);
132  }
133  trkLoad.setTime(match.getTimeMUS().getTimeStamp());
134  if (mTracker->LoadTrack(trkLoad)) {
135  continue;
136  }
137  ++nTracksLoaded;
138  LOGF(DEBUG, "Loaded track %i with time %f", nTracksLoaded, trkLoad.getTime());
139  }
140 
141  for (int iTrklt = 0; iTrklt < nTracklets; ++iTrklt) {
142  auto trklt = trackletsTRD[iTrklt];
143  unsigned int trkltWord = 0; // DUMMY
144  GPUTRDTrackletWord trkltLoad;
145  trkltLoad.SetId(iTrklt);
146  trkltLoad.SetHCId(trklt.getHCID());
147  trkltLoad.SetTrackletWord(trkltWord);
148  if (mTracker->LoadTracklet(trkltLoad) > 0) {
149  LOG(WARNING) << "Could not load tracklet " << iTrklt;
150  }
151  }
152  mTracker->SetTriggerRecordTimes(&(trdTriggerTimes[0]));
153  mTracker->SetTriggerRecordIndices(&(trdTriggerIndices[0]));
154  mTracker->SetNCollisions(nCollisions);
155  //mTracker->DumpTracks();
156  mTracker->DoTracking(mChainTracking);
157  //mTracker->DumpTracks();
158 
159  std::vector<GPUTRDTrack> tracksOut(mTracker->NTracks());
160  std::copy(mTracker->Tracks(), mTracker->Tracks() + mTracker->NTracks(), tracksOut.begin());
161  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MATCHTRD", 0, Lifetime::Timeframe}, tracksOut);
162 
163  mTimer.Stop();
164 }

172 DataProcessorSpec getTRDGlobalTrackingSpec(bool useMC)
173 {
174  std::vector<InputSpec> inputs;
175  std::vector<OutputSpec> outputs;
176  inputs.emplace_back("tpcitstrack", "GLO", "TPCITS", 0, Lifetime::Timeframe);
177  inputs.emplace_back("trdtracklets", o2::header::gDataOriginTRD, "TRACKLETS", 0, Lifetime::Timeframe);
178  inputs.emplace_back("trdtriggerrec", o2::header::gDataOriginTRD, "TRKTRGRD", 0, Lifetime::Timeframe);
179 
180  if (useMC) {
181  LOG(FATAL) << "MC usage must be disabled for this workflow, since it is not yet implemented";
182  }
183 
184  outputs.emplace_back(o2::header::gDataOriginTRD, "MATCHTRD", 0, Lifetime::Timeframe);
185 
186  return DataProcessorSpec{
187  "trd-globaltracking",
188  inputs,
189  outputs,
190  AlgorithmSpec{adaptFromTask<TRDGlobalTracking>(useMC)},
191  Options{}};
192 }
