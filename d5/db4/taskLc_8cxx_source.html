82  registry.fill(HIST("hptprong2"), candidate.ptProng2());
83  registry.fill(HIST("hdeclength"), candidate.decayLength());
84  registry.fill(HIST("hd0Prong0"), candidate.impactParameter0());
85  registry.fill(HIST("hd0Prong1"), candidate.impactParameter1());
86  registry.fill(HIST("hd0Prong2"), candidate.impactParameter2());
87  registry.fill(HIST("hCt"), CtLc(candidate));
88  registry.fill(HIST("hCPA"), candidate.cpa());
89  registry.fill(HIST("hEta"), candidate.eta());

143  registry.fill(HIST("hEtaRecSig"), candidate.eta());
144  } else {
145  registry.fill(HIST("hPtRecBg"), candidate.pt());
146  registry.fill(HIST("hCPARecBg"), candidate.cpa());
147  registry.fill(HIST("hEtaRecBg"), candidate.eta());
