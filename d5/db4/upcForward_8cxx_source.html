46  void init(o2::framework::InitContext&)
47  {
48  TString SelectionCuts[6] = {"NoSelection", "CMup11Trigger", "twotracks", "oppositecharge", "-2.5<Eta<-4", "Pt<1"};
49  for (int i = 0; i < 6; i++) {
50  hSelectionCounter->GetXaxis()->SetBinLabel(i + 1, SelectionCuts[i].Data());
51  }
52  }
53  void process(aod::BC const& bc, aod::Muons const& tracksMuon)
54  {
55  hSelectionCounter->Fill(0);
56 
57  int iMuonTracknumber = 0;
58  TLorentzVector p1, p2, p;
59  bool ispositive = kFALSE;
60  bool isnegative = kFALSE;
61  /*this code below is suggested by evgeny.
62  this code is now hardcoded for run 246392 as we are not sure if trigger id is same for all the runs*/
63  uint64_t classIndexMUP11 = 54; // 246392
64  bool isMUP11fired = bc.triggerMask() & (1ull << classIndexMUP11);
65  if (!isMUP11fired) {
66  return;
67  }
68  LOGF(info, "mup11 fired");
69  hSelectionCounter->Fill(1);
70  for (auto& muon : tracksMuon) {
71  hCharge->Fill(muon.charge());
72  iMuonTracknumber++;
73  if (muon.charge() > 0) {
74  p1.SetXYZM(muon.px(), muon.py(), muon.pz(), mmuon);
75  ispositive = kTRUE;
76  }
77  if (muon.charge() < 0) {
78  p2.SetXYZM(muon.px(), muon.py(), muon.pz(), mmuon);
79  isnegative = kTRUE;
80  }
81  }
82  if (iMuonTracknumber != 2) {
83  return;
84  }
85  hSelectionCounter->Fill(2);
86  if (!ispositive || !isnegative) {
87  return;
88  }
89  hSelectionCounter->Fill(3);
90 
91  if (-4 < p1.Eta() < -2.5 || -4 < p2.Eta() < -2.5) {
92  return;
93  }
94  hSelectionCounter->Fill(4);
95  p = p1 + p2;
96  if (p.Pt() > 1) {
97  return;
98  }
99  hSelectionCounter->Fill(5);
100  hPt->Fill(p.Pt());
101  hPx->Fill(p.Px());
102  hPy->Fill(p.Py());
103  hPz->Fill(p.Pz());
104  hRap->Fill(p.Rapidity());
105  hMass->Fill(p.M());
106  hPhi->Fill(p.Phi());
107 
108  } //end of process
109 };
