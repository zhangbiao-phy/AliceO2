167 template <>
168 inline DeliveryType read(const std::string& str)
169 {
170  if (str == "Raw/Int") {
171  return RAW_INT;
172  } else if (str == "Raw/Uint") {
173  return RAW_UINT;
174  } else if (str == "Raw/Double") {
175  return RAW_DOUBLE;
176  } else if (str == "Raw/Bool") {
177  return RAW_BOOL;
178  } else if (str == "Raw/Char") {
179  return RAW_CHAR;
180  } else if (str == "Raw/String") {
181  return RAW_STRING;
182  } else if (str == "Raw/Binary") {
183  return RAW_BINARY;
184  } else if (str == "Raw/Time") {
185  return RAW_TIME;
186  } else if (str == "DPVAL/Int") {
187  return DPVAL_INT;
188  } else if (str == "DPVAL/Uint") {
189  return DPVAL_UINT;
190  } else if (str == "DPVAL/Double") {
191  return DPVAL_DOUBLE;
192  } else if (str == "DPVAL/Bool") {
193  return DPVAL_BOOL;
194  } else if (str == "DPVAL/Char") {
195  return DPVAL_CHAR;
196  } else if (str == "DPVAL/String") {
197  return DPVAL_STRING;
198  } else if (str == "DPVAL/Binary") {
199  return DPVAL_BINARY;
200  } else if (str == "DPVAL/Time") {
201  return DPVAL_TIME;
202  } else if (str == "Void") {
203  return VOID;
204  } else {
205  throw std::domain_error("\"" + str +
206  "\" doesn't represent a DeliveryType.");
207  }
208 }
