48  void init(o2::framework::InitContext&)
49  {
50  ccdb->setURL(url.value);
51  ccdb->setTimestamp(timestamp.value);
52  ccdb->setCaching(true);
53  ccdb->setLocalObjectValidityChecking();
54  // Not later than now objects
55  ccdb->setCreatedNotAfter(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
56  //
57  const std::vector<float> p = {0.008, 0.008, 0.002, 40.0};
58  response.SetParameters(DetectorResponse::kSigma, p);
59  const std::string fname = paramfile.value;
60  if (!fname.empty()) { // Loading the parametrization from file
61  response.LoadParamFromFile(fname.data(), sigmaname.value, DetectorResponse::kSigma);
62  } else { // Loading it from CCDB
63  const std::string path = "Analysis/PID/TOF";
64  response.LoadParam(DetectorResponse::kSigma, ccdb->getForTimeStamp<Parametrization>(path + "/" + sigmaname.value, timestamp.value));
65  }

150  template <typename T>
151  void makelogaxis(T h)
152  {
153  const int nbins = h->GetNbinsX();
154  double binp[nbins + 1];
155  double max = h->GetXaxis()->GetBinUpEdge(nbins);
156  double min = h->GetXaxis()->GetBinLowEdge(1);
157  if (min <= 0) {
158  min = 0.00001;
159  }
160  double lmin = TMath::Log10(min);
161  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
162  for (int i = 0; i < nbins; i++) {
163  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
164  }
165  binp[nbins] = max + 1;
166  h->GetXaxis()->Set(nbins, binp);
167  }

207  void fillParticleHistos(const T& t, const float tof, const float exp_diff, const float nsigma)

214  void process(aod::Collision const& collision, soa::Join<aod::Tracks, aod::TracksExtra, aod::pidRespTOF, aod::pidRespTOFbeta> const& tracks)
215  {
216  const float collisionTime_ps = collision.collisionTime() * 1000.f;
217  histos.fill(HIST("event/vertexz"), collision.posZ());
218  histos.fill(HIST("event/colltime"), collisionTime_ps);
219 
220  for (auto t : tracks) {
221  //
222  if (t.tofSignal() < 0) { // Skipping tracks without TOF
223  continue;
224  }
225 
226  const float tof = t.tofSignal() - collisionTime_ps;
227 
228  //
229  histos.fill(HIST("event/tofsignal"), t.p(), t.tofSignal());
230  histos.fill(HIST("event/tofbeta"), t.p(), t.beta());
231  //
232  fillParticleHistos<0>(t, tof, t.tofExpSignalDiffEl(), t.tofNSigmaEl());
233  fillParticleHistos<1>(t, tof, t.tofExpSignalDiffMu(), t.tofNSigmaMu());
234  fillParticleHistos<2>(t, tof, t.tofExpSignalDiffPi(), t.tofNSigmaPi());
235  fillParticleHistos<3>(t, tof, t.tofExpSignalDiffKa(), t.tofNSigmaKa());
236  fillParticleHistos<4>(t, tof, t.tofExpSignalDiffPr(), t.tofNSigmaPr());
237  fillParticleHistos<5>(t, tof, t.tofExpSignalDiffDe(), t.tofNSigmaDe());
238  fillParticleHistos<6>(t, tof, t.tofExpSignalDiffTr(), t.tofNSigmaTr());
239  fillParticleHistos<7>(t, tof, t.tofExpSignalDiffHe(), t.tofNSigmaHe());
240  fillParticleHistos<8>(t, tof, t.tofExpSignalDiffAl(), t.tofNSigmaAl());
241  }
242  }
