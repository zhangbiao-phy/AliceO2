71  registry.fill(HIST("hPt"), candidate.pt());

74  registry.fill(HIST("hDecayLength"), candidate.decayLength());

78  registry.fill(HIST("hCPAxy"), candidate.cpaXY());

81  registry.fill(HIST("hImpactParameterProngSqSum"), candidate.impactParameterProngSqSum());
82  registry.fill(HIST("hDecayLengthError"), candidate.errorDecayLength());
