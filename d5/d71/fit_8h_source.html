56 template <typename T>
57 TFitResultPtr fit(const size_t nBins, const T* arr, const T xMin, const T xMax, TF1& func, std::string_view option = "")
58 {
59  Foption_t fitOption;
60  ROOT::Fit::FitOptionsMake(ROOT::Fit::kHistogram, option.data(), fitOption);
61 
62  ROOT::Fit::DataRange range(xMin, xMax);
63  ROOT::Fit::DataOptions opt;
64  ROOT::Fit::BinData fitdata(opt, range);
65  fitdata.Initialize(nBins, 1);
66 
67  // create an empty TFitResult
68  std::shared_ptr<TFitResult> tfr(new TFitResult());
69  // create the fitter from an empty fit result
70  //std::shared_ptr<ROOT::Fit::Fitter> fitter(new ROOT::Fit::Fitter(std::static_pointer_cast<ROOT::Fit::FitResult>(tfr) ) );
71  ROOT::Fit::Fitter fitter(tfr);
72  //ROOT::Fit::FitConfig & fitConfig = fitter->Config();
73 
74  const double binWidth = double(xMax - xMin) / double(nBins);
75 
76  for (Int_t ibin = 0; ibin < nBins; ibin++) {
77  const double x = double(xMin) + double(ibin + 0.5) * binWidth;
78  const double y = double(arr[ibin]);
79  const double ey = std::sqrt(y);
80  fitdata.Add(x, y, ey);
81  }
82 
83  const int special = func.GetNumber();
84  const int npar = func.GetNpar();
85  bool linear = func.IsLinear();
86  if (special == 299 + npar) {
87  linear = kTRUE; // for polynomial functions
88  }
89  // do not use linear fitter in these case
90  if (fitOption.Bound || fitOption.Like || fitOption.Errors || fitOption.Gradient || fitOption.More || fitOption.User || fitOption.Integral || fitOption.Minuit) {
91  linear = kFALSE;
92  }
93 
94  if (special != 0 && !fitOption.Bound && !linear) {
95  if (special == 100) {
96  ROOT::Fit::InitGaus(fitdata, &func); // gaussian
97  } else if (special == 400) {
98  ROOT::Fit::InitGaus(fitdata, &func); // landau (use the same)
99  } else if (special == 200) {
100  ROOT::Fit::InitExpo(fitdata, &func); // exponential
101  }
102  }
103 
104  if ((linear || fitOption.Gradient)) {
105  fitter.SetFunction(ROOT::Math::WrappedMultiTF1(func));
106  } else {
107  fitter.SetFunction(static_cast<const ROOT::Math::IParamMultiFunction&>(ROOT::Math::WrappedMultiTF1(func)));
108  }
109 
110  // standard least square fit
111  const bool fitok = fitter.Fit(fitdata, fitOption.ExecPolicy);
112  if (!fitok) {
113  LOGP(warning, "bad fit");
114  }
115 
116  return TFitResultPtr(tfr);
117 }

140 template <typename T>
141 Double_t fitGaus(const size_t nBins, const T* arr, const T xMin, const T xMax, std::vector<T>& param)
142 {
143  static TLinearFitter fitter(3, "pol2");
144  static TMatrixD mat(3, 3);
145  static Double_t kTol = mat.GetTol();
146  fitter.StoreData(kFALSE);
147  fitter.ClearPoints();
148  TVectorD par(3);
149  TVectorD sigma(3);
150  TMatrixD A(3, 3);
151  TMatrixD b(3, 1);
152  T rms = TMath::RMS(nBins, arr);
153  T max = TMath::MaxElement(nBins, arr);
154  T binWidth = (xMax - xMin) / T(nBins);
155 
156  Float_t meanCOG = 0;
157  Float_t rms2COG = 0;
158  Float_t sumCOG = 0;
159 
160  Float_t entries = 0;
161  Int_t nfilled = 0;
162 
163  param.resize(4);
164  param[0] = 0.;
165  param[1] = 0.;
166  param[2] = 0.;
167  param[3] = 0.;
168 
169  for (Int_t i = 0; i < nBins; i++) {
170  entries += arr[i];
171  if (arr[i] > 0) {
172  nfilled++;
173  }
174  }
175 
176  // TODO: Check why this is needed
177  if (max < 4) {
178  return -4;
179  }
180  if (entries < 12) {
181  return -4;
182  }
183 
184  if (rms < kTol) {
185  return -4;
186  }
187 
188  param[3] = entries;
189 
190  Int_t npoints = 0;
191  for (Int_t ibin = 0; ibin < nBins; ibin++) {
192  Float_t entriesI = arr[ibin];
193  if (entriesI > 1) {
194  Double_t xcenter = xMin + (ibin + 0.5) * binWidth;
195  Double_t error = 1. / TMath::Sqrt(entriesI);
196  Double_t val = TMath::Log(Float_t(entriesI));
197  fitter.AddPoint(&xcenter, val, error);
198  if (npoints < 3) {
199  A(npoints, 0) = 1;
200  A(npoints, 1) = xcenter;
201  A(npoints, 2) = xcenter * xcenter;
202  b(npoints, 0) = val;
203  meanCOG += xcenter * entriesI;
204  rms2COG += xcenter * entriesI * xcenter;
205  sumCOG += entriesI;
206  }
207  npoints++;
208  }
209  }
210 
211  Double_t chi2 = 0;
212  if (npoints >= 3) {
213  if (npoints == 3) {
214  //analytic calculation of the parameters for three points
215  A.Invert();
216  TMatrixD res(1, 3);
217  res.Mult(A, b);
218  par[0] = res(0, 0);
219  par[1] = res(0, 1);
220  par[2] = res(0, 2);
221  chi2 = -3.;
222  } else {
223  // use fitter for more than three points
224  fitter.Eval();
225  fitter.GetParameters(par);
226  fitter.GetCovarianceMatrix(mat);
227  chi2 = fitter.GetChisquare() / Double_t(npoints);
228  }
229  if (TMath::Abs(par[1]) < kTol) {
230  return -4;
231  }
232  if (TMath::Abs(par[2]) < kTol) {
233  return -4;
234  }
235 
236  param[1] = T(par[1] / (-2. * par[2]));
237  param[2] = T(1. / TMath::Sqrt(TMath::Abs(-2. * par[2])));
238  Double_t lnparam0 = par[0] + par[1] * param[1] + par[2] * param[1] * param[1];
239  if (lnparam0 > 307) {
240  return -4;
241  }
242  param[0] = TMath::Exp(lnparam0);
243 
244  return chi2;
245  }
246 
247  if (npoints == 2) {
248  //use center of gravity for 2 points
249  meanCOG /= sumCOG;
250  rms2COG /= sumCOG;
251  param[0] = max;
252  param[1] = meanCOG;
253  param[2] = TMath::Sqrt(TMath::Abs(meanCOG * meanCOG - rms2COG));
254  chi2 = -2.;
255  }
256  if (npoints == 1) {
257  meanCOG /= sumCOG;
258  param[0] = max;
259  param[1] = meanCOG;
260  param[2] = binWidth / TMath::Sqrt(12);
261  chi2 = -1.;
262  }
263  return chi2;
264 }

283 template <typename T>
284 StatisticsData getStatisticsData(const T* arr, const size_t nBins, const double xMin, const double xMax)
285 {
286  double mean = 0;
287  double rms2 = 0;
288  double sum = 0;
289  size_t npoints = 0;
290 
291  double binWidth = (xMax - xMin) / (double)nBins;
292 
293  StatisticsData data;
294  // in case something went wrong the COG is the histogram lower limit
295  data.mCOG = xMin;
296 
297  for (size_t ibin = 0; ibin < nBins; ++ibin) {
298  double entriesI = (double)arr[ibin];
299  double xcenter = xMin + (ibin + 0.5) * binWidth; // +0.5 to shift to bin centre
300  if (entriesI > 0) {
301  mean += xcenter * entriesI;
302  rms2 += xcenter * entriesI * xcenter;
303  sum += entriesI;
304  ++npoints;
305  }
306  }
307  if (sum == 0) {
308  return data;
309  }
310  mean /= sum;
311 
312  data.mCOG = mean;
313  // exception in case of only one bin is filled
314  // set the standard deviation to bin width over sqrt(12)
315  rms2 /= sum;
316  if (npoints == 1) {
317  data.mStdDev = binWidth / std::sqrt(12.);
318  } else {
319  data.mStdDev = std::sqrt(std::abs(rms2 - mean * mean));
320  }
321 
322  data.mSum = sum;
323 
324  return data;
325 }

332 template <typename T, typename R = double>
333 R median(std::vector<T> v)
334 {
335  if (v.empty()) {
336  return R{};
337  }
338  auto n = v.size() / 2;
339  nth_element(v.begin(), v.begin() + n, v.end());
340  auto med = R{v[n]};
341  if (!(v.size() & 1)) { //If the set size is even
342  auto max_it = max_element(v.begin(), v.begin() + n);
343  med = R{(*max_it + med) / 2.0};
344  }
345  return med;
346 }

382 template <typename T>
383 bool LTMUnbinned(const std::vector<T>& data, std::vector<size_t>& index, std::array<float, 7>& params, float fracKeep)
384 {
385  int nPoints = data.size();
386  std::vector<float> w(2 * nPoints);
387  int nKeep = nPoints * fracKeep;
388  if (nKeep > nPoints) {
389  nKeep = nPoints;
390  }
391  if (nKeep < 2) {
392  return false;
393  }
394  // sort in increasing order
395  SortData(data, index);
396  // build cumulants
397  double sum1 = 0.0;
398  double sum2 = 0.0;
399  for (int i = 0; i < nPoints; i++) {
400  double x = data[index[i]];
401  sum1 += x;
402  sum2 += x * x;
403  w[i] = sum1;
404  w[i + nPoints] = sum2;
405  }
406  double maxRMS = sum2 + 1e6;
407  params[0] = nKeep;
408  int limI = nPoints - nKeep + 1; // lowest possible bin to accept
409  for (int i = 0; i < limI; i++) {
410  const int limJ = i + nKeep - 1; // highest accepted bin
411  sum1 = static_cast<double>(w[limJ]) - static_cast<double>(i ? w[i - 1] : 0.);
412  sum2 = static_cast<double>(w[nPoints + limJ]) - static_cast<double>(i ? w[nPoints + i - 1] : 0.);
413  const double mean = sum1 / nKeep;
414  const double rms2 = sum2 / nKeep - mean * mean;
415  if (rms2 > maxRMS) {
416  continue;
417  }
418  maxRMS = rms2;
419  params[1] = mean;
420  params[2] = rms2;
421  params[5] = i;
422  params[6] = limJ;
423  }
424  //
425  if (params[2] < 0) {
426  LOG(error) << "Rounding error: RMS = " << params[2] << " < 0";
427  return false;
428  }
429  params[2] = std::sqrt(params[2]);
430  params[3] = params[2] / std::sqrt(params[0]); // error on mean
431  params[4] = params[3] / std::sqrt(2.0); // error on RMS
432  return true;
433 }

461 template <typename T>
462 bool LTMUnbinnedSig(const std::vector<T>& data, std::vector<size_t>& index, std::array<float, 7>& params, float fracKeepMin, float sigTgt, bool sorted = false)
463 {
464  int nPoints = data.size();
465  std::vector<double> wx(nPoints);
466  std::vector<double> wx2(nPoints);
467 
468  if (!sorted) {
469  // sort in increasing order
470  SortData(data, index);
471  } else {
472  // array is already sorted
473  std::iota(index.begin(), index.end(), 0);
474  }
475  // build cumulants
476  double sum1 = 0.0;
477  double sum2 = 0.0;
478  for (int i = 0; i < nPoints; i++) {
479  double x = data[index[i]];
480  sum1 += x;
481  sum2 += x * x;
482  wx[i] = sum1;
483  wx2[i] = sum2;
484  }
485  int keepMax = nPoints;
486  int keepMin = fracKeepMin * nPoints;
487  if (keepMin > keepMax) {
488  keepMin = keepMax;
489  }
490  float sigTgt2 = sigTgt * sigTgt;
491  //
492  while (true) {
493  double maxRMS = wx2.back() + 1e6;
494  int keepN = (keepMax + keepMin) / 2;
495  if (keepN < 2) {
496  return false;
497  }
498  params[0] = keepN;
499  int limI = nPoints - keepN + 1;
500  for (int i = 0; i < limI; ++i) {
501  const int limJ = i + keepN - 1;
502  sum1 = wx[limJ] - (i ? wx[i - 1] : 0.);
503  sum2 = wx2[limJ] - (i ? wx2[i - 1] : 0.);
504  const double mean = sum1 / keepN;
505  const double rms2 = sum2 / keepN - mean * mean;
506  if (rms2 > maxRMS) {
507  continue;
508  }
509  maxRMS = rms2;
510  params[1] = mean;
511  params[2] = rms2;
512  params[5] = i;
513  params[6] = limJ;
514  }
515  if (maxRMS < sigTgt2) {
516  keepMin = keepN;
517  } else {
518  keepMax = keepN;
519  }
520  if (keepMin >= keepMax - 1) {
521  break;
522  }
523  }
524  params[2] = std::sqrt(params[2]);
525  params[3] = params[2] / std::sqrt(params[0]); // error on mean
526  params[4] = params[3] / std::sqrt(2.0); // error on RMS
527  return true;
528 }
