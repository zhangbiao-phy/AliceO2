70  template <typename T>
71  void makelogaxis(T h)
72  {
73  const int nbins = h->GetNbinsX();
74  double binp[nbins + 1];
75  double max = h->GetXaxis()->GetBinUpEdge(nbins);
76  double min = h->GetXaxis()->GetBinLowEdge(1);
77  if (min <= 0) {
78  min = 0.00001;
79  }
80  double lmin = TMath::Log10(min);
81  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
82  for (int i = 0; i < nbins; i++) {
83  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
84  }
85  binp[nbins] = max + 1;
86  h->GetXaxis()->Set(nbins, binp);
87  }

120  template <uint8_t pidIndex, typename T, typename TT>
121  void fillNsigma(const T& track, const TT& mcParticles, const float& nsigma)
122  {
123  if (abs(track.mcParticle().pdgCode()) == PDGs[pidIndex]) {
124  histos.fill(HIST(hnsigmaMC[pidIndex]), track.pt(), nsigma);
125 
126  if (MC::isPhysicalPrimary(mcParticles, track.mcParticle())) { // Selecting primaries
127  histos.fill(HIST(hnsigmaMCprm[pidIndex]), track.pt(), nsigma);
128  } else {
129  histos.fill(HIST(hnsigmaMCsec[pidIndex]), track.pt(), nsigma);
130  }
131  }
132  }

134  void process(aod::Collision const& collision,
135  soa::Join<aod::Tracks, aod::TracksExtra, aod::pidRespTOF, aod::McTrackLabels> const& tracks,
136  aod::McParticles& mcParticles)
137  {
138  const float collisionTime_ps = collision.collisionTime() * 1000.f;
139  for (auto t : tracks) {
140  //
141  if (t.tofSignal() < 0) { // Skipping tracks without TOF
142  continue;
143  }
144  float nsigma = -999.f;
145  if constexpr (pid_type == 0) {
146  nsigma = t.tofNSigmaEl();
147  } else if constexpr (pid_type == 1) {
148  nsigma = t.tofNSigmaMu();
149  } else if constexpr (pid_type == 2) {
150  nsigma = t.tofNSigmaPi();
151  } else if constexpr (pid_type == 3) {
152  nsigma = t.tofNSigmaKa();
153  } else if constexpr (pid_type == 4) {
154  nsigma = t.tofNSigmaPr();
155  } else if constexpr (pid_type == 5) {
156  nsigma = t.tofNSigmaDe();
157  } else if constexpr (pid_type == 6) {
158  nsigma = t.tofNSigmaTr();
159  } else if constexpr (pid_type == 7) {
160  nsigma = t.tofNSigmaHe();
161  } else if constexpr (pid_type == 8) {
162  nsigma = t.tofNSigmaAl();
163  }
164 
165  // Fill for all
166  histos.fill(HIST(hnsigma[pid_type]), t.pt(), nsigma);
167  if (MC::isPhysicalPrimary(mcParticles, t.mcParticle())) { // Selecting primaries
168  histos.fill(HIST(hnsigmaprm[pid_type]), t.pt(), nsigma);
169  } else {
170  histos.fill(HIST(hnsigmasec[pid_type]), t.pt(), nsigma);
171  }
172  // Fill with PDG codes
173  fillNsigma<0>(t, mcParticles, nsigma);
174  fillNsigma<1>(t, mcParticles, nsigma);
175  fillNsigma<2>(t, mcParticles, nsigma);
176  fillNsigma<3>(t, mcParticles, nsigma);
177  fillNsigma<4>(t, mcParticles, nsigma);
178  fillNsigma<5>(t, mcParticles, nsigma);
179  fillNsigma<6>(t, mcParticles, nsigma);
180  fillNsigma<7>(t, mcParticles, nsigma);
181  fillNsigma<8>(t, mcParticles, nsigma);
182  }
183  }
