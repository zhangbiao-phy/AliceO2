183 template <typename T>
184 void displayTable()

187  auto label = MetadataTrait<T>::metadata::tableLabel();
188  fmt::printf(R"(%s[color="%s" cellpadding="0" fillcolor="%s" fontcolor="%s" label = <
189 <TABLE cellpadding='2' cellspacing='0' cellborder='0' ><TH cellpadding='0' bgcolor="black"><TD bgcolor="%s"><font color="%s">%s</font></TD></TH>)",
190  label, style.color, style.background, style.fontcolor, style.headerbgcolor, style.headerfontcolor, label);
191  if (pack_size(typename T::iterator::persistent_columns_t{}) -
192  pack_size(typename T::iterator::external_index_columns_t{}) >
193  0) {
194  displayColumns(typename T::iterator::persistent_columns_t{}, style.color, style.background);
195  fmt::printf("%s", "HR");
196  }
197  if (pack_size(typename T::iterator::dynamic_columns_t{})) {
198  displayColumns(typename T::iterator::dynamic_columns_t{}, style.methodcolor, style.methodbgcolor);
199  fmt::printf("%s", "HR");
200  }
201  displayIndexColumns(typename T::iterator::external_index_columns_t{}, style.indexcolor, style.indexbgcolor);
202  fmt::printf("%s", "</TABLE>\n>]\n");
203  dumpIndex<T>(typename T::iterator::external_index_columns_t{});
204 }
