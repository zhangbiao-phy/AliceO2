47  template <typename I, typename D, typename T>
48  class _Iter
49  {
50  public:
51  using difference_type = int64_t;
52  using value_type = T;
53  using pointer = const T*;
54  using reference = const T&;
55  using iterator_category = std::random_access_iterator_tag;
56 
57  _Iter(const gsl::span<const D>& data, bool end = false) : mData(data), mIndex(end ? data.size() : 0){};
58  _Iter() = default;
59 
60  const I& operator++()
61  {
62  ++mIndex;

64  }
65 
66  const I& operator--()

82  bool operator!=(const I& other) const { return mIndex != other.mIndex; }
83  bool operator==(const I& other) const { return mIndex == other.mIndex; }
84  bool operator>(const I& other) const { return mIndex > other.mIndex; }
85  bool operator<(const I& other) const { return mIndex < other.mIndex; }

93  // BC difference wrt previous if in the same orbit, otherwise the abs.value.
94  // For the very 1st entry return 0 (diff wrt 1st BC in the CTF header)
95  class Iter_bcIncTrig : public _Iter<Iter_bcIncTrig, TriggerRecord, uint16_t>
96  {
97  public:
98  using _Iter<Iter_bcIncTrig, TriggerRecord, uint16_t>::_Iter;
99  value_type operator*() const
100  {
101  if (mIndex) {
102  if (mData[mIndex].getBCData().orbit == mData[mIndex - 1].getBCData().orbit) {
103  return mData[mIndex].getBCData().bc - mData[mIndex - 1].getBCData().bc;
104  } else {
105  return mData[mIndex].getBCData().bc;
106  }
107  }
108  return 0;
109  }
110  };
