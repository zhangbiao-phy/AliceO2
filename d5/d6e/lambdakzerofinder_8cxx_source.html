11 // V0 Finder task
12 // ==============
13 //
14 // This code loops over positive and negative tracks and finds
15 // valid V0 candidates from scratch using a certain set of
16 // minimum (configurable) selection criteria.
17 //
18 // It is different than the producer: the producer merely
19 // loops over an *existing* list of V0s (pos+neg track
20 // indices) and calculates the corresponding full V0 information
21 //
22 // In both cases, any analysis should loop over the "V0Data"
23 // table as that table contains all information.
24 //
25 // Comments, questions, complaints, suggestions?
26 // Please write to:
27 // david.dobrigkeit.chinellato@cern.ch
28 //

81 struct lambdakzeroprefilter {
82  Configurable<float> dcanegtopv{"dcanegtopv", .1, "DCA Neg To PV"};
83  Configurable<float> dcapostopv{"dcapostopv", .1, "DCA Pos To PV"};
84  Configurable<int> mincrossedrows{"mincrossedrows", 70, "min crossed rows"};
85  Configurable<int> tpcrefit{"tpcrefit", 1, "demand TPC refit"};
86 
87  Produces<aod::V0GoodPosTracks> v0GoodPosTracks;
88  Produces<aod::V0GoodNegTracks> v0GoodNegTracks;
89 
90  //still exhibiting issues? To be checked
91  //Partition<soa::Join<aod::FullTracks, aod::TracksExtended>> goodPosTracks = aod::track::signed1Pt > 0.0f && aod::track::dcaXY > dcapostopv;
92  //Partition<soa::Join<aod::FullTracks, aod::TracksExtended>> goodNegTracks = aod::track::signed1Pt < 0.0f && aod::track::dcaXY < -dcanegtopv;
93 
94  void process(aod::Collision const& collision,
95  soa::Join<aod::FullTracks, aod::TracksExtended> const& tracks)
96  {
97  for (auto& t0 : tracks) {
98  if (tpcrefit) {
99  if (!(t0.trackType() & o2::aod::track::TPCrefit)) {
100  continue; //TPC refit
101  }
102  }
103  if (t0.tpcNClsCrossedRows() < mincrossedrows) {
104  continue;
105  }
106  if (t0.signed1Pt() > 0.0f) {
107  if (fabs(t0.dcaXY()) < dcapostopv) {
108  continue;
109  }
110  v0GoodPosTracks(t0.globalIndex(), t0.collisionId(), t0.dcaXY());
111  }
112  if (t0.signed1Pt() < 0.0f) {
113  if (fabs(t0.dcaXY()) < dcanegtopv) {
114  continue;
115  }
116  v0GoodNegTracks(t0.globalIndex(), t0.collisionId(), -t0.dcaXY());
117  }
118  }
119  }
120 };

122 struct lambdakzerofinder {
123  Produces<aod::StoredV0Datas> v0data;
124 
125  HistogramRegistry registry{
126  "registry",
127  {
128  {"hCandPerEvent", "hCandPerEvent", {HistType::kTH1F, {{1000, 0.0f, 1000.0f}}}},
129  },
130  };
131 
132  //Configurables
133  Configurable<double> d_bz{"d_bz", +5.0, "bz field"};
134  Configurable<double> d_UseAbsDCA{"d_UseAbsDCA", kTRUE, "Use Abs DCAs"};
135 
136  //Selection criteria
137  Configurable<double> v0cospa{"v0cospa", 0.995, "V0 CosPA"}; //double -> N.B. dcos(x)/dx = 0 at x=0)
138  Configurable<float> dcav0dau{"dcav0dau", 1.0, "DCA V0 Daughters"};
139  Configurable<float> v0radius{"v0radius", 5.0, "v0radius"};
140 
141  void process(aod::Collision const& collision, aod::FullTracks const& tracks,
142  aod::V0GoodPosTracks const& ptracks, aod::V0GoodNegTracks const& ntracks)
143  {
144  //Define o2 fitter, 2-prong
145  o2::vertexing::DCAFitterN<2> fitter;
146  fitter.setBz(d_bz);
147  fitter.setPropagateToPCA(true);
148  fitter.setMaxR(200.);
149  fitter.setMinParamChange(1e-3);
150  fitter.setMinRelChi2Change(0.9);
151  fitter.setMaxDZIni(1e9);
152  fitter.setMaxChi2(1e9);
153  fitter.setUseAbsDCA(d_UseAbsDCA);
154 
155  Long_t lNCand = 0;
156 
157  std::array<float, 3> pVtx = {collision.posX(), collision.posY(), collision.posZ()};
158 
159  for (auto& t0id : ptracks) { //FIXME: turn into combination(...)
160  auto t0 = t0id.goodTrack_as<aod::FullTracks>();
161  auto Track1 = getTrackParCov(t0);
162  for (auto& t1id : ntracks) {
163  auto t1 = t1id.goodTrack_as<aod::FullTracks>();
164  auto Track2 = getTrackParCov(t1);
165 
166  //Try to progate to dca
167  int nCand = fitter.process(Track1, Track2);
168  if (nCand == 0) {
169  continue;
170  }
171  const auto& vtx = fitter.getPCACandidate();
172 
173  //Fiducial: min radius
174  auto thisv0radius = TMath::Sqrt(TMath::Power(vtx[0], 2) + TMath::Power(vtx[1], 2));
175  if (thisv0radius < v0radius) {
176  continue;
177  }
178 
179  //DCA V0 daughters
180  auto thisdcav0dau = fitter.getChi2AtPCACandidate();
181  if (thisdcav0dau > dcav0dau) {
182  continue;
183  }
184 
185  std::array<float, 3> pos = {0.};
186  std::array<float, 3> pvec0;
187  std::array<float, 3> pvec1;
188  for (int i = 0; i < 3; i++) {
189  pos[i] = vtx[i];
190  }
191  fitter.getTrack(0).getPxPyPzGlo(pvec0);
192  fitter.getTrack(1).getPxPyPzGlo(pvec1);
193 
194  auto thisv0cospa = RecoDecay::CPA(array{collision.posX(), collision.posY(), collision.posZ()},
195  array{vtx[0], vtx[1], vtx[2]}, array{pvec0[0] + pvec1[0], pvec0[1] + pvec1[1], pvec0[2] + pvec1[2]});
196  if (thisv0cospa < v0cospa) {
197  continue;
198  }
199 
200  lNCand++;
201  v0data(t0.globalIndex(), t1.globalIndex(), t0.collisionId(),
202  fitter.getTrack(0).getX(), fitter.getTrack(1).getX(),
203  pos[0], pos[1], pos[2],
204  pvec0[0], pvec0[1], pvec0[2],
205  pvec1[0], pvec1[1], pvec1[2],
206  fitter.getChi2AtPCACandidate(),
207  t0id.dcaXY(), t1id.dcaXY());
208  }
209  }
210  registry.fill(HIST("hCandPerEvent"), lNCand);
211  }
212 };

240  Filter preFilterV0 = nabs(aod::v0data::dcapostopv) > dcapostopv&& nabs(aod::v0data::dcanegtopv) > dcanegtopv&& aod::v0data::dcaV0daughters < dcav0dau;
241 
243  void process(soa::Join<aod::Collisions, aod::EvSels, aod::Cents>::iterator const& collision,
244  soa::Filtered<aod::V0Datas> const& fullV0s)
245  {
246  if (!collision.alias()[kINT7]) {
247  return;
248  }
249  if (!collision.sel7()) {
250  return;
251  }
252 
253  Long_t lNCand = 0;
254  for (auto& v0 : fullV0s) {
255  if (v0.v0radius() > v0radius && v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()) > v0cospa) {
256  registry.fill(HIST("hV0Radius"), v0.v0radius());
257  registry.fill(HIST("hV0CosPA"), v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()));
258  registry.fill(HIST("hDCAPosToPV"), v0.dcapostopv());
259  registry.fill(HIST("hDCANegToPV"), v0.dcanegtopv());
260  registry.fill(HIST("hDCAV0Dau"), v0.dcaV0daughters());
261 
262  if (TMath::Abs(v0.yLambda()) < 0.5) {
263  registry.fill(HIST("h3dMassLambda"), collision.centV0M(), v0.pt(), v0.mLambda());
264  registry.fill(HIST("h3dMassAntiLambda"), collision.centV0M(), v0.pt(), v0.mAntiLambda());
265  }
266  if (TMath::Abs(v0.yK0Short()) < 0.5) {
267  registry.fill(HIST("h3dMassK0Short"), collision.centV0M(), v0.pt(), v0.mK0Short());
268  }
269  lNCand++;
270  }
271  }
272  registry.fill(HIST("hCandPerEvent"), lNCand);
273  }
274 };
