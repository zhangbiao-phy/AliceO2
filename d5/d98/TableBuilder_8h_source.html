99  template <typename T>
100  static arrow::Status appendToList(std::unique_ptr<arrow::FixedSizeListBuilder>& builder, T* data, int size = 1)
101  {
102  using ArrowType = typename detail::ConversionTraits<std::decay_t<T>>::ArrowType;
103  using BuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;
104  size_t numElements = static_cast<const arrow::FixedSizeListType*>(builder->type().get())->list_size();
105 
106  auto status = builder->AppendValues(size);
107  auto ValueBuilder = static_cast<BuilderType*>(builder->value_builder());
108  status &= ValueBuilder->AppendValues(data, numElements * size, nullptr);
109 
110  return status;
111  }

173  template <typename HolderType, typename PTR>
174  static arrow::Status bulkAppendChunked(HolderType& holder, BulkInfo<PTR> info)
175  {
176  // Appending nullptr is a no-op.
177  if (info.ptr == nullptr) {
178  return arrow::Status::OK();
179  }
180  if constexpr (std::is_same_v<decltype(holder.builder), std::unique_ptr<arrow::FixedSizeListBuilder>>) {
181  if (appendToList<std::remove_pointer_t<decltype(info.ptr)>>(holder.builder, info.ptr, info.size).ok() == false) {
182  throw runtime_error("Unable to append to column");
183  } else {
184  return arrow::Status::OK();
185  }
186  } else {
187  if (holder.builder->AppendValues(info.ptr, info.size, nullptr).ok() == false) {
188  throw runtime_error("Unable to append to column");
189  } else {
190  return arrow::Status::OK();
191  }
192  }
193  }

195  template <typename HolderType, typename ITERATOR>
196  static arrow::Status append(HolderType& holder, std::pair<ITERATOR, ITERATOR> ip)
197  {
198  using ArrowType = typename detail::ConversionTraits<typename ITERATOR::value_type>::ArrowType;
199  using ValueBuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;
200  // FIXME: for the moment we do not fill things.
201  auto status = holder.builder->Append();
202  auto valueBuilder = reinterpret_cast<ValueBuilderType*>(holder.builder->value_builder());
203  return status & valueBuilder->AppendValues(&*ip.first, std::distance(ip.first, ip.second));
204  }

206  // Lists do not have UnsafeAppend so we need to use the slow path in any case.
207  template <typename HolderType, typename ITERATOR>
208  static void unsafeAppend(HolderType& holder, std::pair<ITERATOR, ITERATOR> ip)
209  {
210  using ArrowType = typename detail::ConversionTraits<typename ITERATOR::value_type>::ArrowType;
211  using ValueBuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;

213  auto status = holder.builder->Append();
214  auto valueBuilder = reinterpret_cast<ValueBuilderType*>(holder.builder->value_builder());
215  status &= valueBuilder->AppendValues(&*ip.first, std::distance(ip.first, ip.second));
216  if (!status.ok()) {
217  throw runtime_error("Unable to append values to valueBuilder!");
218  }
219  return;
220  }

381 template <typename T>
382 struct CachedInsertion {
383  static constexpr int CHUNK_SIZE = 256;
384 
385  template <typename BUILDER>
386  arrow::Status append(BUILDER& builder, T value)
387  {
388  cache[pos % CHUNK_SIZE] = value;
389  ++pos;
390  if (pos % CHUNK_SIZE == 0) {
391  return builder->AppendValues(cache, CHUNK_SIZE, nullptr);
392  }
393  return arrow::Status::OK();
394  }
395 
396  template <typename BUILDER>
397  arrow::Status flush(BUILDER& builder)
398  {
399  if (pos % CHUNK_SIZE != 0) {
400  return builder->AppendValues(cache, pos % CHUNK_SIZE, nullptr);
401  }
402  return arrow::Status::OK();
403  }
404  T cache[CHUNK_SIZE];
405  int pos = 0;
406 };

574  template <typename... ARGS>
575  auto makeBuilders(std::vector<std::string> const& columnNames, size_t nRows)
576  {
577  mSchema = std::make_shared<arrow::Schema>(TableBuilderHelpers::makeFields<ARGS...>(columnNames));
578 
579  auto holders = new HoldersTuple<ARGS...>(typename HolderTrait<ARGS>::Holder(mMemoryPool)...);
580  if (nRows != -1) {
581  auto seq = std::make_index_sequence<sizeof...(ARGS)>{};
582  TableBuilderHelpers::reserveAll(*holders, nRows, seq);
583  }
584  mHolders = holders; // We store the builders
585  }

589  {

591  return TableBuilderHelpers::finalize(arrays, *(HoldersTuple<ARGS...>*)holders, std::make_index_sequence<sizeof...(ARGS)>{});
592  };

604  template <typename... ARGS>
605  auto persist(std::vector<std::string> const& columnNames)
606  {
607  using args_pack_t = framework::pack<ARGS...>;
608  if constexpr (sizeof...(ARGS) == 1 &&
609  is_bounded_array<pack_element_t<0, args_pack_t>>::value == false &&
610  std::is_arithmetic_v<pack_element_t<0, args_pack_t>> == false) {
611  using objType_t = pack_element_t<0, framework::pack<ARGS...>>;
612  using argsPack_t = decltype(tuple_to_pack(framework::to_tuple(std::declval<objType_t>())));
613  auto persister = persistTuple(argsPack_t{}, columnNames);
614  return [persister = persister](unsigned int slot, objType_t const& obj) -> void {
615  auto t = to_tuple(obj);
616  persister(slot, t);
617  };
618  } else if constexpr (sizeof...(ARGS) == 1 &&
619  is_bounded_array<pack_element_t<0, args_pack_t>>::value == true) {
620  using objType_t = pack_element_t<0, framework::pack<ARGS...>>;
621  auto persister = persistTuple(framework::pack<objType_t>{}, columnNames);
622  // Callback used to fill the builders
623  return [persister = persister](unsigned int slot, typename BuilderMaker<objType_t>::FillType const& arg) -> void {
624  persister(slot, std::forward_as_tuple(arg));
625  };
626  } else if constexpr (sizeof...(ARGS) >= 1) {
627  auto persister = persistTuple(framework::pack<ARGS...>{}, columnNames);
628  // Callback used to fill the builders
629  return [persister = persister](unsigned int slot, typename BuilderMaker<ARGS>::FillType... args) -> void {
630  persister(slot, std::forward_as_tuple(args...));
631  };
632  } else {
633  static_assert(o2::framework::always_static_assert_v<ARGS...>, "Unmanaged case");
634  }
635  }
636 
638  template <typename... ARGS>
639  auto persistTuple(framework::pack<ARGS...>, std::vector<std::string> const& columnNames)
640  {
641  constexpr int nColumns = sizeof...(ARGS);
642  validate(nColumns, columnNames);
643  mArrays.resize(nColumns);
644  makeBuilders<ARGS...>(columnNames, 1000);
645  makeFinalizer<ARGS...>();
646 
647  // Callback used to fill the builders
648  using FillTuple = std::tuple<typename BuilderMaker<ARGS>::FillType...>;
649  return [holders = mHolders](unsigned int slot, FillTuple const& t) -> void {
650  auto status = TableBuilderHelpers::append(*(HoldersTuple<ARGS...>*)holders, std::index_sequence_for<ARGS...>{}, t);
651  if (status == false) {
652  throwError(runtime_error("Unable to append"));
653  }
654  };
655  }

772 template <typename... C>
773 auto spawner(framework::pack<C...> columns, arrow::Table* atable)
774 {
775  static auto new_schema = o2::soa::createSchemaFromColumns(columns);
776  static auto projectors = framework::expressions::createProjectors(columns, atable->schema());
777 
778  if (atable->num_rows() == 0) {
779  return makeEmptyTable<soa::Table<C...>>();
780  }
781 
782  arrow::TableBatchReader reader(*atable);
783  std::shared_ptr<arrow::RecordBatch> batch;
784  arrow::ArrayVector v;
785  std::array<arrow::ArrayVector, sizeof...(C)> chunks;
786  std::vector<std::shared_ptr<arrow::ChunkedArray>> arrays;
787 
788  while (true) {
789  auto s = reader.ReadNext(&batch);
790  if (!s.ok()) {
791  throw runtime_error_f("Cannot read batches from table: %s", s.ToString().c_str());
792  }
793  if (batch == nullptr) {
794  break;
795  }
796  s = projectors->Evaluate(*batch, arrow::default_memory_pool(), &v);
797  if (!s.ok()) {
798  throw runtime_error_f("Cannot apply projector: %s", s.ToString().c_str());
799  }
800  for (auto i = 0u; i < sizeof...(C); ++i) {
801  chunks[i].emplace_back(v.at(i));
802  }
803  }
804 
805  for (auto i = 0u; i < sizeof...(C); ++i) {
806  arrays.push_back(std::make_shared<arrow::ChunkedArray>(chunks[i]));
807  }
808 
809  return arrow::Table::Make(new_schema, arrays);
810 }

829 {

831  int step;
832  auto count = start.mMaxRow - start.globalIndex();
833 
834  while (count > 0) {
835  step = count / 2;
836  start.moveByIndex(step);
837  if (start.template getId<Key>() <= value) {
838  count -= step + 1;
839  } else {
840  start.moveByIndex(-step);
841  count = step;
842  }
843  }
