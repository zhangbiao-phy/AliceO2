98 void PHOSCalibCollector::scanClusters(o2::framework::ProcessingContext& pc)
99 {
100 
101  // auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime; // is this the timestamp of the current TF?
102 
103  auto clusters = pc.inputs().get<std::vector<o2::phos::FullCluster>>("clusters");
104  auto cluTR = pc.inputs().get<gsl::span<o2::phos::TriggerRecord>>("cluTR");
105  LOG(INFO) << "Processing TF with " << clusters.size() << " clusters and " << cluTR.size() << " TriggerRecords";
106 
107  for (auto& tr : cluTR) {
108 
109  //Mark new event
110  //First goes new event marker + BC (16 bit), next word orbit (32 bit)
111  EventHeader h = {0};
112  h.mMarker = 16383;
113  h.mBC = tr.getBCData().bc;
114  mDigits.push_back(h.mDataWord);
115  mDigits.push_back(tr.getBCData().orbit);
116 
117  int iclu = 0;
118  int firstCluInEvent = tr.getFirstEntry();
119  int lastCluInEvent = firstCluInEvent + tr.getNumberOfObjects();
120 
121  mBuffer->startNewEvent(); // mark stored clusters to be used for Mixing
122  for (int i = firstCluInEvent; i < lastCluInEvent; i++) {
123  const FullCluster& clu = clusters[i];
124 
125  fillTimeMassHisto(clu);
126  bool isGood = checkCluster(clu);
127 
128  auto cluList = clu.getElementList();
129  for (auto ce = cluList->begin(); ce != cluList->end(); ce++) {
130  short absId = ce->absId;
131  //Fill cells from cluster for next iterations
132  short adcCounts = ce->energy / mCalibParams->getGain(absId);
133  // Need to chale LG gain too to fit dynamic range
134  if (!ce->isHG) {
135  adcCounts /= mCalibParams->getHGLGRatio(absId);
136  }
137  CalibDigit d = {0};
138  d.mAddress = absId;
139  d.mAdcAmp = adcCounts;
140  d.mHgLg = ce->isHG;
141  d.mBadChannel = isGood;
142  d.mCluster = (i - firstCluInEvent) % kMaxCluInEvent;
143  mDigits.push_back(d.mDataWord);
144  if (i - firstCluInEvent > kMaxCluInEvent) {
145  //Normally this is not critical as indexes are used "locally", i.e. are compared to previous/next
146  LOG(INFO) << "Too many clusters per event:" << i - firstCluInEvent << ", apply more strict selection; clusters with same indexes will appear";
147  }
148  }
149  }
150  }

153 void PHOSCalibCollector::readDigits()
154 {
155  // open files from the list
156  // for each file read digits, calibrate them and form clusters
157  // fill inv mass and time (for check) histograms
158  std::ifstream ifs(mdigitsfilelist, std::ifstream::in);
159  if (!ifs.is_open()) {
160  LOG(ERROR) << "can not open file " << mdigitsfilelist;
161  return;
162  }
163 
164  std::string filename;
165  while (ifs >> filename) {
166  TFile f(filename.data(), "READ");
167  if (!f.IsOpen()) {
168  LOG(ERROR) << "can not read file " << filename;
169  continue;
170  }
171  std::vector<uint32_t>* digits;
172  f.GetObject("Digits", digits);
173 
174  std::vector<uint32_t>::const_iterator digIt = digits->cbegin();
175  std::vector<uint32_t>::const_iterator digEnd = digits->cend();
176  FullCluster clu;
177  bool isNextNewEvent;
178  mBuffer->startNewEvent(); // mark stored clusters to be used for Mixing
179  while (nextCluster(digIt, digEnd, clu, isNextNewEvent)) {
180 
181  fillTimeMassHisto(clu);
182  if (isNextNewEvent) {
183  mBuffer->startNewEvent(); // mark stored clusters to be used for Mixing
184  }
185  }
186  delete digits;
187  f.Close();
188  }
189 }

191 bool PHOSCalibCollector::nextCluster(std::vector<uint32_t>::const_iterator digitIt, std::vector<uint32_t>::const_iterator digitEnd,
192  FullCluster& clu, bool& isNextNewEvent)
193 {
194  //Scan digits belonging to cluster
195  // return true if cluster read
196  isNextNewEvent = false;
197  clu.reset();
198  int cluIndex = -1;
199  while (digitIt != digitEnd) {
200  CalibDigit d = {*digitIt};
201  if (d.mAddress == 16383) { //impossible address, marker of new event
202  if (clu.getMultiplicity() > 0) { //already read cluster: calculate its parameters; do not go into next event
203  isNextNewEvent = true;
204  break;
205  }
206  //If just started cluster, read new event. Two first words event header
207  EventHeader h = {*digitIt};
208  mEvBC = h.mBC; //current event BC
209  digitIt++;
210  mEvOrbit = *digitIt; //current event orbit
211  digitIt++;
212  continue;
213  }
214  if (cluIndex == -1) { //start new cluster
215  cluIndex = d.mCluster;
216  } else {
217  if (cluIndex != d.mCluster) { //digit belongs to text cluster: all digits from current read
218  break;
219  }
220  }
221  short absId = d.mAddress;
222  float energy = d.mAdcAmp * mCalibParams->getGain(absId);
223  if (!d.mHgLg) {
224  energy *= mCalibParams->getHGLGRatio(absId);
225  }
226  clu.addDigit(absId, energy, 0., 0, 1.);
227  digitIt++;
228  }
229  if (digitIt == digitEnd && clu.getMultiplicity() == 0) {
230  return false;
231  }
232  //Evaluate clu parameters
233  clu.purify();
234  clu.evalAll();
235  return true;
236 }

256 void PHOSCalibCollector::fillTimeMassHisto(const FullCluster& clu)
257 {
258  // Fill time distributions only for cells in cluster
259  if (mMode == 0) {
260  auto cluList = clu.getElementList();
261  for (auto ce = cluList->begin(); ce != cluList->end(); ce++) {
262  short absId = ce->absId;
263  if (ce->isHG) {
264  if (ce->energy > mEminHGTime) {
265  mHistos[kTimeHGPerCell].Fill(absId, ce->time);
266  }
267  mHistos[kTimeHGSlewing].Fill(ce->time, ce->energy);
268  } else {
269  if (ce->energy > mEminLGTime) {
270  mHistos[kTimeLGPerCell].Fill(absId, ce->time);
271  }
272  mHistos[kTimeLGSlewing].Fill(ce->time, ce->energy);
273  }
274  }
275  }
276 
277  //Real and Mixed inv mass distributions
278  // prepare TLorentsVector
279  float posX, posZ;
280  clu.getLocalPosition(posX, posZ);
281  TVector3 vec3;
282  mGeom->local2Global(clu.module(), posX, posZ, vec3);
283  vec3 -= mVertex;
284  float e = clu.getEnergy();
285  short absId;
286  mGeom->relPosToAbsId(clu.module(), posX, posZ, absId);
287 
288  TLorentzVector v(vec3.X() * e, vec3.Y() * e, vec3.Z() * e, e);
289  // Fill calibration histograms for all cells, even bad, but partners in inv, mass should be good
290  bool isGood = checkCluster(clu);
291  for (short ip = mBuffer->size(); --ip;) {
292  const TLorentzVector& vp = mBuffer->getEntry(ip);
293  TLorentzVector sum = v + vp;
294  if (mBuffer->isCurrentEvent(ip)) { //same (real) event
295  if (isGood) {
296  mHistos[kReInvMassNonlin].Fill(e, sum.M());
297  }
298  if (sum.Pt() > mPtMin) {
299  mHistos[kReInvMassPerCell].Fill(absId, sum.M());
300  }
301  } else { //Mixed
302  if (isGood) {
303  mHistos[kMiInvMassNonlin].Fill(e, sum.M());
304  }
305  if (sum.Pt() > mPtMin) {
306  mHistos[kMiInvMassPerCell].Fill(absId, sum.M());
307  }
308  }
309  }
310 
311  //Add to list ot partners only if cluster is good
312  if (isGood) {
313  mBuffer->addEntry(v);
314  }
315 }
