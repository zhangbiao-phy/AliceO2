53  void init(o2::framework::InitContext& ic) final
54  {
55 
56  // stopping all stopwatches, since they start counting from the moment they are created
57  for (int idet = 0; idet < kNdetectors; idet++) {
58  mDeltaProcessingDetLoop[idet].Stop();
59  mDeltaProcessingDetLoop[idet].Reset();
60  }

62  std::vector<DPID> pidVect;
63 
64  DPID dpidtmp;
65  DeliveryType typechar = RAW_CHAR;
66  std::string dpAliaschar = "TestChar_0";
67  DPID::FILL(dpidtmp, dpAliaschar, typechar);
68  pidVect.push_back(dpidtmp);
69 
70  //std::vector<int> vectDet{kTest, kTPC}; // only one detector for now
71  std::vector<int> vectDet{kTest};
72  mDetectorPid[dpidtmp] = vectDet;
73 
74  DeliveryType typeint = RAW_INT;
75  for (int i = 0; i < 50000; i++) {
76  std::string dpAliasint = "TestInt_" + std::to_string(i);
77  DPID::FILL(dpidtmp, dpAliasint, typeint);
78  pidVect.push_back(dpidtmp);
79  mDetectorPid[dpidtmp] = vectDet;
80  }
81 
82  DeliveryType typedouble = RAW_DOUBLE;
83  for (int i = 0; i < 4; i++) {
84  std::string dpAliasdouble = "TestDouble_" + std::to_string(i);
85  DPID::FILL(dpidtmp, dpAliasdouble, typedouble);
86  pidVect.push_back(dpidtmp);
87  mDetectorPid[dpidtmp] = vectDet;
88  }
89 
90  DeliveryType typestring = RAW_STRING;
91  std::string dpAliasstring0 = "TestString_0";
92  DPID::FILL(dpidtmp, dpAliasstring0, typestring);
93  pidVect.push_back(dpidtmp);
94  mDetectorPid[dpidtmp] = vectDet;
95 
96  for (int idet = 0; idet < kNdetectors; idet++) {
97  mDCSprocVect[idet].init(pidVect);
98  mDCSprocVect[idet].setMaxCyclesNoFullMap(ic.options().get<int64_t>("max-cycles-no-full-map"));
99  mDCSprocVect[idet].setName("Test1Det");
100  }
101  mProcessFullDeltaMap = ic.options().get<bool>("process-full-delta-map");
102  }

104  void run(o2::framework::ProcessingContext& pc) final
105  {
106  auto tfid = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
107  for (int idet = 0; idet < kNdetectors; idet++) {
108  mDCSprocVect[idet].setTF(tfid);
109  }
110 
111  TStopwatch s;
112  LOG(DEBUG) << "TF: " << tfid << " --> receiving binary data...";
113  mReceiveBinaryData.Start(mFirstTF);
114  auto rawchar = pc.inputs().get<const char*>("input");
115  mReceiveBinaryData.Stop();
116  LOG(DEBUG) << "TF: " << tfid << " --> ...binary data received: realTime = "
117  << mReceiveBinaryData.RealTime() << ", cpuTime = "
118  << mReceiveBinaryData.CpuTime();
119  LOG(DEBUG) << "TF: " << tfid << " --> receiving (delta) binary data...";
120  mDeltaReceiveBinaryData.Start(mFirstTF);
121  auto rawcharDelta = pc.inputs().get<const char*>("inputDelta");
122  mDeltaReceiveBinaryData.Stop();
123  LOG(DEBUG) << "TF: " << tfid << " --> ...binary (delta) data received: realTime = "
124  << mDeltaReceiveBinaryData.RealTime()
125  << ", cpuTime = " << mDeltaReceiveBinaryData.CpuTime();
126 
127  // full map
128  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().get("input").header);
129  auto sz = dh->payloadSize;
130  int nDPs = sz / sizeof(DPCOM);
131  std::unordered_map<DPID, DPVAL> dcsmap;
132  DPCOM dptmp;
133  LOG(DEBUG) << "TF: " << tfid << " --> building unordered_map...";
134  mBuildingUnorderedMap.Start(mFirstTF);
135  for (int i = 0; i < nDPs; i++) {
136  memcpy(&dptmp, rawchar + i * sizeof(DPCOM), sizeof(DPCOM));
137  dcsmap[dptmp.id] = dptmp.data;
138  LOG(DEBUG) << "Reading from generator: i = " << i << ", DPCOM = " << dptmp;
139  LOG(DEBUG) << "Reading from generator: i = " << i << ", DPID = " << dptmp.id;
140  }
141  mBuildingUnorderedMap.Stop();
142  LOG(DEBUG) << "TF: " << tfid << " --> ...unordered_map built = "
143  << mBuildingUnorderedMap.RealTime() << ", cpuTime = " << mBuildingUnorderedMap.CpuTime();
144 
145  // delta map
146  const auto* dhDelta = o2::header::get<o2::header::DataHeader*>(pc.inputs().get("inputDelta").header);
147  auto szDelta = dhDelta->payloadSize;
148  int nDPsDelta = szDelta / sizeof(DPCOM);
149  std::unordered_map<DPID, DPVAL> dcsmapDelta;
150  LOG(DEBUG) << "TF: " << tfid << " --> building (delta) unordered_map...";
151  mDeltaBuildingUnorderedMap.Start(mFirstTF);
152  for (int i = 0; i < nDPsDelta; i++) {
153  memcpy(&dptmp, rawcharDelta + i * sizeof(DPCOM), sizeof(DPCOM));
154  dcsmapDelta[dptmp.id] = dptmp.data;
155  LOG(DEBUG) << "Reading from generator: i = " << i << ", DPCOM = " << dptmp;
156  LOG(DEBUG) << "Reading from generator: i = " << i << ", DPID = " << dptmp.id;
157  }
158  mDeltaBuildingUnorderedMap.Stop();
159  LOG(DEBUG) << "TF: " << tfid << " --> ...unordered_map (delta) built = "
160  << mDeltaBuildingUnorderedMap.RealTime() << ", cpuTime = "
161  << mDeltaBuildingUnorderedMap.CpuTime();
162 
163  if (tfid % 6000 == 0) {
164  LOG(INFO) << "Number of DPs received = " << nDPs;
165  for (int idet = 0; idet < kNdetectors; idet++) {
166  LOG(DEBUG) << "TF: " << tfid << " --> starting processing...";
167  mProcessing[idet].Start(mResetStopwatchProcessing);
168  mDCSprocVect[idet].processMap(dcsmap, false);
169  mProcessing[idet].Stop();
170  LOG(DEBUG) << "TF: " << tfid << " --> ...processing done: realTime = "
171  << mProcessing[idet].RealTime() << ", cpuTime = "
172  << mProcessing[idet].CpuTime();
173  }
174  mResetStopwatchProcessing = false; // from now on, we sum up the processing time
175  mTFsProcessing++;
176  } else {
177  LOG(INFO) << "Number of DPs received (delta map) = " << nDPsDelta;
178  if (mProcessFullDeltaMap) {
179  for (int idet = 0; idet < kNdetectors; idet++) {
180  LOG(DEBUG) << "TF: " << tfid << " --> starting (delta) processing...";
181  mDeltaProcessing[idet].Start(mResetStopwatchDeltaProcessing);
182  mDCSprocVect[idet].processMap(dcsmapDelta, true);
183  mDeltaProcessing[idet].Stop();
184  LOG(DEBUG) << "TF: " << tfid << " --> ...processing (delta) done: realTime = "
185  << mDeltaProcessing[idet].RealTime()
186  << ", cpuTime = " << mDeltaProcessing[idet].CpuTime();
187  }
188  mResetStopwatchDeltaProcessing = false; // from now on, we sum up the processing time
189  mTFsDeltaProcessing++;
190  } else {
191 
192  // processing per DP found in the map, to be done in case of a delta map
193 
194  LOG(DEBUG) << "TF: " << tfid << " --> starting (delta) processing in detector loop...";
195  for (const auto& dpcom : dcsmapDelta) {
196  std::vector<int> detVect = mDetectorPid[dpcom.first];
197  for (int idet = 0; idet < detVect.size(); idet++) {
198  mDeltaProcessingDetLoop[idet].Start(mResetStopwatchDeltaProcessingDetLoop);
199  mDCSprocVect[idet].processDP(dpcom);
200  mDeltaProcessingDetLoop[idet].Stop();
201  }
202  mResetStopwatchDeltaProcessingDetLoop = false; // from now on, we sum up the processing time
203  }
204  for (int idet = 0; idet < kNdetectors; idet++) {
205  LOG(DEBUG) << "TF: " << tfid << " --> ...processing (delta) in detector loop done: realTime = "
206  << mDeltaProcessingDetLoop[idet].RealTime() << ", cpuTime = "
207  << mDeltaProcessingDetLoop[idet].CpuTime();
208  }
209  // now preparing CCDB object
210  for (int idet = 0; idet < kNdetectors; idet++) {
211  std::map<std::string, std::string> md;
212  mDCSprocVect[idet].prepareCCDBobject(mDCSprocVect[idet].getCCDBSimpleMovingAverage(),
213  mDCSprocVect[idet].getCCDBSimpleMovingAverageInfo(),
214  mDCSprocVect[idet].getName() + "/TestDCS/SimpleMovingAverageDPs",
215  tfid, md);
216  }
217  mTFsDeltaProcessingDetLoop++;
218  }
219  }
220  sendOutput(pc.outputs());
221  mFirstTF = false;
222  mTFs++;
223  }

225  void endOfStream(o2::framework::EndOfStreamContext& ec) final
226  {
227  LOG(INFO) << "\n\nTIMING SUMMARY:\n";
228  LOG(INFO) << "Number of processed TF: " << mTFs;
229  LOG(INFO) << "Number of processed TF, processing full map: " << mTFsProcessing;
230  LOG(INFO) << "Number of processed TF, processing delta map: " << mTFsDeltaProcessing;
231  LOG(INFO) << "Number of processed TF, processing delta map per DP: " << mTFsDeltaProcessingDetLoop;
232  LOG(INFO) << "Receiving binary data --> realTime = "
233  << mReceiveBinaryData.RealTime() / mTFs << ", cpuTime = "
234  << mReceiveBinaryData.CpuTime() / mTFs;
235  LOG(INFO) << "Receiving binary data (delta) --> realTime = "
236  << mDeltaReceiveBinaryData.RealTime() / mTFs << ", cpuTime = "
237  << mDeltaReceiveBinaryData.CpuTime() / mTFs;
238  LOG(INFO) << "Building unordered_map --> realTime = "
239  << mBuildingUnorderedMap.RealTime() / mTFs << ", cpuTime = "
240  << mBuildingUnorderedMap.CpuTime() / mTFs;
241  LOG(INFO) << "Building unordered_map (delta) --> realTime = "
242  << mDeltaBuildingUnorderedMap.RealTime() / mTFs << ", cpuTime = "
243  << mDeltaBuildingUnorderedMap.CpuTime() / mTFs;
244  for (int i = 0; i < kNdetectors; i++) {
245  LOG(INFO) << " --> : Detector " << i;
246  if (mTFsProcessing != 0) {
247  LOG(INFO) << "Processing full map (average over " << mTFsProcessing << " TFs) --> realTime = "
248  << mProcessing[i].RealTime() / mTFsProcessing << ", cpuTime = "
249  << mProcessing[i].CpuTime() / mTFsProcessing;
250  } else {
251  LOG(INFO) << "Full DCS map was never processed";
252  }
253  if (mTFsDeltaProcessing != 0) {
254  LOG(INFO) << "Processing full delta map (average over " << mTFsDeltaProcessing << " TFs) --> realTime = "
255  << mDeltaProcessing[i].RealTime() / mTFsDeltaProcessing << ", cpuTime = "
256  << mDeltaProcessing[i].CpuTime() / mTFsDeltaProcessing;
257  } else {
258  LOG(INFO) << "Full delta DCS map was never processed";
259  }
260  if (mTFsDeltaProcessingDetLoop != 0) {
261  LOG(INFO) << "Processing delta map per DP (average over " << mTFsDeltaProcessingDetLoop
262  << " TFs) --> realTime = "
263  << mDeltaProcessingDetLoop[i].RealTime() / mTFsDeltaProcessingDetLoop << ", cpuTime = "
264  << mDeltaProcessingDetLoop[i].CpuTime() / mTFsDeltaProcessingDetLoop;
265  } else {
266  LOG(INFO) << "Delta map was never process DP by DP";
267  }
268  }
269  }

292  void sendOutput(framework::DataAllocator& output)
293  {
294  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
295  // copied from LHCClockCalibratorSpec.cxx
296  using clbUtils = o2::calibration::Utils;
297  for (int idet = 0; idet < kNdetectors; idet++) {
298  const auto& payload = mDCSprocVect[idet].getCCDBSimpleMovingAverage();
299  auto& info = mDCSprocVect[idet].getCCDBSimpleMovingAverageInfo();
300  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
301  LOG(INFO) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
302  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
303 
304  output.snapshot(framework::Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload, 0}, *image.get());
305  output.snapshot(framework::Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo, 0}, info);
306  }

314 DataProcessorSpec getDCSDataProcessorSpec()
315 {
316 
317  using clbUtils = o2::calibration::Utils;
318 
319  std::vector<OutputSpec> outputs;
320  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload});

324  "dcs-data-processor",
325  Inputs{{"input", "DCS", "DATAPOINTS"}, {"inputDelta", "DCS", "DATAPOINTSdelta"}},
326  outputs,
327  AlgorithmSpec{adaptFromTask<o2::dcs::DCSDataProcessor>()},
328  Options{
329  {"max-cycles-no-full-map", VariantType::Int64, 6000ll, {"max num of cycles between the sending of 2 full maps"}},

331 }
