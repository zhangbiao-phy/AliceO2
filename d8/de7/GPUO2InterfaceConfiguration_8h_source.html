100 // Structure with pointers to actual data for input and output
101 // Which ptr is used for input and which for output is defined in GPUO2InterfaceConfiguration::configWorkflow
102 // Inputs and outputs are mutually exclusive.
103 // Inputs which are nullptr are considered empty, and will not throw an error.
104 // Outputs, which point to std::[container] / MCTruthContainer, will be filled and no output
105 // is written if the ptr is a nullptr.
106 // Outputs, which point to other structures are set by GPUCATracking to the location of the output. The previous
107 // value of the pointer is overridden. GPUCATracking will try to place the output in the "void* outputBuffer"
108 // location if it is not a nullptr.
109 struct GPUO2InterfaceIOPtrs {
110  // Input: TPC clusters in cluster native format, or digits, or list of ZS pages - const as it can only be input
111  const o2::tpc::ClusterNativeAccess* clusters = nullptr;
112  const std::array<gsl::span<const o2::tpc::Digit>, o2::tpc::constants::MAXSECTOR>* o2Digits = nullptr;
113  const std::array<const o2::dataformats::ConstMCTruthContainerView<o2::MCCompLabel>*, o2::tpc::constants::MAXSECTOR>* o2DigitsMC = nullptr;
114  const o2::gpu::GPUTrackingInOutZS* tpcZS = nullptr;
115 
116  // Input / Output for Merged TPC tracks, two ptrs, for the tracks themselves, and for the MC labels.
117 #ifdef MS_GSL_V3
118  gsl::span<o2::tpc::TrackTPC> outputTracks = {nullptr, (gsl::span<o2::tpc::TrackTPC>::size_type)0};
119  gsl::span<uint32_t> outputClusRefs = {nullptr, (gsl::span<uint32_t>::size_type)0};
120  gsl::span<o2::MCCompLabel> outputTracksMCTruth = {nullptr, (gsl::span<o2::MCCompLabel>::size_type)0};
121 #else
122  gsl::span<o2::tpc::TrackTPC> outputTracks = {nullptr, (gsl::span<o2::tpc::TrackTPC>::index_type)0};
123  gsl::span<uint32_t> outputClusRefs = {nullptr, (gsl::span<uint32_t>::index_type)0};
124  gsl::span<o2::MCCompLabel> outputTracksMCTruth = {nullptr, (gsl::span<o2::MCCompLabel>::index_type)0};
125 #endif
126  // Output for entropy-reduced clusters of TPC compression
127  const o2::tpc::CompressedClustersFlat* compressedClusters = nullptr;
128 };
