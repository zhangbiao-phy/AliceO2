131 void CompressedDecodingTask::run(ProcessingContext& pc)
132 {
133  mTimer.Start(false);
134 
135  if (pc.inputs().getNofParts(0) && !mConetMode && 0) { // it doesn't work
136  //RS set the 1st orbit of the TF from the O2 header, relying on rdhHandler is not good (in fact, the RDH might be eliminated in the derived data)
137  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().getByPos(0).header);
138  mInitOrbit = dh->firstTForbit;
139  }
140 
141  // mDecoder.setFirstIR({0, mInitOrbit});
142 
144  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
145  if (!iit.isValid()) {
146  continue;
147  }
148 
150  for (auto const& ref : iit) {
151  const auto* headerIn = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
152  auto payloadIn = ref.payload;
153  auto payloadInSize = headerIn->payloadSize;
154 
155  DecoderBase::setDecoderBuffer(payloadIn);
156  DecoderBase::setDecoderBufferSize(payloadInSize);
157  DecoderBase::run();
158  }
159  }
160 
161  if ((mNCrateOpenTF > 0 || mConetMode) && mNCrateOpenTF == mNCrateCloseTF) {
162  mHasToBePosted = true;
163  }
164 
165  if (mHasToBePosted) {
166  postData(pc);
167  }
168  mTimer.Stop();
169 }

348 void CompressedDecodingTask::rdhHandler(const o2::header::RAWDataHeader* rdh)
349 {
350  const auto& rdhr = *rdh;
351  // set first orbtt here (to be check in future), please not remove this!!!
352  mCurrentOrbit = RDHUtils::getHeartBeatOrbit(rdhr);
353  if (mNCrateOpenTF == 0) {
354  mInitOrbit = mCurrentOrbit;
355  mDecoder.setFirstIR({0, mInitOrbit});
356  }
357 
358  // rdh close
359  if (RDHUtils::getStop(rdhr) && RDHUtils::getHeartBeatOrbit(rdhr) == o2::raw::HBFUtils::Instance().getNOrbitsPerTF() - 1 + mInitOrbit) {
360  mNCrateCloseTF++;
361  return;
362  }
363 
364  // rdh open
365  if ((RDHUtils::getPageCounter(rdhr) == 0) && (RDHUtils::getTriggerType(rdhr) & o2::trigger::TF)) {
366  mNCrateOpenTF++;
367  mInitOrbit = RDHUtils::getHeartBeatOrbit(rdhr); // RSTODO this may be eliminated once the framework will start to propagated the dh.firstTForbit
368  // printf("New TF open RDH %d\n", int(rdh->feeId));
369  }
