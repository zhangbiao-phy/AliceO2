29 /*
30  Fast 2D histo class which can be messages as
31  FlatHisto2D<float> histo(nbinsX, xmin, xmax, nbinsY, ymin, ymax);
32  histo.fill(...);
33  pc.outputs().snapshot(Output{"Origin", "Desc", 0, Lifetime::Timeframe}, histo.getBase());
34 
35  and received (read only!) as
36  const auto hdata = pc.inputs().get<gsl::span<float>>("histodata");
37  FlatHisto2D<float> histoView;
38  histoView.adoptExternal(hdata);
39  or directly
40  FlatHisto2D<float> histoView(pc.inputs().get<gsl::span<float>>("histodata"));
41 */
42 
43 template <typename T = float>
44 class FlatHisto2D
45 {
46  static_assert(std::is_same<T, float>::value || std::is_same<T, double>::value, "T must be float or double");
47 
48  public:
49  enum { NBinsX,
50  NBinsY,
51  XMin,
52  XMax,
53  YMin,
54  YMax,
55  BinSizeX,
56  BinSizeY,
57  BinSizeXInv,
58  BinSizeYInv,
59  NServiceSlots };
60 
61  FlatHisto2D() = default;

63  FlatHisto2D(int nbx, T xmin, T xmax, int nby, T ymin, T ymax)
64  {
65  assert(nbx > 0 && xmin < xmax);
66  assert(nby > 0 && ymin < ymax);
67  mData.resize(nbx * nby + NServiceSlots, 0.);
68  mData[NBinsX] = nbx;
69  mData[NBinsY] = nby;
70  mData[XMin] = xmin;
71  mData[XMax] = xmax;
72  mData[YMin] = ymin;
73  mData[YMax] = ymax;
74  mData[BinSizeX] = (xmax - xmin) / nbx;
75  mData[BinSizeXInv] = nbx / (xmax - xmin);
76  mData[BinSizeY] = (ymax - ymin) / nby;
77  mData[BinSizeYInv] = nby / (ymax - ymin);
78  init();
79  }

135  void add(const FlatHisto2D& other)
136  {
137  assert(getNBinsX() == other.getNBinsX() && getXMin() == other.getXMin() && getXMax() == other.getXMax() &&
138  getNBinsY() == other.getNBinsY() && getYMin() == other.getYMin() && getYMax() == other.getYMax() &&
139  canFill());
140  int last = NServiceSlots + getNBins();
141  const auto& otherView = other.getView();
142  for (int i = NServiceSlots; i < last; i++) {
143  mData[i] += otherView[i];
144  }
145  }
146 
147  void subtract(const FlatHisto2D& other)
148  {
149  assert(getNBinsX() == other.getNBinsX() && getXMin() == other.getXMin() && getXMax() == other.getXMax() &&
150  getNBinsY() == other.getNBinsY() && getYMin() == other.getYMin() && getYMax() == other.getYMax() &&
151  canFill());
152  int last = NServiceSlots + getNBins();
153  const auto& otherView = other.getView();
154  for (int i = NServiceSlots; i < last; i++) {
155  mData[i] -= otherView[i];
156  }
157  }

159  void setBinContent(uint32_t bin, T w)
160  {
161  assert(canFill());
162  if (bin < getNBins()) {
163  mData[bin + NServiceSlots] = w;
164  }
165  }
166 
167  void setBinContent(uint32_t binX, uint32_t binY, T w)
168  {
169  assert(canFill());
170  auto bin = getGlobalBin(binX, binY);
171  if (bin < getNBins()) {
172  mData[+NServiceSlots] = w;
173  }
174  }
175 
176  void clear()
177  {
178  assert(canFill());
179  memset(mData.data() + NServiceSlots, 0, sizeof(T) * getNBins());
180  }
181 
182  T getSum() const
183  {
184  T sum = 0;
185  for (int i = getNBins(); i--;) {
186  sum += getBinContent(i);
187  }
188  return sum;
189  }

191  void adoptExternal(const gsl::span<const T> ext)
192  {
193  assert(ext.size() > NServiceSlots);
194  mData.clear();
195  mDataView = ext;
196  mNBinsX = (int)mDataView[NBinsX];
197  mNBinsY = (int)mDataView[NBinsY];
198  mNBins = mNBinsX * mNBinsY;
199  }
200 
201  void init()
202  { // when reading from file, need to call this method to make it operational
203  assert(mData.size() > NServiceSlots);
204  mDataView = gsl::span<const T>(mData.data(), mData.size());
205  mNBinsX = (int)mData[NBinsX];
206  mNBinsY = (int)mData[NBinsY];
207  mNBins = mNBinsX * mNBinsY;
208  }
209 
210  void fill(T x, T y)
211  {
212  uint32_t bin = getBin(x, y);
213  if (bin < mNBins) {
214  mData[NServiceSlots + bin]++;
215  }
216  }
217 
218  void fill(T x, T y, T w)
219  {
220  uint32_t bin = getBin(x, y);
221  if (bin < mNBins) {
222  mData[NServiceSlots + bin] += w;
223  }
224  }
225 
226  uint32_t getBinX(T x) const
227  {
228  auto dx = x - getXMin();
229  return dx < 0 ? 0xffffffff : uint32_t(dx * getBinSizeXInv());
230  }
231 
232  uint32_t getBinY(T y) const
233  {
234  auto dy = y - getYMin();
235  return dy < 0 ? 0xffffffff : uint32_t(dy * getBinSizeYInv());
236  }
237 
238  uint32_t getBin(T x, T y) const
239  {
240  auto bx = getBinX(x), by = getBinY(y);
241  return bx < getNBinsX() && by < getNBinsY() ? getGlobalBin(bx, by) : 0xffffffff;
242  }
243 
244  bool canFill() const
245  {
246  // histo can be filled only if hase its own data, otherwise only query can be done on the view
247  return mData.size() > NServiceSlots;
248  }
249 
250  gsl::span<const T> getSliceY(uint32_t binX) const
251  {
252  int offs = NServiceSlots + binX * getNBinsY();
253  return binX < getNBinsX() ? gsl::span<const T>(&mDataView[offs], getNBinsY()) : gsl::span<const T>();
254  }

256  TH1F createSliceYTH1F(uint32_t binX, const std::string& name = "histo2dslice") const
257  {
258  TH1F h(name.c_str(), name.c_str(), getNBinsY(), getYMin(), getYMax());
259  if (binX < getNBinsX()) {
260  for (int i = getNBinsY(); i--;) {
261  h.SetBinContent(i + 1, getBinContent(binX, i));
262  }
263  }
264  return h;
265  }
266 
267  TH2F createTH2F(const std::string& name = "histo2d")
268  {
269  TH2F h(name.c_str(), name.c_str(), getNBinsX(), getXMin(), getXMax(), getNBinsY(), getYMin(), getYMax());
270  for (int i = getNBinsX(); i--;) {
271  for (int j = getNBinsY(); j--;) {
272  auto w = getBinContent(i, j);
273  if (w) {
274  h.SetBinContent(i + 1, j + 1, w);
275  }
276  }
277  }
278  return h;
279  }
280 
281  const std::vector<T>& getBase() const { return mData; }
282  gsl::span<const T> getView() const { return mDataView; }
283 
284  int getGlobalBin(uint32_t binX, uint32_t binY) const { return binX * getNBinsY() + binY; }
285 
286  protected:
287  T getBinSizeXInv() const { return mDataView[BinSizeXInv]; }
288  T getBinSizeYInv() const { return mDataView[BinSizeYInv]; }
289 
290  std::vector<T> mData; // data to fill
291  gsl::span<const T> mDataView;
292  int mNBinsX = 0;
293  int mNBinsY = 0;
294  int mNBins = 0;
295 
296  ClassDefNV(FlatHisto2D, 1);
297 };
