100  // TODO: make mixing binning configurable
101  std::vector<float> fCentLimsHashing{0.0f, 10.0f, 20.0f, 30.0f, 50.0f, 70.0f, 90.0f};
102 
103  Configurable<std::string> fConfigEventCuts{"cfgEventCuts", "eventStandard", "Event selection"};

105  int getHash(float centrality)
106  {
107  if ((centrality < *fCentLimsHashing.begin()) || (centrality > *(fCentLimsHashing.end() - 1))) {
108  return -1;
109  }
110  auto cent = std::lower_bound(fCentLimsHashing.begin(), fCentLimsHashing.end(), centrality);
111  return (cent - fCentLimsHashing.begin());
112  }

186  void DefineCuts()
187  {
188  // Defines track cuts for both the dielectron candidates and also for the dilepton - hadron task
189  TString cutNamesStr = fConfigElectronCuts.value;
190  if (!cutNamesStr.IsNull()) {
191  std::unique_ptr<TObjArray> objArray(cutNamesStr.Tokenize(","));
192  for (int icut = 0; icut < objArray->GetEntries(); ++icut) {
193  fTrackCuts.push_back(*dqcuts::GetCompositeCut(objArray->At(icut)->GetName()));
194  }
195  }
196 
197  AnalysisCompositeCut correlationsHadronCut("hadronCut", "hadronCut", true);
198  correlationsHadronCut.AddCut(dqcuts::GetAnalysisCut("electronStandardQuality"));
199  correlationsHadronCut.AddCut(dqcuts::GetAnalysisCut("standardPrimaryTrack"));
200  AnalysisCut hadronKine;
201  hadronKine.AddCut(VarManager::kPt, 4.0, 100.0);
202  correlationsHadronCut.AddCut(&hadronKine);
203  fTrackCuts.push_back(correlationsHadronCut);
204 
205  VarManager::SetUseVars(AnalysisCut::fgUsedVars); // provide the list of required variables so that VarManager knows what to fill
206  }

208  void process(MyEvents::iterator const& event, MyBarrelTracks const& tracks)
209  {
210  VarManager::ResetValues(0, VarManager::kNBarrelTrackVariables, fValues);
211  // fill event information which might be needed in histograms that combine track and event properties
212  VarManager::FillEvent<gkEventFillMap>(event, fValues);
213 
214  uint8_t filterMap = uint8_t(0);
215 
216  trackSel.reserve(tracks.size());
217 
218  for (auto& track : tracks) {
219  filterMap = uint8_t(0);
220  VarManager::FillTrack<gkTrackFillMap>(track, fValues);
221  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", fValues);
222 
223  int i = 0;
224  for (auto cut = fTrackCuts.begin(); cut != fTrackCuts.end(); cut++, i++) {
225  if ((*cut).IsSelected(fValues)) {
226  filterMap |= (uint8_t(1) << i);
227  fHistMan->FillHistClass(Form("TrackBarrel_%s", (*cut).GetName()), fValues);
228  }
229  }
230  trackSel(filterMap);
231  }
232  }

330  void process(soa::Filtered<MyEventsHashSelected>& events, soa::Filtered<MyBarrelTracksSelected> const& tracks, soa::Filtered<MyMuonTracksSelected> const& muons)
331  {
332  uint8_t twoTrackFilter = 0;
333 
334  events.bindExternalIndices(&tracks);
335  events.bindExternalIndices(&muons);
336  auto tracksTuple = std::make_tuple(tracks);
337  auto muonsTuple = std::make_tuple(muons);
338  AnalysisDataProcessorBuilder::GroupSlicer slicerTracks(events, tracksTuple);
339  AnalysisDataProcessorBuilder::GroupSlicer slicerMuons(events, muonsTuple);
340 
341  // Strictly upper categorised collisions, for 100 combinations per bin, skipping those in entry -1
342  for (auto& [event1, event2] : selfCombinations("fMixingHash", 100, -1, events, events)) {
343 
344  // event informaiton is required to fill histograms where both event and pair information is required (e.g. inv.mass vs centrality)
345  VarManager::ResetValues(0, VarManager::kNVars, fValues);
346  VarManager::FillEvent<gkEventFillMap>(event1, fValues);
347 
348  auto it1 = slicerTracks.begin();
349  auto it2 = slicerTracks.begin();
350  for (auto& slice : slicerTracks) {
351  if (slice.groupingElement().index() == event1.index()) {
352  it1 = slice;
353  break;
354  }
355  }
356  for (auto& slice : slicerTracks) {
357  if (slice.groupingElement().index() == event2.index()) {
358  it2 = slice;
359  break;
360  }
361  }
362 
363  auto tracks1 = std::get<soa::Filtered<MyBarrelTracksSelected>>(it1.associatedTables());
364  tracks1.bindExternalIndices(&events);
365  auto tracks2 = std::get<soa::Filtered<MyBarrelTracksSelected>>(it2.associatedTables());
366  tracks2.bindExternalIndices(&events);
367 
368  for (auto& track1 : tracks1) {
369  for (auto& track2 : tracks2) {
370  twoTrackFilter = track1.isBarrelSelected() & track2.isBarrelSelected() & fTwoTrackFilterMask;
371 
372  if (!twoTrackFilter) { // the tracks must have at least one filter bit in common to continue
373  continue;
374  }
375  VarManager::FillPair(track1, track2, fValues);
376 
377  for (int i = 0; i < fCutNames.size(); ++i) {
378  if (twoTrackFilter & (uint8_t(1) << i)) {
379  if (track1.charge() * track2.charge() < 0) {
380  fHistMan->FillHistClass(Form("PairsBarrelMEPM_%s", fCutNames[i].Data()), fValues);
381  } else {
382  if (track1.charge() > 0) {
383  fHistMan->FillHistClass(Form("PairsBarrelMEPP_%s", fCutNames[i].Data()), fValues);
384  } else {
385  fHistMan->FillHistClass(Form("PairsBarrelMEMM_%s", fCutNames[i].Data()), fValues);
386  }
387  }
388  } // end if (filter bits)
389  } // end for (cuts)
390  } // end for (track2)
391  } // end for (track1)
392 
393  auto im1 = slicerMuons.begin();
394  auto im2 = slicerMuons.begin();
395  for (auto& slice : slicerMuons) {
396  if (slice.groupingElement().index() == event1.index()) {
397  im1 = slice;
398  break;
399  }
400  }
401  for (auto& slice : slicerMuons) {
402  if (slice.groupingElement().index() == event2.index()) {
403  im2 = slice;
404  break;
405  }
406  }
407 
408  auto muons1 = std::get<soa::Filtered<MyMuonTracksSelected>>(im1.associatedTables());
409  muons1.bindExternalIndices(&events);
410  auto muons2 = std::get<soa::Filtered<MyMuonTracksSelected>>(im2.associatedTables());
411  muons2.bindExternalIndices(&events);
412 
413  for (auto& muon1 : muons1) {
414  for (auto& muon2 : muons2) {
415  twoTrackFilter = muon1.isMuonSelected() & muon2.isMuonSelected();
416  if (!twoTrackFilter) { // the tracks must have at least one filter bit in common to continue
417  continue;
418  }
419  VarManager::FillPair(muon1, muon2, fValues);
420  if (muon1.charge() * muon2.charge() < 0) {
421  fHistMan->FillHistClass("PairsMuonMEPM", fValues);
422  } else {
423  if (muon1.charge() > 0) {
424  fHistMan->FillHistClass("PairsMuonMEPP", fValues);
425  } else {
426  fHistMan->FillHistClass("PairsMuonMEMM", fValues);
427  }
428  }
429  } // end for (muon2)
430  } // end for (muon1)
431  } // end for (event combinations)
432  } // end process()
433 };

479  void process(soa::Filtered<MyEventsVtxCovSelected>::iterator const& event, soa::Filtered<MyBarrelTracksSelected> const& tracks, soa::Filtered<MyMuonTracksSelected> const& muons)
480  {
481  if (!event.isEventSelected()) {
482  return;
483  }
484  // Reset the fValues array
485  VarManager::ResetValues(0, VarManager::kNVars, fValues);
486  VarManager::FillEvent<gkEventFillMap>(event, fValues);
487 
488  // Run the same event pairing for barrel tracks
489  uint8_t twoTrackFilter = 0;
490  uint16_t dileptonFilterMap = 0;
491  for (auto& [t1, t2] : combinations(tracks, tracks)) {
492  twoTrackFilter = t1.isBarrelSelected() & t2.isBarrelSelected() & fTwoTrackFilterMask;
493  if (!twoTrackFilter) { // the tracks must have at least one filter bit in common to continue
494  continue;
495  }
496  dileptonFilterMap = uint16_t(twoTrackFilter);
497  VarManager::FillPair(t1, t2, fValues);
498  dileptonList(event, fValues[VarManager::kMass], fValues[VarManager::kPt], fValues[VarManager::kEta], fValues[VarManager::kPhi], t1.charge() + t2.charge(), dileptonFilterMap);
499  for (int i = 0; i < fCutNames.size(); ++i) {
500  if (twoTrackFilter & (uint8_t(1) << i)) {
501  if (t1.charge() * t2.charge() < 0) {
502  fHistMan->FillHistClass(Form("PairsBarrelSEPM_%s", fCutNames[i].Data()), fValues);
503  } else {
504  if (t1.charge() > 0) {
505  fHistMan->FillHistClass(Form("PairsBarrelSEPP_%s", fCutNames[i].Data()), fValues);
506  } else {
507  fHistMan->FillHistClass(Form("PairsBarrelSEMM_%s", fCutNames[i].Data()), fValues);
508  }
509  }
510  }
511  }
512  } // end loop over barrel track pairs
513 
514  // same event pairing for muons
515  for (auto& [muon1, muon2] : combinations(muons, muons)) {
516  twoTrackFilter = muon1.isMuonSelected() & muon2.isMuonSelected();
517  if (!twoTrackFilter) { // the muons must have at least one filter bit in common to continue
518  continue;
519  }
520  // NOTE: The dimuons in this task ae pushed in the same table as the dielectrons.
521  // In order to discriminate them, the dileptonFilterMap uses the first 8 bits for dielectrons and the last 8 for dimuons.
522  // TBD: Other implementations may be possible, for example add a column to the dilepton table to specify the pair type (dielectron, dimuon, electron-muon, etc.)
523  dileptonFilterMap = uint16_t(twoTrackFilter) << 8;
524  VarManager::FillPair(muon1, muon2, fValues);
525  dileptonList(event, fValues[VarManager::kMass], fValues[VarManager::kPt], fValues[VarManager::kEta], fValues[VarManager::kPhi], muon1.charge() + muon2.charge(), dileptonFilterMap);
526  if (muon1.charge() * muon2.charge() < 0) {
527  fHistMan->FillHistClass("PairsMuonSEPM", fValues);
528  } else {
529  if (muon1.charge() > 0) {
530  fHistMan->FillHistClass("PairsMuonSEPP", fValues);
531  } else {
532  fHistMan->FillHistClass("PairsMuonSEMM", fValues);
533  }
534  }
535  } // end loop over muon track pairs
536  }

539 struct DileptonHadronAnalysis {
540  //
541  // This task combines dilepton candidates with a track and could be used for example
542  // in analyses with the dilepton as one of the decay products of a higher mass resonance (e.g. B0 -> Jpsi + K)
543  // or in dilepton + hadron correlations, etc.
544  // It requires the TableReader task to be in the workflow and produce the dilepton table
545  //
546  // The barrel and muon track filtering tasks can produce multiple parallel decisions, which are used to produce
547  // dileptons which inherit the logical intersection of the track level decisions (see the TableReader task). This can be used
548  // also in the dilepton-hadron correlation analysis. However, in this model of the task, we use all the dileptons produced in the
549  // TableReader task to combine them with the hadrons selected by the barrel track selection.
550  OutputObj<THashList> fOutputList{"output"};
551  HistogramManager* fHistMan;
552 
553  // use two values array to avoid mixing up the quantities
554  float* fValuesDilepton;
555  float* fValuesHadron;
556 
557  Filter eventFilter = aod::reducedevent::isEventSelected == 1;
558  Filter dileptonFilter = aod::reducedpair::mass > 2.92f && aod::reducedpair::mass<3.16f && aod::reducedpair::pt> 0.0f && aod::reducedpair::charge == 0;
559  // NOTE: the barrel track filter is shared between the filters for dilepton electron candidates (first n-bits)
560  // and the associated hadrons (n+1 bit) --> see the barrel track selection task
561  // The current condition should be replaced when bitwise operators will become available in Filter expresions
562  // NOTE: the name of this configurable has to be the same as the one used in the barrel track selection task
563  Configurable<std::string> fConfigElectronCuts{"cfgElectronCuts", "jpsiPID1", "Comma separated list of barrel track cuts"};
564  int fNHadronCutBit;

568  void init(o2::framework::InitContext&)

571  fValuesHadron = new float[VarManager::kNVars];

578  VarManager::SetUseVars(fHistMan->GetUsedVars());

580 
581  TString configCutNamesStr = fConfigElectronCuts.value;
582  if (!configCutNamesStr.IsNull()) {
583  std::unique_ptr<TObjArray> objArray(configCutNamesStr.Tokenize(","));
584  fNHadronCutBit = objArray->GetEntries();
585  } else {
586  fNHadronCutBit = 0;

590  void process(soa::Filtered<MyEventsVtxCovSelected>::iterator const& event, MyBarrelTracksSelected const& hadrons, soa::Filtered<aod::Dileptons> const& dileptons)
591  {
592  VarManager::ResetValues(0, VarManager::kNVars, fValuesHadron);
593  VarManager::ResetValues(0, VarManager::kNVars, fValuesDilepton);
594  // fill event information which might be needed in histograms that combine track/pair and event properties
595  VarManager::FillEvent<gkEventFillMap>(event, fValuesHadron);
596  VarManager::FillEvent<gkEventFillMap>(event, fValuesDilepton); // TODO: check if needed (just for dilepton QA which might be depending on event wise variables)
597 
598  // loop once over dileptons for QA purposes
599  for (auto dilepton : dileptons) {
600  VarManager::FillTrack<fgDileptonFillMap>(dilepton, fValuesDilepton);
601  fHistMan->FillHistClass("DileptonsSelected", fValuesDilepton);
602  }
603 
604  // loop over hadrons
605  for (auto& hadron : hadrons) {
606  if (!(hadron.isBarrelSelected() & (uint8_t(1) << fNHadronCutBit))) {
607  continue;
608  }
609  for (auto dilepton : dileptons) {
610  // TODO: At the moment there is no check on whether this hadron is one of the dilepton daughters!
611  VarManager::FillDileptonHadron(dilepton, hadron, fValuesHadron);
612  fHistMan->FillHistClass("DileptonHadronInvMass", fValuesHadron);
613  fHistMan->FillHistClass("DileptonHadronCorrelation", fValuesHadron);
614  }
615  }
616  }
