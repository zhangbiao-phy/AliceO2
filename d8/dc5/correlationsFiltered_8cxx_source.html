162  continue;
163  }

169  continue;

178  deltaPhi -= TMath::TwoPi();

187  }

191  template <typename T>
192  bool conversionCuts(T const& track1, T const& track2)
193  {
194  // skip if like sign
195  if (track1.charge() * track2.charge() > 0) {
196  return false;
197  }
198 
199  bool decision = false;
200 
201  if (conversionCut(track1, track2, Photon, cfgPairCutPhoton)) {
202  decision = true;
203  }
204  if (conversionCut(track1, track2, K0, cfgPairCutK0)) {
205  decision = true;
206  }
207  if (conversionCut(track1, track2, Lambda, cfgPairCutLambda)) {
208  decision = true;
209  }
210  if (conversionCut(track2, track1, Lambda, cfgPairCutLambda)) {
211  decision = true;
212  }
213  if (conversionCut(track1, track2, Phi, cfgPairCutPhi)) {
214  decision = true;
215  }
216  if (conversionCut(track1, track2, Rho, cfgPairCutRho)) {
217  decision = true;
218  }
219 
220  return decision;
221  }

223  template <typename T>
224  bool conversionCut(T const& track1, T const& track2, PairCuts conv, double cut)
225  {
226  //LOGF(info, "pt is %f %f", track1.pt(), track2.pt());
227 
228  if (cut < 0) {
229  return false;
230  }
231 
232  double massD1, massD2, massM;
233 
234  switch (conv) {
235  case Photon:
236  massD1 = 0.51e-3;
237  massD2 = 0.51e-3;
238  massM = 0;
239  break;
240  case K0:
241  massD1 = 0.1396;
242  massD2 = 0.1396;
243  massM = 0.4976;
244  break;
245  case Lambda:
246  massD1 = 0.9383;
247  massD2 = 0.1396;
248  massM = 1.115;
249  break;
250  case Phi:
251  massD1 = 0.4937;
252  massD2 = 0.4937;
253  massM = 1.019;
254  break;
255  case Rho:
256  massD1 = 0.1396;
257  massD2 = 0.1396;
258  massM = 0.770;
259  break;
260  }
261 
262  auto massC = getInvMassSquaredFast(track1, massD1, track2, massD2);
263 
264  if (TMath::Abs(massC - massM * massM) > cut * 5) {
265  return false;
266  }
267 
268  massC = getInvMassSquared(track1, massD1, track2, massD2);
269  qa.mControlConvResoncances->Fill(static_cast<int>(conv), massC - massM * massM);
270  if (massC > (massM - cut) * (massM - cut) && massC < (massM + cut) * (massM + cut)) {
271  return true;
272  }
273 
274  return false;
275  }

277  template <typename T>
278  double getInvMassSquared(T const& track1, double m0_1, T const& track2, double m0_2)
279  {
280  // calculate inv mass squared
281  // same can be achieved, but with more computing time with
282  /*TLorentzVector photon, p1, p2;
283  p1.SetPtEtaPhiM(triggerParticle->Pt(), triggerEta, triggerParticle->Phi(), 0.510e-3);
284  p2.SetPtEtaPhiM(particle->Pt(), eta[j], particle->Phi(), 0.510e-3);
285  photon = p1+p2;
286  photon.M()*/
287 
288  float tantheta1 = 1e10;
289 
290  if (track1.eta() < -1e-10 || track1.eta() > 1e-10) {
291  float expTmp = TMath::Exp(-track1.eta());
292  tantheta1 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
293  }
294 
295  float tantheta2 = 1e10;
296  if (track2.eta() < -1e-10 || track2.eta() > 1e-10) {
297  float expTmp = TMath::Exp(-track2.eta());
298  tantheta2 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
299  }
300 
301  float e1squ = m0_1 * m0_1 + track1.pt() * track1.pt() * (1.0 + 1.0 / tantheta1 / tantheta1);
302  float e2squ = m0_2 * m0_2 + track2.pt() * track2.pt() * (1.0 + 1.0 / tantheta2 / tantheta2);
303 
304  float mass2 = m0_1 * m0_1 + m0_2 * m0_2 + 2 * (TMath::Sqrt(e1squ * e2squ) - (track1.pt() * track2.pt() * (TMath::Cos(track1.phi() - track2.phi()) + 1.0 / tantheta1 / tantheta2)));
305 
306  // Printf(Form("%f %f %f %f %f %f %f %f %f", pt1, eta1, phi1, pt2, eta2, phi2, m0_1, m0_2, mass2));
307 
308  return mass2;
309  }

311  template <typename T>
312  double getInvMassSquaredFast(T const& track1, double m0_1, T const& track2, double m0_2)
313  {
314  // calculate inv mass squared approximately
315 
316  const float eta1 = track1.eta();
317  const float eta2 = track2.eta();
318  const float phi1 = track1.phi();
319  const float phi2 = track2.phi();
320  const float pt1 = track1.pt();
321  const float pt2 = track2.pt();
322 
323  float tantheta1 = 1e10;
324 
325  if (eta1 < -1e-10 || eta1 > 1e-10) {
326  float expTmp = 1.0 - eta1 + eta1 * eta1 / 2 - eta1 * eta1 * eta1 / 6 + eta1 * eta1 * eta1 * eta1 / 24;
327  tantheta1 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
328  }
329 
330  float tantheta2 = 1e10;
331  if (eta2 < -1e-10 || eta2 > 1e-10) {
332  float expTmp = 1.0 - eta2 + eta2 * eta2 / 2 - eta2 * eta2 * eta2 / 6 + eta2 * eta2 * eta2 * eta2 / 24;
333  tantheta2 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
334  }
335 
336  float e1squ = m0_1 * m0_1 + pt1 * pt1 * (1.0 + 1.0 / tantheta1 / tantheta1);
337  float e2squ = m0_2 * m0_2 + pt2 * pt2 * (1.0 + 1.0 / tantheta2 / tantheta2);
338 
339  // fold onto 0...pi
340  float deltaPhi = TMath::Abs(phi1 - phi2);
341  while (deltaPhi > TMath::TwoPi()) {
342  deltaPhi -= TMath::TwoPi();
343  }
344  if (deltaPhi > TMath::Pi()) {
345  deltaPhi = TMath::TwoPi() - deltaPhi;
346  }
347 
348  float cosDeltaPhi = 0;
349  if (deltaPhi < TMath::Pi() / 3) {
350  cosDeltaPhi = 1.0 - deltaPhi * deltaPhi / 2 + deltaPhi * deltaPhi * deltaPhi * deltaPhi / 24;
351  } else if (deltaPhi < 2 * TMath::Pi() / 3) {
352  cosDeltaPhi = -(deltaPhi - TMath::Pi() / 2) + 1.0 / 6 * TMath::Power((deltaPhi - TMath::Pi() / 2), 3);
353  } else {
354  cosDeltaPhi = -1.0 + 1.0 / 2.0 * (deltaPhi - TMath::Pi()) * (deltaPhi - TMath::Pi()) - 1.0 / 24.0 * TMath::Power(deltaPhi - TMath::Pi(), 4);
355  }
356 
357  double mass2 = m0_1 * m0_1 + m0_2 * m0_2 + 2 * (TMath::Sqrt(e1squ * e2squ) - (pt1 * pt2 * (cosDeltaPhi + 1.0 / tantheta1 / tantheta2)));
358 
359  // Printf(Form("%f %f %f %f %f %f %f %f %f", pt1, eta1, phi1, pt2, eta2, phi2, m0_1, m0_2, mass2));
360 
361  return mass2;
362  }

364  template <typename T>
365  bool twoTrackCut(T const& track1, T const& track2, int bSign)
366  {
367  // the variables & cuthave been developed by the HBT group
368  // see e.g. https://indico.cern.ch/materialDisplay.py?contribId=36&sessionId=6&materialId=slides&confId=142700
369 
370  auto deta = track1.eta() - track2.eta();
371 
372  // optimization
373  if (TMath::Abs(deta) < cfgTwoTrackCut * 2.5 * 3) {
374  // check first boundaries to see if is worth to loop and find the minimum
375  float dphistar1 = getDPhiStar(track1, track2, cfgTwoTrackCutMinRadius, bSign);
376  float dphistar2 = getDPhiStar(track1, track2, 2.5, bSign);
377 
378  const float kLimit = cfgTwoTrackCut * 3;
379 
380  if (TMath::Abs(dphistar1) < kLimit || TMath::Abs(dphistar2) < kLimit || dphistar1 * dphistar2 < 0) {
381  float dphistarminabs = 1e5;
382  float dphistarmin = 1e5;
383  for (Double_t rad = cfgTwoTrackCutMinRadius; rad < 2.51; rad += 0.01) {
384  float dphistar = getDPhiStar(track1, track2, rad, bSign);
385 
386  float dphistarabs = TMath::Abs(dphistar);
387 
388  if (dphistarabs < dphistarminabs) {
389  dphistarmin = dphistar;
390  dphistarminabs = dphistarabs;
391  }
392  }
393 
394  qa.mTwoTrackDistancePt[0]->Fill(deta, dphistarmin, TMath::Abs(track1.pt() - track2.pt()));
395 
396  if (dphistarminabs < cfgTwoTrackCut && TMath::Abs(deta) < cfgTwoTrackCut) {
397  //Printf("Removed track pair %ld %ld with %f %f %f %f %d %f %f %d %d", track1.index(), track2.index(), deta, dphistarminabs, track1.phi2(), track1.pt(), track1.charge(), track2.phi2(), track2.pt(), track2.charge(), bSign);
398  return true;
399  }
400 
401  qa.mTwoTrackDistancePt[1]->Fill(deta, dphistarmin, TMath::Abs(track1.pt() - track2.pt()));
402  }
403  }
404 
405  return false;
406  }

408  template <typename T>
409  float getDPhiStar(T const& track1, T const& track2, float radius, float bSign)
410  {
411  //
412  // calculates dphistar
413  //
414 
415  auto phi1 = track1.phi();
416  auto pt1 = track1.pt();
417  auto charge1 = track1.charge();
418 
419  auto phi2 = track2.phi();
420  auto pt2 = track2.pt();
421  auto charge2 = track2.charge();
422 
423  float dphistar = phi1 - phi2 - charge1 * bSign * TMath::ASin(0.075 * radius / pt1) + charge2 * bSign * TMath::ASin(0.075 * radius / pt2);
424 
425  static const Double_t kPi = TMath::Pi();
426 
427  if (dphistar > kPi) {
428  dphistar = kPi * 2 - dphistar;
429  }
430  if (dphistar < -kPi) {
431  dphistar = -kPi * 2 - dphistar;
432  }
433  if (dphistar > kPi) { // might look funny but is needed
434  dphistar = kPi * 2 - dphistar;
435  }
436 
437  return dphistar;
438  }
