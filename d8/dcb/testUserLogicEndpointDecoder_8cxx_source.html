105 SampaChannelHandler handlePacket(std::string& result)
106 {
107  return [&result](DsElecId dsId, uint8_t channel, SampaCluster sc) {
108  result += fmt::format("{}-ch-{}-ts-{}-q", asString(dsId), channel, sc.sampaTime);
109  if (sc.isClusterSum()) {
110  result += fmt::format("-{}-cs-{}", sc.chargeSum, sc.clusterSize);
111  } else {
112  for (auto s : sc.samples) {
113  result += fmt::format("-{}", s);
114  }
115  }
116  result += "\n";
117  };

128 std::vector<std::byte> convertBuffer2PayloadBuffer(gsl::span<const std::byte> buffer,
129  std::optional<size_t> insertSync = std::nullopt)
130 {
131  // some gym to go from a buffer coming from the encoder,
132  // which holds (DataBlockHeader,Payload) pairs in a byte buffer
133  // to a buffer that holds just Payload information
134  // to be fed to the decoder
135 
136  // strip the headers
137  std::vector<std::byte> b8;
138  forEachDataBlockRef(
139  buffer, [&](const DataBlockRef& r) {
140  auto& b = r.block;
141  b8.insert(b8.end(), b.payload.begin(), b.payload.end());
142  });
143 
144  // convert to a 64-bits buffer to be able to insert sync if needed
145  std::vector<uint64_t> b64;
146  impl::copyBuffer(b8, b64);
147 
148  // insert a sync at the given position if required
149  if (insertSync.has_value() && insertSync.value() < b64.size()) {
150  uint64_t prefix = b64[0] & 0xFFFC00000000000F;
151  b64.insert(b64.begin() + insertSync.value(), prefix | sampaSyncWord);
152  }
153 
154  impl::dumpBuffer<o2::mch::raw::UserLogicFormat>(b64);
155 
156  // get back to byte buffer to return
157  std::vector<std::byte> bytes;
158  impl::copyBuffer(b64, bytes);
159  return bytes;
160 }

162 template <typename CHARGESUM>
163 std::string decodeBuffer(int feeId, gsl::span<const std::byte> buffer)
164 {
165  std::string results;
166  auto fee2solar = createFeeLink2SolarMapper<ElectronicMapperGenerated>();
167  DecodedDataHandlers handlers;
168  handlers.sampaChannelHandler = handlePacket(results);
169  handlers.sampaErrorHandler = handleError(results);
170  UserLogicEndpointDecoder<CHARGESUM> dec(feeId, fee2solar, handlers);
171  dec.append(buffer);
172  return results;
173 }

175 template <typename CHARGESUM>
176 std::string testPayloadDecode(DsElecId ds1,
177  int ch1,
178  const std::vector<SampaCluster>& clustersFirstChannel,
179  DsElecId ds2 = DsElecId{0, 0, 0},
180  int ch2 = 47,
181  const std::vector<SampaCluster>& clustersSecondChannel = {},
182  std::optional<size_t> insertSync = std::nullopt)
183 {
184  auto encoder = createPayloadEncoder<UserLogicFormat, CHARGESUM, true>();
185 
186  encoder->startHeartbeatFrame(0, 0);
187 
188  auto solar2feelink = createSolar2FeeLinkMapper<ElectronicMapperGenerated>();
189 
190  uint16_t feeId{0};
191 
192  auto f1 = solar2feelink(ds1.solarId());
193  if (!f1.has_value()) {
194  throw std::invalid_argument("invalid solarId for ds1");
195  }
196  if (!clustersSecondChannel.empty()) {
197  auto f2 = solar2feelink(ds2.solarId());
198  if (!f2.has_value()) {
199  throw std::invalid_argument("invalid solarId for ds2");
200  }
201  if (f2->feeId() != f1->feeId()) {
202  throw std::invalid_argument("this test is only meant to work with 2 solars in the same cru endpoint");
203  }
204  }
205 
206  feeId = f1->feeId();
207 
208  encoder->addChannelData(ds1, ch1, clustersFirstChannel);
209  if (!clustersSecondChannel.empty()) {
210  encoder->addChannelData(ds2, ch2, clustersSecondChannel);
211  }
212 
213  std::vector<std::byte> buffer;
214  encoder->moveToBuffer(buffer);
215  auto payloadBuffer = convertBuffer2PayloadBuffer(buffer, insertSync);
216 
217  return decodeBuffer<CHARGESUM>(feeId, payloadBuffer);
218 }

220 std::string testPayloadDecodeCruPages(gsl::span<const uint64_t> page)
221 {
222  auto solar2feelink = createSolar2FeeLinkMapper<ElectronicMapperGenerated>();
223 
224  const void* rdhP = reinterpret_cast<const void*>(page.data());
225  uint16_t feeId = o2::raw::RDHUtils::getFEEID(rdhP);
226  auto rdhSize = o2::raw::RDHUtils::getHeaderSize(rdhP);
227  auto payloadSize = o2::raw::RDHUtils::getMemorySize(rdhP) - rdhSize;
228 
229  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(page.data()), page.size() * 8);
230  gsl::span<const std::byte> payloadBuffer = buffer.subspan(rdhSize, payloadSize);
231 
232  const uint16_t CRUID_MASK = 0xFF;
233  const uint16_t CHARGESUM_MASK = 0x100;
234  if (feeId & CHARGESUM_MASK) {
235  return decodeBuffer<ChargeSumMode>(feeId & CRUID_MASK, payloadBuffer);
236  } else {
237  return decodeBuffer<SampleMode>(feeId & CRUID_MASK, payloadBuffer);
238  }
239 }
