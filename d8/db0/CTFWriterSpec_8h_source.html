82 // process data of particular detector
83 template <typename C>
84 void CTFWriterSpec::processDet(o2::framework::ProcessingContext& pc, DetID det, CTFHeader& header, TTree* tree)
85 {
86  if (!isPresent(det) || !pc.inputs().isValid(det.getName())) {
87  return;
88  }
89  auto ctfBuffer = pc.inputs().get<gsl::span<o2::ctf::BufferType>>(det.getName());
90  const auto ctfImage = C::getImage(ctfBuffer.data());
91  ctfImage.print(o2::utils::concat_string(det.getName(), ": "));
92  if (mWriteCTF) {
93  ctfImage.appendToTree(*tree, det.getName());
94  header.detectors.set(det);
95  }
96  if (mCreateDict) {
97  if (!mFreqsAccumulation[det].size()) {
98  mFreqsAccumulation[det].resize(C::getNBlocks());
99  mFreqsMetaData[det].resize(C::getNBlocks());
100  }
101  if (!mHeaders[det]) { // store 1st header
102  mHeaders[det] = ctfImage.cloneHeader();
103  }
104  for (int ib = 0; ib < C::getNBlocks(); ib++) {
105  const auto& bl = ctfImage.getBlock(ib);
106  if (bl.getNDict()) {
107  auto& freq = mFreqsAccumulation[det][ib];
108  auto& mdSave = mFreqsMetaData[det][ib];
109  const auto& md = ctfImage.getMetadata(ib);
110  freq.addFrequencies(bl.getDict(), bl.getDict() + bl.getNDict(), md.min, md.max);
111  mdSave = o2::ctf::Metadata{0, 0, md.coderType, md.streamSize, md.probabilityBits, md.opt, freq.getMinSymbol(), freq.getMaxSymbol(), (int)freq.size(), 0, 0};
112  }
113  }
114  }
115 }

117 // store dictionary of a particular detector
118 template <typename C>
119 void CTFWriterSpec::storeDictionary(DetID det, CTFHeader& header)
120 {
121  if (!isPresent(det) || !mFreqsAccumulation[det].size()) {
122  return;
123  }
124  prepareDictionaryTreeAndFile(det);
125  // create vector whose data contains dictionary in CTF format (EncodedBlock)
126  auto dictBlocks = C::createDictionaryBlocks(mFreqsAccumulation[det], mFreqsMetaData[det]);
127  auto& h = C::get(dictBlocks.data())->getHeader();
128  h = *reinterpret_cast<typename std::remove_reference<decltype(h)>::type*>(mHeaders[det].get());
129  C::get(dictBlocks.data())->print(o2::utils::concat_string("Storing dictionary for ", det.getName(), ": "));
130  C::get(dictBlocks.data())->appendToTree(*mDictTreeOut.get(), det.getName()); // cast to EncodedBlock
131  // mFreqsAccumulation[det].clear();
132  // mFreqsMetaData[det].clear();
133  if (mDictPerDetector) {
134  header.detectors.reset();
135  }
136  header.detectors.set(det);
137  if (mDictPerDetector) {
138  closeDictionaryTreeAndFile(header);
139  }
140 }
