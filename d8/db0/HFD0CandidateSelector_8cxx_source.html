78  template <typename T>
79  int getpTBin(T candpT)

81  double pTBins[npTBins + 1] = {0, 0.5, 1., 1.5, 2., 2.5, 3., 3.5, 4., 4.5, 5., 5.5, 6., 6.5, 7., 7.5, 8., 9., 10., 12., 16., 20., 24., 36., 50., 100.};

112  template <typename T>
113  bool selectionTopol(const T& hfCandProng2)
114  {
115  auto candpT = hfCandProng2.pt();
116  int pTBin = getpTBin(candpT);
117  if (pTBin == -1) {
118  return false;
119  }
120 
121  if (candpT < d_pTCandMin || candpT >= d_pTCandMax) {
122  return false; //check that the candidate pT is within the analysis range
123  }
124  if (hfCandProng2.impactParameterProduct() > cuts[pTBin][7]) {
125  return false; //product of daughter impact parameters
126  }
127  if (hfCandProng2.cpa() < cuts[pTBin][8]) {
128  return false; //cosine of pointing angle
129  }
130  if (hfCandProng2.cpaXY() < cuts[pTBin][9]) {
131  return false; //cosine of pointing angle XY
132  }
133  if (hfCandProng2.decayLengthXYNormalised() < cuts[pTBin][10]) {
134  return false; //normalised decay length in XY plane
135  }
136  // if (hfCandProng2.dca() > cuts[pTBin][1]) return false; //candidate DCA
137  //if (hfCandProng2.chi2PCA() > cuts[pTBin][1]) return false; //candidate DCA
138 
139  //decay exponentail law, with tau = beta*gamma*ctau
140  //decay length > ctau retains (1-1/e)
141  double decayLengthCut = TMath::Min((hfCandProng2.p() * 0.0066) + 0.01, 0.06);
142  if (TMath::Abs(hfCandProng2.impactParameterNormalised0()) < 0.5 || TMath::Abs(hfCandProng2.impactParameterNormalised1()) < 0.5) {
143  return false;
144  }
145  if (hfCandProng2.decayLength() * hfCandProng2.decayLength() < decayLengthCut * decayLengthCut) {
146  return false;
147  }
148  if (hfCandProng2.decayLengthNormalised() * hfCandProng2.decayLengthNormalised() < 1.0) {
149  //return false; //add back when getter fixed
150  }
151  return true;
152  }

160  template <typename T1, typename T2>
161  bool selectionTopolConjugate(const T1& hfCandProng2, const T2& trackPion, const T2& trackKaon)
162  {
163 
164  auto candpT = hfCandProng2.pt();
165  int pTBin = getpTBin(candpT);
166  if (pTBin == -1) {
167  return false;
168  }
169 
170  if (trackPion.charge() > 0) { //invariant mass cut
171  if (TMath::Abs(InvMassD0(hfCandProng2) - RecoDecay::getMassPDG(421)) > cuts[pTBin][0]) {
172  return false;
173  }
174  } else {
175  if (TMath::Abs(InvMassD0bar(hfCandProng2) - RecoDecay::getMassPDG(421)) > cuts[pTBin][0]) {
176  return false;
177  }
178  }
179 
180  if (trackPion.pt() < cuts[pTBin][4] || trackKaon.pt() < cuts[pTBin][3]) {
181  return false; //cut on daughter pT
182  }
183  if (TMath::Abs(trackPion.dcaPrim0()) > cuts[pTBin][6] || TMath::Abs(trackKaon.dcaPrim0()) > cuts[pTBin][5]) {
184  return false; //cut on daughter dca - need to add secondary vertex constraint here
185  }
186 
187  if (trackPion.charge() > 0) { //cut on cos(theta *)
188  if (TMath::Abs(CosThetaStarD0(hfCandProng2)) > cuts[pTBin][2]) {
189  return false;
190  }
191  } else {
192  if (TMath::Abs(CosThetaStarD0bar(hfCandProng2)) > cuts[pTBin][2]) {
193  return false;
194  }
195  }
196 
197  return true;
198  }

274  template <typename T>
275  int selectionPID(const T& track, int nPDG)
276  {
277  int statusTPC = -1;
278  int statusTOF = -1;
279 
280  if (validTPCPID(track)) {
281  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPC)) {
282  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPCCombined)) {
283  statusTPC = 0; //rejected by PID
284  } else {
285  statusTPC = 1; //potential to be acceepted if combined with TOF
286  }
287  } else {
288  statusTPC = 2; //positive PID
289  }
290  } else {
291  statusTPC = -1; //no PID info
292  }
293 
294  if (validTOFPID(track)) {
295  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOF)) {
296  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOFCombined)) {
297  statusTOF = 0; //rejected by PID
298  } else {
299  statusTOF = 1; //potential to be acceepted if combined with TOF
300  }
301  } else {
302  statusTOF = 2; //positive PID
303  }
304  } else {
305  statusTOF = -1; //no PID info
306  }
307 
308  if (statusTPC == 2 || statusTOF == 2) {
309  return 1; //what if we have 2 && 0 ?
310  } else if (statusTPC == 1 && statusTOF == 1) {
311  return 1;
312  } else if (statusTPC == 0 || statusTOF == 0) {
313  return 0;
314  } else {
315  return -1;
316  }

319  void process(aod::HfCandProng2 const& hfCandProng2s, aod::BigTracksPID const& tracks)
320  {
321  int statusD0, statusD0bar; // final selection flag : 0-rejected 1-accepted
322  bool topolD0, topolD0bar;
323  int pidD0, pidD0bar, pionPlus, pionMinus, kaonPlus, kaonMinus;
324 
325  for (auto& hfCandProng2 : hfCandProng2s) { //looping over 2 prong candidates
326 
327  statusD0 = 0;
328  statusD0bar = 0;
329 
330  if (!(hfCandProng2.hfflag() & 1 << D0ToPiK)) {
331  hfSelD0Candidate(statusD0, statusD0bar);
332  continue;
333  }
334 
335  auto trackPos = hfCandProng2.index0_as<aod::BigTracksPID>(); //positive daughter
336  auto trackNeg = hfCandProng2.index1_as<aod::BigTracksPID>(); //negative daughter
337 
338  topolD0 = true;
339  topolD0bar = true;
340  pidD0 = -1;
341  pidD0bar = -1;
342  pionPlus = -1;
343  pionMinus = -1;
344  kaonPlus = -1;
345  kaonMinus = -1;
346 
347  // daughter track validity selection
348  if (!daughterSelection(trackPos) || !daughterSelection(trackNeg)) {
349  hfSelD0Candidate(statusD0, statusD0bar);
350  continue;
351  }
352 
353  //implement filter bit 4 cut - should be done before this task at the track selection level
354  //need to add special cuts (additional cuts on decay length and d0 norm)
355 
356  //conjugate independent topological selection
357  if (!selectionTopol(hfCandProng2)) {
358  hfSelD0Candidate(statusD0, statusD0bar);
359  continue;
360  }
361 
362  //conjugate dependent toplogical selection for D0
363  topolD0 = selectionTopolConjugate(hfCandProng2, trackPos, trackNeg);
364  //conjugate dependent toplogical selection for D0bar
365  topolD0bar = selectionTopolConjugate(hfCandProng2, trackNeg, trackPos);
366 
367  if (!topolD0 && !topolD0bar) {
368  hfSelD0Candidate(statusD0, statusD0bar);
369  continue;
370  }
371 
372  pionPlus = selectionPID(trackPos, 211);
373  kaonMinus = selectionPID(trackNeg, 321);
374  pionMinus = selectionPID(trackNeg, 211);
375  kaonPlus = selectionPID(trackPos, 321);
376 
377  if (pionPlus == 0 || kaonMinus == 0 || pionMinus == 1 || kaonPlus == 1) {
378  pidD0 = 0; //exclude D0
379  }
380  if (pionPlus == 1 || kaonMinus == 1 || pionMinus == 0 || kaonPlus == 0) {
381  pidD0bar = 0; //exclude D0bar
382  }
383  if (pionPlus == 1 && kaonMinus == 1) {
384  pidD0 = 1; //accept D0
385  }
386  if (pionMinus == 1 && kaonPlus == 1) {
387  pidD0bar = 1; //accept D0bar
388  }
389 
390  if (pidD0 == 0 && pidD0bar == 0) {
391  hfSelD0Candidate(statusD0, statusD0bar);
392  continue;
393  }
394 
395  if ((pidD0 == -1 || pidD0 == 1) && topolD0) {
396  statusD0 = 1; //identified as D0
397  }
398  if ((pidD0bar == -1 || pidD0bar == 1) && topolD0bar) {
399  statusD0bar = 1; //identified as D0bar
400  }
401 
402  hfSelD0Candidate(statusD0, statusD0bar);
403  }
404  }
