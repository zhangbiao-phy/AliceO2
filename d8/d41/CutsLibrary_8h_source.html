162 AnalysisCut* o2::aod::dqcuts::GetAnalysisCut(const char* cutName)
163 {
164  //
165  // define here cuts which are likely to be used often
166  //
167  AnalysisCut* cut = new AnalysisCut(cutName, cutName);
168  std::string nameStr = cutName;
169 
170  if (!nameStr.compare("eventStandard")) {
171  cut->AddCut(VarManager::kVtxZ, -10.0, 10.0);
172  cut->AddCut(VarManager::kIsINT7, 0.5, 1.5);
173  return cut;
174  }
175 
176  if (!nameStr.compare("int7vtxZ5")) {
177  cut->AddCut(VarManager::kVtxZ, -5.0, 5.0);
178  cut->AddCut(VarManager::kIsINT7, 0.5, 1.5);
179  return cut;
180  }
181 
182  if (!nameStr.compare("jpsiStandardKine")) {
183  cut->AddCut(VarManager::kPt, 1.0, 1000.0);
184  cut->AddCut(VarManager::kEta, -0.9, 0.9);
185  return cut;
186  }
187 
188  if (!nameStr.compare("lmeeStandardKine")) {
189  cut->AddCut(VarManager::kPt, 0.2, 10.0);
190  cut->AddCut(VarManager::kEta, -0.8, 0.8);
191  return cut;
192  }
193 
194  if (!nameStr.compare("lmeeLowBKine")) {
195  cut->AddCut(VarManager::kPt, 0.075, 10.0);
196  cut->AddCut(VarManager::kEta, -0.8, 0.8);
197  return cut;
198  }
199 
200  if (!nameStr.compare("TightGlobalTrack")) {
201  cut->AddCut(VarManager::kIsSPDfirst, 0.5, 1.5);
202  cut->AddCut(VarManager::kIsITSrefit, 0.5, 1.5);
203  cut->AddCut(VarManager::kIsTPCrefit, 0.5, 1.5);
204  cut->AddCut(VarManager::kTPCchi2, 0.0, 4.0);
205  cut->AddCut(VarManager::kITSchi2, 0.0, 5.0);
206  cut->AddCut(VarManager::kTPCnclsCR, 80.0, 161.);
207  cut->AddCut(VarManager::kITSncls, 3.5, 7.5);
208  return cut;
209  }
210 
211  if (!nameStr.compare("electronStandardQuality")) {
212  cut->AddCut(VarManager::kIsSPDany, 0.5, 1.5);
213  cut->AddCut(VarManager::kIsITSrefit, 0.5, 1.5);
214  cut->AddCut(VarManager::kIsTPCrefit, 0.5, 1.5);
215  cut->AddCut(VarManager::kTPCchi2, 0.0, 4.0);
216  cut->AddCut(VarManager::kITSchi2, 0.1, 36.0);
217  cut->AddCut(VarManager::kTPCncls, 100.0, 161.);
218  return cut;
219  }
220 
221  if (!nameStr.compare("standardPrimaryTrack")) {
222  cut->AddCut(VarManager::kTrackDCAxy, -1.0, 1.0);
223  cut->AddCut(VarManager::kTrackDCAz, -3.0, 3.0);
224  return cut;
225  }
226 
227  TF1* cutLow1 = new TF1("cutLow1", "pol1", 0., 10.);
228  cutLow1->SetParameters(130., -40.0);
229  if (!nameStr.compare("electronPID1")) {
230  cut->AddCut(VarManager::kTPCsignal, 70., 100.);
231  cut->AddCut(VarManager::kTPCsignal, cutLow1, 100.0, false, VarManager::kPin, 0.5, 3.0);
232  return cut;
233  }
234 
235  if (!nameStr.compare("electronPID2")) {
236  cut->AddCut(VarManager::kTPCsignal, 73., 100.);
237  cut->AddCut(VarManager::kTPCsignal, cutLow1, 100.0, false, VarManager::kPin, 0.5, 3.0);
238  return cut;
239  }
240 
241  if (!nameStr.compare("tpc_pion_rejection")) {
242  TF1* f1maxPi = new TF1("f1maxPi", "[0]+[1]*x", 0, 10);
243  f1maxPi->SetParameters(85, -50);
244  cut->AddCut(VarManager::kTPCsignal, 70, f1maxPi, true, VarManager::kPin, 0.0, 0.4, false);
245  return cut;
246  }
247 
248  if (!nameStr.compare("tpc_kaon_rejection")) {
249  TF1* f1minKa = new TF1("f1minKa", "[0]+[1]*x", 0, 10);
250  f1minKa->SetParameters(220, -300);
251  TF1* f1maxKa = new TF1("f1maxKa", "[0]+[1]*x", 0, 10);
252  f1maxKa->SetParameters(182.5, -150);
253  cut->AddCut(VarManager::kTPCsignal, f1minKa, f1maxKa, true, VarManager::kPin, 0.4, 0.8, false);
254  return cut;
255  }
256 
257  if (!nameStr.compare("tpc_proton_rejection")) {
258  TF1* f1minPr = new TF1("f1minPr", "[0]+[1]*x", 0, 10);
259  f1minPr->SetParameters(170, -100);
260  TF1* f1maxPr = new TF1("f1maxPr", "[0]+[1]*x", 0, 10);
261  f1maxPr->SetParameters(175, -75);
262  cut->AddCut(VarManager::kTPCsignal, f1minPr, f1maxPr, true, VarManager::kPin, 0.8, 1.4, false);
263  return cut;
264  }
265 
266  if (!nameStr.compare("tpc_electron")) {
267  cut->AddCut(VarManager::kTPCsignal, 70, 90, false, VarManager::kPin, 0.0, 1e+10, false);
268  return cut;
269  }
270 
271  if (!nameStr.compare("tof_electron")) {
272  cut->AddCut(VarManager::kTOFbeta, 0.99, 1.01, false, VarManager::kPin, 0.0, 1e+10, false);
273  return cut;
274  }
275 
276  if (!nameStr.compare("muonQualityCuts")) {
277  cut->AddCut(VarManager::kEta, -4.0, -2.5);
278  cut->AddCut(VarManager::kMuonRAtAbsorberEnd, 200.0, 900.0);
279  cut->AddCut(VarManager::kMuonPDca, 0.0, 300.0);
280  return cut;
281  }
282 
283  if (!nameStr.compare("pairNoCut")) {
284  cut->AddCut(VarManager::kMass, 0.0, 1000.0);
285  return cut;
286  }
287 
288  if (!nameStr.compare("pairMassLow")) {
289  cut->AddCut(VarManager::kMass, 2.5, 1000.0);
290  return cut;
291  }
292 
293  if (!nameStr.compare("pairJpsi")) {
294  cut->AddCut(VarManager::kMass, 2.8, 3.3);
295  return cut;
296  }
297 
298  if (!nameStr.compare("pairPsi2S")) {
299  cut->AddCut(VarManager::kMass, 3.4, 3.9);
300  return cut;
301  }
302 
303  if (!nameStr.compare("pairUpsilon")) {
304  cut->AddCut(VarManager::kMass, 8.0, 11.0);
305  return cut;
306  }
307 
308  if (!nameStr.compare("pairPtLow1")) {
309  cut->AddCut(VarManager::kPt, 2.0, 1000.0);
310  return cut;
311  }
312 
313  if (!nameStr.compare("pairPtLow2")) {
314  cut->AddCut(VarManager::kPt, 5.0, 1000.0);
315  return cut;
316  }
317 
318  delete cut;
319  return nullptr;
320 }
