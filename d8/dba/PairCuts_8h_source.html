88 template <typename T>
89 bool PairCuts::conversionCuts(T const& track1, T const& track2)
90 {
91  // skip if like sign
92  if (track1.charge() * track2.charge() > 0) {
93  return false;
94  }
95 
96  for (int i = 0; i < static_cast<int>(ParticlesLastEntry); i++) {
97  Particle particle = static_cast<Particle>(i);
98  if (mCuts[i] > 0) {
99  if (conversionCut(track1, track2, particle, mCuts[i])) {
100  return true;
101  }
102  if (particle == Lambda) {
103  if (conversionCut(track2, track1, particle, mCuts[i])) {
104  return true;
105  }
106  }
107  }
108  }
109 
110  return false;
111 }

113 template <typename T>
114 bool PairCuts::twoTrackCut(T const& track1, T const& track2, int bSign)
115 {
116  // the variables & cut have been developed in Run 1 by the CF - HBT group
117  //
118  // Parameters:
119  // bSign: sign of B field
120 
121  auto deta = track1.eta() - track2.eta();
122 
123  // optimization
124  if (TMath::Abs(deta) < mTwoTrackDistance * 2.5 * 3) {
125  // check first boundaries to see if is worth to loop and find the minimum
126  float dphistar1 = getDPhiStar(track1, track2, mTwoTrackRadius, bSign);
127  float dphistar2 = getDPhiStar(track1, track2, 2.5, bSign);
128 
129  const float kLimit = mTwoTrackDistance * 3;
130 
131  if (TMath::Abs(dphistar1) < kLimit || TMath::Abs(dphistar2) < kLimit || dphistar1 * dphistar2 < 0) {
132  float dphistarminabs = 1e5;
133  float dphistarmin = 1e5;
134  for (Double_t rad = mTwoTrackRadius; rad < 2.51; rad += 0.01) {
135  float dphistar = getDPhiStar(track1, track2, rad, bSign);
136 
137  float dphistarabs = TMath::Abs(dphistar);
138 
139  if (dphistarabs < dphistarminabs) {
140  dphistarmin = dphistar;
141  dphistarminabs = dphistarabs;
142  }
143  }
144 
145  if (histogramRegistry != nullptr) {
146  histogramRegistry->fill(HIST("TwoTrackDistancePt_0"), deta, dphistarmin, TMath::Abs(track1.pt() - track2.pt()));
147  }
148 
149  if (dphistarminabs < mTwoTrackDistance && TMath::Abs(deta) < mTwoTrackDistance) {
150  //LOGF(debug, "Removed track pair %ld %ld with %f %f %f %f %d %f %f %d %d", track1.index(), track2.index(), deta, dphistarminabs, track1.phi2(), track1.pt(), track1.charge(), track2.phi2(), track2.pt(), track2.charge(), bSign);
151  return true;
152  }
153 
154  if (histogramRegistry != nullptr) {
155  histogramRegistry->fill(HIST("TwoTrackDistancePt_1"), deta, dphistarmin, TMath::Abs(track1.pt() - track2.pt()));
156  }
157  }
158  }
159 
160  return false;
161 }

163 template <typename T>
164 bool PairCuts::conversionCut(T const& track1, T const& track2, Particle conv, double cut)
165 {
166  //LOGF(info, "pt is %f %f", track1.pt(), track2.pt());
167 
168  if (cut < 0) {
169  return false;
170  }
171 
172  double massD1, massD2, massM;
173 
174  switch (conv) {
175  case Photon:
176  massD1 = 0.51e-3;
177  massD2 = 0.51e-3;
178  massM = 0;
179  break;
180  case K0:
181  massD1 = 0.1396;
182  massD2 = 0.1396;
183  massM = 0.4976;
184  break;
185  case Lambda:
186  massD1 = 0.9383;
187  massD2 = 0.1396;
188  massM = 1.115;
189  break;
190  case Phi:
191  massD1 = 0.4937;
192  massD2 = 0.4937;
193  massM = 1.019;
194  break;
195  case Rho:
196  massD1 = 0.1396;
197  massD2 = 0.1396;
198  massM = 0.770;
199  break;
200  default:
201  LOGF(fatal, "Particle now known");
202  return false;
203  break;
204  }
205 
206  auto massC = getInvMassSquaredFast(track1, massD1, track2, massD2);
207 
208  if (TMath::Abs(massC - massM * massM) > cut * 5) {
209  return false;
210  }
211 
212  massC = getInvMassSquared(track1, massD1, track2, massD2);
213 
214  if (histogramRegistry != nullptr) {
215  histogramRegistry->fill(HIST("ControlConvResonances"), static_cast<int>(conv), massC - massM * massM);
216  }
217 
218  if (massC > (massM - cut) * (massM - cut) && massC < (massM + cut) * (massM + cut)) {
219  return true;
220  }
221 
222  return false;
223 }

225 template <typename T>
226 double PairCuts::getInvMassSquared(T const& track1, double m0_1, T const& track2, double m0_2)
227 {
228  // calculate inv mass squared
229  // same can be achieved, but with more computing time with
230  /*TLorentzVector photon, p1, p2;
231  p1.SetPtEtaPhiM(triggerParticle->Pt(), triggerEta, triggerParticle->Phi(), 0.510e-3);
232  p2.SetPtEtaPhiM(particle->Pt(), eta[j], particle->Phi(), 0.510e-3);
233  photon = p1+p2;
234  photon.M()*/
235 
236  float tantheta1 = 1e10;
237 
238  if (track1.eta() < -1e-10 || track1.eta() > 1e-10) {
239  float expTmp = TMath::Exp(-track1.eta());
240  tantheta1 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
241  }
242 
243  float tantheta2 = 1e10;
244  if (track2.eta() < -1e-10 || track2.eta() > 1e-10) {
245  float expTmp = TMath::Exp(-track2.eta());
246  tantheta2 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
247  }
248 
249  float e1squ = m0_1 * m0_1 + track1.pt() * track1.pt() * (1.0 + 1.0 / tantheta1 / tantheta1);
250  float e2squ = m0_2 * m0_2 + track2.pt() * track2.pt() * (1.0 + 1.0 / tantheta2 / tantheta2);
251 
252  float mass2 = m0_1 * m0_1 + m0_2 * m0_2 + 2 * (TMath::Sqrt(e1squ * e2squ) - (track1.pt() * track2.pt() * (TMath::Cos(track1.phi() - track2.phi()) + 1.0 / tantheta1 / tantheta2)));
253 
254  // LOGF(debug, "%f %f %f %f %f %f %f %f %f", pt1, eta1, phi1, pt2, eta2, phi2, m0_1, m0_2, mass2);
255 
256  return mass2;
257 }

259 template <typename T>
260 double PairCuts::getInvMassSquaredFast(T const& track1, double m0_1, T const& track2, double m0_2)
261 {
262  // calculate inv mass squared approximately
263 
264  const float eta1 = track1.eta();
265  const float eta2 = track2.eta();
266  const float phi1 = track1.phi();
267  const float phi2 = track2.phi();
268  const float pt1 = track1.pt();
269  const float pt2 = track2.pt();
270 
271  float tantheta1 = 1e10;
272 
273  if (eta1 < -1e-10 || eta1 > 1e-10) {
274  float expTmp = 1.0 - eta1 + eta1 * eta1 / 2 - eta1 * eta1 * eta1 / 6 + eta1 * eta1 * eta1 * eta1 / 24;
275  tantheta1 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
276  }
277 
278  float tantheta2 = 1e10;
279  if (eta2 < -1e-10 || eta2 > 1e-10) {
280  float expTmp = 1.0 - eta2 + eta2 * eta2 / 2 - eta2 * eta2 * eta2 / 6 + eta2 * eta2 * eta2 * eta2 / 24;
281  tantheta2 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
282  }
283 
284  float e1squ = m0_1 * m0_1 + pt1 * pt1 * (1.0 + 1.0 / tantheta1 / tantheta1);
285  float e2squ = m0_2 * m0_2 + pt2 * pt2 * (1.0 + 1.0 / tantheta2 / tantheta2);
286 
287  // fold onto 0...pi
288  float deltaPhi = TMath::Abs(phi1 - phi2);
289  while (deltaPhi > M_PI * 2) {
290  deltaPhi -= M_PI * 2;
291  }
292  if (deltaPhi > M_PI) {
293  deltaPhi = M_PI * 2 - deltaPhi;
294  }
295 
296  float cosDeltaPhi = 0;
297  if (deltaPhi < M_PI / 3) {
298  cosDeltaPhi = 1.0 - deltaPhi * deltaPhi / 2 + deltaPhi * deltaPhi * deltaPhi * deltaPhi / 24;
299  } else if (deltaPhi < 2 * M_PI / 3) {
300  cosDeltaPhi = -(deltaPhi - M_PI / 2) + 1.0 / 6 * TMath::Power((deltaPhi - M_PI / 2), 3);
301  } else {
302  cosDeltaPhi = -1.0 + 1.0 / 2.0 * (deltaPhi - M_PI) * (deltaPhi - M_PI) - 1.0 / 24.0 * TMath::Power(deltaPhi - M_PI, 4);
303  }
304 
305  double mass2 = m0_1 * m0_1 + m0_2 * m0_2 + 2 * (TMath::Sqrt(e1squ * e2squ) - (pt1 * pt2 * (cosDeltaPhi + 1.0 / tantheta1 / tantheta2)));
306 
307  //LOGF(debug, "%f %f %f %f %f %f %f %f %f", pt1, eta1, phi1, pt2, eta2, phi2, m0_1, m0_2, mass2);
308 
309  return mass2;
310 }

312 template <typename T>
313 float PairCuts::getDPhiStar(T const& track1, T const& track2, float radius, float bSign)
314 {
315  //
316  // calculates dphistar
317  //
318 
319  auto phi1 = track1.phi();
320  auto pt1 = track1.pt();
321  auto charge1 = track1.charge();
322 
323  auto phi2 = track2.phi();
324  auto pt2 = track2.pt();
325  auto charge2 = track2.charge();
326 
327  float dphistar = phi1 - phi2 - charge1 * bSign * TMath::ASin(0.075 * radius / pt1) + charge2 * bSign * TMath::ASin(0.075 * radius / pt2);
328 
329  if (dphistar > M_PI) {
330  dphistar = M_PI * 2 - dphistar;
331  }
332  if (dphistar < -M_PI) {
333  dphistar = -M_PI * 2 - dphistar;
334  }
335  if (dphistar > M_PI) { // might look funny but is needed
336  dphistar = M_PI * 2 - dphistar;
337  }
338 
339  return dphistar;
340 }
