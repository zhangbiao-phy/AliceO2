28 void Encoder::init(const char* filename, bool perLink, int verbosity, bool debugMode)
29 {
31 
32  CrateMasks masks;
33  auto gbtIds = mFEEIdConfig.getConfiguredGBTIds();
34 
35  mRawWriter.setVerbosity(verbosity);
36  int lcnt = 0;
37  for (auto& gbtId : gbtIds) {
38  auto feeId = mFEEIdConfig.getFeeId(gbtId);
39  mRawWriter.registerLink(feeId, mFEEIdConfig.getCRUId(gbtId), raw::sUserLogicLinkID, mFEEIdConfig.getEndPointId(gbtId), perLink ? fmt::format("{:s}_L{:d}.raw", filename, lcnt) : fmt::format("{:s}.raw", filename));
40  mGBTEncoders[feeId].setFeeId(feeId);
41  mGBTEncoders[feeId].setMask(masks.getMask(feeId));
42  mGBTIds[feeId] = gbtId;
43  lcnt++;
44 
45  // Initializes the trigger response to be added to the empty HBs
46  auto ir = getOrbitIR(0);
47  mGBTEncoders[feeId].processTrigger(ir, raw::sORB);
48  mGBTEncoders[feeId].flush(mOrbitResponse[feeId], ir);
49  mOrbitResponseWord[feeId] = mOrbitResponse[feeId];
50  completeWord(mOrbitResponseWord[feeId]);
51  }
52 
53  mRawWriter.setEmptyPageCallBack(this);
54 
55  mConverter.setDebugMode(debugMode);

65 void Encoder::onOrbitChange(uint32_t orbit)
66 {
68  auto ir = getOrbitIR(orbit);
69  for (uint16_t feeId = 0; feeId < crateparams::sNGBTs; ++feeId) {
70  // Write the data corresponding to the previous orbit

77 void Encoder::completeWord(std::vector<char>& buffer)
78 {
80  size_t dataSize = buffer.size();
81  size_t cruWord = 2 * o2::raw::RDHUtils::GBTWord;
82  size_t modulo = dataSize % cruWord;
83  if (modulo) {
84  dataSize += cruWord - modulo;
85  buffer.resize(dataSize, static_cast<char>(0));
86  }
87 }

89 void Encoder::writePayload(uint16_t feeId, const InteractionRecord& ir)
90 {
92 
93  // The first part of data is the answer to the orbit trigger

95  // Then we flush the received data
96  mGBTEncoders[feeId].flush(buf, ir);
97  completeWord(buf);
98  mRawWriter.addData(feeId, mFEEIdConfig.getCRUId(mGBTIds[feeId]), raw::sUserLogicLinkID, mFEEIdConfig.getEndPointId(mGBTIds[feeId]), ir, buf);

101 void Encoder::finalize(bool closeFile)
102 {
104  if (mLastIR.isDummy()) {
105  mLastIR.bc = mRawWriter.getHBFUtils().bcFirst;
106  mLastIR.orbit = mRawWriter.getHBFUtils().orbitFirst;
107  }
108  auto ir = getOrbitIR(mLastIR.orbit);
109  for (uint16_t feeId = 0; feeId < crateparams::sNGBTs; ++feeId) {
110  // Write the last payload
111  writePayload(feeId, ir);
112  if (!mGBTEncoders[feeId].isEmpty()) {
113  // Since the regional response comes after few clocks,
114  // we might have the corresponding regional cards in the next orbit.
115  // If this is the case, we flush all data of the next orbit
116  ++ir.orbit;
117  writePayload(feeId, ir);
118  }
119  }
120  if (closeFile) {
121  mRawWriter.close();
122  }
123 }

126 {
128  if (ir.orbit != mLastIR.orbit) {

131 
132  mConverter.process(data);
133

136  }
137  mLastIR = ir;
