31 struct Header {
32  static constexpr uint32_t MagicWord = 0xFC;
33 
34  union {
35  uint64_t word0 = 0;
36  struct {
37  uint64_t bitMaskLow : 64;
38  };
39  };
40  union {
42  uint64_t word1 = 0;
43  struct {
44  uint64_t bitMaskHigh : 16;
45  uint32_t bunchCrossing : 12;
46  uint32_t numWordsPayload : 4;
47  uint32_t syncOffsetBC : 8;
48  uint32_t syncOffsetCRUCycles : 16;
49  uint32_t magicWord : 8;
50  };
51  };

53  std::bitset<80> getChannelBits() const
54  {
55  return std::bitset<80>((std::bitset<80>(bitMaskHigh) << 64) | std::bitset<80>(bitMaskLow));
56  }
57 
58  bool isFillWord() const { return (word0 == 0xffffffffffffffff) && (word1 == 0xffffffffffffffff); }

60  bool hasCorrectMagicWord() const { return magicWord == MagicWord; }

75 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2>
76 struct Data {
77  static constexpr uint32_t ChannelsPerWord = DataWordSizeBits / DataBitSizeT;
78  static constexpr uint32_t ChannelsPerHalfWord = ChannelsPerWord / 2;
79  static constexpr uint32_t DataBitSize = DataBitSizeT;
80  static constexpr uint32_t SignificantBits = SignificantBitsT;
81  static constexpr uint64_t BitMask = ((uint64_t(1) << DataBitSize) - 1);
82  static constexpr float FloatConversion = 1.f / float(1 << SignificantBits);
83 
84  uint64_t adcValues[2]{};
85 
87  void setADCValue(uint32_t pos, uint64_t value)
88  {
89  const uint32_t word = pos / ChannelsPerHalfWord;
90  const uint32_t posInWord = pos % ChannelsPerHalfWord;
91 
92  const uint64_t set = (value & BitMask) << (posInWord * DataBitSize);
93  const uint64_t mask = (0xFFFFFFFFFFFFFFFF ^ (BitMask << (posInWord * DataBitSize)));
94 
95  adcValues[word] &= mask;
96  adcValues[word] |= set;
97  }
98 
100  void setADCValueFloat(uint32_t pos, float value)
101  {
102  setADCValue(pos, uint64_t((value + 0.5f * FloatConversion) / FloatConversion));
103  }
104 
106  uint32_t getADCValue(uint32_t pos) const
107  {
108  const uint32_t word = pos / ChannelsPerHalfWord;
109  const uint32_t posInWord = pos % ChannelsPerHalfWord;
110 
111  return (adcValues[word] >> (posInWord * DataBitSize)) & BitMask;
112  }
113 
115  float getADCValueFloat(uint32_t pos) const
116  {
117  return float(getADCValue(pos)) * FloatConversion;
118  }
119 
121  void reset()
122  {
123  adcValues[0] = 0;
124  adcValues[1] = 0;
125  }
126 };

169 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2, bool HasHeaderT = true>
170 struct Container {
171  static constexpr uint32_t ChannelsPerWord = DataWordSizeBits / DataBitSizeT;
172 
173  ContainerT<DataBitSizeT, SignificantBitsT, HasHeaderT> cont;
174 
176  uint32_t getADCValue(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValue(word % ChannelsPerWord); }
177 
179  float getADCValueFloat(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValueFloat(word % ChannelsPerWord); }
180 
182  void setADCValue(uint32_t word, uint64_t value) { cont.data[word / ChannelsPerWord].setADCValue(word % ChannelsPerWord, value); }
183 
185  void setADCValueFloat(uint32_t word, float value) { cont.data[word / ChannelsPerWord].setADCValueFloat(word % ChannelsPerWord, value); }
186 
188  void reset()
189  {
190  for (int i = 0; i < cont.getDataWords(); ++i) {
191  cont.data[i].reset();
192  }
193  }
194 
195  uint32_t getBunchCrossing() const { return cont.getBunchCrossing(); }
196  uint32_t getDataWords() const { return cont.getDataWords(); }
197  uint32_t getTotalWords() const { return cont.getTotalWords(); }
198  std::bitset<80> getChannelBits() const { return cont.getChannelBits(); }
199 
201  size_t getTotalSizeBytes() const { return getTotalWords() * DataWordSizeBytes; }
202 
204  Container* next() const
205  {
206  return (Container*)((const char*)this + getTotalWords() * DataWordSizeBytes);
207  }
