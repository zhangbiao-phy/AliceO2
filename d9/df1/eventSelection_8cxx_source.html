70 struct EventSelectionTask {
71  Produces<aod::EvSels> evsel;
72  Service<o2::ccdb::BasicCCDBManager> ccdb;
73  Configurable<bool> isMC{"isMC", 0, "0 - data, 1 - MC"};

75  EvSelParameters par;

77  void init(InitContext&)
78  {
79  ccdb->setURL("http://ccdb-test.cern.ch:8080");
80  ccdb->setCaching(true);
81  ccdb->setLocalObjectValidityChecking();

84  void process(aod::Run2MatchedSparse::iterator const& collision, aod::BCsWithTimestamps const&, aod::Zdcs const& zdcs, aod::FV0As const& fv0as, aod::FV0Cs const& fv0cs, aod::FT0s const& ft0s, aod::FDDs const& fdds)
85  {
86  auto bc = collision.bc_as<aod::BCsWithTimestamps>();
87  LOGF(debug, "timestamp=%llu", bc.timestamp());
88  TriggerAliases* aliases = ccdb->getForTimeStamp<TriggerAliases>("Trigger/TriggerAliases", bc.timestamp());
89  if (!aliases) {
90  LOGF(fatal, "Trigger aliases are not available in CCDB for run=%d at timestamp=%llu", bc.runNumber(), bc.timestamp());
91  }
92  uint64_t triggerMask = bc.triggerMask();
93  LOGF(debug, "triggerMask=%llu", triggerMask);
94 
95  // fill fired aliases
96  int32_t alias[kNaliases] = {0};
97  for (auto& al : aliases->GetAliasToClassIdsMap()) {
98  for (auto& classIndex : al.second) {
99  alias[al.first] |= (triggerMask & (1ull << classIndex)) > 0;
100  }
101  }
102 
103  float timeZNA = collision.has_zdc() ? collision.zdc().timeZNA() : -999.f;
104  float timeZNC = collision.has_zdc() ? collision.zdc().timeZNC() : -999.f;
105  float timeV0A = collision.has_fv0a() ? collision.fv0a().time() : -999.f;
106  float timeV0C = collision.has_fv0c() ? collision.fv0c().time() : -999.f;
107  float timeT0A = collision.has_ft0() ? collision.ft0().timeA() : -999.f;
108  float timeT0C = collision.has_ft0() ? collision.ft0().timeC() : -999.f;
109  float timeFDA = collision.has_fdd() ? collision.fdd().timeA() : -999.f;
110  float timeFDC = collision.has_fdd() ? collision.fdd().timeC() : -999.f;
111 
112  LOGF(debug, "timeZNA=%f timeZNC=%f", timeZNA, timeZNC);
113  LOGF(debug, "timeV0A=%f timeV0C=%f", timeV0A, timeV0C);
114  LOGF(debug, "timeFDA=%f timeFDC=%f", timeFDA, timeFDC);
115  LOGF(debug, "timeT0A=%f timeT0C=%f", timeT0A, timeT0C);
116 
117  bool bbZNA = timeZNA > par.fZNABBlower && timeZNA < par.fZNABBupper;
118  bool bbZNC = timeZNC > par.fZNCBBlower && timeZNC < par.fZNCBBupper;
119  bool bbV0A = timeV0A > par.fV0ABBlower && timeV0A < par.fV0ABBupper;
120  bool bbV0C = timeV0C > par.fV0CBBlower && timeV0C < par.fV0CBBupper;
121  bool bgV0A = timeV0A > par.fV0ABGlower && timeV0A < par.fV0ABGupper;
122  bool bgV0C = timeV0C > par.fV0CBGlower && timeV0C < par.fV0CBGupper;
123  bool bbFDA = timeFDA > par.fFDABBlower && timeFDA < par.fFDABBupper;
124  bool bbFDC = timeFDC > par.fFDCBBlower && timeFDC < par.fFDCBBupper;
125  bool bgFDA = timeFDA > par.fFDABGlower && timeFDA < par.fFDABGupper;
126  bool bgFDC = timeFDC > par.fFDCBGlower && timeFDC < par.fFDCBGupper;
127  bool bbT0A = timeT0A > par.fT0ABBlower && timeT0A < par.fT0ABBupper;
128  bool bbT0C = timeT0C > par.fT0CBBlower && timeT0C < par.fT0CBBupper;
129 
130  if (isMC) {
131  bbZNA = 1;
132  bbZNC = 1;
133  }
134 
135  // Fill event selection columns
136  int64_t foundFT0 = -1; // this column is not used in run2 analysis
137  evsel(alias, bbT0A, bbT0C, bbV0A, bbV0C, bgV0A, bgV0C, bbZNA, bbZNC, bbFDA, bbFDC, bgFDA, bgFDC, foundFT0);
138  }
139 };

141 struct EventSelectionTaskRun3 {
142  Produces<aod::EvSels> evsel;
143 
144  EvSelParameters par;

146  void process(aod::Collision const& collision, soa::Join<aod::BCs, aod::Run3MatchedToBCSparse> const& bct0s, aod::Zdcs const& zdcs, aod::FV0As const& fv0as, aod::FT0s const& ft0s, aod::FDDs const& fdds)
147  {
148  int64_t ft0Dist;
149  int64_t foundFT0 = -1;
150  float timeA = -999.f;
151  float timeC = -999.f;
152 
153  auto bcIter = collision.bc_as<soa::Join<aod::BCs, aod::Run3MatchedToBCSparse>>();
154 
155  uint64_t apprBC = bcIter.globalBC();
156  uint64_t meanBC = apprBC - std::lround(collision.collisionTime() / o2::constants::lhc::LHCBunchSpacingNS);
157  int deltaBC = std::ceil(collision.collisionTimeRes() / o2::constants::lhc::LHCBunchSpacingNS * 4);
158 
159  int moveCount = 0;
160  while (bcIter != bct0s.end() && bcIter.globalBC() <= meanBC + deltaBC && bcIter.globalBC() >= meanBC - deltaBC) {
161  if (bcIter.has_ft0()) {
162  ft0Dist = bcIter.globalBC() - meanBC;
163  foundFT0 = bcIter.ft0().globalIndex();
164  break;
165  }
166  ++bcIter;
167  ++moveCount;
168  }
169 
170  bcIter.moveByIndex(-moveCount);
171  while (bcIter != bct0s.begin() && bcIter.globalBC() <= meanBC + deltaBC && bcIter.globalBC() >= meanBC - deltaBC) {
172  --bcIter;
173  if (bcIter.has_ft0() && (meanBC - bcIter.globalBC()) < ft0Dist) {
174  foundFT0 = bcIter.ft0().globalIndex();
175  break;
176  }
177  if ((meanBC - bcIter.globalBC()) >= ft0Dist) {
178  break;
179  }
180  }
181 
182  if (foundFT0 != -1) {
183  auto ft0 = ft0s.iteratorAt(foundFT0);
184  timeA = ft0.timeA();
185  timeC = ft0.timeC();
186  }
187 
188  bool bbZNA = 1;
189  bool bbZNC = 1;
190  bool bbV0A = 0;
191  bool bbV0C = 0;
192  bool bgV0A = 0;
193  bool bgV0C = 0;
194  bool bbFDA = 0;
195  bool bbFDC = 0;
196  bool bgFDA = 0;
197  bool bgFDC = 0;
198  bool bbT0A = timeA > par.fT0ABBlower && timeA < par.fT0ABBupper;
199  bool bbT0C = timeC > par.fT0CBBlower && timeC < par.fT0CBBupper;
200 
201  int32_t alias[kNaliases] = {0};
202  // Fill event selection columns
203  // saving FT0 row index (foundFT0) for further analysis
204  evsel(alias, bbT0A, bbT0C, bbV0A, bbV0C, bgV0A, bgV0C, bbZNA, bbZNC, bbFDA, bbFDC, bgFDA, bgFDC, foundFT0);
205  }
206 };
