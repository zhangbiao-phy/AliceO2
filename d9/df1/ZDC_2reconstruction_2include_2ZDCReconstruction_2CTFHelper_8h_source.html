56  template <typename I, typename D, typename T, int M = 1>
57  class _Iter
58  {
59  public:
60  using difference_type = int64_t;
61  using value_type = T;
62  using pointer = const T*;
63  using reference = const T&;
64  using iterator_category = std::random_access_iterator_tag;
65 
66  _Iter(const gsl::span<const D>& data, bool end = false) : mData(data), mIndex(end ? M * data.size() : 0){};
67  _Iter() = default;
68 
69  const I& operator++()
70  {
71  ++mIndex;
72  return (I&)(*this);
73  }
74 
75  const I& operator--()
76  {
77  mIndex--;
78  return (I&)(*this);
79  }
80 
81  difference_type operator-(const I& other) const { return mIndex - other.mIndex; }
82 
83  difference_type operator-(size_t idx) const { return mIndex - idx; }
84 
85  const I& operator-(size_t idx)
86  {
87  mIndex -= idx;
88  return (I&)(*this);
89  }
90 
91  bool operator!=(const I& other) const { return mIndex != other.mIndex; }
92  bool operator==(const I& other) const { return mIndex == other.mIndex; }
93  bool operator>(const I& other) const { return mIndex > other.mIndex; }
94  bool operator<(const I& other) const { return mIndex < other.mIndex; }
95 
96  protected:
97  gsl::span<const D> mData{};
98  size_t mIndex = 0;
99  };

102  // BC difference wrt previous if in the same orbit, otherwise the abs.value.
103  // For the very 1st entry return 0 (diff wrt 1st BC in the CTF header)
104  class Iter_bcIncTrig : public _Iter<Iter_bcIncTrig, BCData, uint16_t>
105  {
106  public:
107  using _Iter<Iter_bcIncTrig, BCData, uint16_t>::_Iter;
108  value_type operator*() const
109  {
110  if (mIndex) {
111  if (mData[mIndex].ir.orbit == mData[mIndex - 1].ir.orbit) {
112  return mData[mIndex].ir.bc - mData[mIndex - 1].ir.bc;
113  } else {
114  return mData[mIndex].ir.bc;
115  }
116  }
117  return 0;
118  }
119  };
