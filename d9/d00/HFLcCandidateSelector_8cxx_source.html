64  template <typename T>
65  int getpTBin(T candpT)
66  {
67  double pTBins[npTBins + 1] = {0, 1., 2., 3., 4., 5., 6., 8., 12., 24., 36.};
68  if (candpT < pTBins[0] || candpT >= pTBins[npTBins]) {
69  return -1;
70  }
71  for (int i = 0; i < npTBins; i++) {
72  if (candpT < pTBins[i + 1]) {
73  return i;
74  }
75  }
76  return -1;
77  }

131  template <typename T1, typename T2>
132  bool selectionTopolConjugate(const T1& hfCandProng3, const T2& trackProton, const T2& trackKaon, const T2& trackPion)
133  {
134 
135  auto candpT = hfCandProng3.pt();
136  int pTBin = getpTBin(candpT);
137  if (pTBin == -1) {
138  return false;
139  }
140 
141  if (trackProton.pt() < cuts[pTBin][1] || trackKaon.pt() < cuts[pTBin][2] || trackPion.pt() < cuts[pTBin][3]) {
142  return false; //cut on daughter pT
143  }
144 
145  if (trackProton.globalIndex() == hfCandProng3.index0Id()) {
146  if (TMath::Abs(InvMassLcpKpi(hfCandProng3) - RecoDecay::getMassPDG(4122)) > cuts[pTBin][0]) {
147  return false;
148  }
149  } else {
150  if (TMath::Abs(InvMassLcpiKp(hfCandProng3) - RecoDecay::getMassPDG(4122)) > cuts[pTBin][0]) {
151  return false;
152  }
153  }
154 
155  return true;
156  }

236  template <typename T>
237  int selectionPID(const T& track, int nPDG)
238  {
239  int statusTPC = -1;
240  int statusTOF = -1;
241 
242  if (validTPCPID(track)) {
243  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPC)) {
244  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPCCombined)) {
245  statusTPC = 0; //rejected by PID
246  } else {
247  statusTPC = 1; //potential to be acceepted if combined with TOF
248  }
249  } else {
250  statusTPC = 2; //positive PID
251  }
252  } else {
253  statusTPC = -1; //no PID info
254  }
255 
256  if (validTOFPID(track)) {
257  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOF)) {
258  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOFCombined)) {
259  statusTOF = 0; //rejected by PID
260  } else {
261  statusTOF = 1; //potential to be acceepted if combined with TOF
262  }
263  } else {
264  statusTOF = 2; //positive PID
265  }
266  } else {
267  statusTOF = -1; //no PID info
268  }
269 
270  if (statusTPC == 2 || statusTOF == 2) {
271  return 1; //what if we have 2 && 0 ?
272  } else if (statusTPC == 1 && statusTOF == 1) {
273  return 1;
274  } else if (statusTPC == 0 || statusTOF == 0) {
275  return 0;
276  } else {
277  return -1;

281  void process(aod::HfCandProng3 const& hfCandProng3s, aod::BigTracksPID const& tracks)
282  {
283  int statusLcpKpi, statusLcpiKp; // final selection flag : 0-rejected 1-accepted
284  int pidLcpKpi, pidLcpiKp;
285 
286  for (auto& hfCandProng3 : hfCandProng3s) { //looping over 3 prong candidates
287 
288  statusLcpKpi = 0;
289  statusLcpiKp = 0;
290 
291  if (!(hfCandProng3.hfflag() & 1 << LcToPKPi)) {
292  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
293  continue;
294  }
295 
296  auto trackPos1 = hfCandProng3.index0_as<aod::BigTracksPID>(); //positive daughter (negative for the antiparticles)
297  auto trackNeg1 = hfCandProng3.index1_as<aod::BigTracksPID>(); //negative daughter (positive for the antiparticles)
298  auto trackPos2 = hfCandProng3.index2_as<aod::BigTracksPID>(); //positive daughter (negative for the antiparticles)
299 
300  pidLcpKpi = -1;
301  pidLcpiKp = -1;
302 
303  // daughter track validity selection
304  if (!daughterSelection(trackPos1) || !daughterSelection(trackNeg1) || !daughterSelection(trackPos2)) {
305  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
306  continue;
307  }
308 
309  //implement filter bit 4 cut - should be done before this task at the track selection level
310 
311  //conjugate independent topological selection
312  if (!selectionTopol(hfCandProng3)) {
313  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
314  continue;
315  }
316 
317  //conjugate dependent toplogical selection for Lc
318 
319  bool topolLcpKpi = selectionTopolConjugate(hfCandProng3, trackPos1, trackNeg1, trackPos2);
320  bool topolLcpiKp = selectionTopolConjugate(hfCandProng3, trackPos2, trackNeg1, trackPos1);
321 
322  if (!topolLcpKpi && !topolLcpiKp) {
323  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
324  continue;
325  }
326 
327  if (!d_FilterPID) {
328  // PID non applied
329  pidLcpKpi = 1;
330  pidLcpiKp = 1;
331  } else {
332  int proton1 = selectionPID(trackPos1, kProton);
333  int proton2 = selectionPID(trackPos2, kProton);
334  int kaonMinus = selectionPID(trackNeg1, kKPlus);
335  int pionPlus1 = selectionPID(trackPos1, kPiPlus);
336  int pionPlus2 = selectionPID(trackPos2, kPiPlus);
337 
338  if (proton1 == 0 || kaonMinus == 0 || pionPlus2 == 0) {
339  pidLcpKpi = 0; //exclude LcpKpi
340  }
341  if (proton1 == 1 && kaonMinus == 1 && pionPlus2 == 1) {
342  pidLcpKpi = 1; //accept LcpKpi
343  }
344  if (proton2 == 0 || kaonMinus == 0 || pionPlus1 == 0) {
345  pidLcpiKp = 0; //exclude LcpiKp
346  }
347  if (proton2 == 1 && kaonMinus == 1 && pionPlus1 == 1) {
348  pidLcpiKp = 1; //accept LcpiKp
349  }
350  }
351 
352  if (pidLcpKpi == 0 && pidLcpiKp == 0) {
353  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
354  continue;
355  }
356 
357  if ((pidLcpKpi == -1 || pidLcpKpi == 1) && topolLcpKpi) {
358  statusLcpKpi = 1; //identified as LcpKpi
359  }
360  if ((pidLcpiKp == -1 || pidLcpiKp == 1) && topolLcpiKp) {
361  statusLcpiKp = 1; //identified as LcpiKp
362  }
363 
364  hfSelLcCandidate(statusLcpKpi, statusLcpiKp);
365  }
366  }
