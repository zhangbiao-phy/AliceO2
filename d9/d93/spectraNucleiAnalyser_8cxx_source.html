31 struct NucleiSpectraAnalyserTask {
32 
33  Configurable<float> yMin{"yMin", -0.8, "Maximum rapidity"};
34  Configurable<float> yMax{"yMax", 0.8, "Minimum rapidity"};
35  Configurable<float> yBeam{"yBeam", 0., "Beam rapidity"};
36 
37  Configurable<float> nSigmaCutLow{"nSigmaCutLow", -30.0, "Value of the Nsigma cut"};
38  Configurable<float> nSigmaCutHigh{"nSigmaCutHigh", +3., "Value of the Nsigma cut"};
39 
40  HistogramRegistry spectra{"spectra", {}, OutputObjHandlingPolicy::AnalysisObject, true, true};
41 
42  void init(o2::framework::InitContext&)
43  {
44  spectra.add("fCollZpos", "collision z position", HistType::kTH1F, {{600, -20., +20., "z position (cm)"}});
45  spectra.add("fHePt", "He pT", HistType::kTH1F, {{60, 0.0, 30., "#it{p}_{T} (GeV/#it{c})"}});
46  }
47  void process(aod::LFCollision const& collision, aod::LFNucleiTracks const& tracks)
48  {
49 
50  spectra.fill(HIST("fCollZpos"), collision.posZ());
51 
52  for (auto track : tracks) { // start loop over tracks
53  TLorentzVector cutVector{};
54  cutVector.SetPtEtaPhiM(track.pt() * 2.0, track.eta(), track.phi(), constants::physics::MassHelium3);
55  if (cutVector.Rapidity() < yMin + yBeam || cutVector.Rapidity() > yMax + yBeam) {
56  continue;
57  }
58  float nsigma = track.tpcNSigmaHe();
59  if (nsigma < nSigmaCutLow || nsigma > nSigmaCutHigh) {
60  continue;
61  }
62  spectra.fill(HIST("fHePt"), track.pt());
63  }
64  }
65 };
66 
67 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
68 {
69  return WorkflowSpec{
70  adaptAnalysisTask<NucleiSpectraAnalyserTask>(cfgc, "nucleispectra-task-skim-analyser")};
71 }
