49 template <typename coder_T, typename stream_T, typename source_T>

51 const stream_IT LiteralEncoder<coder_T, stream_T, source_T>::process(const stream_IT outputBegin, const stream_IT outputEnd, const source_IT inputBegin, const source_IT inputEnd, std::vector<source_T>& literals) const
52 {
53  using namespace internal;
54  using ransCoder = internal::Encoder<coder_T, stream_T>;
55  LOG(trace) << "start encoding";
56  RANSTimer t;
57  t.start();
58 
59  static_assert(std::is_same<typename std::iterator_traits<source_IT>::value_type, source_T>::value);
60  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
61 
62  if (inputBegin == inputEnd) {
63  LOG(warning) << "passed empty message to encoder, skip encoding";
64  return outputEnd;
65  }
66 
67  if (outputBegin == outputEnd) {
68  const std::string errorMessage("Unallocated encode buffer passed to encoder. Aborting");
69  LOG(error) << errorMessage;
70  throw std::runtime_error(errorMessage);
71  }
72 
73  ransCoder rans0, rans1;
74 
75  stream_IT outputIter = outputBegin;
76  source_IT inputIT = inputEnd;
77 
78  const auto inputBufferSize = std::distance(inputBegin, inputEnd);
79 
80  auto encode = [&literals, this](source_IT symbolIter, stream_IT outputIter, ransCoder& coder) {
81  const source_T symbol = *symbolIter;
82  const auto& encoderSymbol = (*this->mSymbolTable)[symbol];
83  if (this->mSymbolTable->isRareSymbol(symbol)) {
84  literals.push_back(symbol);
85  }
86  return std::tuple(symbolIter, coder.putSymbol(outputIter, encoderSymbol, this->mProbabilityBits));
87  };
88 
89  // odd number of bytes?
90  if (inputBufferSize & 1) {
91  std::tie(inputIT, outputIter) = encode(--inputIT, outputIter, rans0);
92  assert(outputIter < outputEnd);
93  }
94 
95  while (inputIT != inputBegin) { // NB: working in reverse!
96  std::tie(inputIT, outputIter) = encode(--inputIT, outputIter, rans1);
97  std::tie(inputIT, outputIter) = encode(--inputIT, outputIter, rans0);
98  assert(outputIter < outputEnd);
99  }
100  outputIter = rans1.flush(outputIter);
101  outputIter = rans0.flush(outputIter);
102  // first iterator past the range so that sizes, distances and iterators work correctly.
103  ++outputIter;
104 
105  assert(!(outputIter > outputEnd));
106 
107  // deal with overflow
108  if (outputIter > outputEnd) {
109  const std::string exceptionText = [&]() {
110  std::stringstream ss;
111  ss << __func__ << " detected overflow in encode buffer: allocated:" << std::distance(outputBegin, outputEnd) << ", used:" << std::distance(outputBegin, outputIter);
112  return ss.str();
113  }();
114 
115  LOG(error) << exceptionText;
116  throw std::runtime_error(exceptionText);
117  }
118 
119  t.stop();
120  LOG(debug1) << "Encoder::" << __func__ << " {ProcessedBytes: " << inputBufferSize * sizeof(source_T) << ","
121  << " inclusiveTimeMS: " << t.getDurationMS() << ","
122  << " BandwidthMiBPS: " << std::fixed << std::setprecision(2) << (inputBufferSize * sizeof(source_T) * 1.0) / (t.getDurationS() * 1.0 * (1 << 20)) << "}";
123 
124 // advanced diagnostics for debug builds
125 #if !defined(NDEBUG)
126 
127  const auto inputBufferSizeB = inputBufferSize * sizeof(source_T);
128  const auto outputBufferSizeB = std::distance(outputBegin, outputIter) * sizeof(stream_T);
129 
130  LOG(debug2) << "EncoderProperties: {"
131  << "sourceTypeB: " << sizeof(source_T) << ", "
132  << "streamTypeB: " << sizeof(stream_T) << ", "
133  << "coderTypeB: " << sizeof(coder_T) << ", "
134  << "probabilityBits: " << this->mProbabilityBits << ", "
135  << "inputBufferSizeB: " << inputBufferSizeB << ", "
136  << "outputBufferSizeB: " << outputBufferSizeB << ", "
137  << "compressionFactor: " << std::fixed << std::setprecision(2) << static_cast<double>(inputBufferSizeB) / static_cast<double>(outputBufferSizeB) << "}";
138 #endif
139 
140  LOG(trace) << "done encoding";
141 
142  return outputIter;
143 };
