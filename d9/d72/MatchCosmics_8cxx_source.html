38 //________________________________________________________
39 void MatchCosmics::process(const o2::globaltracking::RecoContainer& data)
40 {
41  updateTimeDependentParams();
42  mRecords.clear();
43  mWinners.clear();
44  mCosmicTracks.clear();
45  mCosmicTracksLbl.clear();
46 
47  createSeeds(data);
48  int ntr = mSeeds.size();
49 
50  // propagate to DCA to origin
51  const o2::math_utils::Point3D<float> v{0., 0., 0};
52  for (int i = 0; i < ntr; i++) {
53  auto& trc = mSeeds[i];
54  if (!o2::base::Propagator::Instance()->propagateToDCABxByBz(v, trc, mMatchParams->maxStep, mMatchParams->matCorr)) {
55  trc.matchID = Reject; // reject track
56  }
57  }
58 
59  // sort in time bracket lower edge
60  std::vector<int> sortID(ntr);
61  std::iota(sortID.begin(), sortID.end(), 0);
62  std::sort(sortID.begin(), sortID.end(), [this](int a, int b) { return mSeeds[a].tBracket.getMin() < mSeeds[b].tBracket.getMin(); });
63 
64  for (int i = 0; i < ntr; i++) {
65  for (int j = i + 1; j < ntr; j++) {
66  if (!checkPair(sortID[i], sortID[j])) {
67  break;
68  }
69  }
70  }
71 
72  selectWinners();
73  refitWinners(data);
74 }

77 void MatchCosmics::refitWinners(const o2::globaltracking::RecoContainer& data)
78 {
79  LOG(INFO) << "Refitting " << mWinners.size() << " winner matches";
80  int count = 0;
81  auto tpcTBinMUSInv = 1. / mTPCTBinMUS;
82  if (!mTPCTransform) { // eventually, should be updated at every TF?
83  mTPCTransform = o2::tpc::TPCFastTransformHelperO2::instance()->create(0);
84  }
85  const auto& tpcClusRefs = data.getTPCTracksClusterRefs();
86  const auto& tpcClusShMap = data.clusterShMapTPC;
87  std::unique_ptr<o2::gpu::GPUO2InterfaceRefit> tpcRefitter;
88  if (data.inputsTPCclusters) {
89  tpcRefitter = std::make_unique<o2::gpu::GPUO2InterfaceRefit>(&data.inputsTPCclusters->clusterIndex,
90  mTPCTransform.get(), mBz,
91  tpcClusRefs.data(), tpcClusShMap.data(),
92  nullptr, o2::base::Propagator::Instance());
93  }
94 
95  const auto& itsClusters = prepareITSClusters(data);
96  // RS FIXME: this is probably a temporary solution, since ITS tracking over boundaries will likely change the TrackITS format
97  std::vector<int> itsTracksROF;
98 
99  const auto& itsTracksROFRec = data.getITSTracksROFRecords<o2::itsmft::ROFRecord>();
100  itsTracksROF.resize(data.getITSTracks<o2::its::TrackITS>().size());
101  for (unsigned irf = 0, cnt = 0; irf < itsTracksROFRec.size(); irf++) {
102  int ntr = itsTracksROFRec[irf].getNEntries();
103  for (int itr = 0; itr < ntr; itr++) {
104  itsTracksROF[cnt++] = irf;
105  }
106  }
107 
108  auto refitITSTrack = [this, &data, &itsTracksROF, &itsClusters](o2::track::TrackParCov& trFit, GTrackID gidx, float& chi2, bool inward = false) {
109  const auto& itsTrOrig = data.getITSTrack<o2::its::TrackITS>(gidx);
110  int nclRefit = 0, ncl = itsTrOrig.getNumberOfClusters(), rof = itsTracksROF[gidx.getIndex()];
111  const auto& itsClustersROFRec = data.getITSClustersROFRecords<o2::itsmft::ROFRecord>();
112  const auto& itsTrackClusRefs = data.getITSTracksClusterRefs();
113  int clusIndOffs = itsClustersROFRec[rof].getFirstEntry(), clEntry = itsTrOrig.getFirstClusterEntry();
114  const auto propagator = o2::base::Propagator::Instance();
115  const auto geomITS = o2::its::GeometryTGeo::Instance();
116  int from = ncl - 1, to = -1, step = -1;
117  if (inward) {
118  from = 0;
119  to = ncl;
120  step = 1;
121  }
122  for (int icl = from; icl != to; icl += step) { // ITS clusters are referred in layer decreasing order
123  const auto& clus = itsClusters[clusIndOffs + itsTrackClusRefs[clEntry + icl]];
124  float alpha = geomITS->getSensorRefAlpha(clus.getSensorID()), x = clus.getX();
125  if (!trFit.rotate(alpha) || !propagator->propagateToX(trFit, x, propagator->getNominalBz(), this->mMatchParams->maxSnp, this->mMatchParams->maxStep, this->mMatchParams->matCorr)) {
126  break;
127  }
128  chi2 += trFit.getPredictedChi2(clus);
129  if (!trFit.update(clus)) {
130  break;
131  }
132  nclRefit++;
133  }
134  return nclRefit == ncl ? ncl : -1;
135  };
136 
137  for (auto winRID : mWinners) {
138  const auto& rec = mRecords[winRID];
139  int poolEntryID[2] = {rec.id0, rec.id1};
140  const o2::track::TrackParCov outerLegs[2] = {data.getTrackParamOut(mSeeds[rec.id0].origID), data.getTrackParamOut(mSeeds[rec.id1].origID)};
141  auto tmin = std::max(mSeeds[rec.id0].tBracket.getMin(), mSeeds[rec.id1].tBracket.getMin());
142  auto tmax = std::min(mSeeds[rec.id0].tBracket.getMax(), mSeeds[rec.id1].tBracket.getMax());
143  auto t0 = 0.5 * (tmin + tmax);
144  auto pnt0 = outerLegs[0].getXYZGlo(), pnt1 = outerLegs[1].getXYZGlo();
145  int btm = 0, top = 1;
146  // we fit topward from bottom
147  if (pnt0.Y() > pnt1.Y()) {
148  btm = 1;
149  top = 0;
150  }
151  LOG(DEBUG) << "Winner " << count++ << " Record " << winRID << " Partners:"
152  << " B: " << mSeeds[poolEntryID[btm]].origID << "/" << mSeeds[poolEntryID[btm]].origID.getSourceName()
153  << " U: " << mSeeds[poolEntryID[top]].origID << "/" << mSeeds[poolEntryID[top]].origID.getSourceName()
154  << " | T[" << tmin << ":" << tmax << "]";
155 
156  float chi2 = 0;
157  int nclTot = 0;
158 
159  // Start from bottom leg inward refit
160  o2::track::TrackParCov trCosm(mSeeds[poolEntryID[btm]]); // copy of the btm track
161  // The bottom leg needs refit only if it is an unconstrained TPC track, otherwise it is already refitted as inner param
162  if (mSeeds[poolEntryID[btm]].origID.getSource() == GTrackID::TPC) {
163  const auto& tpcTrOrig = data.getTPCTrack<o2::tpc::TrackTPC>(mSeeds[poolEntryID[btm]].origID);
164  trCosm = outerLegs[btm];
165  int retVal = tpcRefitter->RefitTrackAsTrackParCov(trCosm, tpcTrOrig.getClusterRef(), t0 * tpcTBinMUSInv, &chi2, false, true); // inward refit, reset
166  if (retVal < 0) { // refit failed
167  LOG(DEBUG) << "Inward refit of btm TPC track failed.";
168  continue;
169  }
170  nclTot += retVal;
171  LOG(DEBUG) << "chi2 after btm TPC refit with " << retVal << " clusters : " << chi2 << " orig.chi2 was " << tpcTrOrig.getChi2();
172  } else { // just collect NClusters and chi2
173  auto gidxListBtm = data.getSingleDetectorRefs(mSeeds[poolEntryID[btm]].origID);
174  if (gidxListBtm[GTrackID::TPC].isIndexSet()) {
175  const auto& tpcTrOrig = data.getTPCTrack<o2::its::TrackITS>(gidxListBtm[GTrackID::TPC]);
176  nclTot += tpcTrOrig.getNClusters();
177  chi2 += tpcTrOrig.getChi2();
178  }
179  if (gidxListBtm[GTrackID::ITS].isIndexSet()) {
180  const auto& itsTrOrig = data.getITSTrack<o2::its::TrackITS>(gidxListBtm[GTrackID::ITS]);
181  nclTot += itsTrOrig.getNClusters();
182  chi2 += itsTrOrig.getChi2();
183  }
184  }
185  trCosm.invert();
186  if (!trCosm.rotate(mSeeds[poolEntryID[top]].getAlpha()) ||
187  !o2::base::Propagator::Instance()->PropagateToXBxByBz(trCosm, mSeeds[poolEntryID[top]].getX(), mMatchParams->maxSnp, mMatchParams->maxStep, mMatchParams->matCorr)) {
188  LOG(DEBUG) << "Rotation/propagation of btm-track to top-track frame failed.";
189  continue;
190  }
191  // save bottom parameter at merging point
192  auto trCosmBtm = trCosm;
193  int nclBtm = nclTot;
194 
195  // Continue with top leg outward refit
196  auto gidxListTop = data.getSingleDetectorRefs(mSeeds[poolEntryID[top]].origID);
197 
198  // is there ITS sub-track?
199  if (gidxListTop[GTrackID::ITS].isIndexSet()) {
200  auto nclfit = refitITSTrack(trCosm, gidxListTop[GTrackID::ITS], chi2, false);
201  if (nclfit < 0) {
202  continue;
203  }
204  LOG(DEBUG) << "chi2 after top ITS refit with " << nclfit << " clusters : " << chi2 << " orig.chi2 was " << data.getITSTrack<o2::its::TrackITS>(gidxListTop[GTrackID::ITS]).getChi2();
205  nclTot += nclfit;
206  } // ITS refit
207  //
208  if (gidxListTop[GTrackID::TPC].isIndexSet()) { // outward refit in TPC
209  float xtogo = 0;
210  if (!trCosm.getXatLabR(o2::globaltracking::MatchTPCITS::XTPCInnerRef, xtogo, mBz, o2::track::DirOutward) ||
211  !o2::base::Propagator::Instance()->PropagateToXBxByBz(trCosm, xtogo, mMatchParams->maxSnp, mMatchParams->maxStep, mMatchParams->matCorr)) {
212  LOG(DEBUG) << "Propagation to inner TPC boundary X=" << xtogo << " failed";
213  continue;
214  }
215  const auto& tpcTrOrig = data.getTPCTrack<o2::tpc::TrackTPC>(gidxListTop[GTrackID::TPC]);
216  int retVal = tpcRefitter->RefitTrackAsTrackParCov(trCosm, tpcTrOrig.getClusterRef(), t0 * tpcTBinMUSInv, &chi2, true, false); // outward refit, no reset
217  if (retVal < 0) { // refit failed
218  LOG(DEBUG) << "Outward refit of top TPC track failed.";
219  continue;
220  } // outward refit in TPC
221  LOG(DEBUG) << "chi2 after top TPC refit with " << retVal << " clusters : " << chi2 << " orig.chi2 was " << tpcTrOrig.getChi2();
222  nclTot += retVal;
223  }
224 
225  // inward refit of top leg for evaluation in DCA
226  float chi2Dummy = 0;
227  auto trCosmTop = outerLegs[top];
228  if (gidxListTop[GTrackID::TPC].isIndexSet()) { // inward refit in TPC
229  const auto& tpcTrOrig = data.getTPCTrack<o2::tpc::TrackTPC>(gidxListTop[GTrackID::TPC]);
230  int retVal = tpcRefitter->RefitTrackAsTrackParCov(trCosmTop, tpcTrOrig.getClusterRef(), t0 * tpcTBinMUSInv, &chi2Dummy, false, true); // inward refit, reset
231  if (retVal < 0) { // refit failed
232  LOG(DEBUG) << "Outward refit of top TPC track failed.";
233  continue;
234  } // inward refit in TPC
235  }
236  // is there ITS sub-track ?
237  if (gidxListTop[GTrackID::ITS].isIndexSet()) {
238  auto nclfit = refitITSTrack(trCosmTop, gidxListTop[GTrackID::ITS], chi2Dummy, true);
239  if (nclfit < 0) {
240  continue;
241  }
242  nclTot += nclfit;
243  } // ITS refit
244  // propagate to bottom param
245  if (!trCosmTop.rotate(trCosmBtm.getAlpha()) ||
246  !o2::base::Propagator::Instance()->PropagateToXBxByBz(trCosmTop, trCosmBtm.getX(), mMatchParams->maxSnp, mMatchParams->maxStep, mMatchParams->matCorr)) {
247  LOG(DEBUG) << "Rotation/propagation of top-track to bottom-track frame failed.";
248  continue;
249  }
250  // calculate weighted average of 2 legs and chi2
251  o2::track::TrackParCov::MatrixDSym5 cov5;
252  float chi2Match = trCosmBtm.getPredictedChi2(trCosmTop, cov5);
253  if (!trCosmBtm.update(trCosmTop, cov5)) {
254  LOG(DEBUG) << "Top/Bottom update failed";
255  continue;
256  }
257  // create final track
258  mCosmicTracks.emplace_back(mSeeds[poolEntryID[btm]].origID, mSeeds[poolEntryID[top]].origID, trCosmBtm, trCosmTop, chi2, chi2Match, nclTot);
259  if (mUseMC) {
260  o2::MCCompLabel lbl[2] = {data.getTrackMCLabel(mSeeds[poolEntryID[btm]].origID), data.getTrackMCLabel(mSeeds[poolEntryID[top]].origID)};
261  auto& tlb = mCosmicTracksLbl.emplace_back((nclBtm > nclTot - nclBtm ? lbl[0] : lbl[1]));
262  tlb.setFakeFlag(lbl[0] != lbl[1]);
263  }
264  }
265  LOG(INFO) << "Validated " << mCosmicTracks.size() << " top-bottom tracks";
266 }

268 //________________________________________________________
269 void MatchCosmics::selectWinners()
270 {
271  // select mutually best matches
272  int ntr = mSeeds.size(), iter = 0, nValidated = 0;
273  mWinners.reserve(mRecords.size() / 2); // there are 2 records per match candidate
274  do {
275  nValidated = 0;
276  int nRemaining = 0;
277  for (int i = 0; i < ntr; i++) {
278  if (mSeeds[i].matchID < 0 || mRecords[mSeeds[i].matchID].next == Validated) { // either have no match or already validated
279  continue;
280  }
281  nRemaining++;
282  if (validateMatch(i)) {
283  mWinners.push_back(mSeeds[i].matchID);
284  nValidated++;
285  continue;
286  }
287  }
288  LOGF(INFO, "iter %d Validated %d of %d remaining matches", iter, nValidated, nRemaining);
289  iter++;
290  } while (nValidated);
291 }

293 //________________________________________________________
294 bool MatchCosmics::validateMatch(int partner0)
295 {
296  // make sure that the best partner of seed_i has also seed_i as a best partner
297  auto& matchRec = mRecords[mSeeds[partner0].matchID];
298  auto partner1 = matchRec.id1;
299  auto& patnerRec = mRecords[mSeeds[partner1].matchID];
300  if (patnerRec.next == Validated) { // partner1 was already validated with other partner0
301  return false;
302  }
303  if (patnerRec.id1 == partner0) { // mutually best
304  // unlink winner partner0 from all other mathes
305  auto next0 = matchRec.next;
306  while (next0 > MinusOne) {
307  auto& nextRec = mRecords[next0];
308  suppressMatch(partner0, nextRec.id1);
309  next0 = nextRec.next;
310  }
311  matchRec.next = Validated;
312 
313  // unlink winner partner1 from all other matches
314  auto next1 = patnerRec.next;
315  while (next1 > MinusOne) {
316  auto& nextRec = mRecords[next1];
317  suppressMatch(partner1, nextRec.id1);
318  next1 = nextRec.next;
319  }
320  patnerRec.next = Validated;
321  return true;
322  }
323  return false;
324 }

326 //________________________________________________________
327 void MatchCosmics::suppressMatch(int partner0, int partner1)
328 {
329  // suppress reference to partner0 from partner1 match record
330  if (mSeeds[partner1].matchID < 0 || mRecords[mSeeds[partner1].matchID].next == Validated) {
331  LOG(WARNING) << "Attempt to remove null or validated partner match " << mSeeds[partner1].matchID;
332  return;
333  }
334  int topID = MinusOne, next = mSeeds[partner1].matchID;
335  while (next > MinusOne) {
336  auto& matchRec = mRecords[next];
337  if (matchRec.id1 == partner0) {
338  if (topID < 0) { // best match
339  mSeeds[partner1].matchID = matchRec.next; // exclude best match link
340  } else { // not the 1st link in the chain
341  mRecords[topID].next = matchRec.next;
342  }
343  return;
344  }
345  topID = next;
346  next = matchRec.next;
347  }
348 }

350 //________________________________________________________
351 bool MatchCosmics::checkPair(int i, int j)
352 {
353  // if validated with given chi2, register match
354  auto& seed0 = mSeeds[i];
355  auto& seed1 = mSeeds[j];
356  if (seed0.matchID == Reject) {
357  return false;
358  }
359  if (seed1.matchID == Reject) {
360  return true;
361  }
362 
363  LOG(DEBUG) << "Seed " << i << " [" << seed0.tBracket.getMin() << " : " << seed0.tBracket.getMax() << "] | "
364  << "Seed " << j << " [" << seed1.tBracket.getMin() << " : " << seed1.tBracket.getMax() << "] | ";
365  LOG(DEBUG) << seed0.origID << " | " << seed0.o2::track::TrackPar::asString();
366  LOG(DEBUG) << seed1.origID << " | " << seed1.o2::track::TrackPar::asString();
367 
368  if (seed1.tBracket > seed0.tBracket) {
369  return false; // since the brackets are sorted in tmin, all following tbj will also exceed tbi
370  }
371  float chi2 = 0.f;
372  // check
373  // 1) crude check on tgl and q/pt (if B!=0). Note: back-to-back tracks will have mutually params (see TrackPar::invertParam)
374  while (1) {
375  auto dTgl = seed0.getTgl() + seed1.getTgl();
376  if (dTgl * dTgl > (mMatchParams->systSigma2[o2::track::kTgl] + seed0.getSigmaTgl2() + seed1.getSigmaTgl2()) * mMatchParams->crudeNSigma2Cut[o2::track::kTgl]) {
377  break;
378  }
379  if (mFieldON) {
380  auto dQ2Pt = seed0.getQ2Pt() + seed1.getQ2Pt();
381  if (dQ2Pt * dQ2Pt > (mMatchParams->systSigma2[o2::track::kQ2Pt] + seed0.getSigma1Pt2() + seed1.getSigma1Pt2()) * mMatchParams->crudeNSigma2Cut[o2::track::kQ2Pt]) {
382  break;
383  }
384  }
385  o2::track::TrackParCov seed1Inv = seed1;
386  seed1Inv.invert();
387  for (int i = 0; i < o2::track::kNParams; i++) { // add systematic error
388  seed1Inv.updateCov(mMatchParams->systSigma2[i], o2::track::DiagMap[i]);
389  }
390 
391  if (!seed1Inv.rotate(seed0.getAlpha()) ||
392  !o2::base::Propagator::Instance()->PropagateToXBxByBz(seed1Inv, seed0.getX(), mMatchParams->maxSnp, mMatchParams->maxStep, mMatchParams->matCorr)) {
393  break;
394  }
395  auto dSnp = seed0.getSnp() - seed1Inv.getSnp();
396  if (dSnp * dSnp > (seed0.getSigmaSnp2() + seed1Inv.getSigmaSnp2()) * mMatchParams->crudeNSigma2Cut[o2::track::kSnp]) {
397  break;
398  }
399  auto dY = seed0.getY() - seed1Inv.getY();
400  if (dY * dY > (seed0.getSigmaY2() + seed1Inv.getSigmaY2()) * mMatchParams->crudeNSigma2Cut[o2::track::kY]) {
401  break;
402  }
403  bool ignoreZ = seed0.origID.getSource() == o2d::GlobalTrackID::TPC || seed1.origID.getSource() == o2d::GlobalTrackID::TPC;
404  if (!ignoreZ) { // cut on Z makes no sense for TPC only tracks
405  auto dZ = seed0.getZ() - seed1Inv.getZ();
406  if (dZ * dZ > (seed0.getSigmaZ2() + seed1Inv.getSigmaZ2()) * mMatchParams->crudeNSigma2Cut[o2::track::kZ]) {
407  break;
408  }
409  } else { // inflate Z error
410  seed1Inv.setCov(250. * 250., o2::track::DiagMap[o2::track::kZ]);
411  seed1Inv.setCov(0., o2::track::CovarMap[o2::track::kZ][o2::track::kY]); // set all correlation terms for Z error to 0
412  seed1Inv.setCov(0., o2::track::CovarMap[o2::track::kZ][o2::track::kSnp]);
413  seed1Inv.setCov(0., o2::track::CovarMap[o2::track::kZ][o2::track::kTgl]);
414  seed1Inv.setCov(0., o2::track::CovarMap[o2::track::kZ][o2::track::kQ2Pt]);
415  }
416  // calculate chi2 (expensive)
417  chi2 = seed0.getPredictedChi2(seed1Inv);
418  if (chi2 > mMatchParams->crudeChi2Cut) {
419  break;
420  }
421 
422  registerMatch(i, j, chi2);
423  registerMatch(j, i, chi2); // the reverse reference can be also done in a separate loop
424  LOG(DEBUG) << "Chi2 = " << chi2 << " NMatches " << mRecords.size();
425  break;
426  }
427  return true;
428 }

430 //________________________________________________________
431 void MatchCosmics::registerMatch(int i, int j, float chi2)
432 {
434  int newRef = mRecords.size();
435  auto& matchRec = mRecords.emplace_back(MatchRecord{i, j, chi2, MinusOne});
436  auto* best = &mSeeds[i].matchID;
437  while (*best > MinusOne) {
438  auto& oldMatchRec = mRecords[*best];
439  if (oldMatchRec.chi2 > chi2) { // insert new match in front of the old one
440  matchRec.next = *best; // new record will refer to the one it is superseding
441  *best = newRef; // the reference on the superseded record should now refer to new one
442  break;
443  }
444  best = &oldMatchRec.next;
445  }
446  if (matchRec.next == MinusOne) { // did not supersed any other record
447  *best = newRef;
448  }
449 }

451 //________________________________________________________
452 void MatchCosmics::createSeeds(const o2::globaltracking::RecoContainer& data)
453 {
454  // Scan all inputs and create seeding tracks
455 
456  mSeeds.clear();
457 
458  std::function<void(const o2::track::TrackParCov& _tr, float t0, float terr, GTrackID _origID)> creator =
459  [this](const o2::track::TrackParCov& _tr, float t0, float terr, GTrackID _origID) {
460  if (std::abs(_tr.getQ2Pt()) > this->mQ2PtCutoff) {
461  return;
462  }
463  if (_origID.getSource() == GTrackID::TPC) { // convert TPC bins to \mus
464  t0 *= this->mTPCTBinMUS;
