174  void process(MyEvents::iterator const& event, MyBarrelTracks const& tracks, aod::BCs const& bcs)
175  {
176  uint8_t filterMap = uint8_t(0);
177  trackSel.reserve(tracks.size());
178 
179  VarManager::ResetValues(0, VarManager::kNBarrelTrackVariables, fValues);
180  // fill event information which might be needed in histograms that combine track and event properties
181  VarManager::FillEvent<gkEventFillMap>(event, fValues);
182 
183  for (auto& track : tracks) {
184  filterMap = uint8_t(0);
185  // TODO: if a condition on the event selection is applied, the histogram output is missing
186  VarManager::FillTrack<gkTrackFillMap>(track, fValues);
187  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", fValues);
188  int i = 0;
189  for (auto cut = fTrackCuts.begin(); cut != fTrackCuts.end(); ++cut, ++i) {
190  if ((*cut).IsSelected(fValues)) {
191  filterMap |= (uint8_t(1) << i);
192  fHistMan->FillHistClass(Form("TrackBarrel_%s", (*cut).GetName()), fValues);
193  }
194  }
195  trackSel(filterMap);
196  }
197  }
198 };

270  void process(MyEventsSelected::iterator const& event, MyBarrelTracksSelected const& tracks, aod::BCs const& bcs)
271  {
272  uint64_t filter = 0;
273 
274  if (event.isDQEventSelected() == 1) {
275  // Reset the fValues array
276  VarManager::ResetValues(0, VarManager::kNVars, fValues);
277  VarManager::FillEvent<gkEventFillMap>(event, fValues);
278 
279  uint8_t* pairsCount = new uint8_t[fNPairCuts * fNTrackCuts];
280  for (int i = 0; i < fNPairCuts * fNTrackCuts; i++) {
281  pairsCount[i] = 0;
282  }
283 
284  uint8_t cutFilter = 0;
285  for (auto tpos : posTracks) {
286  for (auto tneg : negTracks) { // +- pairs
287  cutFilter = tpos.isBarrelSelected() & tneg.isBarrelSelected();
288  if (!cutFilter) { // the tracks must have at least one filter bit in common to continue
289  continue;
290  }
291  VarManager::FillPair(tpos, tneg, fValues); // compute pair quantities
292  for (int i = 0; i < fNTrackCuts; ++i) {
293  if (!(cutFilter & (uint8_t(1) << i))) {
294  continue;
295  }
296  fHistMan->FillHistClass(Form("PairsBarrelPM_%s", fTrkCutsNameArray->At(i)->GetName()), fValues);
297  int j = 0;
298  for (auto cut = fPairCuts.begin(); cut != fPairCuts.end(); cut++, j++) {
299  if ((*cut).IsSelected(fValues)) {
300  pairsCount[j + i * fNPairCuts] += 1;
301  }
302  }
303  }
304  }
305  }
306 
307  for (int i = 0; i < fNTrackCuts; i++) {
308  for (int j = 0; j < fNPairCuts; j++) {
309  if (pairsCount[j + i * fNPairCuts] > 0) {
310  filter |= (uint64_t(1) << (j + i * fNPairCuts));
311  fStats->Fill(j, i);
312  }
313  }
314  }
315  delete[] pairsCount;
316  }
317  eventFilter(filter);
318  }
319 };

329 void DefineHistograms(HistogramManager* histMan, TString histClasses)
330 {
331  //
332  // Define here the histograms for all the classes required in analysis.
333  // The histogram classes are provided in the histClasses string, separated by semicolon ";"
334  // NOTE: Additional histograms to those predefined in the library can be added here !!
335  const int kNRuns = 25;
336  int runs[kNRuns] = {
337  285009, 285011, 285012, 285013, 285014, 285015, 285064, 285065, 285066, 285106,
338  285108, 285125, 285127, 285165, 285200, 285202, 285203, 285222, 285224, 285327,
339  285328, 285347, 285364, 285365, 285396};
340  VarManager::SetRunNumbers(kNRuns, runs);
341 
342  std::unique_ptr<TObjArray> objArray(histClasses.Tokenize(";"));
343  for (Int_t iclass = 0; iclass < objArray->GetEntries(); ++iclass) {
344  TString classStr = objArray->At(iclass)->GetName();
345  histMan->AddHistClass(classStr.Data());
346 
347  if (classStr.Contains("Event")) {
348  dqhistograms::DefineHistograms(histMan, objArray->At(iclass)->GetName(), "event", "trigger,vtxPbPb");
349  }
350 
351  if (classStr.Contains("Track")) {
352  dqhistograms::DefineHistograms(histMan, objArray->At(iclass)->GetName(), "track", "its,tpcpid,dca");
353  }
354 
355  if (classStr.Contains("Pairs")) {
356  dqhistograms::DefineHistograms(histMan, objArray->At(iclass)->GetName(), "pair");
357  }
358  }
