39 DetID::mask_t DetID::getMask(const std::string_view detList)
40 {
41  DetID::mask_t mask;
42  std::string ss(detList), sname{};
43  if (ss.find(NONE) != std::string::npos) {
44  return mask;
45  }
46  if (ss.find(ALL) != std::string::npos) {
47  mask.set();
48  return mask;
49  }
50  std::replace(ss.begin(), ss.end(), ' ', ',');
51  std::stringstream sss(ss);
52  while (getline(sss, sname, ',')) {
53  for (auto id = DetID::First; id <= DetID::Last; id++) {
54  if (sname == getName(id)) {
55  mask.set(id);
56  sname = "";
57  break;
58  }
59  }
60  if (!sname.empty()) {
61  throw std::runtime_error(fmt::format("Wrong entry {:s} in detectors list {:s}", sname, detList));
62  }
63  }
64  return mask;
65 }
