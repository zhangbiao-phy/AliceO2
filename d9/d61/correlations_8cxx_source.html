139  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels, aod::Cents>>::iterator const& collision, aod::BCsWithTimestamps const&, myTracks const& tracks)
140  {
141  auto bc = collision.bc_as<aod::BCsWithTimestamps>();
142  if (cfgEfficiencyTrigger.value.empty() == false) {
143  cfg.mEfficiencyTrigger = ccdb->getForTimeStamp<THnT<float>>(cfgEfficiencyTrigger, bc.timestamp());
144  LOGF(info, "Loaded efficiency histogram for trigger particles from %s (%p)", cfgEfficiencyTrigger.value.c_str(), (void*)cfg.mEfficiencyTrigger);
145  }
146  if (cfgEfficiencyAssociated.value.empty() == false) {
147  cfg.mEfficiencyAssociated = ccdb->getForTimeStamp<THnT<float>>(cfgEfficiencyAssociated, bc.timestamp());
148  LOGF(info, "Loaded efficiency histogram for associated particles from %s (%p)", cfgEfficiencyAssociated.value.c_str(), (void*)cfg.mEfficiencyAssociated);
149  }
150 
151  LOGF(info, "Tracks for collision: %d | Vertex: %.1f | INT7: %d | V0M: %.1f", tracks.size(), collision.posZ(), collision.sel7(), collision.centV0M());
152 
153  const auto centrality = collision.centV0M();
154 
155  same->fillEvent(centrality, CorrelationContainer::kCFStepAll);
156 
157  if (!collision.alias()[kINT7]) {
158  return;
159  }
160  if (!collision.sel7()) {
161  return;
162  }
163 
164  same->fillEvent(centrality, CorrelationContainer::kCFStepTriggered);
165 
166  // vertex already checked as filter
167  same->fillEvent(centrality, CorrelationContainer::kCFStepVertex);
168 
169  same->fillEvent(centrality, CorrelationContainer::kCFStepReconstructed);
170 
171  int bSign = 1; // TODO magnetic field from CCDB
172 
173  // Cache efficiency for particles (too many FindBin lookups)
174  float* efficiencyAssociated = nullptr;
175  if (cfg.mEfficiencyAssociated) {
176  efficiencyAssociated = new float[tracks.size()];
177  int i = 0;
178  for (auto& track1 : tracks) {
179  efficiencyAssociated[i++] = getEfficiency(cfg.mEfficiencyAssociated, track1.eta(), track1.pt(), centrality, collision.posZ());
180  }
181  }
182 
183  for (auto& track1 : tracks) {
184  // LOGF(info, "Track %f | %f | %f %d %d", track1.eta(), track1.phi(), track1.pt(), track1.isGlobalTrack(), track1.isGlobalTrackSDD());
185 
186  registry.fill(HIST("yields"), centrality, track1.pt(), track1.eta());
187  registry.fill(HIST("etaphi"), centrality, track1.eta(), track1.phi());
188 
189  if (cfgTriggerCharge != 0 && cfgTriggerCharge * track1.charge() < 0) {
190  continue;
191  }
192 
193  float triggerWeight = 1.0;
194  if (cfg.mEfficiencyTrigger) {
195  triggerWeight = getEfficiency(cfg.mEfficiencyTrigger, track1.eta(), track1.pt(), centrality, collision.posZ());
196  }
197 
198  same->getTriggerHist()->Fill(CorrelationContainer::kCFStepReconstructed, track1.pt(), centrality, collision.posZ(), triggerWeight);
199  //mixed->getTriggerHist()->Fill(eventValues, CorrelationContainer::kCFStepReconstructed);
200 
201  int i = -1;
202  for (auto& track2 : tracks) {
203  i++; // HACK
204  if (track1 == track2) {
205  continue;
206  }
207 
208  if (cfgPtOrder != 0 && track2.pt() >= track1.pt()) {
209  continue;
210  }
211 
212  if (cfgAssociatedCharge != 0 && cfgAssociatedCharge * track2.charge() < 0) {
213  continue;
214  }
215  if (cfgPairCharge != 0 && cfgPairCharge * track1.charge() * track2.charge() < 0) {
216  continue;
217  }
218 
219  if (cfg.mPairCuts && mPairCuts.conversionCuts(track1, track2)) {
220  continue;
221  }
222 
223  if (cfgTwoTrackCut > 0 && mPairCuts.twoTrackCut(track1, track2, bSign)) {
224  continue;
225  }
226 
227  float associatedWeight = 1.0;
228  if (cfg.mEfficiencyAssociated) {
229  associatedWeight = efficiencyAssociated[i];
230  }
231 
232  float deltaPhi = track1.phi() - track2.phi();
233  if (deltaPhi > 1.5 * TMath::Pi()) {
234  deltaPhi -= TMath::TwoPi();
235  }
236  if (deltaPhi < -0.5 * TMath::Pi()) {
237  deltaPhi += TMath::TwoPi();
238  }
239 
240  same->getPairHist()->Fill(CorrelationContainer::kCFStepReconstructed,
241  track1.eta() - track2.eta(), track2.pt(), track1.pt(), centrality, deltaPhi, collision.posZ(),
242  triggerWeight * associatedWeight);
243  //mixed->getPairHist()->Fill(values, CorrelationContainer::kCFStepReconstructed);
244  }
245  }
246 
247  delete[] efficiencyAssociated;
248  }

255  const auto centrality = collision.centV0M();

272  //LOGF(info, "Combination %d %d", track1.index(), track2.index());
273

275  continue;

278  continue;
279  }

281  continue;
282  }
283

285  continue;
286  }
287 
288  if (cfgTwoTrackCut > 0 && mPairCuts.twoTrackCut(track1, track2, bSign)) {
289  continue;
290  }
291 
292  float deltaPhi = track1.phi() - track2.phi();
293  if (deltaPhi > 1.5 * TMath::Pi()) {
294  deltaPhi -= TMath::TwoPi();
295  }
296  if (deltaPhi < -0.5 * TMath::Pi()) {
297  deltaPhi += TMath::TwoPi();
298  }
299 
300  same->getPairHist()->Fill(CorrelationContainer::kCFStepReconstructed,
301  track1.eta() - track2.eta(), track2.pt(), track1.pt(), centrality, deltaPhi, collision.posZ());
302  //mixed->getPairHist()->Fill(values, CorrelationContainer::kCFStepReconstructed);
303  }

306  double getEfficiency(THn* eff, float eta, float pt, float centrality, float posZ)
307  {
308  int effVars[4];
309  effVars[0] = eff->GetAxis(0)->FindBin(eta);

311  effVars[2] = eff->GetAxis(2)->FindBin(centrality);
312  effVars[3] = eff->GetAxis(3)->FindBin(posZ);
313  return eff->GetBinContent(effVars);
