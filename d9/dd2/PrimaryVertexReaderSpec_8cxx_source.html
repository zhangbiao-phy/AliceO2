33 void PrimaryVertexReader::run(ProcessingContext& pc)
34 {
35  auto ent = mTree->GetReadEntry() + 1;
36  assert(ent < mTree->GetEntries()); // this should not happen
37  mTree->GetEntry(ent);
38  LOG(INFO) << "Pushing " << mVerticesPtr->size() << " vertices at entry " << ent;
39 
40  pc.outputs().snapshot(Output{"GLO", "PVTX", 0, Lifetime::Timeframe}, mVertices);
41  pc.outputs().snapshot(Output{"GLO", "PVTX_TRMTC", 0, Lifetime::Timeframe}, mPV2MatchIdx);
42  pc.outputs().snapshot(Output{"GLO", "PVTX_TRMTCREFS", 0, Lifetime::Timeframe}, mPV2MatchIdxRef);
43 
44  if (mUseMC) {
45  pc.outputs().snapshot(Output{"GLO", "PVTX_MCTR", 0, Lifetime::Timeframe}, mLabels);
46  }
47 
48  if (mVerbose) {
49  int cnt = 0;
50  for (const auto& vtx : mVertices) {
51  Label lb;
52  if (mUseMC) {
53  lb = mLabels[cnt];
54  }
55  LOG(INFO) << "#" << cnt << " " << vtx << " | MC:" << lb;
56  LOG(INFO) << "References: " << mPV2MatchIdxRef[cnt];
57  for (int is = 0; is < GIndex::NSources; is++) {
58  LOG(INFO) << GIndex::getSourceName(is) << " : " << mPV2MatchIdxRef[cnt].getEntriesOfSource(is) << " attached:";
59  int idMin = mPV2MatchIdxRef[cnt].getFirstEntryOfSource(is), idMax = idMin + mPV2MatchIdxRef[cnt].getEntriesOfSource(is);
60  std::string trIDs;
61  int cntT = 0;
62  for (int i = idMin; i < idMax; i++) {
63  trIDs += mPV2MatchIdx[i].asString() + " ";
64  if (!((++cntT) % 15)) {
65  LOG(INFO) << trIDs;
66  trIDs = "";
67  }
68  }
69  if (!trIDs.empty()) {
70  LOG(INFO) << trIDs;
71  }
72  }
73  cnt++;
74  }
75  }
76 
77  if (mTree->GetReadEntry() + 1 >= mTree->GetEntries()) {
78  pc.services().get<ControlService>().endOfStream();
79  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
80  }
81 }

83 void PrimaryVertexReader::connectTree()
84 {
85  mTree.reset(nullptr); // in case it was already loaded
86  mFile.reset(TFile::Open(mFileName.c_str()));
87  assert(mFile && !mFile->IsZombie());
88  mTree.reset((TTree*)mFile->Get(mVertexTreeName.c_str()));
89  assert(mTree);
90  assert(mTree->GetBranch(mVertexBranchName.c_str()));
91  assert(mTree->GetBranch(mVertexTrackIDsBranchName.c_str()));
92  assert(mTree->GetBranch(mVertex2TrackIDRefsBranchName.c_str()));
93 
94  mTree->SetBranchAddress(mVertexBranchName.c_str(), &mVerticesPtr);
95  mTree->SetBranchAddress(mVertexTrackIDsBranchName.c_str(), &mPV2MatchIdxPtr);
96  mTree->SetBranchAddress(mVertex2TrackIDRefsBranchName.c_str(), &mPV2MatchIdxRefPtr);
97 
98  if (mUseMC) {
99  assert(mTree->GetBranch(mVertexLabelsBranchName.c_str()));
100  mTree->SetBranchAddress(mVertexLabelsBranchName.c_str(), &mLabelsPtr);
101  }
102 
103  LOG(INFO) << "Loaded " << mVertexTreeName << " tree from " << mFileName << " with " << mTree->GetEntries() << " entries";
104 }

106 DataProcessorSpec getPrimaryVertexReaderSpec(bool useMC)
107 {
108  std::vector<OutputSpec> outputs;
109  outputs.emplace_back("GLO", "PVTX", 0, Lifetime::Timeframe);
110  outputs.emplace_back("GLO", "PVTX_TRMTC", 0, Lifetime::Timeframe);
111  outputs.emplace_back("GLO", "PVTX_TRMTCREFS", 0, Lifetime::Timeframe);
112 
113  if (useMC) {
114  outputs.emplace_back("GLO", "PVTX_MCTR", 0, Lifetime::Timeframe);
115  }
116 
117  return DataProcessorSpec{
118  "primary-vertex-reader",
119  Inputs{},
120  outputs,
121  AlgorithmSpec{adaptFromTask<PrimaryVertexReader>(useMC)},
122  Options{
123  {"primary-vertex-infile", VariantType::String, "o2_primary_vertex.root", {"Name of the input primary vertex file"}},
124  {"vertex-track-matches-infile", VariantType::String, "o2_pvertex_track_matches.root", {"Name of the input file with primary vertex - tracks matches"}}}};
125 }
