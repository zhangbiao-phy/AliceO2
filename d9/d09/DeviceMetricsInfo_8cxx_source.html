26 // Parses a metric in the form
27 //
28 // [METRIC] <name>,<type> <value> <timestamp> [<tag>,<tag>]
29 bool DeviceMetricsHelper::parseMetric(std::string_view const s, ParsedMetricMatch& match)
30 {
33  constexpr size_t PREFIXSIZE = 9;
34  if (s.size() < PREFIXSIZE) {
35  return false;
36  }
37  if (memcmp(s.data(), "[METRIC] ", 9) != 0) {
38  return false;
39  }
40  char const* comma; // first comma
41  char const* spaces[256]; // list of spaces
42  char const** space = spaces; // first element to fill
43 
44  comma = (char const*)memchr((void*)s.data(), ',', s.size());
45  if (comma == nullptr) {
46  return false;
47  }
48  // Find all spaces
49  char const* nextSpace = s.data();
50  while (space - spaces < 256) {
51  *space = strchr(nextSpace, ' ');
52  if (*space == nullptr) {
53  break;
54  }
55  nextSpace = *space + 1;
56  space++;
57  }
58  // First space should always come before comma
59  if (spaces[0] > comma) {
60  return false;
61  }
62  match.beginKey = spaces[0] + 1;
63  match.endKey = comma;
64  // type is alway 1 char after the comma, followed by space
65  if ((spaces[1] - comma) != 2) {
66  return false;
67  }
68  char* ep = nullptr;
69  match.type = static_cast<MetricType>(strtol(comma + 1, &ep, 10));
70  if (ep != spaces[1]) {
71  return false;
72  }
73  // We need at least 4 spaces
74  if (space - spaces < 4) {
75  return false;
76  }
77  // Value is between the second and the last but one space
78  switch (match.type) {
79  case MetricType::Int:
80  match.intValue = strtol(spaces[1] + 1, &ep, 10);
81  if (ep != *(space - 2)) {
82  return false;
83  }
84  break;
85  case MetricType::Float:
86  match.floatValue = strtof(spaces[1] + 1, &ep);
87  if (ep != *(space - 2)) {
88  return false;
89  }
90  break;
91  case MetricType::String:
92  match.beginStringValue = spaces[1] + 1;
93  match.endStringValue = *(space - 2);
94  break;
95  case MetricType::Uint64:
96  match.uint64Value = strtoul(spaces[1] + 1, &ep, 10);
97  if (ep != *(space - 2)) {
98  return false;
99  }
100  break;
101 
102  default:
103  return false;
104  }
105  // Timestamp is between the last but one and the last space
106  match.timestamp = strtol(*(space - 2) + 1, &ep, 10);
107  if (ep != *(space - 1)) {
108  return false;
109  }
110  return true;
111 }

113 bool DeviceMetricsHelper::processMetric(ParsedMetricMatch& match,
114  DeviceMetricsInfo& info,
115  DeviceMetricsHelper::NewMetricCallback newMetricsCallback)
116 {
117  // get the type
118  size_t metricIndex = -1;
119 
120  StringMetric stringValue;
121  switch (match.type) {
122  case MetricType::Float:
123  case MetricType::Int:
124  case MetricType::Uint64:
125  break;
126  case MetricType::String: {
127  auto lastChar = std::min(match.endStringValue - match.beginStringValue, StringMetric::MAX_SIZE - 1);
128  memcpy(stringValue.data, match.beginStringValue, lastChar);
129  stringValue.data[lastChar] = '\0';
130  } break;
131  default:
132  return false;
133  break;
134  };
135 
136  // Find the metric based on the label. Create it if not found.
137  auto cmpFn = [namePtr = match.beginKey,
138  nameSize = match.endKey - match.beginKey](MetricLabelIndex const& a, MetricLabelIndex const& b)
139  -> bool {
140  return strncmp(a.label, namePtr, nameSize) < 0;
141  };
142  auto mi = std::lower_bound(info.metricLabelsIdx.begin(),
143  info.metricLabelsIdx.end(),
144  MetricLabelIndex{},
145  cmpFn);
146 
147  // We could not find the metric, lets insert a new one.
148  if (mi == info.metricLabelsIdx.end() || (strncmp(mi->label, match.beginKey, std::min(match.endKey - match.beginKey, (long)MetricLabelIndex::MAX_METRIC_LABEL_SIZE - 1)) != 0)) {
149  // Create a new metric
150  MetricInfo metricInfo;
151  metricInfo.pos = 0;
152  metricInfo.type = match.type;
153  metricInfo.filledMetrics = 0;
154  // Add a new empty buffer for it of the correct kind
155  switch (match.type) {
156  case MetricType::Int:
157  metricInfo.storeIdx = info.intMetrics.size();
158  info.intMetrics.emplace_back(std::array<int, 1024>{});
159  break;
160  case MetricType::String:
161  metricInfo.storeIdx = info.stringMetrics.size();
162  info.stringMetrics.emplace_back(std::array<StringMetric, 32>{});
163  break;
164  case MetricType::Float:
165  metricInfo.storeIdx = info.floatMetrics.size();
166  info.floatMetrics.emplace_back(std::array<float, 1024>{});
167  break;
168  case MetricType::Uint64:
169  metricInfo.storeIdx = info.uint64Metrics.size();
170  info.uint64Metrics.emplace_back(std::array<uint64_t, 1024>{});
171  break;
172 
173  default:
174  return false;
175  };
176  // Add the timestamp buffer for it
177  info.timestamps.emplace_back(std::array<size_t, 1024>{});
178  info.max.push_back(std::numeric_limits<float>::lowest());
179  info.min.push_back(std::numeric_limits<float>::max());
180  info.maxDomain.push_back(std::numeric_limits<size_t>::lowest());
181  info.minDomain.push_back(std::numeric_limits<size_t>::max());
182  info.changed.push_back(false);
183 
184  // Add the index by name in the correct position
185  // this will require moving the tail of the index,
186  // but inserting should happen only once for each metric,
187  // so who cares.
188  MetricLabelIndex metricLabelIdx;
189  auto lastChar = std::min(match.endKey - match.beginKey, (ptrdiff_t)MetricLabelIndex::MAX_METRIC_LABEL_SIZE - 1);
190  memcpy(metricLabelIdx.label, match.beginKey, lastChar);
191  metricLabelIdx.label[lastChar] = '\0';
192  metricLabelIdx.size = lastChar;
193  metricLabelIdx.index = info.metrics.size();
194  info.metricLabelsIdx.insert(mi, metricLabelIdx);
195  // Add the the actual Metric info to the store
196  metricIndex = info.metrics.size();
197  assert(metricInfo.storeIdx != -1);
198  assert(metricLabelIdx.label[0] != '\0');
199  if (newMetricsCallback != nullptr) {
200  newMetricsCallback(metricLabelIdx.label, metricInfo, match.intValue, metricIndex);
201  }
202  info.metrics.push_back(metricInfo);
203  } else {
204  metricIndex = mi->index;
205  }
206  assert(metricIndex != -1);
207  // We are now guaranteed our metric is present at metricIndex.
208  MetricInfo& metricInfo = info.metrics[metricIndex];
209 
210  // auto mod = info.timestamps[metricIndex].size();
211  info.minDomain[metricIndex] = std::min(info.minDomain[metricIndex], (size_t)match.timestamp);
212  info.maxDomain[metricIndex] = std::max(info.maxDomain[metricIndex], (size_t)match.timestamp);
213 
214  switch (metricInfo.type) {
215  case MetricType::Int: {
216  info.intMetrics[metricInfo.storeIdx][metricInfo.pos] = match.intValue;
217  info.max[metricIndex] = std::max(info.max[metricIndex], (float)match.intValue);
218  info.min[metricIndex] = std::min(info.min[metricIndex], (float)match.intValue);
219  // Save the timestamp for the current metric we do it here
220  // so that we do not update timestamps for broken metrics
221  info.timestamps[metricIndex][metricInfo.pos] = match.timestamp;
222  // Update the position where to write the next metric
223  metricInfo.pos = (metricInfo.pos + 1) % info.intMetrics[metricInfo.storeIdx].size();
224  ++metricInfo.filledMetrics;
225  } break;
226  case MetricType::String: {
227  info.stringMetrics[metricInfo.storeIdx][metricInfo.pos] = stringValue;
228  // Save the timestamp for the current metric we do it here
229  // so that we do not update timestamps for broken metrics
230  info.timestamps[metricIndex][metricInfo.pos] = match.timestamp;
231  metricInfo.pos = (metricInfo.pos + 1) % info.stringMetrics[metricInfo.storeIdx].size();
232  ++metricInfo.filledMetrics;
233  } break;
234  case MetricType::Float: {
235  info.floatMetrics[metricInfo.storeIdx][metricInfo.pos] = match.floatValue;
236  info.max[metricIndex] = std::max(info.max[metricIndex], match.floatValue);
237  info.min[metricIndex] = std::min(info.min[metricIndex], match.floatValue);
238  // Save the timestamp for the current metric we do it here
239  // so that we do not update timestamps for broken metrics
240  info.timestamps[metricIndex][metricInfo.pos] = match.timestamp;
241  metricInfo.pos = (metricInfo.pos + 1) % info.floatMetrics[metricInfo.storeIdx].size();
242  ++metricInfo.filledMetrics;
243  } break;
244  case MetricType::Uint64: {
245  info.uint64Metrics[metricInfo.storeIdx][metricInfo.pos] = match.uint64Value;
246  info.max[metricIndex] = std::max(info.max[metricIndex], (float)match.uint64Value);
247  info.min[metricIndex] = std::min(info.min[metricIndex], (float)match.uint64Value);
248  // Save the timestamp for the current metric we do it here
249  // so that we do not update timestamps for broken metrics
250  info.timestamps[metricIndex][metricInfo.pos] = match.timestamp;
251  // Update the position where to write the next metric
252  metricInfo.pos = (metricInfo.pos + 1) % info.uint64Metrics[metricInfo.storeIdx].size();
253  ++metricInfo.filledMetrics;
254  } break;
255 
256  default:
257  return false;
258  break;
259  };
260  // Note that we updated a given metric.
261  info.changed[metricIndex] = true;
262  return true;
263 }

265 size_t DeviceMetricsHelper::metricIdxByName(const std::string& name, const DeviceMetricsInfo& info)
266 {
267  size_t i = 0;
268  while (i < info.metricLabelsIdx.size()) {
269  auto& metricName = info.metricLabelsIdx[i];
270  // We check the size first and then the last character because that's
271  // likely to be different for multi-index metrics
272  if (metricName.size == name.size() && metricName.label[metricName.size - 1] == name[metricName.size - 1] && memcmp(metricName.label, name.c_str(), metricName.size) == 0) {
273  return metricName.index;
274  }
275  ++i;
276  }
277  return i;
278 }
