68 bool RawWriter::processTrigger(const gsl::span<o2::cpv::Digit> digitsbranch, const o2::cpv::TriggerRecord& trg)
69 {
70 
71  //Array used to sort properly digits
72  for (int i = kNcc; i--;) {
73  for (int j = kNDilogic; j--;) {
74  for (int k = kNGasiplex; k--;) {
75  mPadCharge[i][j][k].clear();
76  }
77  }
78  }
79 
80  for (auto& dig : gsl::span(digitsbranch.data() + trg.getFirstEntry(), trg.getNumberOfObjects())) {
81 
82  short absId = dig.getAbsId();
83  short ccId, dil, gas, pad;
84  o2::cpv::Geometry::absIdToHWaddress(absId, ccId, dil, gas, pad);
85 
86  //Convert Amp to ADC counts
87  short charge = dig.getAmplitude() / mCalibParams->getGain(absId);
88  if (charge > 4095) {
89  charge = 4095;
90  }
91  mPadCharge[ccId][dil][gas].emplace_back(charge, pad);
92  }
93 
94  //Do through DLLs and fill raw words in proper order
95  mPayload.clear();
96 
97  int ccWordCounter = 0;
98  int gbtWordCounter = 0;
99  for (char ccId = 0; ccId < kNcc; ccId++) {
100  for (char dil = 0; dil < kNDilogic; dil++) {
101  for (char gas = 0; gas < kNGasiplex; gas++) {
102  for (padCharge& pc : mPadCharge[ccId][dil][gas]) {
103  // Generate 3 CC words, add CC header and empty bits to complete 128 bits;
104  PadWord currentword = {0};
105  currentword.charge = pc.charge;
106  currentword.address = pc.pad;
107  currentword.gas = gas;
108  currentword.dil = dil;
109  mPayload.push_back(currentword.bytes[0]);
110  mPayload.push_back(currentword.bytes[1]);
111  mPayload.push_back(currentword.bytes[2]);
112  ccWordCounter++;
113  if (ccWordCounter % 3 == 0) { // complete 3 channels (72 bit) + CC index (8 bits) + 6 empty bits = Generate 128 bits of data
114  mPayload.push_back(ccId);
115  for (int i = 6; i--;) {
116  mPayload.push_back(char(0));
117  }
118  gbtWordCounter++;
119  }
120  }
121  if (ccWordCounter % 3 != 0) {
122  while (ccWordCounter % 3 != 0) {
123  mPayload.push_back(char(255));
124  mPayload.push_back(char(255));
125  mPayload.push_back(char(255));
126  ccWordCounter++;
127  }
128  mPayload.push_back(ccId);
129  for (int i = 6; i--;) {
130  mPayload.push_back(char(0));
131  }
132  gbtWordCounter++;
133  }
134  }
135  }
136  }
137  cpvtrailer tr(gbtWordCounter); //cout GBT words
138  for (int i = 0; i < 16; i++) {
139  mPayload.push_back(tr.bytes[i]);
140  }
141 
142  // register output data
143  LOG(DEBUG1) << "Adding payload with size " << mPayload.size() << " char words)";
144 
145  mRawWriter->addData(0, 0, 0, 0, trg.getBCData(), gsl::span<char>(mPayload.data(), mPayload.size()));
146  return true;
147 }
148 int RawWriter::carryOverMethod(const header::RDHAny* rdh, const gsl::span<char> data,
149  const char* ptr, int maxSize, int splitID,
150  std::vector<char>& trailer, std::vector<char>& header) const
151 {
152 
153  constexpr int cpvTrailerSize = 36;
154  int offs = ptr - &data[0]; // offset wrt the head of the payload
155  assert(offs >= 0 && size_t(offs + maxSize) <= data.size()); // make sure ptr and end of the suggested block are within the payload
156  int leftBefore = data.size() - offs; // payload left before this splitting
157  int leftAfter = leftBefore - maxSize; // what would be left after the suggested splitting
158  int actualSize = maxSize;
159  if (leftAfter && leftAfter <= cpvTrailerSize) { // avoid splitting the trailer or writing only it.
160  actualSize -= (cpvTrailerSize - leftAfter) + 4; // (as we work with int, not char in decoding)
161  }
162  return actualSize;
163 }
