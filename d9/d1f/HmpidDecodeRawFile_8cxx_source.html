88 bool HmpidDecodeRawFile::getBlockFromStream(uint32_t** streamPtr, uint32_t Size)
89 {
90  if (Size > MAXRAWFILEBUFFER)
91  return (false);
92  int nr = fread(mFileBuffer, sizeof(int32_t), HEADERDIMENSION_W, fh);
93  if (nr != HEADERDIMENSION_W) {
94  throw TH_WRONGFILELEN;
95  }
96  Size = ((mFileBuffer[2] & 0x0000FFFF) / sizeof(int32_t)) - HEADERDIMENSION_W;
97  nr = fread(mFileBuffer + HEADERDIMENSION_W, sizeof(int32_t), Size, fh);
98  LOG(DEBUG) << " getBlockFromStream read " << nr << " of " << Size + HEADERDIMENSION_W << " words !" << FairLogger::endl;
99  if (nr != Size) {
100  throw TH_WRONGFILELEN;
101  }
102  *streamPtr = mFileBuffer;
103  mStartStreamPtr = mFileBuffer;
104  mActualStreamPtr = mFileBuffer;
105  mEndStreamPtr = mFileBuffer + Size;
106  return (true);
107 }
