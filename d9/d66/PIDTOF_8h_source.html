69 template <typename Coll, typename Trck, o2::track::PID::ID id>
70 float Beta<Coll, Trck, id>::GetBeta(const float length, const float tofSignal, const float collisionTime)
71 {
72  if (tofSignal <= 0) {
73  return -999.f;
74  }
75  return length / (tofSignal - collisionTime) / kCSPEED;
76 }

80 float Beta<Coll, Trck, id>::GetExpectedSigma(const float& length, const float& tofSignal, const float& collisionTime, const float& time_reso)

90 float Beta<Coll, Trck, id>::GetExpectedSignal(const float& mom, const float& mass)
91 {
92  if (mom > 0) {

120 template <typename Coll, typename Trck, o2::track::PID::ID id>
121 float ExpTimes<Coll, Trck, id>::ComputeExpectedTime(const float& tofExpMom, const float& length, const float& massZ)
122 {
123  const float energy = sqrt((massZ * massZ) + (tofExpMom * tofExpMom));
124  return length * energy / (kCSPEED * tofExpMom);
125 }
