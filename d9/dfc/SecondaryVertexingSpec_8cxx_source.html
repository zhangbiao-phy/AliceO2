40 void SecondaryVertexingSpec::init(InitContext& ic)
41 {
42  //-------- init geometry and field --------//
43  o2::base::GeometryManager::loadGeometry();
44  o2::base::Propagator::initFieldFromGRP("o2sim_grp.root");
45  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
46  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
47  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
48  if (o2::base::NameConf::pathExists(matLUTFile)) {
49  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
50  o2::base::Propagator::Instance()->setMatLUT(lut);
51  LOG(INFO) << "Loaded material LUT from " << matLUTFile;
52  } else {
53  LOG(INFO) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
54  }
55  mTimer.Stop();
56  mTimer.Reset();
57 
58  mVertexer.init();
59 }

61 void SecondaryVertexingSpec::run(ProcessingContext& pc)
62 {
63  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
64  mTimer.Start(false);
65 
66  const auto tracksITSTPC = pc.inputs().get<gsl::span<o2::dataformats::TrackTPCITS>>("tpcits");
67  const auto tracksTPC = pc.inputs().get<gsl::span<o2::tpc::TrackTPC>>("tpc");
68  const auto tracksITS = pc.inputs().get<gsl::span<o2::its::TrackITS>>("its");
69 
70  const auto pvertices = pc.inputs().get<gsl::span<o2::dataformats::PrimaryVertex>>("pvtx");
71  const auto pvtxTracks = pc.inputs().get<gsl::span<o2::dataformats::VtxTrackIndex>>("pvtx_cont");
72  const auto pvtxTrackRefs = pc.inputs().get<gsl::span<o2::dataformats::VtxTrackRef>>("pvtx_tref");
73 
74  std::vector<V0> v0s;
75  std::vector<RRef> pv2v0ref;
76 
77  o2::dataformats::GlobalTrackAccessor tracksPool;
78  tracksPool.registerContainer(tracksITSTPC, GIndex::ITSTPC);
79  tracksPool.registerContainer(tracksITS, GIndex::ITS);
80  tracksPool.registerContainer(tracksTPC, GIndex::TPC);
81 
82  mVertexer.process(pvertices, pvtxTracks, pvtxTrackRefs, tracksPool, v0s, pv2v0ref);
83 
84  pc.outputs().snapshot(Output{"GLO", "V0s", 0, Lifetime::Timeframe}, v0s);
85  pc.outputs().snapshot(Output{"GLO", "PVTX_V0REFS", 0, Lifetime::Timeframe}, pv2v0ref);
86 
87  mTimer.Stop();
88  LOG(INFO) << "Found " << v0s.size() << " V0s, timing: CPU: "
89  << mTimer.CpuTime() - timeCPU0 << " Real: " << mTimer.RealTime() - timeReal0 << " s";
90 }

98 DataProcessorSpec getSecondaryVertexingSpec()
99 {
100  std::vector<InputSpec> inputs;
101  std::vector<OutputSpec> outputs;
102 
103  inputs.emplace_back("pvtx", "GLO", "PVTX", 0, Lifetime::Timeframe); // prim.vertices
104  inputs.emplace_back("pvtx_cont", "GLO", "PVTX_TRMTC", 0, Lifetime::Timeframe); // global ids of associated tracks
105  inputs.emplace_back("pvtx_tref", "GLO", "PVTX_TRMTCREFS", 0, Lifetime::Timeframe); // vertex - trackID refs
106  //
107  inputs.emplace_back("tpcits", "GLO", "TPCITS", 0, Lifetime::Timeframe); // matched ITS-TPC tracks
108  inputs.emplace_back("its", "ITS", "TRACKS", 0, Lifetime::Timeframe); // standalone ITS tracks
109  inputs.emplace_back("tpc", "TPC", "TRACKS", 0, Lifetime::Timeframe); // standalone TPC tracks
110  //
111  outputs.emplace_back("GLO", "V0s", 0, Lifetime::Timeframe); // found V0s
112  outputs.emplace_back("GLO", "PVTX_V0REFS", 0, Lifetime::Timeframe); // prim.vertex -> V0s refs
113 
114  return DataProcessorSpec{
115  "secondary-vertexing",
116  inputs,
117  outputs,
118  AlgorithmSpec{adaptFromTask<SecondaryVertexingSpec>()},
119  Options{{"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
120 }
