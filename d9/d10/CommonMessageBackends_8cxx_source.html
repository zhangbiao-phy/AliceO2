134 o2::framework::ServiceSpec CommonMessageBackends::arrowBackendSpec()
135 {
136  using o2::monitoring::Metric;
137  using o2::monitoring::Monitoring;
138  using o2::monitoring::tags::Key;
139  using o2::monitoring::tags::Value;
140 
141  return ServiceSpec{"arrow-backend",
142  CommonMessageBackendsHelpers<ArrowContext>::createCallback(),
143  CommonServices::noConfiguration(),
144  CommonMessageBackendsHelpers<ArrowContext>::clearContext(),
145  CommonMessageBackendsHelpers<ArrowContext>::sendCallback(),
146  nullptr,
147  nullptr,
148  CommonMessageBackendsHelpers<ArrowContext>::clearContextEOS(),
149  CommonMessageBackendsHelpers<ArrowContext>::sendCallbackEOS(),
150  nullptr,
151  nullptr,
152  nullptr,
153  nullptr,
154  nullptr,
155  [](ServiceRegistry& registry,
156  std::vector<DeviceMetricsInfo>& allDeviceMetrics,
157  std::vector<DeviceSpec>& specs,
158  std::vector<DeviceInfo>& infos,
159  DeviceMetricsInfo& driverMetrics,
160  size_t timestamp) {
161  int64_t totalBytesCreated = 0;
162  int64_t totalBytesDestroyed = 0;
163  int64_t totalMessagesCreated = 0;
164  int64_t totalMessagesDestroyed = 0;
165  static RateLimitingState currentState = RateLimitingState::UNKNOWN;
166  static auto stateMetric = DeviceMetricsHelper::createNumericMetric<uint64_t>(driverMetrics, "rate-limit-state");
167  static auto totalBytesCreatedMetric = DeviceMetricsHelper::createNumericMetric<uint64_t>(driverMetrics, "total-arrow-bytes-created");
168  static auto totalBytesDestroyedMetric = DeviceMetricsHelper::createNumericMetric<uint64_t>(driverMetrics, "total-arrow-bytes-destroyed");
169  static auto totalMessagesCreatedMetric = DeviceMetricsHelper::createNumericMetric<uint64_t>(driverMetrics, "total-arrow-messages-created");
170  static auto totalMessagesDestroyedMetric = DeviceMetricsHelper::createNumericMetric<uint64_t>(driverMetrics, "total-arrow-messages-destroyed");
171  static auto totalBytesDeltaMetric = DeviceMetricsHelper::createNumericMetric<int>(driverMetrics, "arrow-bytes-delta");
172  static auto totalSignalsMetric = DeviceMetricsHelper::createNumericMetric<uint64_t>(driverMetrics, "aod-reader-signals");
173  static auto skippedSignalsMetric = DeviceMetricsHelper::createNumericMetric<uint64_t>(driverMetrics, "aod-skipped-signals");
174  static auto remainingBytes = DeviceMetricsHelper::createNumericMetric<uint64_t>(driverMetrics, "aod-remaining-bytes");
175 
176  bool changed = false;
177  bool hasMetrics = false;
178  // Find the last timestamp when we signaled.
179  size_t signalIndex = DeviceMetricsHelper::metricIdxByName("aod-reader-signals", driverMetrics);
180  size_t lastSignalTimestamp = 0;
181  if (signalIndex < driverMetrics.metrics.size()) {
182  MetricInfo info = driverMetrics.metrics.at(signalIndex);
183  lastSignalTimestamp = driverMetrics.timestamps[signalIndex][info.pos - 1];
184  }
185 
186  size_t lastTimestamp = 0;
187  size_t firstTimestamp = -1;
188  size_t lastDecision = 0;
189  for (auto& deviceMetrics : allDeviceMetrics) {
190  {
191  size_t index = DeviceMetricsHelper::metricIdxByName("arrow-bytes-created", deviceMetrics);
192  if (index < deviceMetrics.metrics.size()) {
193  hasMetrics = true;
194  changed |= deviceMetrics.changed.at(index);
195  MetricInfo info = deviceMetrics.metrics.at(index);
196  auto& data = deviceMetrics.uint64Metrics.at(info.storeIdx);
197  totalBytesCreated += (int64_t)data.at((info.pos - 1) % data.size());
198  lastTimestamp = std::max(lastTimestamp, deviceMetrics.timestamps[index][info.pos - 1]);
199  firstTimestamp = std::min(lastTimestamp, firstTimestamp);
200  }
201  }
202  {
203  size_t index = DeviceMetricsHelper::metricIdxByName("arrow-bytes-destroyed", deviceMetrics);
204  if (index < deviceMetrics.metrics.size()) {
205  hasMetrics = true;
206  changed |= deviceMetrics.changed.at(index);
207  MetricInfo info = deviceMetrics.metrics.at(index);
208  auto& data = deviceMetrics.uint64Metrics.at(info.storeIdx);
209  totalBytesDestroyed += (int64_t)data.at((info.pos - 1) % data.size());
210  firstTimestamp = std::min(lastTimestamp, firstTimestamp);
211  }
212  }
213  {
214  size_t index = DeviceMetricsHelper::metricIdxByName("arrow-messages-created", deviceMetrics);
215  if (index < deviceMetrics.metrics.size()) {
216  MetricInfo info = deviceMetrics.metrics.at(index);
217  auto& data = deviceMetrics.uint64Metrics.at(info.storeIdx);
218  totalMessagesCreated += (int64_t)data.at((info.pos - 1) % data.size());
219  }
220  }
221  {
222  size_t index = DeviceMetricsHelper::metricIdxByName("arrow-messages-destroyed", deviceMetrics);
223  if (index < deviceMetrics.metrics.size()) {
224  MetricInfo info = deviceMetrics.metrics.at(index);
225  auto& data = deviceMetrics.uint64Metrics.at(info.storeIdx);
226  totalMessagesDestroyed += (int64_t)data.at((info.pos - 1) % data.size());
227  }
228  }
229  }
230  if (changed) {
231  totalBytesCreatedMetric(driverMetrics, totalBytesCreated, timestamp);
232  totalBytesDestroyedMetric(driverMetrics, totalBytesDestroyed, timestamp);
233  totalMessagesCreatedMetric(driverMetrics, totalMessagesCreated, timestamp);
234  totalMessagesDestroyedMetric(driverMetrics, totalMessagesDestroyed, timestamp);
235  totalBytesDeltaMetric(driverMetrics, totalBytesCreated - totalBytesDestroyed, timestamp);
236  }
237  bool done = false;
238  static int stateTransitions = 0;
239  static int signalsCount = 0;
240  static int skippedCount = 0;
241  static uint64_t now = 0;
242  static uint64_t lastSignal = 0;
243  now = uv_hrtime();
244  while (!done) {
245  stateMetric(driverMetrics, (uint64_t)(currentState), stateTransitions++);
246  switch (currentState) {
247  case RateLimitingState::UNKNOWN: {
248  for (auto& deviceMetrics : allDeviceMetrics) {
249  size_t index = DeviceMetricsHelper::metricIdxByName("arrow-bytes-created", deviceMetrics);
250  if (index < deviceMetrics.metrics.size()) {
251  currentState = RateLimitingState::STARTED;
252  }
253  }
254  done = true;
255  } break;
256  case RateLimitingState::STARTED: {
257  for (size_t di = 0; di < specs.size(); ++di) {
258  if (specs[di].name == "internal-dpl-aod-reader") {
259  if (di < infos.size() && (now - lastSignal > 10000000)) {
260  kill(infos[di].pid, SIGUSR1);
261  totalSignalsMetric(driverMetrics, signalsCount++, timestamp);
262  lastSignal = now;
263  } else {
264  skippedSignalsMetric(driverMetrics, skippedCount++, timestamp);
265  }
266  }
267  }
268  changed = false;
269  currentState = RateLimitingState::CHANGED;
270  } break;
271  case RateLimitingState::CHANGED: {
272  remainingBytes(driverMetrics, totalBytesCreated <= (totalBytesDestroyed + memLimit) ? (totalBytesDestroyed + memLimit) - totalBytesCreated : 0, timestamp);
273  if (totalBytesCreated <= totalBytesDestroyed) {
274  currentState = RateLimitingState::EMPTY;
275  } else if (totalBytesCreated <= (totalBytesDestroyed + memLimit)) {
276  currentState = RateLimitingState::BELOW_LIMIT;
277  } else {
278  currentState = RateLimitingState::ABOVE_LIMIT;
279  }
280  changed = false;
281  } break;
282  case RateLimitingState::EMPTY: {
283  for (size_t di = 0; di < specs.size(); ++di) {
284  if (specs[di].name == "internal-dpl-aod-reader") {
285  if (di < infos.size()) {
286  kill(infos[di].pid, SIGUSR1);
287  totalSignalsMetric(driverMetrics, signalsCount++, timestamp);
288  lastSignal = now;
289  }
290  }
291  }
292  changed = false;
293  currentState = RateLimitingState::NEXT_ITERATION_FROM_BELOW;
294  }
295  case RateLimitingState::BELOW_LIMIT: {
296  for (size_t di = 0; di < specs.size(); ++di) {
297  if (specs[di].name == "internal-dpl-aod-reader") {
298  if (di < infos.size() && (now - lastSignal > 10000000)) {
299  kill(infos[di].pid, SIGUSR1);
300  totalSignalsMetric(driverMetrics, signalsCount++, timestamp);
301  lastSignal = now;
302  } else {
303  skippedSignalsMetric(driverMetrics, skippedCount++, timestamp);
304  }
305  }
306  }
307  changed = false;
308  currentState = RateLimitingState::NEXT_ITERATION_FROM_BELOW;
309  } break;
310  case RateLimitingState::NEXT_ITERATION_FROM_BELOW: {
311  if (!changed) {
312  done = true;
313  } else {
314  currentState = RateLimitingState::CHANGED;
315  }
316  } break;
317  case RateLimitingState::ABOVE_LIMIT: {
318  if (!changed) {
319  done = true;
320  } else if (totalBytesCreated > (totalBytesDestroyed + memLimit / 3)) {
321  done = true;
322  } else {
323  currentState = RateLimitingState::CHANGED;
324  }
325  };
326  };
327  }
328  },
329  [](ProcessingContext& ctx, void* service) {
330  using DataHeader = o2::header::DataHeader;
331  ArrowContext* arrow = reinterpret_cast<ArrowContext*>(service);
332  auto totalBytes = 0;
333  auto totalMessages = 0;
334  for (auto& input : ctx.inputs()) {
335  if (input.header == nullptr) {
336  continue;
337  }
338  auto dh = o2::header::get<DataHeader*>(input.header);
339  if (dh->serialization != o2::header::gSerializationMethodArrow) {
340  continue;
341  }
342  auto dph = o2::header::get<DataProcessingHeader*>(input.header);
343  bool forwarded = false;
344  for (auto const& forward : ctx.services().get<DeviceSpec const>().forwards) {
345  if (DataSpecUtils::match(forward.matcher, dh->dataOrigin, dh->dataDescription, dh->subSpecification)) {
346  forwarded = true;
347  break;
348  }
349  }
350  if (forwarded) {
351  continue;
352  }
353  totalBytes += dh->payloadSize;
354  totalMessages += 1;
355  }
356  arrow->updateBytesDestroyed(totalBytes);
357  arrow->updateMessagesDestroyed(totalMessages);
358  auto& monitoring = ctx.services().get<Monitoring>();
359  monitoring.send(Metric{(uint64_t)arrow->bytesDestroyed(), "arrow-bytes-destroyed"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
360  monitoring.send(Metric{(uint64_t)arrow->messagesDestroyed(), "arrow-messages-destroyed"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
361  monitoring.flushBuffer();
362  },
363  nullptr,
364  ServiceKind::Serial};
365 }

367 o2::framework::ServiceSpec CommonMessageBackends::fairMQBackendSpec()
368 {
369  return ServiceSpec{"fairmq-backend",
370  [](ServiceRegistry& services, DeviceState&, fair::mq::ProgOptions&) -> ServiceHandle {
371  auto& device = services.get<RawDeviceService>();
372  auto context = new MessageContext(FairMQDeviceProxy{device.device()});
373  auto& spec = services.get<DeviceSpec const>();
374 
375  auto dispatcher = [&device](FairMQParts&& parts, std::string const& channel, unsigned int index) {
376  DataProcessor::doSend(*device.device(), std::move(parts), channel.c_str(), index);
377  };
378 
379  auto matcher = [policy = spec.dispatchPolicy](o2::header::DataHeader const& header) {
380  if (policy.triggerMatcher == nullptr) {
381  return true;
382  }
383  return policy.triggerMatcher(Output{header});
384  };
385 
386  if (spec.dispatchPolicy.action == DispatchPolicy::DispatchOp::WhenReady) {
387  context->init(DispatchControl{dispatcher, matcher});
388  }
389  return ServiceHandle{TypeIdHelpers::uniqueId<MessageContext>(), context};
390  },
