52 // HACK: In order to be able to deduce which kind of aod object is transmitted to the templated VarManager::Fill functions
53 // a constexpr static bit map must be defined and sent as template argument
54 // The user has to include in this bit map all the tables needed in analysis, as defined in VarManager::ObjTypes
55 // Additionally, one should make sure that the requested tables are actually provided in the process() function,
56 // otherwise a compile time error will be thrown.
57 // This is a temporary fix until the arrow/ROOT issues are solved, at which point it will be possible
58 // to automatically detect the object types transmitted to the VarManager
59 constexpr static uint32_t gkEventFillMap = VarManager::ObjTypes::BC | VarManager::ObjTypes::Collision | VarManager::ObjTypes::CollisionCent;

134  void process(MyEvent const& collision, aod::Muons const& tracksMuon, aod::BCs const& bcs, soa::Filtered<MyBarrelTracks> const& tracksBarrel)
135  {
136  uint64_t tag = 0;
137  uint32_t triggerAliases = 0;
138  for (int i = 0; i < kNaliases; i++) {
139  if (collision.alias()[i] > 0) {
140  triggerAliases |= (uint32_t(1) << i);
141  }
142  }
143 
144  VarManager::ResetValues(0, VarManager::kNEventWiseVariables, fValues);
145  VarManager::FillEvent<eventFillMap>(collision, fValues); // extract event information and place it in the fgValues array
146  fHistMan->FillHistClass("Event_BeforeCuts", fValues); // automatically fill all the histograms in the class Event
147 
148  if (!fEventCut->IsSelected(fValues)) {
149  return;
150  }
151 
152  fHistMan->FillHistClass("Event_AfterCuts", fValues);
153 
154  event(tag, collision.bc().runNumber(), collision.posX(), collision.posY(), collision.posZ(), collision.numContrib());
155  eventExtended(collision.bc().globalBC(), collision.bc().triggerMask(), triggerAliases, fValues[VarManager::kCentVZERO]);
156  eventVtxCov(collision.covXX(), collision.covXY(), collision.covXZ(), collision.covYY(), collision.covYZ(), collision.covZZ(), collision.chi2());
157 
158  uint64_t trackFilteringTag = 0;
159  trackBasic.reserve(tracksBarrel.size());
160  trackBarrel.reserve(tracksBarrel.size());
161  trackBarrelCov.reserve(tracksBarrel.size());
162  trackBarrelPID.reserve(tracksBarrel.size());
163 
164  for (auto& track : tracksBarrel) {
165  VarManager::FillTrack<gkTrackFillMap>(track, fValues);
166  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", fValues);
167  if (!fTrackCut->IsSelected(fValues)) {
168  continue;
169  }
170  fHistMan->FillHistClass("TrackBarrel_AfterCuts", fValues);
171 
172  if (track.isGlobalTrack()) {
173  trackFilteringTag |= (uint64_t(1) << 0);
174  }
175  if (track.isGlobalTrackSDD()) {
176  trackFilteringTag |= (uint64_t(1) << 1);
177  }
178  trackBasic(event.lastIndex(), track.globalIndex(), trackFilteringTag, track.pt(), track.eta(), track.phi(), track.charge());
179  trackBarrel(track.tpcInnerParam(), track.flags(), track.itsClusterMap(), track.itsChi2NCl(),
180  track.tpcNClsFindable(), track.tpcNClsFindableMinusFound(), track.tpcNClsFindableMinusCrossedRows(),
181  track.tpcNClsShared(), track.tpcChi2NCl(),
182  track.trdChi2(), track.tofChi2(),
183  track.length(), track.dcaXY(), track.dcaZ());
184  trackBarrelCov(track.cYY(), track.cZZ(), track.cSnpSnp(), track.cTglTgl(), track.c1Pt21Pt2());
185  trackBarrelPID(track.tpcSignal(),
186  track.tpcNSigmaEl(), track.tpcNSigmaMu(),
187  track.tpcNSigmaPi(), track.tpcNSigmaKa(), track.tpcNSigmaPr(),
188  track.tpcNSigmaDe(), track.tpcNSigmaTr(), track.tpcNSigmaHe(), track.tpcNSigmaAl(),
189  track.tofSignal(), track.beta(),
190  track.tofNSigmaEl(), track.tofNSigmaMu(),
191  track.tofNSigmaPi(), track.tofNSigmaKa(), track.tofNSigmaPr(),
192  track.tofNSigmaDe(), track.tofNSigmaTr(), track.tofNSigmaHe(), track.tofNSigmaAl(),
193  track.trdSignal());
194  }
195 
196  muonBasic.reserve(selectedMuons.size());
197  muonExtended.reserve(selectedMuons.size());
198  for (auto& muon : selectedMuons) {
199  // TODO: add proper information for muon tracks
200  if (muon.bcId() != collision.bcId()) {
201  continue;
202  }
203  VarManager::FillTrack<gkMuonFillMap>(muon, fValues);
204  fHistMan->FillHistClass("Muons_BeforeCuts", fValues);
205  // TODO: the trackFilteringTag will not be needed to encode whether the track is a muon since there is a dedicated table for muons
206  trackFilteringTag |= (uint64_t(1) << 0); // this is a MUON arm track TODO: to be taken out
207  if (!fMuonCut->IsSelected(fValues)) {
208  continue;
209  }
210  fHistMan->FillHistClass("Muons_AfterCuts", fValues);
211 
212  muonBasic(event.lastIndex(), trackFilteringTag, muon.pt(), muon.eta(), muon.phi(), muon.charge());
213  muonExtended(muon.inverseBendingMomentum(), muon.thetaX(), muon.thetaY(), muon.zMu(), muon.bendingCoor(), muon.nonBendingCoor(), muon.chi2(), muon.chi2MatchTrigger());
214  }
215  }

217  void DefineHistograms(TString histClasses)
218  {
219  std::unique_ptr<TObjArray> objArray(histClasses.Tokenize(";"));
220  for (Int_t iclass = 0; iclass < objArray->GetEntries(); ++iclass) {
221  TString classStr = objArray->At(iclass)->GetName();
222  fHistMan->AddHistClass(classStr.Data());
223 
224  // NOTE: The level of detail for histogramming can be controlled via configurables
225  if (classStr.Contains("Event")) {
226  dqhistograms::DefineHistograms(fHistMan, objArray->At(iclass)->GetName(), "event", "trigger,cent");
227  }
228 
229  if (classStr.Contains("Track")) {
230  dqhistograms::DefineHistograms(fHistMan, objArray->At(iclass)->GetName(), "track", "tpcpid");
231  }
232  if (classStr.Contains("Muons")) {
233  dqhistograms::DefineHistograms(fHistMan, objArray->At(iclass)->GetName(), "track", "muon");
234  }
235  }
236  }
