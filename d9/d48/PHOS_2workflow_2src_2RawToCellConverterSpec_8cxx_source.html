93 void RawToCellConverterSpec::run(framework::ProcessingContext& ctx)
94 {
95  // Cache cells from bunch crossings as the component reads timeframes from many links consecutively
96  std::map<o2::InteractionRecord, std::shared_ptr<std::vector<o2::phos::Cell>>> cellBuffer; // Internal cell buffer/
97  int firstEntry = 0;
98  mOutputHWErrors.clear();
99  if (mFillChi2) {
100  mOutputFitChi.clear();
101  }
102 
103  for (const auto& rawData : framework::InputRecordWalker(ctx.inputs())) {
104 
105  o2::phos::RawReaderMemory rawreader(o2::framework::DataRefUtils::as<const char>(rawData));
106 
107  // loop over all the DMA pages
108  while (rawreader.hasNext()) {
109  try {
110  rawreader.next();
111  } catch (RawDecodingError::ErrorType_t e) {
112  LOG(ERROR) << "Raw decoding error " << (int)e;
113  //add error list
114  mOutputHWErrors.emplace_back(14, (int)e, 1); //Put general errors to non-existing DDL14
115  //if problem in header, abandon this page
116  if (e == RawDecodingError::ErrorType_t::PAGE_NOTFOUND ||
117  e == RawDecodingError::ErrorType_t::HEADER_DECODING ||
118  e == RawDecodingError::ErrorType_t::HEADER_INVALID) {
119  break;
120  }
121  //if problem in payload, try to continue
122  continue;
123  }
124  auto& header = rawreader.getRawHeader();
125  auto triggerBC = o2::raw::RDHUtils::getTriggerBC(header);
126  auto triggerOrbit = o2::raw::RDHUtils::getTriggerOrbit(header);
127  auto ddl = o2::raw::RDHUtils::getFEEID(header);
128 
129  o2::InteractionRecord currentIR(triggerBC, triggerOrbit);
130  std::shared_ptr<std::vector<o2::phos::Cell>> currentCellContainer;
131  auto found = cellBuffer.find(currentIR);
132  if (found == cellBuffer.end()) {
133  currentCellContainer = std::make_shared<std::vector<o2::phos::Cell>>();
134  cellBuffer[currentIR] = currentCellContainer;
135  } else {
136  currentCellContainer = found->second;
137  }
138  if (ddl > o2::phos::Mapping::NDDL) { //only 14 correct DDLs
139  LOG(ERROR) << "DDL=" << ddl;
140  mOutputHWErrors.emplace_back(15, 16, char(ddl)); //Add non-existing DDL as DDL 15
141  continue; //skip STU ddl
142  }
143  // use the altro decoder to decode the raw data, and extract the RCU trailer
144  o2::phos::AltroDecoder decoder(rawreader);
145  AltroDecoderError::ErrorType_t err = decoder.decode();
146 
147  if (err != AltroDecoderError::kOK) {
148  //TODO handle severe errors
149  //TODO: probably careful conversion of decoder errors to Fitter errors?
150  char e = (char)err;
151  mOutputHWErrors.emplace_back(ddl, 16, e); //assign general header errors to non-existing FEE 16
152  }
153  auto& rcu = decoder.getRCUTrailer();
154  auto& channellist = decoder.getChannels();
155  // Loop over all the channels for this RCU
156  for (auto& chan : channellist) {
157  short absId;
158  Mapping::CaloFlag caloFlag;
159  short fee;
160  char e2 = CheckHWAddress(ddl, chan.getHardwareAddress(), fee);
161  if (e2) {
162  mOutputHWErrors.emplace_back(ddl, fee, e2);
163  continue;
164  }
165  Mapping::ErrorStatus s = mMapping->hwToAbsId(ddl, chan.getHardwareAddress(), absId, caloFlag);
166  if (s != Mapping::ErrorStatus::kOK) {
167  mOutputHWErrors.emplace_back(ddl, 15, (char)s); //use non-existing FEE 15 for general errors
168  continue;
169  }
170  if (caloFlag != Mapping::kTRU) { //HighGain or LowGain
171  CaloRawFitter::FitStatus fitResults = mRawFitter->evaluate(chan.getBunches());
172  if (fitResults == CaloRawFitter::FitStatus::kNoTime) {
173  mOutputHWErrors.emplace_back(ddl, fee, (char)5); //Time evaluation error occured
174  }
175  if (mFillChi2) {
176  for (int is = 0; is < mRawFitter->getNsamples(); is++) {
177  if (!mRawFitter->isOverflow(is)) { //Overflow is will show wrong chi2
178  short chiAddr = absId;
179  chiAddr |= caloFlag << 14;
180  mOutputFitChi.emplace_back(chiAddr);
181  mOutputFitChi.emplace_back(short(mRawFitter->getChi2(is)));
182  }
183  }
184  }
185  if (fitResults == CaloRawFitter::FitStatus::kOK || fitResults == CaloRawFitter::FitStatus::kNoTime) {
186  //TODO: which results should be accepted? full configurable list
187  if (!mPedestalRun) {
188  for (int is = 0; is < mRawFitter->getNsamples(); is++) {
189  if (caloFlag == Mapping::kHighGain && !mRawFitter->isOverflow(is)) {
190  currentCellContainer->emplace_back(absId, mRawFitter->getAmp(is),
191  mRawFitter->getTime(is) * o2::phos::PHOSSimParams::Instance().mTimeTick, (ChannelType_t)caloFlag);
192  }
193  if (caloFlag == Mapping::kLowGain) {
194  currentCellContainer->emplace_back(absId, mRawFitter->getAmp(is),
195  mRawFitter->getTime(is) * o2::phos::PHOSSimParams::Instance().mTimeTick, (ChannelType_t)caloFlag);
196  }
197  }
198  } else { //pedestal, to store RMS, scale in by 1.e-7 to fit range
199  currentCellContainer->emplace_back(absId, mRawFitter->getAmp(0), 1.e-7 * mRawFitter->getTime(0), (ChannelType_t)caloFlag);
200  }
201  }
202  }
203  }
204  } //RawReader::hasNext
205  }
206 
207  // Loop over BCs, sort cells with increasing cell ID and write to output containers
208  mOutputCells.clear();
209  mOutputTriggerRecords.clear();
210  for (auto [bc, cells] : cellBuffer) {
211  int prevCellSize = mOutputCells.size();
212  if (cells->size()) {
213  // Sort cells according to cell ID
214  std::sort(cells->begin(), cells->end(), [](o2::phos::Cell& lhs, o2::phos::Cell& rhs) { return lhs.getAbsId() < rhs.getAbsId(); });
215 
216  if (mCombineGHLG && !mPedestalRun) { // combine for normal data, do not combine e.g. for LED run and pedestal
217  //Combine HG and LG sells
218  //Should be next to each other after sorting
219  auto it1 = cells->begin();
220  auto it2 = cells->begin();
221  it2++;
222  while (it1 != cells->end()) {
223  if (it2 != cells->end()) {
224  if ((*it1).getAbsId() == (*it2).getAbsId()) { //HG and LG channels, if both, copy only HG as more precise
225  if ((*it1).getType() == o2::phos::HIGH_GAIN) {
226  mOutputCells.push_back(*it1);
227  } else {
228  mOutputCells.push_back(*it2);
229  }
230  ++it1; //yes increase twice
231  ++it2;
232  } else { //no double cells, copy this one
233  mOutputCells.push_back(*it1);
234  }
235  } else { //just copy last one
236  mOutputCells.push_back(*it1);
237  }
238  ++it1;
239  ++it2;
240  }
241  } else {
242  for (auto& cell : *cells) {
243  mOutputCells.push_back(cell);
244  }
245  }
246  }
247 
248  mOutputTriggerRecords.emplace_back(bc, prevCellSize, mOutputCells.size() - prevCellSize);
249  }
250  cellBuffer.clear();
251 
252  LOG(INFO) << "[PHOSRawToCellConverter - run] Writing " << mOutputCells.size() << " cells ...";
253  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLS", 0, o2::framework::Lifetime::Timeframe}, mOutputCells);
254  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggerRecords);
255  ctx.outputs().snapshot(o2::framework::Output{"PHS", "RAWHWERRORS", 0, o2::framework::Lifetime::Timeframe}, mOutputHWErrors);
256  if (mFillChi2) {
257  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLFITQA", 0, o2::framework::Lifetime::Timeframe}, mOutputFitChi);
258  }
259 }

261 char RawToCellConverterSpec::CheckHWAddress(short ddl, short hwAddr, short& fee)
262 {
263 
264  if (ddl < 0 || ddl > o2::phos::Mapping::NDDL) {
265  return (char)4;
266  }
267  // short chan = hwAddr & 0xf;
268  short chip = hwAddr >> 4 & 0x7;
269  short fec = hwAddr >> 7 & 0xf;
270  short branch = hwAddr >> 11 & 0x1;
271 
272  if (branch < 0 || branch > 1) {
273  return (char)1;
274  }
275  if (fec < 0 || fec > 15) {
276  return (char)2;
277  }
278  if (fec != 0 && (chip < 0 || chip > 4 || chip == 1)) { //Do not check for TRU (fec=0)
279  return (char)3;
280  }
281  return (char)0;
282 }
